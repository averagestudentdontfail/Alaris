69 public static PricingRegime DetermineRegime(double riskFreeRate, double dividendYield, bool isCall)
70 {
71 if (riskFreeRate >= 0)
72 {
73 // Check for call double boundary: 0 < r < q
74 if (isCall && dividendYield > riskFreeRate)
75 {
76 return PricingRegime.DoubleBoundary;
77 }
78 // Standard regime: positive rates, single boundary
79 return PricingRegime.PositiveRates;
80 }
81 else
82 {
83 // Negative rate regime
84 if (!isCall && dividendYield < riskFreeRate)
85 {
86 // Put double boundary regime: q < r < 0
87 return PricingRegime.DoubleBoundary;
88 }
89 else
90 {
91 // Negative rates but single boundary
92 return PricingRegime.NegativeRatesSingleBoundary;
93 }
94 }

100 public Task<OptionPricing> PriceOption(STDT003A parameters)
101 {
102 ArgumentNullException.ThrowIfNull(parameters);
103 parameters.Validate();
104
105 return Task.Run(() =>
106 {
107 try
108 {
109 bool isCall = parameters.OptionType == OptionType.Call;
110 PricingRegime regime = DetermineRegime(parameters.RiskFreeRate, parameters.DividendYield, isCall);
111 double timeToExpiry = parameters.TimeToExpiry();
112
113 SafeLog(() =>
114 {
115 string paramsStr = $"r={parameters.RiskFreeRate:F4}, q={parameters.DividendYield:F4}, Ïƒ={parameters.ImpliedVolatility:F4}, T={timeToExpiry:F4}";
116 LogPricingOption(_logger!, parameters.UnderlyingPrice, parameters.Strike, paramsStr,
117 isCall ? "Call" : "Put", regime, null);
118 });
119
120 // Price using native FD engine (works for all regimes)
121 double price = _nativeEngine.Price(
122 parameters.UnderlyingPrice,
123 parameters.Strike,
124 timeToExpiry,
125 parameters.RiskFreeRate,
126 parameters.DividendYield,
127 parameters.ImpliedVolatility,
128 parameters.OptionType);
129
130 // Calculate Greeks
131 double delta = _nativeEngine.Delta(
132 parameters.UnderlyingPrice, parameters.Strike, timeToExpiry,
133 parameters.RiskFreeRate, parameters.DividendYield, parameters.ImpliedVolatility,
134 parameters.OptionType);
135
136 double gamma = _nativeEngine.Gamma(
137 parameters.UnderlyingPrice, parameters.Strike, timeToExpiry,
138 parameters.RiskFreeRate, parameters.DividendYield, parameters.ImpliedVolatility,
139 parameters.OptionType);
140
141 double vega = _nativeEngine.Vega(
142 parameters.UnderlyingPrice, parameters.Strike, timeToExpiry,
143 parameters.RiskFreeRate, parameters.DividendYield, parameters.ImpliedVolatility,
144 parameters.OptionType);
145
146 double theta = _nativeEngine.Theta(
147 parameters.UnderlyingPrice, parameters.Strike, timeToExpiry,
148 parameters.RiskFreeRate, parameters.DividendYield, parameters.ImpliedVolatility,
149 parameters.OptionType);
150
151 double rho = _nativeEngine.Rho(
152 parameters.UnderlyingPrice, parameters.Strike, timeToExpiry,
153 parameters.RiskFreeRate, parameters.DividendYield, parameters.ImpliedVolatility,
154 parameters.OptionType);
155
156 double moneyness = parameters.UnderlyingPrice / parameters.Strike;
157
158 return new OptionPricing
159 {
160 Price = price,
161 Delta = delta,
162 Gamma = gamma,
163 Vega = vega,
164 Theta = theta,
165 Rho = rho,
166 ImpliedVolatility = parameters.ImpliedVolatility,
167 TimeToExpiry = timeToExpiry,
168 Moneyness = moneyness
169 };
170 }
171 catch (ArgumentException ex)
172 {
173 SafeLog(() => LogErrorPricing(_logger!, ex));
174 throw;
175 }
176 catch (InvalidOperationException ex)
177 {
178 SafeLog(() => LogErrorPricing(_logger!, ex));
179 throw;
180 }
181 });
182 }

187 public async Task<STPR001APricing> PriceSTPR001A(STPR001AParameters parameters)
188 {
189 ArgumentNullException.ThrowIfNull(parameters);
190 parameters.Validate();
191
192 bool isCall = parameters.OptionType == OptionType.Call;
193 PricingRegime regime = DetermineRegime(parameters.RiskFreeRate, parameters.DividendYield, isCall);
194
195 SafeLog(() => LogPricingSTPR001A(_logger!, regime, isCall ? "Call" : "Put", parameters.Strike,
196 CalculateDaysToExpiry(parameters.ValuationDate, parameters.FrontExpiry),
197 CalculateDaysToExpiry(parameters.ValuationDate, parameters.BackExpiry), null));
198
199 // Price both legs of the spread
200 OptionPricing frontPricing = await PriceOption(CreateSTDT003A(parameters, parameters.FrontExpiry)).ConfigureAwait(false);
201 OptionPricing backPricing = await PriceOption(CreateSTDT003A(parameters, parameters.BackExpiry)).ConfigureAwait(false);
202
203 // Calculate spread metrics
204 double spreadCost = backPricing.Price - frontPricing.Price;
205 double maxProfit = await CalculateMaxProfit(parameters, spreadCost).ConfigureAwait(false);
206 double breakEven = await CalculateBreakEven(parameters, spreadCost).ConfigureAwait(false);
207
208 return BuildSTPR001APricing(frontPricing, backPricing, spreadCost, maxProfit, breakEven);
209 }

287 private async Task<double> CalculateMaxProfit(STPR001AParameters parameters, double spreadCost)
288 {
289 // At front expiration, max profit occurs when underlying is at strike
290 // The back option still has time value
291 double frontTimeToExpiry = DayCounters.Actual365Fixed.YearFraction(parameters.ValuationDate, parameters.FrontExpiry);
292
293 // Create params for back option at front expiry
294 CRTM005A frontExpiryDate = parameters.FrontExpiry;
295
296 STDT003A backAtFrontExpiry = new STDT003A
297 {
298 UnderlyingPrice = parameters.Strike, // ATM at expiry
299 Strike = parameters.Strike,
300 Expiry = parameters.BackExpiry,
301 ImpliedVolatility = parameters.ImpliedVolatility,
302 RiskFreeRate = parameters.RiskFreeRate,
303 DividendYield = parameters.DividendYield,
304 OptionType = parameters.OptionType,
305 ValuationDate = frontExpiryDate
306 };
307
308 OptionPricing backValue = await PriceOption(backAtFrontExpiry).ConfigureAwait(false);
309
310 // Max profit = back option value at front expiry (when ATM) - initial debit
311 return backValue.Price - spreadCost;
312 }
