106 public void Update_MultipleSteps_ReducesVariance()
107 {
108 // Arrange
109 var filter = new Alaris.Strategy.Core.STKF001A();
110 filter.Reset(0.20, 0.05);
111
112 double initialVariance = filter.Variance;
113
114 // Act

116 {
117 filter.Update(0.21 + (0.01 * Math.Sin(i)), sampleSize: 30);
118 }
119
120 // Assert

122 "repeated measurements should reduce estimation uncertainty");
123 }
