38 // ========== Single Option Timing ==========
39
40 [Fact]
41 public void SingleOption_SpeedComparison()
42 {
43 const int warmupIterations = 10;
44 const int testIterations = 100;
45
46 // Warm up JIT
47 for (int i = 0; i < warmupIterations; i++)
48 {
49 _ = _spectralFast.Price(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
50 _ = _spectralAccurate.Price(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
51 _ = _fdEngine.Price(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
52 }
53
54 // Spectral Fast timing
55 var swFast = Stopwatch.StartNew();
56 for (int i = 0; i < testIterations; i++)
57 {
58 _ = _spectralFast.Price(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
59 }
60 swFast.Stop();
61 double fastAvgUs = swFast.Elapsed.TotalMicroseconds / testIterations;
62
63 // Spectral Accurate timing
64 var swAccurate = Stopwatch.StartNew();
65 for (int i = 0; i < testIterations; i++)
66 {
67 _ = _spectralAccurate.Price(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
68 }
69 swAccurate.Stop();
70 double accurateAvgUs = swAccurate.Elapsed.TotalMicroseconds / testIterations;
71
72 // FD timing
73 var swFD = Stopwatch.StartNew();
74 for (int i = 0; i < testIterations; i++)
75 {
76 _ = _fdEngine.Price(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
77 }
78 swFD.Stop();
79 double fdAvgUs = swFD.Elapsed.TotalMicroseconds / testIterations;
80
81 _output.WriteLine("== Single Option Timing (µs) ==");
82 _output.WriteLine($"Spectral Fast: {fastAvgUs:F1}");
83 _output.WriteLine($"Spectral Accurate: {accurateAvgUs:F1}");
84 _output.WriteLine($"Finite Difference: {fdAvgUs:F1}");
85 _output.WriteLine($"FD/Spectral Ratio: {fdAvgUs / accurateAvgUs:F1}x");
86
87 // Both should be reasonably fast
88 Assert.True(accurateAvgUs < 10000, $"Spectral should be < 10ms, got {accurateAvgUs}µs");
89 }

91 // ========== Portfolio Throughput ==========
92
93 [Fact]
94 public void Portfolio_ThroughputComparison()
95 {
96 const int portfolioSize = 500;
97 var random = new Random(42);
98
99 // Generate portfolio
100 var options = Enumerable.Range(0, portfolioSize).Select(_ => (
101 Spot: 100.0,
102 Strike: 80.0 + (random.NextDouble() * 40.0),
103 Tau: 0.1 + (random.NextDouble() * 2.0),
104 R: 0.01 + (random.NextDouble() * 0.08),
105 Q: random.NextDouble() * 0.05,
106 Sigma: 0.10 + (random.NextDouble() * 0.40)
107 )).ToArray();
108
109 // Warm up
110 _ = _spectralAccurate.Price(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
111 _ = _fdEngine.Price(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
112
113 // Spectral
114 var swSpectral = Stopwatch.StartNew();
115 foreach (var opt in options)
116 {
117 _ = _spectralAccurate.Price(opt.Spot, opt.Strike, opt.Tau, opt.R, opt.Q, opt.Sigma, OptionType.Put);
118 }
119 swSpectral.Stop();
120
121 // FD
122 var swFD = Stopwatch.StartNew();
123 foreach (var opt in options)
124 {
125 _ = _fdEngine.Price(opt.Spot, opt.Strike, opt.Tau, opt.R, opt.Q, opt.Sigma, OptionType.Put);
126 }
127 swFD.Stop();
128
129 double spectralThroughput = portfolioSize / swSpectral.Elapsed.TotalSeconds;
130 double fdThroughput = portfolioSize / swFD.Elapsed.TotalSeconds;
131
132 _output.WriteLine($"== Portfolio Throughput (N={portfolioSize}) ==");
133 _output.WriteLine($"Spectral: {spectralThroughput:F0} options/sec ({swSpectral.ElapsedMilliseconds}ms total)");
134 _output.WriteLine($"FD: {fdThroughput:F0} options/sec ({swFD.ElapsedMilliseconds}ms total)");
135 _output.WriteLine($"Ratio: {spectralThroughput / fdThroughput:F2}x");
136
137 // Should process reasonable throughput
138 Assert.True(spectralThroughput > 50, $"Should process > 50 options/sec, got {spectralThroughput:F0}");
139 }

141 // ========== Accuracy vs Speed Tradeoff ==========
142
143 [Fact]
144 public void AccuracySpeedTradeoff_SchemeComparison()
145 {
146 // Reference: use high precision spectral as benchmark
147 double reference = _spectralHighPrecision.Price(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
148
149 // Measure each scheme
150 var schemes = new (string Name, CREN004A Engine)[]
151 {
152 ("Fast", _spectralFast),
153 ("Accurate", _spectralAccurate),
154 ("HighPrecision", _spectralHighPrecision)
155 };
156
157 _output.WriteLine("== Accuracy vs Speed Tradeoff ==");
158 _output.WriteLine($"Reference (HighPrecision): {reference:F8}");
159 _output.WriteLine("");
160
161 foreach (var (name, engine) in schemes)
162 {
163 const int iterations = 50;
164
165 // Time it
166 var sw = Stopwatch.StartNew();
167 double price = 0;
168 for (int i = 0; i < iterations; i++)
169 {
170 price = engine.Price(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
171 }
172 sw.Stop();
173
174 double error = System.Math.Abs(price - reference);
175 double avgUs = sw.Elapsed.TotalMicroseconds / iterations;
176
177 _output.WriteLine($"{name,-15}: Price={price:F6}, Error={error:E2}, Time={avgUs:F0}µs");
178 }
179 }

197 _ = _spectralAccurate.Gamma(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
198 _ = _spectralAccurate.Vega(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
199 }
200 swSpectral.Stop();
201
202 // FD Greeks
203 var swFD = Stopwatch.StartNew();
204 for (int i = 0; i < iterations; i++)
205 {
206 _ = _fdEngine.Delta(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
207 _ = _fdEngine.Gamma(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
208 _ = _fdEngine.Vega(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
209 }
210 swFD.Stop();
211
212 _output.WriteLine("== Greeks Computation (Δ + Γ + ν) ==");
213 _output.WriteLine($"Spectral: {swSpectral.ElapsedMilliseconds}ms for {iterations} iterations");
214 _output.WriteLine($"FD: {swFD.ElapsedMilliseconds}ms for {iterations} iterations");
