275 private static List<PriceBar> CreateSamplePriceData()
276 {
277 var prices = new List<PriceBar>();
278 var baseDate = DateTime.Today.AddDays(-400);
279 double basePrice = 100.0;
280
281 for (int i = 0; i < 400; i++)
282 {
283 // Skip weekends
284 var date = baseDate.AddDays(i);
285 if (date.DayOfWeek == DayOfWeek.Saturday || date.DayOfWeek == DayOfWeek.Sunday)
286 {
287 continue;
288 }
289
290 // Simulate price with random walk
291 double dailyReturn = (i % 90 < 2) ? 0.05 * (i % 2 == 0 ? 1 : -1) : 0.01 * ((i % 3) - 1);
292 double close = basePrice * (1 + dailyReturn);
293
294 prices.Add(new PriceBar
295 {
296 Date = date,
297 Open = basePrice,
298 High = Math.Max(basePrice, close) * 1.005,
299 Low = Math.Min(basePrice, close) * 0.995,
300 Close = close,
301 Volume = 1000000
302 });
303
304 basePrice = close;
305 }
306
307 return prices;
308 }

310 private static List<DateTime> CreateSampleEarningsDates()
311 {
312 var dates = new List<DateTime>();
313 var baseDate = DateTime.Today;
314
315 // Create quarterly earnings dates going back 3 years
316 for (int q = 1; q <= 12; q++)
317 {
318 var earningsDate = baseDate.AddDays(-90 * q);
319 // Ensure not weekend
320 if (earningsDate.DayOfWeek == DayOfWeek.Saturday)
321 {
322 earningsDate = earningsDate.AddDays(-1);
323 }
324 else if (earningsDate.DayOfWeek == DayOfWeek.Sunday)
325 {
326 earningsDate = earningsDate.AddDays(-2);
327 }
328 dates.Add(earningsDate);
329 }
330
331 return dates;
332 }
