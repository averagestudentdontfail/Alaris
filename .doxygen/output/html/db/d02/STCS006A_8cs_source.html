103 public STCS007A Validate(
104 Signal signal,
105 STCS002A frontLegParams,
106 STCS002A backLegParams)
107 {
108 ArgumentNullException.ThrowIfNull(signal);
109 ArgumentNullException.ThrowIfNull(frontLegParams);
110 ArgumentNullException.ThrowIfNull(backLegParams);
111
112 try
113 {
114 // Compute spread execution costs
115 STCS004A spreadCost = _costModel.ComputeSpreadCost(frontLegParams, backLegParams);
116
117 // Compute post-cost IV/RV ratio
118 // The execution cost degrades the effective edge
119 double preCostRatio = signal.IVRVRatio;
120 double postCostRatio = ComputePostCostRatio(preCostRatio, spreadCost);
121
122 // Validate against thresholds
123 bool passesRatioThreshold = postCostRatio >= _minimumPostCostIVRVRatio;
124 bool passesSlippageThreshold = spreadCost.SlippagePercent <= _maximumSlippagePercent;
125 bool passesCostThreshold = spreadCost.ExecutionCostPercent <= _maximumExecutionCostPercent;
126
127 // Check for high slippage warning
128 if (!passesSlippageThreshold)
129 {
130 SafeLog(() => LogSlippageWarning(
131 _logger!,
132 signal.Symbol,
133 spreadCost.SlippagePercent,
134 _maximumSlippagePercent,
135 null));
136 }
137
138 bool overallPass = passesRatioThreshold && passesSlippageThreshold && passesCostThreshold;
139
140 var result = new STCS007A
141 {
142 Symbol = signal.Symbol,
143 PreCostIVRVRatio = preCostRatio,
144 PostCostIVRVRatio = postCostRatio,
145 MinimumRequiredRatio = _minimumPostCostIVRVRatio,
146 SpreadCost = spreadCost,
147 PassesRatioThreshold = passesRatioThreshold,
148 PassesSlippageThreshold = passesSlippageThreshold,
149 PassesCostThreshold = passesCostThreshold,
150 OverallPass = overallPass,
151 CostModel = _costModel.ModelName
152 };
153
154 SafeLog(() => LogValidationResult(
155 _logger!,
156 signal.Symbol,
157 preCostRatio,
158 postCostRatio,
159 overallPass,
160 null));
161
162 return result;
163 }
164 catch (InvalidOperationException ex)
165 {
166 SafeLog(() => LogValidationError(_logger!, signal.Symbol, ex));
167 throw;
168 }
169 }

178 private static double ComputePostCostRatio(double preCostRatio, STCS004A spreadCost)
179 {
180 // Execution cost degrades the ratio proportionally
181 // If costs are 5% of capital, the effective ratio is reduced by 5%
182 double costFactor = 1.0 - (spreadCost.ExecutionCostPercent / 100.0);
183
184 // Ensure factor is non-negative
185 costFactor = Math.Max(0.0, costFactor);
186
187 return preCostRatio * costFactor;
188 }
