1// CREN002A.cs - Unified American Option Pricing Engine
2// Component ID: CREN002A
3//
4// Purpose: Unified pricing engine interface for all rate regimes
5// - Standard regime (r >= 0, q >= 0): Uses CREN001A Crank-Nicolson FD
6// - Negative rate regime: Uses Alaris.Double QD+ methodology
7//
8// Mathematical Improvements over CREN001A:
9// - ASINH grid distribution: denser points near spot, sparser at boundaries
10// - Gamma = 0 boundary conditions (Neumann): more universal than fixed delta
11//
12// References:
13// - Copenhagen FinTech lectures (Saxo PUK) on FD methods
14// - Wilmott, P. "Paul Wilmott on Quantitative Finance"
15// - Healy (2021) QD+ for negative rates
16// - Alaris.Governance/Coding.md

95 double spot,
96 double strike,
97 double timeToExpiry,
98 double riskFreeRate,
99 double dividendYield,
100 double volatility,
101 OptionType optionType)
102 {

131 private double PriceWithCrankNicolson(
132 double spot,
133 double strike,
134 double timeToExpiry,
135 double riskFreeRate,
136 double dividendYield,
137 double volatility,
138 OptionType optionType)
139 {
140 double sigma = volatility;
141 double sigma2 = sigma * sigma;
142 double dt = timeToExpiry / _timeSteps;
143 double r = riskFreeRate;
144 double q = dividendYield;
145
146 int n = _spotSteps;
147
148 // Rent arrays from pool
149 double[] x = ArrayPool<double>.Shared.Rent(n);
150 double[] dx = ArrayPool<double>.Shared.Rent(n);
151 double[] v = ArrayPool<double>.Shared.Rent(n);
152 double[] vNew = ArrayPool<double>.Shared.Rent(n);
153 double[] a = ArrayPool<double>.Shared.Rent(n);
154 double[] b = ArrayPool<double>.Shared.Rent(n);
155 double[] c = ArrayPool<double>.Shared.Rent(n);
156 double[] d = ArrayPool<double>.Shared.Rent(n);
157
158 try
159 {
160 // Build ASINH-distributed grid (denser near spot)
161 BuildAsinhGrid(x, dx, n, spot, strike, sigma, timeToExpiry);
162
163 // Terminal condition: V(T, S) = payoff(S)
164 for (int i = 0; i < n; i++)
165 {
166 double spotAtNode = System.Math.Exp(x[i]);
167 v[i] = CalculatePayoff(spotAtNode, strike, optionType);
168 }
169
170 // Backward induction
171 for (int step = 0; step < _timeSteps; step++)
172 {
173 // Build tridiagonal system with variable grid spacing
174 BuildTridiagonalSystemVariable(n, dt, sigma2, r, q, x, dx, v, a, b, c, d);
175
176 // Apply Neumann boundary conditions (gamma = 0)
177 ApplyNeumannBoundaries(n, a, b, c, d, v);
178
179 // Solve tridiagonal system
180 SolveTridiagonal(n, a, b, c, d, vNew);
181
182 // Early exercise check
183 for (int i = 0; i < n; i++)
184 {
185 double spotAtNode = System.Math.Exp(x[i]);
186 double intrinsic = CalculatePayoff(spotAtNode, strike, optionType);
187 vNew[i] = System.Math.Max(vNew[i], intrinsic);
188 }
189
190 // Swap for next iteration
191 (v, vNew) = (vNew, v);
192 }
193
194 // Interpolate to get price at current spot
195 double logSpot = System.Math.Log(spot);
196 return InterpolatePrice(x, v, n, logSpot);
197 }
198 finally
199 {
200 ArrayPool<double>.Shared.Return(x);
201 ArrayPool<double>.Shared.Return(dx);
202 ArrayPool<double>.Shared.Return(v);
203 ArrayPool<double>.Shared.Return(vNew);
204 ArrayPool<double>.Shared.Return(a);
205 ArrayPool<double>.Shared.Return(b);
206 ArrayPool<double>.Shared.Return(c);
207 ArrayPool<double>.Shared.Return(d);
208 }
209 }

224 [MethodImpl(MethodImplOptions.AggressiveInlining)]
225 private void BuildAsinhGrid(double[] x, double[] dx, int n, double spot, double strike, double sigma, double T)
226 {
227 // Grid in log-spot space centered at log(spot)
228 double xMid = System.Math.Log(spot);
229
230 // Scale controls how many standard deviations the grid spans
231 double scale = 10.0; // kwinto-cuda uses 50, we use 10 for more focus near ATM
232 double density = _gridConcentration; // Controls point concentration (0.1-0.5 typical)
233
234 // Grid boundaries in log-space
235 double xMin = xMid - (scale * sigma * System.Math.Sqrt(T));
236 double xMax = xMid + (scale * sigma * System.Math.Sqrt(T));
237
238 // Transform to y-space using asinh
239 double yMin = Asinh((xMin - xMid) / density);
240 double yMax = Asinh((xMax - xMid) / density);
241
242 // Generate grid using sinh interpolation (QuantLib formula)
243 double dy = 1.0 / (n - 1);
244 for (int i = 0; i < n; i++)
245 {
246 double xi = i * dy; // Uniform [0, 1]
247 double y = (yMin * (1.0 - xi)) + (yMax * xi); // Linear interpolate in y-space
248 x[i] = xMid + (density * System.Math.Sinh(y)); // Transform back to x-space
249 }
250
251 // Compute local grid spacing
252 for (int i = 0; i < n - 1; i++)
253 {
254 dx[i] = x[i + 1] - x[i];
255 }
256 dx[n - 1] = dx[n - 2]; // Extrapolate last
257 }

272 private void BuildTridiagonalSystemVariable(
273 int n,
274 double dt,
275 double sigma2,
276 double r,
277 double q,
278 double[] x,
279 double[] dx,
280 double[] v,
281 double[] aOut,
282 double[] bOut,
283 double[] cOut,
284 double[] dOut)
285 {
286 double oneMinusTheta = 1.0 - _theta;
287
288 // PDE coefficients (matching Black-Scholes in log-space)
289 double a0 = -r; // kill term
290 double ax = r - q - (0.5 * sigma2); // drift/convection
291 double axx = 0.5 * sigma2; // diffusion (σ²/2, NOT σ²)
292
293 for (int i = 1; i < n - 1; i++)
294 {
295 double dxMinus = dx[i - 1];
296 double dxPlus = dx[i];
297
298 // Variable-spacing finite difference coefficients
299 // Following kwinto-cuda convention
300 double inv_dxm = 1.0 / (dxMinus + dxPlus); // central diff denominator
301 double inv_dx2u = 2.0 / (dxPlus * (dxMinus + dxPlus));
302 double inv_dx2m = 2.0 / (dxMinus * dxPlus);
303 double inv_dx2l = 2.0 / (dxMinus * (dxMinus + dxPlus));
304
305 // Tridiagonal coefficients for A operator
306 double al = (-inv_dxm * ax) + (inv_dx2l * axx); // lower
307 double am = a0 - (inv_dx2m * axx); // middle
308 double au = (inv_dxm * ax) + (inv_dx2u * axx); // upper
309
310 // LHS: (I - θ·dt·A)
311 aOut[i] = -_theta * dt * al;
312 bOut[i] = 1.0 - (_theta * dt * am);
313 cOut[i] = -_theta * dt * au;
314
315 // RHS: (I + (1-θ)·dt·A)·V
316 double convectionTerm = oneMinusTheta * dt * ax * inv_dxm * (v[i + 1] - v[i - 1]);
317 double diffusionTerm = oneMinusTheta * dt * axx * ((inv_dx2u * v[i + 1]) - (inv_dx2m * v[i]) + (inv_dx2l * v[i - 1]));
318 dOut[i] = ((1.0 + (oneMinusTheta * dt * a0)) * v[i]) + convectionTerm + diffusionTerm;
319 }
320 }

334 private static void ApplyNeumannBoundaries(
335 int n,
336 double[] a,
337 double[] b,
338 double[] c,
339 double[] d,
340 double[] v)
341 {
342 // Left boundary: gamma = 0 → V[0] - 2*V[1] + V[2] = 0
343 // Rearranged: V[0] = 2*V[1] - V[2]
344 a[0] = 0;
345 b[0] = 1;
346 c[0] = 0;
347 d[0] = (2 * v[1]) - v[2];
348
349 // Right boundary: gamma = 0 → V[N-3] - 2*V[N-2] + V[N-1] = 0
350 // Rearranged: V[N-1] = 2*V[N-2] - V[N-3]
351 a[n - 1] = 0;
352 b[n - 1] = 1;
353 c[n - 1] = 0;
354 d[n - 1] = (2 * v[n - 2]) - v[n - 3];
355 }

360 private static void SolveTridiagonal(
361 int n,
362 double[] a,
363 double[] b,
364 double[] c,
365 double[] d,
366 double[] x)
367 {
368 double[] cPrime = ArrayPool<double>.Shared.Rent(n);
369 double[] dPrime = ArrayPool<double>.Shared.Rent(n);
370
371 try
372 {
373 cPrime[0] = c[0] / b[0];
374 dPrime[0] = d[0] / b[0];
375
376 for (int i = 1; i < n; i++)
377 {
378 double denom = b[i] - (a[i] * cPrime[i - 1]);
379 if (System.Math.Abs(denom) < 1e-15)
380 {
381 denom = 1e-15;
382 }
383 cPrime[i] = c[i] / denom;
384 dPrime[i] = (d[i] - (a[i] * dPrime[i - 1])) / denom;
385 }
386
387 x[n - 1] = dPrime[n - 1];
388 for (int i = n - 2; i >= 0; i--)
389 {
390 x[i] = dPrime[i] - (cPrime[i] * x[i + 1]);
391 }
392 }
393 finally
394 {
395 ArrayPool<double>.Shared.Return(cPrime);
396 ArrayPool<double>.Shared.Return(dPrime);
397 }
398 }

400 private static double InterpolatePrice(double[] x, double[] v, int n, double logSpot)
401 {
402 int i = 0;
403 for (int j = 0; j < n - 1; j++)
404 {
405 if (x[j] <= logSpot && logSpot <= x[j + 1])
406 {
407 i = j;
408 break;
409 }
410 }
411
412 double dxLocal = x[i + 1] - x[i];
413 if (System.Math.Abs(dxLocal) < 1e-15)
414 {
415 return v[i];
416 }
417 double t = (logSpot - x[i]) / dxLocal;
418 return ((1 - t) * v[i]) + (t * v[i + 1]);
419 }

440 public double Delta(double spot, double strike, double timeToExpiry, double riskFreeRate, double dividendYield, double volatility, OptionType optionType)
441 {
442 const double epsilon = 0.01;
443 double up = Price(spot * (1 + epsilon), strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
444 double down = Price(spot * (1 - epsilon), strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
445 return (up - down) / (2 * spot * epsilon);
446 }

448 public double Gamma(double spot, double strike, double timeToExpiry, double riskFreeRate, double dividendYield, double volatility, OptionType optionType)
449 {
450 const double epsilon = 0.01;
451 double up = Price(spot * (1 + epsilon), strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
452 double mid = Price(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
453 double down = Price(spot * (1 - epsilon), strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
454 double h = spot * epsilon;
455 return (up - (2 * mid) + down) / (h * h);
456 }

458 public double Vega(double spot, double strike, double timeToExpiry, double riskFreeRate, double dividendYield, double volatility, OptionType optionType)
459 {
460 const double epsilon = 0.001;
461 double up = Price(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility + epsilon, optionType);
462 double down = Price(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility - epsilon, optionType);
463 return (up - down) / (2 * epsilon);
464 }

466 public double Theta(double spot, double strike, double timeToExpiry, double riskFreeRate, double dividendYield, double volatility, OptionType optionType)
467 {
468 const double dtDays = 1.0 / 365.0;
469 double now = Price(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
470 double later = Price(spot, strike, timeToExpiry - dtDays, riskFreeRate, dividendYield, volatility, optionType);
471 return (later - now) / dtDays;
472 }
473
474 public double Rho(double spot, double strike, double timeToExpiry, double riskFreeRate, double dividendYield, double volatility, OptionType optionType)
475 {
476 const double epsilon = 0.0001;
477 double up = Price(spot, strike, timeToExpiry, riskFreeRate + epsilon, dividendYield, volatility, optionType);
478 double down = Price(spot, strike, timeToExpiry, riskFreeRate - epsilon, dividendYield, volatility, optionType);
479 return (up - down) / (2 * epsilon);
