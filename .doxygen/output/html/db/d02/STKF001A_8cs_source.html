118 public KalmanVolatilityEstimate Update(double yangZhangEstimate, int sampleSize = 30)
119 {
120 if (!_isInitialised)
121 {
122 Reset(yangZhangEstimate, 0.02);
123 }
124
125 // Predict Step (Time Update)
126
127 // State prediction: x̂_{k|k-1} = F × x̂_{k-1|k-1}
128 double sigmaPred = _sigma + (_params.DeltaT * _sigmaDot);
129 double sigmaDotPred = _params.Phi * _sigmaDot;
130
131 // Covariance prediction: P_{k|k-1} = F × P_{k-1|k-1} × Fᵀ + Q
132 double p11Pred = _p11 + (2 * _params.DeltaT * _p12)
133 + (_params.DeltaT * _params.DeltaT * _p22) + _params.QSigma;
134 double p12Pred = _params.Phi * (_p12 + (_params.DeltaT * _p22));
135 double p22Pred = (_params.Phi * _params.Phi * _p22) + _params.QSigmaDot;
136
137 // Update Step (Measurement Update)
138
139 // Measurement noise: R_k = Var(ŝ_YZ) ≈ σ²/(2n×η)
140 double measurementNoise = ComputeMeasurementNoise(yangZhangEstimate, sampleSize);
141
142 // Innovation: y_k = z_k - H × x̂_{k|k-1}
143 double innovation = yangZhangEstimate - sigmaPred;
144
145 // Innovation covariance: S_k = H × P_{k|k-1} × Hᵀ + R_k
146 // For H = [1, 0], this simplifies to S = P[1,1] + R
147 double innovationCov = p11Pred + measurementNoise;
148
149 // Kalman gain: K_k = P_{k|k-1} × Hᵀ × S_k⁻¹
150 double k1 = p11Pred / innovationCov;
151 double k2 = p12Pred / innovationCov;
152
153 // State update: x̂_{k|k} = x̂_{k|k-1} + K_k × y_k
154 _sigma = sigmaPred + (k1 * innovation);
155 _sigmaDot = sigmaDotPred + (k2 * innovation);
156
157 // Covariance update (Joseph form for numerical stability):
158 // P_{k|k} = (I - K×H) × P_{k|k-1} × (I - K×H)ᵀ + K × R × Kᵀ
159 // For scalar measurement, simplifies considerably
160 double oneMinusK1 = 1.0 - k1;
161 _p11 = (oneMinusK1 * oneMinusK1 * p11Pred) + (k1 * k1 * measurementNoise);
162 _p12 = (oneMinusK1 * p12Pred) - (k1 * k2 * p11Pred) + (k2 * oneMinusK1 * p12Pred);
163 _p22 = p22Pred - (k2 * p12Pred) - (k2 * (p12Pred - (k2 * p22Pred)));
164
165 // Ensure positive semi-definiteness
166 _p11 = Math.Max(_p11, 1e-10);
167 _p22 = Math.Max(_p22, 1e-10);
168
169 // Store diagnostics
170 _lastYangZhang = yangZhangEstimate;
171 _lastKalmanGain = k1;
172 _lastInnovation = innovation;
173 _updateCount++;
174
175 SafeLog(() => LogFilterUpdate(_logger!, _sigma, k1, innovation, null));
176
177 return new KalmanVolatilityEstimate(
178 Volatility: _sigma,
179 VolatilityDrift: _sigmaDot,
180 Variance: _p11,
181 StandardError: Math.Sqrt(Math.Max(0, _p11)),
182 YangZhangRaw: yangZhangEstimate,
183 KalmanGain: k1,
184 Innovation: innovation,
185 MeasurementNoise: measurementNoise);
186 }

236 public (double Lower, double Upper) GetConfidenceInterval(double confidenceLevel = 0.95)
237 {
238 // For 95%: z = 1.96
239 double z = confidenceLevel switch
240 {
241 0.99 => 2.576,
242 0.95 => 1.96,
243 0.90 => 1.645,
244 _ => 1.96
245 };
246
247 double se = StandardError;
248 return (_sigma - (z * se), _sigma + (z * se));
249 }

251 private double ComputeMeasurementNoise(double measurement, int sampleSize)
252 {
253 // Yang-Zhang variance: Var(σ̂_YZ) ≈ σ²/(2n×η) where η ≈ 8
254 const double yangZhangEfficiency = 8.0;
255
256 double variance = measurement * measurement / (2.0 * sampleSize * yangZhangEfficiency);
257
258 // Apply minimum to prevent numerical issues
259 return Math.Max(variance, 1e-8);
260 }
