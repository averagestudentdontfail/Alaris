4// Classifies the interest rate regime for American option pricing.
5// Based on Healy (2021) "American Options Under Negative Rates"
6//
7// Three regimes:
8// 1. Standard (r ≥ 0 OR r < q < 0): Single boundary FD works
9// 2. Double Boundary (q < r < 0 for puts, 0 < r < q for calls): Two exercise boundaries
10// 3. The third case implicitly handled by the classification logic

58 public static RateRegime Classify(double r, double q, bool isCall)
59 {
60 if (isCall)
61 {
62 // Call: Double boundary when 0 < r < q
63 // This means dividend yield exceeds positive risk-free rate
64 if (r > 0 && r < q)
65 {
66 return RateRegime.DoubleBoundary;
67 }
68 }
69 else
70 {
71 // Put: Double boundary when q < r < 0
72 // This means both rates negative, but dividend MORE negative than rate
73 if (r < 0 && q < r)
74 {
75 return RateRegime.DoubleBoundary;
76 }
77 }
78
79 return RateRegime.Standard;
80 }
