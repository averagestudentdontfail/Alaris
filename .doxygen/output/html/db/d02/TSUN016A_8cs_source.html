283 private static List<Trade> GenerateHistoricalTrades(int count)
284 {
285 List<Trade> trades = new List<Trade>();
286 DateTime baseDate = DateTime.Today.AddDays(-count * 2.5);
287
288 for (int i = 0; i < count; i++)
289 {
290 DateTime entryDate = baseDate.AddDays(i * 2.5);
291 int holdingPeriod = 5 + (i % 3 - 1);
292
293 trades.Add(new Trade
294 {
295 Symbol = i % 2 == 0 ? "AAPL" : "MSFT",
296 EntryDate = entryDate,
297 ExitDate = entryDate.AddDays(holdingPeriod),
298 ProfitLoss = (i % 3 == 0) ? -100 : 150
299 });
300 }
301
302 return trades;
303 }
