53 [Fact]
54 public void SolveCharacteristic_NegativeRates_DoubleBoundaryRegime()
55 {
56 // Arrange: Healy (2021) double boundary case: q < r < 0
57 double r = -0.004;
58 double q = -0.006;
59 double σ = 0.30;
60
61 // Act
62 var (λ1, λ2) = CRMF002A.SolveCharacteristic(r, q, σ);
63
64 // Assert: Characteristic equation has two real roots
65 // λ1 > λ2 by convention (as per SolveCharacteristic implementation)
66 Assert.True(λ1 > λ2, $"λ1={λ1} should be > λ2={λ2}");
67
68 // Note: For negative r, the product of roots is λ1*λ2 = -2r/σ² = 2|r|/σ² > 0
69 // This means both roots have the same sign. Combined with sum being positive
70 // (for typical σ), both roots can be positive in negative rate regimes.
71
72 // Validate roots satisfy characteristic equation
73 Assert.True(CRMF002A.ValidateRoot(r, q, σ, λ1, Tolerance));
74 Assert.True(CRMF002A.ValidateRoot(r, q, σ, λ2, Tolerance));
75 }

99 [Theory]
100 [InlineData(0.05, 0.02, 0.30)]
101 [InlineData(-0.004, -0.006, 0.30)] // Use σ=0.30 to ensure real roots
102 [InlineData(0.00, 0.02, 0.30)]
103 public void SolveCharacteristic_VietasFormulas_SumOfRoots(double r, double q, double σ)
104 {
105 // Vieta's formula for quadratic aλ² + bλ + c = 0:
106 // λ1 + λ2 = -b/a = -(r - q - σ²/2) / (σ²/2) = (q - r + σ²/2) / (σ²/2)
107 double σ2 = σ * σ;
108 double expectedSum = (q - r + σ2 / 2.0) / (σ2 / 2.0);
109
110 // Act
111 var (λ1, λ2) = CRMF002A.SolveCharacteristic(r, q, σ);
112 double actualSum = λ1 + λ2;
113
114 // Assert
115 Assert.True(System.Math.Abs(actualSum - expectedSum) < 1e-8,
116 $"Sum of roots {actualSum} should equal {expectedSum}");
117 }

119 [Theory]
120 [InlineData(0.05, 0.02, 0.30)]
121 [InlineData(-0.004, -0.006, 0.30)] // Use σ=0.30 to ensure real roots
122 [InlineData(0.10, 0.00, 0.40)]
123 public void SolveCharacteristic_VietasFormulas_ProductOfRoots(double r, double q, double σ)
