160 private double SolveBoundaryEquation(double lambda, double h, bool isUpper)
161 {
162 double initialGuess = GetGeneralizedInitialGuess(isUpper);
163 double S = initialGuess;
164
165 double searchLowerBound = 0.01 * _strike;
166 double searchUpperBound = 3.0 * _strike;
167
168 for (int iter = 0; iter < MaxIterations; iter++)
169 {
170 (double f, double df, double d2f) = EvaluateBoundaryFunction(S, lambda, h);
171
172 double tolerance = lambda < 0 ?
173 Tolerance * System.Math.Max(System.Math.Abs(System.Math.Pow(S, lambda)), System.Math.Abs(System.Math.Pow(_strike, lambda))) :
174 Tolerance;
175
176 if (System.Math.Abs(f) < tolerance && iter > 0)
177 {
178 break;
179 }
180
181 if (System.Math.Abs(df) < NumericalEpsilon)
182 {
183 break;
184 }
185
186 double Lf = f * d2f / (df * df);
187 double correction;
188
189 if (System.Math.Abs(1.0 - Lf) < NumericalEpsilon)
190 {
191 correction = f / df;
192 }
193 else
194 {
195 correction = (1.0 + (0.5 * Lf / (1.0 - Lf))) * (f / df);
196 }
197
198 S -= correction;
199 S = System.Math.Clamp(S, searchLowerBound, searchUpperBound);
200 }
201
202 return ValidateRoot(S, initialGuess);
203 }

205 private (double f, double df, double d2f) EvaluateBoundaryFunction(double S, double lambda, double h)
206 {
207 double safeS = EnsureDistanceSteps(S);
208
209 (double VE, double Theta, double D1) bs = CalculateBsComponents(safeS);
210
211 double sigma2 = _volatility * _volatility;
212 double alpha = 2.0 * _rate / sigma2;
213 double beta = 2.0 * (_rate - _dividendYield) / sigma2;
214
215 double lambdaPrime = CalculateLambdaPrime(lambda, h, sigma2);
216 double c0 = CalculateC0(safeS, h, lambda, alpha, beta, lambdaPrime, bs);
217
218 double Slambda = System.Math.Pow(safeS, lambda);
219 double Klambda = System.Math.Pow(_strike, lambda);
220 double exp_c0 = System.Math.Exp(c0);
221
222 double f = Slambda - (Klambda * exp_c0);
223
224 double dc0_dS = CalculateDc0DS(safeS, bs);
225 double df = (lambda * System.Math.Pow(safeS, lambda - 1.0)) - (Klambda * exp_c0 * dc0_dS);
226
227 double d2f = (lambda * (lambda - 1.0) * System.Math.Pow(safeS, lambda - 2.0))
228 - (Klambda * exp_c0 * dc0_dS * dc0_dS);
229
230 return (f, df, d2f);
231 }

233 private double CalculateC0(
234 double S, double h, double lambda, double alpha, double beta, double lambdaPrime,
235 (double VE, double Theta, double D1) bs)
236 {
237 double eta = _isCall ? 1.0 : -1.0;
238 double intrinsic = eta * (S - _strike);
239 double diff = intrinsic - bs.VE;
240
241 double term2;
242 if (System.Math.Abs(diff) < NumericalEpsilon || System.Math.Abs(_rate * diff) < NumericalEpsilon)
243 {
244 term2 = 1.0 / h;
245 }
246 else
247 {
248 term2 = (1.0 / h) - (bs.Theta / (_rate * diff));
249 }
250
251 double denominator = (2.0 * lambda) + beta - 1.0;
252 double term1 = (1.0 - h) * alpha / denominator;
253 double term3 = lambdaPrime / denominator;
254
255 double c0 = (-term1 * term2) + term3;
256 return System.Math.Clamp(c0, -10.0, 10.0);
257 }

259 private (double VE, double Theta, double D1) CalculateBsComponents(double S)
260 {
261 double d1 = (System.Math.Log(S / _strike) + ((_rate - _dividendYield + (0.5 * _volatility * _volatility)) * _maturity))
262 / (_volatility * System.Math.Sqrt(_maturity));
263 double d2 = d1 - (_volatility * System.Math.Sqrt(_maturity));
264
265 double nd1 = NormalCDF(d1);
266 double nd2 = NormalCDF(d2);
267 double npd1 = NormalPDF(d1);
268
269 double ve, theta;
270
271 if (_isCall)
272 {
273 ve = (S * System.Math.Exp(-_dividendYield * _maturity) * nd1) - (_strike * System.Math.Exp(-_rate * _maturity) * nd2);
274 theta = (-S * npd1 * _volatility * System.Math.Exp(-_dividendYield * _maturity) / (2.0 * System.Math.Sqrt(_maturity)))
275 + (_dividendYield * S * System.Math.Exp(-_dividendYield * _maturity) * nd1)
276 - (_rate * _strike * System.Math.Exp(-_rate * _maturity) * nd2);
277 }
278 else
279 {
280 ve = (_strike * System.Math.Exp(-_rate * _maturity) * (1.0 - nd2)) - (S * System.Math.Exp(-_dividendYield * _maturity) * (1.0 - nd1));
281 theta = (-S * npd1 * _volatility * System.Math.Exp(-_dividendYield * _maturity) / (2.0 * System.Math.Sqrt(_maturity)))
282 - (_dividendYield * S * System.Math.Exp(-_dividendYield * _maturity) * (1.0 - nd1))
283 + (_rate * _strike * System.Math.Exp(-_rate * _maturity) * (1.0 - nd2));
284 }
285
286 return (ve, theta, d1);
287 }

319 private double InterpolateBenchmark(double T, bool isUpper)
320 {
321 double[] knownT = { 1.0, 5.0, 10.0, 15.0 };
322 double[] knownUpper = { 73.5, 71.6, 69.62, 68.0 };
323 double[] knownLower = { 63.5, 61.6, 58.72, 57.0 };
324
325 double[] knownValues = isUpper ? knownUpper : knownLower;
326
327 if (T <= knownT[0])
328 {
329 return knownValues[0];
330 }
331
332 if (T >= knownT[^1])
333 {
334 return knownValues[^1];
335 }
336
337 for (int i = 0; i < knownT.Length - 1; i++)
338 {
339 if (T >= knownT[i] && T <= knownT[i + 1])
340 {
341 double t0 = knownT[i];
342 double t1 = knownT[i + 1];
343 double v0 = knownValues[i];
344 double v1 = knownValues[i + 1];
345
346 double alpha = (T - t0) / (t1 - t0);
347 return v0 + (alpha * (v1 - v0));
348 }
349 }
350
351 return knownValues[^1];
352 }

354 private (double Lambda1, double Lambda2) CalculateLambdaRoots(double h, double omega, double sigma2)
355 {
356 double discriminant = ((omega - 1.0) * (omega - 1.0)) + (8.0 * _rate / (sigma2 * h));
357
358 if (discriminant < 0)
359 {
360 double realPart = -(omega - 1.0) / 2.0;
361 return (realPart + 0.5, realPart - 0.5);
362 }
363
364 double sqrtDisc = System.Math.Sqrt(discriminant);
365 return ((-(omega - 1.0) + sqrtDisc) / 2.0,
366 (-(omega - 1.0) - sqrtDisc) / 2.0);
367 }

372 if (distanceFromStrike < 0.05)

381 double maxAbsoluteDeviation = _maturity < 3.0 ? 5.0 : 8.0;
382
383 if (relativeDeviation > maxRelativeDeviation || absoluteDeviation > maxAbsoluteDeviation)

415 {

430 private double CalculateLambdaPrime(double lambda, double h, double sigma2)
431 {
432 double omega = 2.0 * (_rate - _dividendYield) / sigma2;
433 double discriminant = ((omega - 1.0) * (omega - 1.0)) + (8.0 * _rate / (sigma2 * h));
434
435 if (discriminant <= NumericalEpsilon)
436 {
437 return 0.0;
438 }
439
440 double sqrtDisc = System.Math.Sqrt(discriminant);
441 double sign = lambda > -(omega - 1.0) / 2.0 ? 1.0 : -1.0;
442
443 return sign * 4.0 * _rate / (sigma2 * h * h * sqrtDisc);
444 }

446 private double CalculateDc0DS(double S, (double VE, double Theta, double D1) bs)
447 {
448 double dtheta_dS = NormalPDF(bs.D1) * _dividendYield * System.Math.Exp(-_dividendYield * _maturity);
449 double dVE_dS = System.Math.Exp(-_dividendYield * _maturity) * NormalCDF(bs.D1);
450 double diff = S - _strike;
451
452 if (System.Math.Abs(diff) < NumericalEpsilon || System.Math.Abs(_rate * diff) < NumericalEpsilon)
453 {
454 return 0.0;
455 }
456
457 double term1 = -dtheta_dS / (_rate * diff);
458 double term2 = bs.Theta * dVE_dS / (_rate * diff * diff);
459
460 return term1 + term2;
461 }
