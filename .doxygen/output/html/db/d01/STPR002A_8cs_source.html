50 public static (double Value, double Error) IntegrateToInfinity(
51 Func<double, double> f,
52 double a,
53 double absoluteTolerance = 1e-8,
54 double relativeTolerance = 1e-6)
55 {
56 ArgumentNullException.ThrowIfNull(f);
57
58 double totalSum = 0;
59 double currentStart = a;
60 int chunkCount = 0;
61 double currentStepSize = IntegrationStepSize;
62 double previousChunkMagnitude = double.MaxValue;
63
64 while (chunkCount < MaxChunks)
65 {
66 double currentEnd = currentStart + currentStepSize;
67
68 // Integrate this chunk
69 double chunkValue = GaussLegendreRule.Integrate(f, currentStart, currentEnd, GaussLegendreOrder);
70 double chunkMagnitude = Math.Abs(chunkValue);
71
72 totalSum += chunkValue;
73
74 // Check convergence: if the latest chunk added almost nothing, we are done.
75 // Characteristic functions decay exponentially, so this happens relatively fast.
76 if (chunkMagnitude < absoluteTolerance && chunkCount > 0)
77 {
78 return (totalSum, absoluteTolerance); // Converged
79 }
80
81 // Adaptive step sizing: if magnitude dropped significantly, double the step
82 // This accelerates convergence for well-behaved decaying integrands
83 if (chunkCount > 2 && chunkMagnitude < previousChunkMagnitude * 0.1)
84 {
85 currentStepSize = Math.Min(currentStepSize * 2.0, IntegrationStepSize * 16);
86 }
87
88 previousChunkMagnitude = chunkMagnitude;
89 currentStart = currentEnd;
90 chunkCount++;
91 }
92
93 // Return best effort if we hit max chunks
94 return (totalSum, 1.0);
95 }
