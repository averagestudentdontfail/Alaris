37 [SuppressMessage("StyleCop.CSharp.NamingRules", "SA1312:VariableNamesMustBeginWithLowerCaseLetter", Justification = "Greek letters")]
38 public static (double λ1, double λ2) SolveCharacteristic(double r, double q, double σ)
39 {
40 System.ArgumentOutOfRangeException.ThrowIfNegativeOrZero(σ, nameof(σ));
41
42 double σ2 = σ * σ;
43
44 // Characteristic equation coefficients
45 // aλ² + bλ + c = 0
46 double a = σ2 / 2.0;
47 double b = r - q - (σ2 / 2.0);
48 double c = -r;
49
50 // Quadratic formula: λ = (-b ± √(b² - 4ac)) / 2a
51 double discriminant = (b * b) - (4.0 * a * c);
52
53 if (discriminant < 0)
54 {
55 // Should not happen for valid parameters, but handle gracefully
56 throw new ArgumentException("Complex roots detected for the provided parameter combination.");
57 }
58
59 double sqrtD = System.Math.Sqrt(discriminant);
60
61 // Initial roots from quadratic formula
62 double λ1Init = (-b + sqrtD) / (2.0 * a);
63 double λ2Init = (-b - sqrtD) / (2.0 * a);
64
65 // Refine using Super-Halley for numerical stability near r ≈ 0
66 double λ1 = SuperHalleyRefine(λ1Init, a, b, c);
67 double λ2 = SuperHalleyRefine(λ2Init, a, b, c);
68
69 // Ensure λ1 > λ2 (convention)
70 if (λ1 < λ2)
71 {
72 (λ1, λ2) = (λ2, λ1);
73 }
74
75 return (λ1, λ2);
76 }

96 [SuppressMessage("StyleCop.CSharp.NamingRules", "SA1312:VariableNamesMustBeginWithLowerCaseLetter", Justification = "Greek letters")]
97 public static double SuperHalleyRefine(
98 double λInit,
99 double a,
100 double b,
101 double c,
102 double tolerance = 1e-12,
103 int maxIterations = 20)
104 {
105 if (tolerance <= 0)
106 {
107 throw new ArgumentOutOfRangeException(nameof(tolerance), "Tolerance must be positive.");
108 }
109
110 ArgumentOutOfRangeException.ThrowIfLessThan(maxIterations, 1);
111 double λ = λInit;
112
113 for (int i = 0; i < maxIterations; i++)
114 {
115 // f(λ) = aλ² + bλ + c
116 double f = (a * λ * λ) + (b * λ) + c;
117
118 // Check convergence
119 if (System.Math.Abs(f) < tolerance)
120 {
121 return λ;
122 }
123
124 // f'(λ) = 2aλ + b
125 double fPrime = (2.0 * a * λ) + b;
126
127 // f''(λ) = 2a
128 double fDoublePrime = 2.0 * a;
129
130 // Super-Halley denominator
131 double denominator = (fPrime * fPrime) - (0.5 * f * fDoublePrime);
132
133 // Guard against division by near-zero - use Brent bisection fallback
134 if (System.Math.Abs(denominator) < 1e-15 || System.Math.Abs(fPrime) < 1e-15)
135 {
136 // Brent bisection: guaranteed convergence
137 return BrentBisectionFallback(λ, a, b, c, tolerance);
138 }
139
140 // Super-Halley update
141 double λNext = λ - ((f * fPrime) / denominator);
142
143 // Check for convergence
144 if (System.Math.Abs(λNext - λ) < tolerance)
145 {
146 return λNext;
147 }
148
149 λ = λNext;
150 }
151
152 return λ;
153 }

181 // Evaluate f(λ) = aλ² + bλ + c
182 static double EvalQuadratic(double λ, double a, double b, double c)
183 {
184 return (a * λ * λ) + (b * λ) + c;
185 }
186
187 // Find bracketing interval around initial estimate
188 double lower = λInit - System.Math.Abs(λInit) - 10.0;
189 double upper = λInit + System.Math.Abs(λInit) + 10.0;
190
191 double fLower = EvalQuadratic(lower, a, b, c);
192 double fUpper = EvalQuadratic(upper, a, b, c);
193
194 // Ensure we have a sign change (root is bracketed)
195 // If not, expand the interval
196 for (int expand = 0; expand < 10 && fLower * fUpper > 0; expand++)
197 {
198 lower -= 10.0;
199 upper += 10.0;
200 fLower = EvalQuadratic(lower, a, b, c);
201 fUpper = EvalQuadratic(upper, a, b, c);
202 }
203
204 // If still not bracketed, return initial estimate (shouldn't happen for valid quadratics)
205 if (fLower * fUpper > 0)
206 {
207 return λInit;
208 }
209
210 // Bisection iteration (guaranteed convergence)
211 for (int i = 0; i < 100; i++)
212 {
213 double mid = (lower + upper) / 2.0;
214 double fMid = EvalQuadratic(mid, a, b, c);
215
216 if (System.Math.Abs(fMid) < tolerance || (upper - lower) / 2.0 < tolerance)
217 {
218 return mid;
219 }
220
221 if (fLower * fMid < 0)
222 {
223 upper = mid;
224 fUpper = fMid;
225 }
226 else
227 {
228 lower = mid;
229 fLower = fMid;
230 }
231 }
232
233 return (lower + upper) / 2.0;
234 }
