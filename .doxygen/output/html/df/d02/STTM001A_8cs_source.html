19 public STTM001AAnalysis Analyze(IReadOnlyList<STTM001APoint> points)
20 {
21 ArgumentNullException.ThrowIfNull(points);
22
23 if (points.Count < 2)
24 {
25 throw new ArgumentException("Need at least 2 points for term structure analysis", nameof(points));
26 }
27
28 int n = points.Count;
29
30 // Rule 5: Use ArrayPool for temporary arrays
31 double[] dte = ArrayPool<double>.Shared.Rent(n);
32 double[] iv = ArrayPool<double>.Shared.Rent(n);
33 int[] indices = ArrayPool<int>.Shared.Rent(n);
34
35 try
36 {
37 // Initialize indices for sorting
38 for (int i = 0; i < n; i++)
39 {
40 indices[i] = i;
41 }
42
43 // Sort indices by DaysToExpiry (avoids creating sorted list)
44 Array.Sort(indices, 0, n, Comparer<int>.Create((a, b) =>
45 points[a].DaysToExpiry.CompareTo(points[b].DaysToExpiry)));
46
47 // Extract sorted values into arrays
48 for (int i = 0; i < n; i++)
49 {
50 int idx = indices[i];
51 dte[i] = points[idx].DaysToExpiry;
52 iv[i] = points[idx].ImpliedVolatility;
53 }
54
55 // Perform linear regression: IV = intercept + slope * DTE
56 // Note: SimpleRegression.Fit only reads the arrays
57 (double intercept, double slope) = SimpleRegression.Fit(
58 dte.AsSpan(0, n).ToArray(),
59 iv.AsSpan(0, n).ToArray());
60
61 double rSquared = CalculateRSquaredOptimized(dte.AsSpan(0, n), iv.AsSpan(0, n), intercept, slope);
62
63 STTM001AAnalysis analysis = new STTM001AAnalysis
64 {
65 Intercept = intercept,
66 Slope = slope,
67 RSquared = rSquared
68 };
69
70 // Add sorted points to the collection
71 for (int i = 0; i < n; i++)
72 {
73 analysis.Points.Add(points[indices[i]]);
74 }
75
76 return analysis;
77 }
78 finally
79 {
80 ArrayPool<double>.Shared.Return(dte);
81 ArrayPool<double>.Shared.Return(iv);
82 ArrayPool<int>.Shared.Return(indices);
83 }
84 }

89 private static double CalculateRSquaredOptimized(ReadOnlySpan<double> x, ReadOnlySpan<double> y, double intercept, double slope)
90 {
91 // Calculate mean
92 double ySum = 0;
93 for (int i = 0; i < y.Length; i++)
94 {
95 ySum += y[i];
96 }
97 double yMean = ySum / y.Length;
98
99 // Calculate SS Total and SS Residual in single pass
100 double ssTotal = 0;
101 double ssResidual = 0;
102 for (int i = 0; i < x.Length; i++)
103 {
104 double yDiff = y[i] - yMean;
105 ssTotal += yDiff * yDiff;
106
107 double predicted = intercept + (slope * x[i]);
108 double residual = y[i] - predicted;
109 ssResidual += residual * residual;
110 }
111
112 return 1.0 - (ssResidual / ssTotal);
113 }
