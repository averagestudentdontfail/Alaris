110 public void Calibrate(IReadOnlyList<TradeOutcome> outcomes)
111 {
112 ArgumentNullException.ThrowIfNull(outcomes);
113
114 var profitable = outcomes.Where(o => o.IsProfitable).Select(o => o.IVRVRatio).ToList();
115 var unprofitable = outcomes.Where(o => !o.IsProfitable).Select(o => o.IVRVRatio).ToList();
116
117 if (profitable.Count < 10 || unprofitable.Count < 10)
118 {
119 throw new ArgumentException(
120 $"Insufficient calibration data: need at least 10 each of profitable ({profitable.Count}) " +
121 $"and unprofitable ({unprofitable.Count}) outcomes");
122 }
123
124 // Compute distribution parameters
125 const double MinSigma = 1e-6;
126 _mu0 = unprofitable.Average();
127 _sigma0 = Math.Max(MinSigma, ComputeStandardDeviation(unprofitable, _mu0));
128
129 _mu1 = profitable.Average();
130 _sigma1 = Math.Max(MinSigma, ComputeStandardDeviation(profitable, _mu1));
131
132 // Compute prior probabilities
133 int total = outcomes.Count;
134 _pi0 = (double)unprofitable.Count / total;
135 _pi1 = (double)profitable.Count / total;
136
137 _isCalibrated = true;
138 _calibrationSampleSize = total;
139
140 SafeLog(() => LogCalibrationComplete(_logger!, profitable.Count, unprofitable.Count, null));
141 SafeLog(() => LogDistributionParameters(_logger!, _mu0, _sigma0, _mu1, _sigma1, null));
142 }

150 public double ComputeOptimalThreshold(double costRatio = 2.0)
151 {
152 if (costRatio <= 0)
153 {
154 throw new ArgumentOutOfRangeException(nameof(costRatio), "Cost ratio must be positive");
155 }
156
157 double threshold;
158
159 // Check for approximately equal variances
160 double varianceRatio = _sigma1 / _sigma0;
161 if (Math.Abs(varianceRatio - 1.0) < 0.1)
162 {
163 // Equal variance approximation: simpler formula
164 double avgSigma = (_sigma0 + _sigma1) / 2;
165 double avgSigmaSq = avgSigma * avgSigma;
166 threshold = ((_mu0 + _mu1) / 2) + (avgSigmaSq / (_mu1 - _mu0) * Math.Log(costRatio * _pi0 / _pi1));
167 }
168 else
169 {
170 // Full quadratic solution for unequal variances
171 threshold = ComputeThresholdUnequelVariances(costRatio);
172 }
173
174 SafeLog(() => LogThresholdComputed(_logger!, threshold, costRatio, null));
175
176 return threshold;
177 }

185 public SignalDetectionResult Evaluate(double observedValue, double? threshold = null)
186 {
187 double actualThreshold = threshold ?? ComputeOptimalThreshold(DefaultCostRatio);
188
189 // Compute likelihood ratio
190 double likelihoodRatio = ComputeLikelihoodRatio(observedValue);
191
192 // Compute posterior probability using Bayes' theorem
193 // P(H₁|x) = p(x|H₁)π₁ / [p(x|H₀)π₀ + p(x|H₁)π₁]
194 double posterior = ComputePosteriorProbability(observedValue);
195
196 bool passes = observedValue >= actualThreshold;
197
198 return new SignalDetectionResult(
199 PassesThreshold: passes,
200 LikelihoodRatio: likelihoodRatio,
201 PosteriorProbability: posterior,
202 Threshold: actualThreshold,
203 ObservedValue: observedValue);
204 }

209 public double ComputeLikelihoodRatio(double x)
210 {
211 double logL0 = LogNormalPdf(x, _mu0, _sigma0);
212 double logL1 = LogNormalPdf(x, _mu1, _sigma1);
213 return Math.Exp(logL1 - logL0);
214 }
215
219 public double ComputePosteriorProbability(double x)
220 {
221 double logP0 = LogNormalPdf(x, _mu0, _sigma0) + Math.Log(_pi0);
222 double logP1 = LogNormalPdf(x, _mu1, _sigma1) + Math.Log(_pi1);
223
224 // Log-sum-exp trick for numerical stability
225 double maxLog = Math.Max(logP0, logP1);
226 double logNorm = maxLog + Math.Log(Math.Exp(logP0 - maxLog) + Math.Exp(logP1 - maxLog));
227
228 return Math.Exp(logP1 - logNorm);
229 }

273 private double ComputeThresholdUnequelVariances(double costRatio)
274 {
275 // Full quadratic solution when σ₀ ≠ σ₁
276 // L(x) = (σ₀/σ₁) × exp[-½((x-μ₁)²/σ₁² - (x-μ₀)²/σ₀²)] > k
277 // Taking log and rearranging gives a quadratic equation
278
279 double sigma0Sq = _sigma0 * _sigma0;
280 double sigma1Sq = _sigma1 * _sigma1;
281
282 double a = (1.0 / sigma0Sq) - (1.0 / sigma1Sq);
283 double b = 2.0 * ((_mu1 / sigma1Sq) - (_mu0 / sigma0Sq));
284 double c = (_mu0 * _mu0 / sigma0Sq) - (_mu1 * _mu1 / sigma1Sq)
285 - (2.0 * Math.Log(_sigma1 / _sigma0 * costRatio * _pi0 / _pi1));
286
287 // Solve ax² + bx + c = 0
288 double discriminant = (b * b) - (4.0 * a * c);
289
290 if (discriminant < 0)
291 {
292 // No real solution - use midpoint
293 return (_mu0 + _mu1) / 2;
294 }
295
296 double sqrtD = Math.Sqrt(discriminant);
297 double x1 = (-b + sqrtD) / (2.0 * a);
298 double x2 = (-b - sqrtD) / (2.0 * a);
299
300 // Choose the solution that falls between the means
301 if (x1 >= _mu0 && x1 <= _mu1)
302 {
303 return x1;
304 }
305
306 if (x2 >= _mu0 && x2 <= _mu1)
307 {
308 return x2;
309 }
310
311 // If neither solution is between means, choose the one closer to the midpoint
312 double midpoint = (_mu0 + _mu1) / 2;
313 return Math.Abs(x1 - midpoint) < Math.Abs(x2 - midpoint) ? x1 : x2;
314 }

316 private static double ComputeStandardDeviation(List<double> values, double mean)
317 {
318 double sumSq = values.Sum(v => (v - mean) * (v - mean));
319 return Math.Sqrt(sumSq / (values.Count - 1));
320 }

322 private static double LogNormalPdf(double x, double mu, double sigma)
323 {
324 double z = (x - mu) / sigma;
325 return (-0.5 * z * z) - Math.Log(sigma) - (0.5 * Math.Log(2 * Math.PI));
326 }

328 private static double NormalCdf(double z)
329 {
330 // Abramowitz and Stegun approximation
331 const double a1 = 0.254829592;
332 const double a2 = -0.284496736;
333 const double a3 = 1.421413741;
334 const double a4 = -1.453152027;
335 const double a5 = 1.061405429;
336 const double p = 0.3275911;
337
338 int sign = z < 0 ? -1 : 1;
339 z = Math.Abs(z) / Math.Sqrt(2);
340
341 double t = 1.0 / (1.0 + (p * z));
342 double polynomial = (((((((a5 * t) + a4) * t) + a3) * t) + a2) * t) + a1;
343 double y = 1.0 - (polynomial * t * Math.Exp(-(z * z)));
344
345 return 0.5 * (1.0 + (sign * y));
346 }
