72 public async Task<IReadOnlyList<UniverseStock>> GetUniverseAsync(
73 DateTime date,
74 CancellationToken cancellationToken = default)
75 {
76 var dateStr = date.ToString("yyyy-MM-dd");
77 var url = $"/v2/aggs/grouped/locale/us/market/stocks/{dateStr}?adjusted=true&apiKey={_apiKey}";
78
79 try
80 {
81 var response = await _httpClient.GetFromJsonAsync<PolygonGroupedResponse>(url, cancellationToken)
82 .ConfigureAwait(false);
83
84 if (response?.Results == null || response.Results.Length == 0)
85 {
86 _logger.LogWarning("No data returned for {Date}", dateStr);
87 return Array.Empty<UniverseStock>();
88 }
89
90 var stocks = response.Results
91 .Where(r => !string.IsNullOrEmpty(r.Ticker))
92 .Select(r => new UniverseStock
93 {
94 Ticker = r.Ticker!,
95 Open = r.Open,
96 High = r.High,
97 Low = r.Low,
98 Close = r.Close,
99 Volume = r.Volume,
100 DollarVolume = r.Close * (decimal)r.Volume,
101 Date = date
102 })
103 .ToList();
104
105 LogUniverseFetched(_logger, date, stocks.Count, null);
106 return stocks;
107 }
108 catch (HttpRequestException ex)
109 {
110 _logger.LogError(ex, "Failed to fetch universe for {Date}", dateStr);
111 throw;
112 }
113 }

123 public async Task<IReadOnlyList<UniverseStock>> GetFilteredUniverseAsync(
124 DateTime date,
125 decimal minDollarVolume = 1_500_000m,
126 decimal minPrice = 5.00m,
127 CancellationToken cancellationToken = default)
128 {
129 var allStocks = await GetUniverseAsync(date, cancellationToken).ConfigureAwait(false);
130
131 var filtered = allStocks
132 .Where(s => s.DollarVolume >= minDollarVolume)
133 .Where(s => s.Close >= minPrice)
134 .OrderByDescending(s => s.DollarVolume)
135 .ToList();
136
137 LogUniverseFiltered(_logger, date, filtered.Count, minDollarVolume, minPrice, null);
138 return filtered;
139 }

150 public async Task<IReadOnlyList<string>> GetUniverseTickersAsync(
