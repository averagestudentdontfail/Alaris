1// CREN001A.cs - Crank-Nicolson Finite Difference American Option Engine
2// Component ID: CREN001A
3//
4// Replaces: QuantLib.FdBlackScholesVanillaEngine
5//
6// Mathematical Specification:
7// - Black-Scholes PDE: ∂V/∂t + (r-q-σ²/2)∂V/∂x + ½σ²∂²V/∂x² - rV = 0
8// - Crank-Nicolson: θ = 0.5 (implicit-explicit average)
9// - Early exercise: V = max(V, intrinsic) at each time step
10// - Thomas algorithm for tridiagonal system O(N) complexity
11//
12// Grid Construction:
13// - Log-spot grid: x ∈ [ln(S/K) - 5σ√T, ln(S/K) + 5σ√T]
14// - Uniform time steps from T to 0 (backward induction)
15//
16// References:
17// - Wilmott, P. "Paul Wilmott on Quantitative Finance" Chapter 77-79
18// - Hull, J.C. "Options, Futures, and Other Derivatives" Chapter 21
19// - QuantLib fdblackscholesvanillaengine.cpp
20// - Alaris.Governance/Coding.md Rule 3 (Bounded Loops), Rule 5 (Zero-allocation)

91 public double Price(
92 double spot,
93 double strike,
94 double timeToExpiry,
95 double riskFreeRate,
96 double dividendYield,
97 double volatility,
98 OptionType optionType)
99 {
100 // Guard clauses (Rule 10: specific exceptions)
101 if (spot <= 0)
102 {
103 throw new ArgumentException("Spot must be positive", nameof(spot));
104 }
105 if (strike <= 0)
106 {
107 throw new ArgumentException("Strike must be positive", nameof(strike));
108 }
109 if (volatility <= 0)
110 {
111 throw new ArgumentException("Volatility must be positive", nameof(volatility));
112 }
113
114 // Handle expired options
115 if (timeToExpiry <= 0)
116 {
117 return CalculatePayoff(spot, strike, optionType);
118 }
119
120 // Grid parameters
121 double sigma = volatility;
122 double sigma2 = sigma * sigma;
123 double dt = timeToExpiry / _timeSteps;
124 double r = riskFreeRate;
125 double q = dividendYield;
126
127 // Log-spot grid: x = ln(S)
128 // Grid centered at ln(spot) with range ±5σ√T
129 double gridWidth = 5.0 * sigma * System.Math.Sqrt(timeToExpiry);
130 double xMin = System.Math.Log(spot) - gridWidth;
131 double xMax = System.Math.Log(spot) + gridWidth;
132 double dx = (xMax - xMin) / (_spotSteps - 1);
133
134 int n = _spotSteps;
135
136 // Rent arrays from pool for zero-allocation (Rule 5)
137 double[] x = ArrayPool<double>.Shared.Rent(n);
138 double[] v = ArrayPool<double>.Shared.Rent(n);
139 double[] vNew = ArrayPool<double>.Shared.Rent(n);
140 double[] a = ArrayPool<double>.Shared.Rent(n);
141 double[] b = ArrayPool<double>.Shared.Rent(n);
142 double[] c = ArrayPool<double>.Shared.Rent(n);
143 double[] d = ArrayPool<double>.Shared.Rent(n);
144
145 try
146 {
147 // Initialize log-spot grid
148 for (int i = 0; i < n; i++)
149 {
150 x[i] = xMin + (i * dx);
151 }
152
153 // Calculate FD coefficients
154 // PDE: ∂V/∂t = -[(r-q-σ²/2)∂V/∂x + ½σ²∂²V/∂x² - rV]
155 // After discretization with centered differences:
156 double alpha = 0.5 * dt * ((sigma2 / (dx * dx)) - ((r - q - (0.5 * sigma2)) / dx));
157 double beta = -dt * ((sigma2 / (dx * dx)) + r);
158 double gamma = 0.5 * dt * ((sigma2 / (dx * dx)) + ((r - q - (0.5 * sigma2)) / dx));
159
160 // Terminal condition: V(T, S) = payoff(S)
161 for (int i = 0; i < n; i++)
162 {
163 double spotAtNode = System.Math.Exp(x[i]);
164 v[i] = CalculatePayoff(spotAtNode, strike, optionType);
165 }
166
167 // Backward induction through time (Rule 3: bounded loop)
168 for (int step = 0; step < _timeSteps; step++)
169 {
170 // Build tridiagonal system for Crank-Nicolson
171 // (I - θ·dt·A)·V^(n) = (I + (1-θ)·dt·A)·V^(n+1)
172 BuildTridiagonalSystem(n, alpha, beta, gamma, v, a, b, c, d);
173
174 // Solve tridiagonal system using Thomas algorithm
175 SolveTridiagonal(n, a, b, c, d, vNew);
176
177 // Apply boundary conditions
178 ApplyBoundaryConditions(n, x, vNew, strike, r, dt, optionType);
179
180 // Early exercise check for American options
181 for (int i = 0; i < n; i++)
182 {
183 double spotAtNode = System.Math.Exp(x[i]);
184 double intrinsic = CalculatePayoff(spotAtNode, strike, optionType);
185 vNew[i] = System.Math.Max(vNew[i], intrinsic);
186 }
187
188 // Swap arrays for next iteration
189 (v, vNew) = (vNew, v);
190 }
191
192 // Interpolate to get price at current spot
193 double logSpot = System.Math.Log(spot);
194 return InterpolatePrice(x, v, n, logSpot);
195 }
196 finally
197 {
198 // Return arrays to pool
199 ArrayPool<double>.Shared.Return(x);
200 ArrayPool<double>.Shared.Return(v);
201 ArrayPool<double>.Shared.Return(vNew);
202 ArrayPool<double>.Shared.Return(a);
203 ArrayPool<double>.Shared.Return(b);
204 ArrayPool<double>.Shared.Return(c);
205 ArrayPool<double>.Shared.Return(d);
206 }
207 }

212 private void BuildTridiagonalSystem(
213 int n,
214 double alpha,
215 double beta,
216 double gamma,
217 double[] v,
218 double[] aOut,
219 double[] bOut,
220 double[] cOut,
221 double[] dOut)
222 {
223 double oneMinusTheta = 1.0 - _theta;
224
225 for (int i = 1; i < n - 1; i++)
226 {
227 // Left hand side: (I - θ·dt·A)
228 aOut[i] = -_theta * alpha;
229 bOut[i] = 1.0 - (_theta * beta);
230 cOut[i] = -_theta * gamma;
231
232 // Right hand side: (I + (1-θ)·dt·A)·V
233 dOut[i] = (oneMinusTheta * alpha * v[i - 1])
234 + ((1.0 + (oneMinusTheta * beta)) * v[i])
235 + (oneMinusTheta * gamma * v[i + 1]);
236 }
237
238 // Boundary conditions will be set separately
239 aOut[0] = 0;
240 bOut[0] = 1;
241 cOut[0] = 0;
242 dOut[0] = v[0];
243
244 aOut[n - 1] = 0;
245 bOut[n - 1] = 1;
246 cOut[n - 1] = 0;
247 dOut[n - 1] = v[n - 1];
248 }

254 private static void SolveTridiagonal(
255 int n,
256 double[] a, // subdiagonal
257 double[] b, // diagonal
258 double[] c, // superdiagonal
259 double[] d, // right-hand side
260 double[] x) // solution
261 {
262 // Forward sweep
263 double[] cPrime = ArrayPool<double>.Shared.Rent(n);
264 double[] dPrime = ArrayPool<double>.Shared.Rent(n);
265
266 try
267 {
268 cPrime[0] = c[0] / b[0];
269 dPrime[0] = d[0] / b[0];
270
271 for (int i = 1; i < n; i++)
272 {
273 double denom = b[i] - (a[i] * cPrime[i - 1]);
274 if (System.Math.Abs(denom) < 1e-15)
275 {
276 denom = 1e-15; // Prevent division by zero
277 }
278 cPrime[i] = c[i] / denom;
279 dPrime[i] = (d[i] - (a[i] * dPrime[i - 1])) / denom;
280 }
281
282 // Back substitution
283 x[n - 1] = dPrime[n - 1];
284 for (int i = n - 2; i >= 0; i--)
285 {
286 x[i] = dPrime[i] - (cPrime[i] * x[i + 1]);
287 }
288 }
289 finally
290 {
291 ArrayPool<double>.Shared.Return(cPrime);
292 ArrayPool<double>.Shared.Return(dPrime);
293 }
294 }

299 private static void ApplyBoundaryConditions(
300 int n,
301 double[] x,
302 double[] v,
303 double strike,
304 double r,
305 double dt,
306 OptionType optionType)
307 {
308 // For puts: V(0) → K (as S → 0, put is worth K)
309 // For calls: V(∞) → 0 (as S → ∞, we use extrapolation)
310
311 if (optionType == OptionType.Put)
312 {
313 // At S = 0, put is worth K·e^(-rT)
314 v[0] = strike * System.Math.Exp(-r * dt);
315 }
316 else
317 {
318 // At S = ∞, call value grows linearly (Neumann condition)
319 // Use linear extrapolation from interior
320 v[n - 1] = (2 * v[n - 2]) - v[n - 3];
321 if (v[n - 1] < 0)
322 {
323 v[n - 1] = 0;
324 }
325 }
326 }

331 private static double InterpolatePrice(double[] x, double[] v, int n, double logSpot)
332 {
333 // Find bracketing indices
334 int i = 0;
335 for (int j = 0; j < n - 1; j++)
336 {
337 if (x[j] <= logSpot && logSpot <= x[j + 1])
338 {
339 i = j;
340 break;
341 }
342 }
343
344 // Linear interpolation
345 double dx = x[i + 1] - x[i];
346 if (System.Math.Abs(dx) < 1e-15)
347 {
348 return v[i];
349 }
350 double t = (logSpot - x[i]) / dx;
351 return ((1 - t) * v[i]) + (t * v[i + 1]);
352 }

357 private static double CalculatePayoff(double spot, double strike, OptionType optionType)
358 {
359 return optionType switch
360 {
361 OptionType.Call => System.Math.Max(spot - strike, 0.0),
362 OptionType.Put => System.Math.Max(strike - spot, 0.0),
363 _ => 0.0
364 };

396 public double Gamma(
397 double spot,
398 double strike,
399 double timeToExpiry,
400 double riskFreeRate,
401 double dividendYield,
402 double volatility,
403 OptionType optionType)
404 {
405 const double epsilon = 0.01;
406 double up = Price(spot * (1 + epsilon), strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
407 double mid = Price(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
408 double down = Price(spot * (1 - epsilon), strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
409 double h = spot * epsilon;
410 return (up - (2 * mid) + down) / (h * h);
411 }
