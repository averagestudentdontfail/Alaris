192 // STRK001A - Net-of-Cost Kelly
193
194 [Fact]
195 public void STRK001A_CalculateNetOfCostKelly_NoCost_ReturnsStandardKelly()
196 {
197 // Arrange: 60% win rate, 2:1 win/loss ratio
198 double winProb = 0.60;
199 double avgWin = 200.0;
200 double avgLoss = 100.0;
201
202 // Standard Kelly: (0.6 * 2 - 0.4) / 2 = 0.4
203 double standardKelly = ((winProb * (avgWin / avgLoss)) - (1 - winProb)) / (avgWin / avgLoss);
204
205 // Act
206 double netKelly = STRK001A.CalculateNetOfCostKelly(winProb, avgWin, avgLoss, roundTripCost: 0);
207
208 // Assert
209 netKelly.Should().BeApproximately(standardKelly, 0.001);
210 }
211
212 [Fact]
213 public void STRK001A_CalculateNetOfCostKelly_HighCost_ReturnsReduced()
