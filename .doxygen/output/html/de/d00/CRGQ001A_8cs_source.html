62 public static (double[] Nodes, double[] Weights) GaussLegendre(int n)
63 {
64 ArgumentOutOfRangeException.ThrowIfLessThan(n, 1);
65 ArgumentOutOfRangeException.ThrowIfGreaterThan(n, MaxLegendreOrder);
66
67 double[] nodes = new double[n];
68 double[] weights = new double[n];
69
70 int m = (n + 1) / 2; // Number of positive roots (symmetric)
71
72 for (int i = 0; i < m; i++)
73 {
74 // Initial guess using asymptotic formula
75 double z = System.Math.Cos(System.Math.PI * (i + 0.75) / (n + 0.5));
76
77 double z1, pp;
78 int iter = 0;
79
80 do
81 {
82 // Evaluate Legendre polynomial and its derivative
83 double p1 = 1.0;
84 double p2 = 0.0;
85
86 for (int j = 0; j < n; j++)
87 {
88 double p3 = p2;
89 p2 = p1;
90 p1 = (((2.0 * j) + 1.0) * z * p2 - j * p3) / (j + 1);
91 }
92
93 // p1 is now P_n(z), derivative using recurrence
94 pp = n * ((z * p1) - p2) / ((z * z) - 1.0);
95
96 z1 = z;
97 z -= p1 / pp;
98 iter++;
99 }
100 while (System.Math.Abs(z - z1) > LegendreRootTolerance && iter < LegendreMaxIterations);
101
102 // Store symmetric pairs
103 nodes[i] = -z;
104 nodes[n - 1 - i] = z;
105
106 // Weight formula
107 double weight = 2.0 / (((1.0 - (z * z)) * pp) * pp);
108 weights[i] = weight;
109 weights[n - 1 - i] = weight;
110 }
111
112 return (nodes, weights);
113 }

122 public static (double[] Nodes, double[] Weights) GaussLegendreAB(int n, double a, double b)
123 {
124 (double[] nodes, double[] weights) = GaussLegendre(n);
125
126 double halfWidth = (b - a) / 2.0;
127 double midPoint = (a + b) / 2.0;
128
129 for (int i = 0; i < n; i++)
130 {
131 nodes[i] = midPoint + (halfWidth * nodes[i]);
132 weights[i] *= halfWidth;
133 }
134
135 return (nodes, weights);
136 }

148 {
149 ArgumentNullException.ThrowIfNull(f);
150 if (a == b)
151 {
152 return 0.0;

154
155 (double[] nodes, double[] weights) = GaussLegendreAB(n, a, b);
156
157 double sum = 0.0;
158 for (int i = 0; i < n; i++)
159 {

180 public static double TanhSinhIntegrate(
181 Func<double, double> f,
182 double a,
183 double b,
184 double tolerance = 1e-10,
185 int maxLevels = 10)
186 {
187 ArgumentNullException.ThrowIfNull(f);
188 ArgumentOutOfRangeException.ThrowIfLessThan(maxLevels, 1);
189 if (tolerance <= 0)
190 {
191 throw new ArgumentOutOfRangeException(nameof(tolerance), "Tolerance must be positive.");
192 }
193
194 if (a == b)
195 {
196 return 0.0;
197 }
198
199 double halfWidth = (b - a) / 2.0;
200 double midPoint = (a + b) / 2.0;
201 double piOverTwo = System.Math.PI / 2.0;
202 double h = 1.0; // Step size
203 double previousResult = 0.0;
204
205 for (int level = 0; level < maxLevels; level++)
206 {
207 double sum = 0.0;
208 for (int k = 0; k <= TanhSinhMaxSamples; k++)
209 {
210 double t = k * h;
211
212 double sinhT = System.Math.Sinh(t);
213 double piHalfSinhT = piOverTwo * sinhT;
214 if (System.Math.Abs(piHalfSinhT) > TanhSinhOverflowLimit)
215 {
216 break;
217 }
218
219 double coshT = System.Math.Cosh(t);
220 double x = System.Math.Tanh(piHalfSinhT);
221 double coshPiHalf = System.Math.Cosh(piHalfSinhT);
222 double w = piOverTwo * coshT / (coshPiHalf * coshPiHalf);
223
224 if (w < TanhSinhWeightCutoff)
225 {
226 break;
227 }
228
229 if (k == 0)
230 {
231 double gVal = f(midPoint + (halfWidth * x));
232 if (!double.IsNaN(gVal) && !double.IsInfinity(gVal))
233 {
234 sum += w * gVal;
235 }
236 }
237 else
238 {
239 double gPos = f(midPoint + (halfWidth * x));
240 double gNeg = f(midPoint - (halfWidth * x));
241
242 if (!double.IsNaN(gPos) && !double.IsInfinity(gPos))
243 {
244 sum += w * gPos;
245 }
246
247 if (!double.IsNaN(gNeg) && !double.IsInfinity(gNeg))
248 {
249 sum += w * gNeg;
250 }
251 }
252 }
253
254 double result = h * sum * halfWidth;
255
256 // Check convergence
257 if (level > 0 && System.Math.Abs(result - previousResult) < tolerance * System.Math.Abs(result))
258 {
259 return result;
260 }
261
262 previousResult = result;
263 h /= 2.0;
264 }
265
266 return previousResult;
267 }

304 private static double AdaptiveIntegrateRecursive(
305 Func<double, double> f,
306 double a,
307 double b,
308 double tolerance,
309 int maxDepth,
310 int depth)
311 {
312 // 7-point Gauss rule
313 double integral7 = IntegrateWithScheme(f, Schemes.Adaptive7, a, b);
314 double integral15 = IntegrateWithScheme(f, Schemes.Adaptive15, a, b);
315
316 double error = System.Math.Abs(integral15 - integral7);
317 double absIntegral = System.Math.Abs(integral15);
318
319 // Check if error is acceptable
320 if (error < tolerance * absIntegral || depth >= maxDepth)
321 {
322 return integral15;
323 }
324
325 // Subdivide
326 double mid = (a + b) / 2.0;
327 double left = AdaptiveIntegrateRecursive(f, a, mid, tolerance / 2.0, maxDepth, depth + 1);
328 double right = AdaptiveIntegrateRecursive(f, mid, b, tolerance / 2.0, maxDepth, depth + 1);
329
330 return left + right;
331 }

341 [MethodImpl(MethodImplOptions.AggressiveInlining)]
342 public static double IntegrateWithScheme(
343 Func<double, double> f,
344 (double[] Nodes, double[] Weights) scheme,
345 double a,
346 double b)
347 {
348 ArgumentNullException.ThrowIfNull(f);
349 if (a == b)
350 {
351 return 0.0;
352 }
353
354 double halfWidth = (b - a) / 2.0;
355 double midPoint = (a + b) / 2.0;
356
357 double sum = 0.0;
358 int n = scheme.Nodes.Length;
359
360 for (int i = 0; i < n; i++)
361 {
362 double x = midPoint + (halfWidth * scheme.Nodes[i]);
363 sum += scheme.Weights[i] * f(x);
364 }
365
366 return halfWidth * sum;
367 }

377 public static (double[] Nodes, double[] Weights) GaussLaguerre(int n)
378 {
379 ArgumentOutOfRangeException.ThrowIfLessThan(n, 1);
380 ArgumentOutOfRangeException.ThrowIfGreaterThan(n, MaxLaguerreOrder);
381
382 double[] nodes = new double[n];
383 double[] weights = new double[n];
384
385 for (int i = 0; i < n; i++)
386 {
387 // Initial guess
388 double z;
389 if (i == 0)
390 {
391 z = 3.0 / (1.0 + (2.4 * n));
392 }
393 else if (i == 1)
394 {
395 z = nodes[0] + (15.0 / (1.0 + (2.5 * n)));
396 }
397 else
398 {
399 double ratio = (1.0 + (2.55 * (i - 1))) / (1.9 * (i - 1));
400 z = nodes[i - 1] + (ratio * (nodes[i - 1] - nodes[i - 2]));
401 }
402
403 // Newton-Raphson
404 int maxIter = 100;
405 for (int iter = 0; iter < maxIter; iter++)
406 {
407 double p1 = 1.0;
408 double p2 = 0.0;
409
410 for (int j = 0; j < n; j++)
411 {
412 double p3 = p2;
413 p2 = p1;
414 p1 = (((((2 * j) + 1) - z) * p2) - (j * p3)) / (j + 1);
415 }
416
417 double pp = (n * p1 - n * p2) / z;
418 double z1 = z;
419 z -= p1 / pp;
420
421 if (System.Math.Abs(z - z1) < 1e-15)
422 {
423 break;
424 }
425 }
426
427 nodes[i] = z;
428
429 // Evaluate L_{n-1} at z
430 double lnm1 = 0.0;
431 double ln = 1.0;
432 for (int j = 0; j < n; j++)
433 {
434 double tmp = ln;
435 ln = (((((2 * j) + 1) - z) * ln) - (j * lnm1)) / (j + 1);
436 lnm1 = tmp;
437 }
438
439 weights[i] = z / ((n + 1) * (n + 1) * lnm1 * lnm1);
440 }
441
442 return (nodes, weights);
443 }

453 public static (double[] Nodes, double[] Weights) GaussHermite(int n)
454 {
455 ArgumentOutOfRangeException.ThrowIfLessThan(n, 1);
456 ArgumentOutOfRangeException.ThrowIfGreaterThan(n, MaxHermiteOrder);
457
458 double[] nodes = new double[n];
459 double[] weights = new double[n];
460
461 int m = (n + 1) / 2;
462
463 for (int i = 0; i < m; i++)
464 {
465 // Initial guess
466 double z;
467 if (i == 0)
468 {
469 z = System.Math.Sqrt(2.0 * n + 1.0) - (1.85575 * System.Math.Pow(2.0 * n + 1.0, -1.0 / 6.0));
470 }
471 else if (i == 1)
472 {
473 z = nodes[0] - (1.14 * System.Math.Pow(n, 0.426) / nodes[0]);
474 }
475 else if (i == 2)
476 {
477 z = (1.86 * nodes[1]) - (0.86 * nodes[0]);
478 }
479 else if (i == 3)
480 {
481 z = (1.91 * nodes[2]) - (0.91 * nodes[1]);
482 }
483 else
484 {
485 z = (2.0 * nodes[i - 1]) - nodes[i - 2];
486 }
487
488 // Newton-Raphson
489 int maxIter = 100;
490 double p2Final = 0.0;
491 for (int iter = 0; iter < maxIter; iter++)
492 {
493 double p1 = System.Math.Sqrt(System.Math.PI) * 0.7511255444649425; // 1/^(1/4)
494 double p2 = 0.0;
495
496 for (int j = 0; j < n; j++)
497 {
498 double p3 = p2;
499 p2 = p1;
500 p1 = (z * System.Math.Sqrt(2.0 / (j + 1)) * p2) - (System.Math.Sqrt((double)j / (j + 1)) * p3);
501 }
502
503 p2Final = p2;
504 double pp = System.Math.Sqrt(2.0 * n) * p2;
505 double z1 = z;
506 z -= p1 / pp;
507
508 if (System.Math.Abs(z - z1) < 1e-15)
509 {
510 break;
511 }
512 }
513
514 nodes[i] = -z;
515 nodes[n - 1 - i] = z;
516
517 double weight = 2.0 / (p2Final * p2Final);
518 weights[i] = weight;
519 weights[n - 1 - i] = weight;
520 }
521
522 return (nodes, weights);
523 }
