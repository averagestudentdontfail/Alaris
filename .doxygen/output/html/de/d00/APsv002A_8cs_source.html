228 public async Task<int> BootstrapEarningsCalendarAsync(
229 DateTime startDate,
230 DateTime endDate,
231 string outputPath,
232 CancellationToken cancellationToken = default)
233 {
234 if (_earningsClient == null)
235 {
236 throw new InvalidOperationException("Earnings client not configured");
237 }
238
239 string nasdaqPath = System.IO.Path.Combine(outputPath, "earnings", "nasdaq");
240 Directory.CreateDirectory(nasdaqPath);
241
242 int totalWeekdays = 0;
243 int downloadedDays = 0;
244 int skippedDays = 0;
245
246 // Count total weekdays
247 for (DateTime d = startDate; d <= endDate; d = d.AddDays(1))
248 {
249 if (d.DayOfWeek != DayOfWeek.Saturday && d.DayOfWeek != DayOfWeek.Sunday)
250 {
251 totalWeekdays++;
252 }
253 }
254
255 _logger?.LogInformation(
256 "Bootstrap earnings calendar: {Start:yyyy-MM-dd} to {End:yyyy-MM-dd} ({Total} weekdays)",
257 startDate, endDate, totalWeekdays);
258
259 await AnsiConsole.Progress()
260 .AutoClear(false)
261 .Columns(
262 new TaskDescriptionColumn(),
263 new ProgressBarColumn(),
264 new PercentageColumn(),
265 new RemainingTimeColumn(),
266 new SpinnerColumn())
267 .StartAsync(async ctx =>
268 {
269 var task = ctx.AddTask("Downloading Earnings Calendar", maxValue: totalWeekdays);
270
271 for (DateTime date = startDate; date <= endDate; date = date.AddDays(1))
272 {
273 cancellationToken.ThrowIfCancellationRequested();
274
275 // Skip weekends
276 if (date.DayOfWeek == DayOfWeek.Saturday || date.DayOfWeek == DayOfWeek.Sunday)
277 {
278 continue;
279 }
280
281 string cachePath = System.IO.Path.Combine(nasdaqPath, $"{date:yyyy-MM-dd}.json");
282
283 // Skip if already cached
284 if (File.Exists(cachePath))
285 {
286 skippedDays++;
287 task.Increment(1);
288 continue;
289 }
290
291 try
292 {
293 // Rate limit: 1 request per second
294 await Task.Delay(TimeSpan.FromSeconds(1), cancellationToken);
295
296 var earnings = await _earningsClient.FetchAndCacheAsync(
297 date, outputPath, cancellationToken);
298
299 downloadedDays++;
300 task.Description = $"Downloaded {date:yyyy-MM-dd} ({earnings.Count} events)";
301 task.Increment(1);
302 }
303 catch (HttpRequestException ex)
304 {
305 _logger?.LogWarning(
306 ex,
307 "Failed to fetch {Date:yyyy-MM-dd}, will retry on next run",
308 date);
309 task.Increment(1);
310 }
311 }
312 });
313
314 _logger?.LogInformation(
315 "Earnings bootstrap complete: {Downloaded} downloaded, {Skipped} skipped (already cached)",
316 downloadedDays, skippedDays);
317
318 return downloadedDays;
319 }

331 public async Task<int> BootstrapOptionsDataAsync(
332 IEnumerable<string> symbols,
333 IEnumerable<DateTime> dates,
334 string sessionDataPath,
335 CancellationToken cancellationToken = default)
336 {
337 var symbolList = symbols.ToList();
338 var dateList = dates.Distinct().OrderBy(d => d).ToList();
339
340 if (symbolList.Count == 0 || dateList.Count == 0)
341 {
342 return 0;
343 }
344
345 var optionsPath = System.IO.Path.Combine(sessionDataPath, "options");
346 Directory.CreateDirectory(optionsPath);
347
348 int totalDownloaded = 0;
349 int totalSkipped = 0;
350 int totalFailed = 0;
351
352 // Apply 2-year limit buffer (Polygon Options Starter plan)
353 var optionsMinDate = DateTime.UtcNow.AddYears(-2).AddMonths(1).Date;
354
355 _logger?.LogInformation(
356 "Bootstrap options: {SymbolCount} symbols Ã— {DateCount} dates",
357 symbolList.Count, dateList.Count);
358
359 await AnsiConsole.Progress()
360 .AutoClear(false)
361 .Columns(
362 new TaskDescriptionColumn(),
363 new ProgressBarColumn(),
364 new PercentageColumn(),
365 new SpinnerColumn())
366 .StartAsync(async ctx =>
367 {
368 var task = ctx.AddTask("Downloading Options Data", maxValue: symbolList.Count * dateList.Count);
369
370 foreach (var symbol in symbolList)
371 {
372 cancellationToken.ThrowIfCancellationRequested();
373
374 foreach (var date in dateList)
375 {
376 cancellationToken.ThrowIfCancellationRequested();
377
378 var symbolLower = symbol.ToLowerInvariant();
379 var dateSuffix = date.ToString("yyyyMMdd");
380 var cachePath = System.IO.Path.Combine(optionsPath, $"{symbolLower}_{dateSuffix}.json");
381
382 // Skip if already cached
383 if (File.Exists(cachePath))
384 {
385 totalSkipped++;
386 task.Increment(1);
387 continue;
388 }
389
390 // Skip dates outside Polygon's 2-year limit
391 if (date < optionsMinDate)
392 {
393 _logger?.LogDebug("Skipping {Symbol} @ {Date}: outside 2-year limit", symbol, date);
394 task.Increment(1);
395 continue;
396 }
397
398 try
399 {
400 task.Description = $"Options: {symbol} @ {date:yyyy-MM-dd}";
401
402 var optionChain = await _polygonClient.GetHistoricalOptionChainAsync(symbol, date, cancellationToken);
403
404 if (optionChain.Contracts.Count > 0)
405 {
406 var json = JsonSerializer.Serialize(optionChain, JsonOptions);
407 await File.WriteAllTextAsync(cachePath, json, cancellationToken);
408 totalDownloaded++;
409 _logger?.LogDebug("Cached {Count} options for {Symbol} @ {Date}",
410 optionChain.Contracts.Count, symbol, date);
411 }
412 else
413 {
414 _logger?.LogDebug("No options found for {Symbol} @ {Date}", symbol, date);
415 }
416 }
417 catch (Exception ex)
418 {
419 totalFailed++;
420 _logger?.LogWarning(ex, "Failed to download options for {Symbol} @ {Date}", symbol, date);
421 }
422
423 task.Increment(1);
424
425 // Rate limit: small delay between requests
426 await Task.Delay(100, cancellationToken);
427 }
428 }
429 });
430
431 _logger?.LogInformation(
432 "Options bootstrap complete: {Downloaded} downloaded, {Skipped} cached, {Failed} failed",
433 totalDownloaded, totalSkipped, totalFailed);
434
435 return totalDownloaded;
436 }

450 public async Task<BootstrapReport> BootstrapSessionDataAsync(
451 Alaris.Core.Model.STDT010A requirements,
452 string sessionDataPath,
453 CancellationToken cancellationToken = default)
454 {
455 ArgumentNullException.ThrowIfNull(requirements);
456 ArgumentException.ThrowIfNullOrWhiteSpace(sessionDataPath);
457
458 var (isValid, error) = requirements.Validate();
459 if (!isValid)
460 throw new ArgumentException(error, nameof(requirements));
461
462 _logger?.LogInformation("Starting unified bootstrap: {Summary}", requirements.GetSummary());
463
464 var report = new BootstrapReport
465 {
466 StartedAt = DateTime.UtcNow,
467 SessionDataPath = sessionDataPath
468 };
469
470 try
471 {
472 // Phase 1: System files (market-hours, symbol-properties, default maps)
473 AnsiConsole.MarkupLine("[blue]Phase 1:[/] Copying system files...");
474 CopySystemFiles(sessionDataPath);
475
476 // Phase 2: Price data with warmup buffer
477 AnsiConsole.MarkupLine("[blue]Phase 2:[/] Downloading price data...");
478 report.PricesDownloaded = await DownloadPriceDataWithBenchmarkAsync(
479 requirements, sessionDataPath, cancellationToken);
480
481 // Phase 3: Earnings calendar with lookahead
482 AnsiConsole.MarkupLine("[blue]Phase 3:[/] Downloading earnings calendar (with lookahead)...");
483 AnsiConsole.MarkupLine($"[grey] Range: {requirements.StartDate:yyyy-MM-dd} to {requirements.EarningsLookaheadEnd:yyyy-MM-dd}[/]");
484
485 if (_earningsClient != null)
486 {
487 report.EarningsDaysDownloaded = await BootstrapEarningsCalendarAsync(
488 requirements.StartDate,
489 requirements.EarningsLookaheadEnd, // Critical: includes lookahead
490 sessionDataPath,
491 cancellationToken);
492 }
493
494 // Phase 4: Compute options-required dates from earnings
495 AnsiConsole.MarkupLine("[blue]Phase 4:[/] Computing options-required dates...");
496 var optionsDates = ComputeOptionsRequiredDates(
497 sessionDataPath,
498 requirements.Symbols,
499 requirements.StartDate,
500 requirements.EndDate,
501 requirements.SignalWindowMinDays,
502 requirements.SignalWindowMaxDays);
503
504 report.OptionsRequiredDatesComputed = optionsDates.Count;
505 AnsiConsole.MarkupLine($"[grey] Found {optionsDates.Count} dates requiring options data[/]");
506
507 // Phase 5: Bootstrap options for each required date
508 if (optionsDates.Count > 0)
509 {
510 AnsiConsole.MarkupLine("[blue]Phase 5:[/] Downloading options data for signal dates...");
511 report.OptionsDownloaded = await BootstrapOptionsDataAsync(
512 requirements.Symbols,
513 optionsDates,
514 sessionDataPath,
515 cancellationToken);
516 }
517
518 // Phase 6: Interest rates (optional)
519 if (_treasuryClient != null)
520 {
521 AnsiConsole.MarkupLine("[blue]Phase 6:[/] Downloading interest rates...");
522 report.InterestRatesDownloaded = await DownloadInterestRatesAsync(
523 requirements, sessionDataPath, cancellationToken);
524 }
525
526 report.Success = true;
527 }
528 catch (Exception ex)
529 {
530 report.Success = false;
531 report.ErrorMessage = ex.Message;
532 _logger?.LogError(ex, "Bootstrap failed");
533 }
534 finally
535 {
536 report.CompletedAt = DateTime.UtcNow;
537 }
538
539 _logger?.LogInformation(
540 "Bootstrap complete: Prices={Prices}, Earnings={Earnings}, Options={Options}",
541 report.PricesDownloaded, report.EarningsDaysDownloaded, report.OptionsDownloaded);
542
543 return report;
544 }

652 private IReadOnlyList<DateTime> ComputeOptionsRequiredDates(
653 string sessionDataPath,
654 IReadOnlyList<string> symbols,
655 DateTime startDate,
656 DateTime endDate,
657 int minDays,
658 int maxDays)
659 {
660 var dates = new HashSet<DateTime>();
661 var nasdaqPath = System.IO.Path.Combine(sessionDataPath, "earnings", "nasdaq");
662
663 if (!Directory.Exists(nasdaqPath))
664 {
665 _logger?.LogWarning("Earnings directory not found: {Path}", nasdaqPath);
666 return Array.Empty<DateTime>();
667 }
668
669 // Read all earnings files and find dates for our symbols
670 foreach (var file in Directory.GetFiles(nasdaqPath, "*.json"))
671 {
672 try
673 {
674 var json = File.ReadAllText(file);
675 var cached = JsonSerializer.Deserialize<CachedEarningsDay>(json);
676
677 if (cached?.Earnings == null)
678 continue;
679
680 foreach (var earning in cached.Earnings)
681 {
682 // Check if this symbol is in our universe
683 if (!symbols.Contains(earning.Symbol, StringComparer.OrdinalIgnoreCase))
684 continue;
685
686 // For this earnings date, compute evaluation dates
687 for (int d = minDays; d <= maxDays; d++)
688 {
689 var evalDate = earning.Date.AddDays(-d);
690
691 // Only include if within session range
692 if (evalDate >= startDate && evalDate <= endDate)
693 {
694 // Skip weekends
695 if (evalDate.DayOfWeek != DayOfWeek.Saturday &&
696 evalDate.DayOfWeek != DayOfWeek.Sunday)
697 {
698 dates.Add(evalDate);
699 }
700 }
701 }
702 }
703 }
704 catch (Exception ex)
705 {
706 _logger?.LogDebug(ex, "Error reading earnings file: {File}", file);
707 }
708 }
709
710 _logger?.LogInformation("Computed {Count} options-required dates from earnings calendar", dates.Count);
711 return dates.OrderBy(d => d).ToList();
712 }

738 private static async Task SaveAsLeanZipAsync(string symbol, IEnumerable<PriceBar> bars, string outputDir)
739 {
740 // LEAN format: ticker.zip containing ticker.csv
741 var ticker = symbol.ToLowerInvariant();
742 var zipPath = System.IO.Path.Combine(outputDir, $"{ticker}.zip");
743
744 using var memoryStream = new MemoryStream();
745 using (var archive = new ZipArchive(memoryStream, ZipArchiveMode.Create, true))
746 {
747 var entry = archive.CreateEntry($"{ticker}.csv");
748 using var entryStream = entry.Open();
749 using var writer = new StreamWriter(entryStream);
750
751 foreach (var bar in bars)
752 {
753 // Format: Date,Open,High,Low,Close,Volume
754 // Date format: yyyyMMdd HH:mm
755 // Prices: Scaled by 10000 ensures compatibility with older LEAN data readers,
756 // but strictly speaking typical daily CSV is
757 // yyyyMMdd 00:00,open*10000,high*10000,low*10000,close*10000,volume
758
759 // Assuming 'bar' is StrategyPriceBar or similar with Open, High, Low, Close, Volume, Date
760 // Since I can't see StrategyPriceBar def right now, I'll assume standard properties.
761 // Using dynamic to avoid circular dependency if Model is in another project,
762 // but ideally should verify type.
763
764 // LEAN Daily resolution expects 'TwelveCharacter' format: "yyyyMMdd HH:mm"
765 // TradeBar.cs ParseEquity uses default scaling (x10000)
766 // CRITICAL: Daily bars must use 00:00 (exchange timezone midnight), not actual UTC timestamp
767 var dateStr = bar.Timestamp.Date.ToString("yyyyMMdd", CultureInfo.InvariantCulture) + " 00:00";
768
769 // Scale by 10000 to match LEAN default scale factor (1/10000)
770 // When LEAN reads this, it divides by 10000 to get the original price.
771 var open = (long)(bar.Open * 10000);
772 var high = (long)(bar.High * 10000);
773 var low = (long)(bar.Low * 10000);
774 var close = (long)(bar.Close * 10000);
775 var volume = (long)bar.Volume;
776
777 await writer.WriteLineAsync(string.Format(CultureInfo.InvariantCulture,
778 "{0},{1},{2},{3},{4},{5}",
779 dateStr, open, high, low, close, volume));
780 }
