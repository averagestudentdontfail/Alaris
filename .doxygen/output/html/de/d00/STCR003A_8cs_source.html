80 private static void CalculateLogReturnsInPlace(
81 IReadOnlyList<PriceBar> priceBars,
82 int startIdx,
83 int window,
84 double[] openReturns,
85 double[] closeReturns,
86 double[] rogersReturns)
87 {
88 for (int i = 0; i < window; i++)
89 {
90 PriceBar current = priceBars[startIdx + i + 1];
91 PriceBar previous = priceBars[startIdx + i];
92
93 // o_i = ln(Open_i / Close_{i-1})
94 openReturns[i] = Math.Log(current.Open / previous.Close);
95
96 // c_i = ln(Close_i / Open_i)
97 double c = Math.Log(current.Close / current.Open);
98 closeReturns[i] = c;
99
100 // Rogers-Satchell component: RS_i = u_i(u_i - c_i) + d_i(d_i - c_i)
101 double u = Math.Log(current.High / current.Open);
102 double d = Math.Log(current.Low / current.Open);
103 rogersReturns[i] = (u * (u - c)) + (d * (d - c));
104 }

110 private static double CalculateSTCR003AVarianceFromSpan(
111 ReadOnlySpan<double> openReturns,
112 ReadOnlySpan<double> closeReturns,
113 ReadOnlySpan<double> rogersReturns,
114 int window)
115 {
116 // Calculate variance components using span-based operations
117 double openVariance = VarianceFromSpan(openReturns);
118 double closeVariance = VarianceFromSpan(closeReturns);
119 double rogersVariance = AverageFromSpan(rogersReturns);
120
121 // Calculate the weight k = 0.34 / (1.34 + (n+1)/(n-1))
122 double k = 0.34 / (1.34 + ((window + 1.0) / (window - 1.0)));
123
124 // Yang-Zhang variance: σ_yz² = σ_o² + k·σ_c² + (1-k)·σ_rs²
125 return openVariance + (k * closeVariance) + ((1 - k) * rogersVariance);
126 }

131 private static double VarianceFromSpan(ReadOnlySpan<double> values)
132 {
133 if (values.Length < 2)
134 {
135 return 0;
136 }
137
138 double mean = AverageFromSpan(values);
139 double sumSquaredDeviations = 0;
140 for (int i = 0; i < values.Length; i++)
141 {
142 double deviation = values[i] - mean;
143 sumSquaredDeviations += deviation * deviation;
144 }
145 return sumSquaredDeviations / (values.Length - 1);
146 }

152 {

159 for (int i = 0; i < values.Length; i++)
160 {

162 }
