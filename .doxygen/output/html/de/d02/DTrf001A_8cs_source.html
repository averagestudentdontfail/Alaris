60 public async Task<decimal> GetCurrentRateAsync(CancellationToken cancellationToken = default)
61 {
62 _logger.LogInformation("Fetching current 3-month T-bill rate");
63
64 try
65 {
66 DateTime today = DateTime.UtcNow.Date;
67 DateTime startDate = today.AddDays(-7);
68
69 TreasurySecurityDto[] securities = await _api.SearchSecuritiesAsync(
70 securityType: "Bill",
71 dateFieldName: "issueDate",
72 startDate: startDate.ToString("yyyy-MM-dd"),
73 endDate: today.ToString("yyyy-MM-dd"),
74 format: "json",
75 cancellationToken);
76
77 if (securities == null || securities.Length == 0)
78 {
79 _logger.LogError("No recent T-bill auctions found. Cannot determine risk-free rate.");
80 throw new InvalidOperationException(
81 "No T-bill auction data found. Treasury Direct API may be unavailable.");
82 }
83
84 // Find most recent 3-month T-bill (91-day maturity)
85 TreasurySecurityDto? threeMonthBill = securities
86 .Where(s => s.Term != null && s.Term.Contains("91"))
87 .OrderByDescending(s => s.IssueDate)
88 .FirstOrDefault();
89
90 if (threeMonthBill == null)
91 {
92 _logger.LogWarning("No 3-month T-bills found in recent auctions");
93
94 // Use any bill as fallback
95 TreasurySecurityDto anyBill = securities
96 .OrderByDescending(s => s.IssueDate)
97 .First();
98
99 decimal fallbackRate = ParseRate(anyBill.InterestRate);
100 _logger.LogInformation("Using fallback rate from {Term}: {Rate:P4}", anyBill.Term, fallbackRate);
101 return fallbackRate;
102 }
103
104 decimal rate = ParseRate(threeMonthBill.InterestRate);
105
106 _logger.LogInformation(
107 "Current 3-month T-bill rate: {Rate:P4} (issue date: {IssueDate:yyyy-MM-dd})",
108 rate,
109 threeMonthBill.IssueDate);
110
111 return rate;
112 }
113 catch (HttpRequestException ex)
114 {
115 _logger.LogError(ex, "HTTP error fetching T-bill rates. Cannot determine risk-free rate.");
116 throw new InvalidOperationException(
117 "Failed to fetch T-bill rates from Treasury Direct API.", ex);
118 }
119 catch (JsonException ex)
120 {
121 _logger.LogError(ex, "JSON parse error fetching T-bill rates.");
122 throw new InvalidOperationException(
123 "Failed to parse Treasury Direct API response.", ex);
124 }
125 }

128 public async Task<IReadOnlyDictionary<DateTime, decimal>> GetHistoricalRatesAsync(
129 DateTime startDate,
130 DateTime endDate,
131 CancellationToken cancellationToken = default)
132 {
133 _logger.LogInformation(
134 "Fetching historical T-bill rates from {StartDate:yyyy-MM-dd} to {EndDate:yyyy-MM-dd}",
135 startDate, endDate);
136
137 try
138 {
139 TreasurySecurityDto[] securities = await _api.SearchSecuritiesAsync(
140 securityType: "Bill",
141 dateFieldName: "issueDate",
142 startDate: startDate.ToString("yyyy-MM-dd"),
143 endDate: endDate.ToString("yyyy-MM-dd"),
144 format: "json",
145 cancellationToken);
146
147 if (securities == null || securities.Length == 0)
148 {
149 _logger.LogError("Treasury API returned no data for range {Start} to {End}. Check Date Range or API Status.", startDate, endDate);
150 throw new InvalidOperationException($"No Treasury data found for {startDate:d}-{endDate:d}");
151 }
152
153 // Group by issue date, prefer 3-month bills
154 Dictionary<DateTime, decimal> ratesByDate = securities
155 .Where(s => !string.IsNullOrWhiteSpace(s.InterestRate))
156 .GroupBy(s => s.IssueDate.Date)
157 .ToDictionary(
158 g => g.Key,
159 g =>
160 {
161 // Prefer 3-month bill if available (Term usually '13-Week' or similar, strict check '91')
162 TreasurySecurityDto preferred = g.FirstOrDefault(s => s.Term != null && s.Term.Contains("91"))
163 ?? g.First();
164 return ParseRate(preferred.InterestRate!);
165 });
166
167 _logger.LogInformation("Retrieved {Count} historical rate observations from Treasury Direct", ratesByDate.Count);
168
169 return ratesByDate;
170 }
171 catch (HttpRequestException ex)
172 {
173 _logger.LogError(ex, "HTTP error fetching historical T-bill rates");
174 throw; // Fail fast, do not use fake data
175 }
176 catch (JsonException ex)
177 {
178 _logger.LogError(ex, "JSON parse error fetching historical T-bill rates");
179 throw;
180 }
181 }

187 private static decimal ParseRate(string? rateString)
188 {
189 if (string.IsNullOrWhiteSpace(rateString))
190 {
191 return 0m;
192 }
193
194 string cleaned = new(rateString.Where(c => char.IsDigit(c) || c == '.' || c == '-').ToArray());
195
196 if (!decimal.TryParse(cleaned, out decimal rate))
197 {
198 return 0m;
199 }
200
201 return rate / 100m;
202 }
