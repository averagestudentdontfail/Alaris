105 public STCS003A ComputeOptionCost(STCS002A parameters)
106 {
107 ArgumentNullException.ThrowIfNull(parameters);
108 parameters.Validate();
109
110 // Compute execution price based on direction
111 double executionPrice = parameters.Direction == OrderDirection.Buy
112 ? parameters.AskPrice
113 : parameters.BidPrice;
114
115 // Compute slippage (difference from mid to execution price)
116 double slippagePerContract = Math.Abs(executionPrice - parameters.MidPrice)
117 * parameters.ContractMultiplier;
118 double totalSlippage = slippagePerContract * parameters.Contracts;
119
120 // Compute fees
121 double commission = _feePerContract * parameters.Contracts;
122 double exchangeFees = _exchangeFeePerContract * parameters.Contracts;
123 double regulatoryFees = _regulatoryFeePerContract * parameters.Contracts;
124
125 var result = new STCS003A
126 {
127 Commission = commission,
128 ExchangeFees = exchangeFees,
129 RegulatoryFees = regulatoryFees,
130 Slippage = totalSlippage,
131 TheoreticalPrice = parameters.MidPrice,
132 ExecutionPrice = executionPrice,
133 Contracts = parameters.Contracts
134 };
135
136 SafeLog(() => LogCostComputed(_logger!, parameters.Symbol, parameters.Contracts, result.TotalCost, null));
137
138 return result;
139 }

142 public STCS004A ComputeSpreadCost(STCS002A frontLegParameters, STCS002A backLegParameters)
143 {
144 ArgumentNullException.ThrowIfNull(frontLegParameters);
145 ArgumentNullException.ThrowIfNull(backLegParameters);
146
147 frontLegParameters.Validate();
148 backLegParameters.Validate();
149
150 // Front leg: SELL (receive bid)
151 STCS002A adjustedFrontParams = frontLegParameters with { Direction = OrderDirection.Sell };
152 STCS003A frontCost = ComputeOptionCost(adjustedFrontParams);
153
154 // Back leg: BUY (pay ask)
155 STCS002A adjustedBackParams = backLegParameters with { Direction = OrderDirection.Buy };
156 STCS003A backCost = ComputeOptionCost(adjustedBackParams);
157
158 // Theoretical debit (mid-price based)
159 double theoreticalDebit = backLegParameters.MidPrice - frontLegParameters.MidPrice;
160
161 // Execution debit (bid-ask adjusted)
162 // Buy back at ask, sell front at bid
163 double executionDebit = backLegParameters.AskPrice - frontLegParameters.BidPrice;
164
165 int contracts = Math.Min(frontLegParameters.Contracts, backLegParameters.Contracts);
166 double contractMultiplier = frontLegParameters.ContractMultiplier;
167
168 var result = new STCS004A
169 {
170 FrontLegCost = frontCost,
171 BackLegCost = backCost,
172 TheoreticalDebit = theoreticalDebit,
173 ExecutionDebit = executionDebit,
174 Contracts = contracts,
175 ContractMultiplier = contractMultiplier
176 };
177
178 SafeLog(() => LogSpreadCostComputed(
179 _logger!,
180 result.TheoreticalDebit,
181 result.ExecutionDebit,
182 result.SlippagePercent,
183 null));
184
185 return result;
186 }
