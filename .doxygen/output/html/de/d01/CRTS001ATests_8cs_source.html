124 [Fact]
125 public void ForwardRate_ConsistentWithDiscountFactors()
126 {
127 var referenceDate = new CRTM005A(1, CRTM005AMonth.January, 2024);
128 double rate = 0.05;
129 double T1 = 1.0;
130 double T2 = 2.0;
131 var curve = new CRTS001AFlatForward(referenceDate, rate);
132
133 double df1 = curve.DiscountFactor(T1);
134 double df2 = curve.DiscountFactor(T2);
135 double impliedFwd = -System.Math.Log(df2 / df1) / (T2 - T1);
136
137 // For flat forward, the instantaneous forward rate at any point equals the flat rate
138 double fwdRate = curve.ForwardRate(T1);
139
140 Assert.Equal(rate, fwdRate, Tolerance);
141 Assert.Equal(rate, impliedFwd, Tolerance);
142 }
