52 [MethodImpl(MethodImplOptions.AggressiveInlining)]
53 public static double Integrate(Func<double, double> f, double a, double b)
54 {
55 ArgumentNullException.ThrowIfNull(f);
56
57 if (a >= b)
58 {
59 return 0.0;
60 }
61
62 double scale = (b - a) * 0.5;
63 double shift = (a + b) * 0.5;
64 double sum = 0.0;
65
66 // Process in batches of 4 using AVX2
67 if (Avx2.IsSupported)
68 {
69 int i = 0;
70 for (; i + 4 <= GLNodes32.Length; i += 4)
71 {
72 // Compute 4 x-values
73 Vector256<double> nodes = Vector256.Create(
74 GLNodes32[i], GLNodes32[i + 1], GLNodes32[i + 2], GLNodes32[i + 3]);
75 Vector256<double> scaleVec = Vector256.Create(scale);
76 Vector256<double> shiftVec = Vector256.Create(shift);
77
78 // x = scale * node + shift
79 Vector256<double> x = Avx.Add(Avx.Multiply(scaleVec, nodes), shiftVec);
80
81 // Evaluate f at 4 points (function call overhead means we batch where we can)
82 double f0 = f(x.GetElement(0));
83 double f1 = f(x.GetElement(1));
84 double f2 = f(x.GetElement(2));
85 double f3 = f(x.GetElement(3));
86
87 // Compute weighted sum
88 sum += (f0 * GLWeights32[i]) + (f1 * GLWeights32[i + 1]) +
89 (f2 * GLWeights32[i + 2]) + (f3 * GLWeights32[i + 3]);
90 }
91
92 // Handle remainder
93 for (; i < GLNodes32.Length; i++)
94 {
95 double xi = (scale * GLNodes32[i]) + shift;
96 sum += f(xi) * GLWeights32[i];
97 }
98 }
99 else
100 {
101 // Scalar fallback
102 for (int i = 0; i < GLNodes32.Length; i++)
103 {
104 double xi = (scale * GLNodes32[i]) + shift;
105 sum += f(xi) * GLWeights32[i];
106 }
107 }
108
109 return scale * sum;
110 }

115 [MethodImpl(MethodImplOptions.AggressiveInlining)]
116 public static (double Value, double Error) IntegrateToInfinity(
117 Func<double, double> f,
118 double a,
119 double absoluteTolerance = 1e-8)
120 {
121 ArgumentNullException.ThrowIfNull(f);
122
123 double totalSum = 0;
124 double currentStart = a;
125 double currentStepSize = IntegrationStepSize;
126 double previousMagnitude = double.MaxValue;
127
128 for (int chunk = 0; chunk < MaxChunks; chunk++)
129 {
130 double currentEnd = currentStart + currentStepSize;
131 double chunkValue = Integrate(f, currentStart, currentEnd);
132 double chunkMagnitude = System.Math.Abs(chunkValue);
133
134 totalSum += chunkValue;
135
136 // Convergence check
137 if (chunkMagnitude < absoluteTolerance && chunk > 0)
138 {
139 return (totalSum, absoluteTolerance);
140 }
141
142 // Adaptive step sizing
143 if (chunk > 2 && chunkMagnitude < previousMagnitude * 0.1)
144 {
145 currentStepSize = System.Math.Min(currentStepSize * 2.0, IntegrationStepSize * 16);
146 }
147
148 previousMagnitude = chunkMagnitude;
149 currentStart = currentEnd;
150 }
151
152 return (totalSum, 1.0);
153 }
