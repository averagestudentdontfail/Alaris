122 public void HighVolatility_AdjustsPositionSize(double vixLevel, double expectedScaling)
123 {
124 // ═══════════════════════════════════════════════════════════
125 // ARRANGE: VIX-based position scaling
126 // ═══════════════════════════════════════════════════════════

128
129 // ═══════════════════════════════════════════════════════════
130 // ACT: Calculate VIX-adjusted risk
131 // ═══════════════════════════════════════════════════════════
132 double vixScaling = vixLevel < 0.30 ? 1.0 :
133 vixLevel < 0.40 ? 0.8 :
134 0.5;
135 double adjustedRisk = baseRisk * vixScaling;
136
137 // ═══════════════════════════════════════════════════════════
138 // ASSERT: Risk scales with VIX regime

141 adjustedRisk.Should().BeLessThanOrEqualTo(baseRisk);
