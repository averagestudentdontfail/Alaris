123 public int DayCount(CRTM005A start, CRTM005A end)
124 {
125 // ISDA 30/360 convention
126 int d1 = start.Day;
127 int d2 = end.Day;
128 int m1 = start.Month;
129 int m2 = end.Month;
130 int y1 = start.Year;
131 int y2 = end.Year;
132
133 // Adjust day values per ISDA rules
134 if (d1 == 31)
135 {
136 d1 = 30;
137 }
138
139 if (d2 == 31 && d1 >= 30)
140 {
141 d2 = 30;
142 }
143
144 return (360 * (y2 - y1)) + (30 * (m2 - m1)) + (d2 - d1);
145 }
146}
