96 public double CalculateExpectedConcurrentFromHistory(
97 IReadOnlyList<Trade> historicalTrades,
98 int analysisWindowDays = 252)
99 {
100 ArgumentNullException.ThrowIfNull(historicalTrades);
101
102 if (historicalTrades.Count < 10)
103 {
104 throw new ArgumentException("Requires at least 10 trades for reliable estimation",
105 nameof(historicalTrades));
106 }
107
108 // Calculate arrival rate (trades per day)
109 DateTime earliest = historicalTrades.Min(t => t.EntryDate);
110 DateTime latest = historicalTrades.Max(t => t.EntryDate);
111 int tradingDays = Math.Max(1, (int)(latest - earliest).TotalDays);
112 double arrivalRate = (double)historicalTrades.Count / tradingDays;
113
114 // Calculate average holding period
115 double avgHoldingPeriod = historicalTrades.Average(t => t.HoldingPeriod);
116
117 return CalculateExpectedConcurrent(arrivalRate, avgHoldingPeriod);
118 }

133 public STRK002A CalculateWithReserve(
134 double portfolioValue,
135 IReadOnlyList<Trade> historicalTrades,
136 double spreadCost,
137 STCR004A signal,
138 int currentOpenPositions,
139 double expectedConcurrent,
140 double reserveBuffer = DefaultReserveBuffer)
141 {
142 ArgumentNullException.ThrowIfNull(historicalTrades);
143 ArgumentNullException.ThrowIfNull(signal);
144
145 if (portfolioValue <= 0)
146 {
147 throw new ArgumentOutOfRangeException(nameof(portfolioValue),
148 "Portfolio value must be positive");
149 }
150
151 if (spreadCost <= 0)
152 {
153 throw new ArgumentOutOfRangeException(nameof(spreadCost),
154 "Spread cost must be positive");
155 }
156
157 if (currentOpenPositions < 0)
158 {
159 throw new ArgumentOutOfRangeException(nameof(currentOpenPositions),
160 "Open positions cannot be negative");
161 }
162
163 if (expectedConcurrent <= 0)
164 {
165 throw new ArgumentOutOfRangeException(nameof(expectedConcurrent),
166 "Expected concurrent positions must be positive");
167 }
168
169 if (reserveBuffer < 1.0)
170 {
171 throw new ArgumentOutOfRangeException(nameof(reserveBuffer),
172 "Reserve buffer must be at least 1.0");
173 }
174
175 // Calculate capital utilisation
176 double targetCapacity = expectedConcurrent * reserveBuffer;
177 double utilisation = currentOpenPositions / targetCapacity;
178
179 // Dynamic quality threshold based on utilisation
180 if (utilisation >= MaxUtilisationThreshold)
181 {
182 SafeLog(() => LogUtilisationFiltered(_logger!, utilisation, signal.Strength.ToString(), null));
183 return GetZeroPosition();
184 }
185
186 if (utilisation >= HighUtilisationThreshold &&
187 signal.Strength != STCR004AStrength.Recommended)
188 {
189 SafeLog(() => LogUtilisationFiltered(_logger!, utilisation, signal.Strength.ToString(), null));
190 return GetZeroPosition();
191 }
192
193 // Get base Kelly allocation
194 STRK002A basePosition = _baseKelly.CalculateFromHistory(
195 portfolioValue, historicalTrades, spreadCost, signal);
196
197 if (basePosition.Contracts == 0)
198 {
199 return basePosition;
200 }
201
202 // Apply reserve adjustment
203 double adjustedAllocationPercent = basePosition.AllocationPercent / targetCapacity;
204
205 // Ensure we don't allocate more than the base Kelly suggests
206 adjustedAllocationPercent = Math.Min(adjustedAllocationPercent, basePosition.AllocationPercent);
207
208 // Recalculate contracts based on adjusted allocation
209 double dollarAllocation = portfolioValue * adjustedAllocationPercent;
210 int contracts = (int)Math.Floor(dollarAllocation / (spreadCost * 100.0));
211
212 SafeLog(() => LogAdjustedAllocation(_logger!, signal.Symbol,
213 basePosition.AllocationPercent, adjustedAllocationPercent, null));
214
215 return new STRK002A
216 {
217 Contracts = Math.Max(contracts, 0),
218 AllocationPercent = adjustedAllocationPercent,
219 DollarAllocation = dollarAllocation,
220 MaxLossPerContract = basePosition.MaxLossPerContract,
221 TotalRisk = contracts * basePosition.MaxLossPerContract,
222 ExpectedProfitPerContract = basePosition.ExpectedProfitPerContract,
223 KellyFraction = basePosition.KellyFraction
224 };
225 }
