199 public static int EncodeTermCurve(ReadOnlySpan<(int DaysToExpiry, double IV, long OIWeight)> points, Span<byte> buffer)
200 {
201 int offset = 0;
202
203 // Version
204 buffer[offset++] = FormatVersion;
205
206 // Count
207 BinaryPrimitives.WriteInt32LittleEndian(buffer[offset..], points.Length);
208 offset += 4;
209
210 // Points
211 foreach (var (days, iv, oi) in points)
212 {
213 int written = EncodeTermPoint(days, iv, oi, buffer[offset..]);
214 offset += written;
215 }
216
217 return offset;
218 }
