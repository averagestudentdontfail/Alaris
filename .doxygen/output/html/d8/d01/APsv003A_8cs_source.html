33 public DataVerificationReport Verify(STDT010A requirements, string sessionDataPath)
34 {
35 ArgumentNullException.ThrowIfNull(requirements);
36 ArgumentException.ThrowIfNullOrWhiteSpace(sessionDataPath);
37
38 var report = new DataVerificationReport
39 {
40 SessionDataPath = sessionDataPath,
41 VerifiedAt = DateTime.UtcNow
42 };
43
44 _logger?.LogInformation("Verifying session data at {Path}", sessionDataPath);
45
46 // Check 1: Price data for all symbols
47 VerifyPriceData(requirements, sessionDataPath, report);
48
49 // Check 2: Map and factor files
50 VerifyMapAndFactorFiles(requirements, sessionDataPath, report);
51
52 // Check 3: Earnings calendar coverage
53 VerifyEarningsCalendar(requirements, sessionDataPath, report);
54
55 // Check 4: Options data for signal generation dates
56 VerifyOptionsData(requirements, sessionDataPath, report);
57
58 // Check 5: Benchmark data
59 VerifyBenchmarkData(requirements, sessionDataPath, report);
60
61 // Compute completeness
62 report.IsComplete =
63 report.MissingPriceData.Count == 0 &&
64 report.MissingMapFiles.Count == 0 &&
65 report.MissingFactorFiles.Count == 0 &&
66 report.MissingEarningsDates.Count == 0 &&
67 report.MissingOptionsData.Count == 0 &&
68 report.BenchmarkAvailable;
69
70 _logger?.LogInformation(
71 "Verification complete: {Status} (Price: {Price}, Maps: {Maps}, Earnings: {Earnings}, Options: {Options})",
72 report.IsComplete ? "PASS" : "FAIL",
73 report.MissingPriceData.Count == 0 ? "OK" : $"{report.MissingPriceData.Count} missing",
74 report.MissingMapFiles.Count == 0 ? "OK" : $"{report.MissingMapFiles.Count} missing",
75 report.MissingEarningsDates.Count == 0 ? "OK" : $"{report.MissingEarningsDates.Count} missing",
76 report.MissingOptionsData.Count == 0 ? "OK" : $"{report.MissingOptionsData.Count} missing");
77
78 return report;
79 }

121 private void VerifyEarningsCalendar(STDT010A requirements, string sessionDataPath, DataVerificationReport report)
122 {
123 var nasdaqPath = Path.Combine(sessionDataPath, "earnings", "nasdaq");
124
125 for (var date = requirements.StartDate; date <= requirements.EarningsLookaheadEnd; date = date.AddDays(1))
126 {
127 // Skip weekends
128 if (date.DayOfWeek == DayOfWeek.Saturday || date.DayOfWeek == DayOfWeek.Sunday)
129 continue;
130
131 var cachePath = Path.Combine(nasdaqPath, $"{date:yyyy-MM-dd}.json");
132 if (!File.Exists(cachePath))
133 {
134 report.MissingEarningsDates.Add(date);
135 }
136 }
137
138 if (report.MissingEarningsDates.Count > 0)
139 {
140 _logger?.LogWarning(
141 "Missing {Count} earnings dates (first: {First}, last: {Last})",
142 report.MissingEarningsDates.Count,
143 report.MissingEarningsDates.Min(),
144 report.MissingEarningsDates.Max());
145 }
146 }

148 private void VerifyOptionsData(STDT010A requirements, string sessionDataPath, DataVerificationReport report)
149 {
150 var optionsPath = Path.Combine(sessionDataPath, "options");
151
152 foreach (var date in requirements.OptionsRequiredDates)
153 {
154 foreach (var symbol in requirements.Symbols) // Benchmark doesn't need options
155 {
156 var ticker = symbol.ToLowerInvariant();
157 var optionFile = Path.Combine(optionsPath, $"{ticker}_{date:yyyyMMdd}.json");
158
159 if (!File.Exists(optionFile))
160 {
161 report.MissingOptionsData.Add((symbol, date));
162 }
163 }
164 }
165
166 if (report.MissingOptionsData.Count > 0)
167 {
168 _logger?.LogWarning(
169 "Missing {Count} options data entries",
170 report.MissingOptionsData.Count);
171 }
172 }

192public sealed class DataVerificationReport

208 public string GetSummary()
209 {
210 if (IsComplete)
211 return "All required data is available.";
212
213 var issues = new List<string>();
214
215 if (MissingPriceData.Count > 0)
216 issues.Add($"Missing price data for: {string.Join(", ", MissingPriceData)}");
217
218 if (MissingMapFiles.Count > 0)
219 issues.Add($"Missing map files for: {string.Join(", ", MissingMapFiles)}");
220
221 if (MissingFactorFiles.Count > 0)
222 issues.Add($"Missing factor files for: {string.Join(", ", MissingFactorFiles)}");
223
224 if (MissingEarningsDates.Count > 0)
225 issues.Add($"Missing {MissingEarningsDates.Count} earnings dates " +
226 $"({MissingEarningsDates.Min():yyyy-MM-dd} to {MissingEarningsDates.Max():yyyy-MM-dd})");
227
228 if (MissingOptionsData.Count > 0)
229 {
230 var uniqueDates = MissingOptionsData.Select(x => x.Date).Distinct().Count();
231 var uniqueSymbols = MissingOptionsData.Select(x => x.Symbol).Distinct().Count();
232 issues.Add($"Missing options data: {uniqueSymbols} symbols {uniqueDates} dates = {MissingOptionsData.Count} entries");
233 }
234
235 if (!BenchmarkAvailable)
236 issues.Add("Missing benchmark (SPY) data");
237
238 return string.Join("\n", issues);
239 }
