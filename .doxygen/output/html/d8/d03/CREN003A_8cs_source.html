156 public double Price(double spot, double strike, double timeToExpiry, double riskFreeRate, double dividendYield, double volatility, OptionType optionType)
157 {
158 bool isCall = optionType == OptionType.Call;
159
160 // Near-expiry handling: use intrinsic blending for Ï„ < 3/252 (3 trading days)
161 if (_nearExpiryHandler.IsNearExpiry(timeToExpiry))
162 {
163 // If intrinsic-only regime, return pure intrinsic
164 if (_nearExpiryHandler.IsIntrinsicOnly(timeToExpiry))
165 {
166 return CREX001A.CalculateIntrinsic(spot, strike, isCall);
167 }
168
169 // Blending zone: compute spectral price and blend with intrinsic
170 double spectralPrice = _spectralEngine.Price(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
171 double blended = _nearExpiryHandler.BlendWithIntrinsic(spectralPrice, spot, strike, isCall, timeToExpiry);
172 return EnforceInvariants(blended, spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, isCall);
173 }
174
175 // Use spectral engine as primary
176 double price = _spectralEngine.Price(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
177
178 // Enforce mathematical invariants if enabled
179 return EnforceInvariants(price, spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, isCall);

185 private double EnforceInvariants(double price, double spot, double strike, double tau, double r, double q, double sigma, bool isCall)
186 {
187 if (!_enforceInvariants)
188 {
189 return price;
190 }
191
192 // Invariant 1: American >= Intrinsic
193 double intrinsic = isCall ? System.Math.Max(0, spot - strike) : System.Math.Max(0, strike - spot);
194 price = System.Math.Max(price, intrinsic);
195
196 // Invariant 2: American >= European
197 double european = BlackScholesEuropean(spot, strike, tau, r, q, sigma, isCall);
198 price = System.Math.Max(price, european);
199
200 return price;
201 }

203 private static double BlackScholesEuropean(double spot, double strike, double tau, double r, double q, double sigma, bool isCall)
204 {
205 if (tau <= 0)
206 {
207 return isCall ? System.Math.Max(0, spot - strike) : System.Math.Max(0, strike - spot);
208 }
209
210 double sqrtT = System.Math.Sqrt(tau);
211 double d1 = (System.Math.Log(spot / strike) + ((r - q + (0.5 * sigma * sigma)) * tau)) / (sigma * sqrtT);
212 double d2 = d1 - (sigma * sqrtT);
213
214 double discountS = System.Math.Exp(-q * tau);
215 double discountK = System.Math.Exp(-r * tau);
216
217 if (isCall)
218 {
219 return (spot * discountS * Math.CRMF001A.NormalCDF(d1)) - (strike * discountK * Math.CRMF001A.NormalCDF(d2));
220 }
221
222 return (strike * discountK * Math.CRMF001A.NormalCDF(-d2)) - (spot * discountS * Math.CRMF001A.NormalCDF(-d1));
223 }

302 public UnifiedPricingResult PriceWithDetails(
303 double spot,
304 double strike,
305 double timeToExpiry,
306 double volatility,
307 double riskFreeRate,
308 double dividendYield,
309 OptionType optionType)
310 {
311 // Validate parameters
312 ValidateParameters(spot, strike, timeToExpiry, volatility);
313
314 bool isCall = optionType == OptionType.Call;
315
316 // Classify regime
317 RateRegime regime = CRRE001A.Classify(riskFreeRate, dividendYield, isCall);
318
319 // Use FD engine for pricing (reliable for all regimes)
320 double price = _fdEngine.Price(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
321 double delta = _fdEngine.Delta(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
322 double gamma = _fdEngine.Gamma(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
323 double theta = _fdEngine.Theta(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
324 double vega = _fdEngine.Vega(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
325 double rho = _fdEngine.Rho(spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, optionType);
326
327 // Calculate early exercise premium (American - European)
328 double europeanPrice = Alaris.Core.Math.CRMF001A.BSPrice(
329 spot, strike, timeToExpiry, volatility, riskFreeRate, dividendYield, isCall);
330 double earlyExercisePremium = price - europeanPrice;
331
332 // Calculate boundaries for double-boundary regime using QD+ approximation
333 double? upperBoundary = null;
334 double? lowerBoundary = null;
335 PricingMethod method = PricingMethod.FiniteDifference;
336
337 if (regime == RateRegime.DoubleBoundary)
338 {
339 // Use QD+ to calculate boundaries for the double-boundary regime
340 // Only attempt if parameters are valid (spot, strike, tau, vol all positive)
341 CRAP001A qdPlusEngine = new CRAP001A(
342 spot, strike, timeToExpiry, riskFreeRate, dividendYield, volatility, isCall);
343 (double upper, double lower) = qdPlusEngine.CalculateBoundaries();
344
345 // Only set boundaries if they are finite and valid
346 if (!double.IsInfinity(upper) && !double.IsNaN(upper))
347 {
348 upperBoundary = upper;
349 }
350
351 if (!double.IsInfinity(lower) && !double.IsNaN(lower))
352 {
353 lowerBoundary = lower;
354 }
355
356 method = PricingMethod.Hybrid; // FD price + QD+ boundaries
357 }
358
359 return new UnifiedPricingResult
360 {
361 Price = price,
362 Delta = delta,
363 Gamma = gamma,
364 Theta = theta,
365 Vega = vega,
366 Rho = rho,
367 Regime = regime,
368 Method = method,
369 EarlyExercisePremium = System.Math.Max(0, earlyExercisePremium),
370 UpperBoundary = upperBoundary,
371 LowerBoundary = lowerBoundary
372 };
373 }
