93 CRVT001A.ComputeGammasBatch(spots[..count], strikes, taus, sigmas, r, q, gammas[..count]);
94 CRVT001A.ComputeVegasBatch(spots[..count], strikes, taus, sigmas, r, q, vegas[..count]);
95
96 // Compute theta as time decay (simplified: -∂V/∂τ per day)
97 // θ = -(V(τ) - V(τ-Δτ)) / Δτ approximated
98 const double daysPerYear = 252.0;
99
100 // Assemble results
101 for (int i = 0; i < count; i++)
102 {
103 // Simplified theta: assume theta ≈ -vega * σ / (2 * sqrt(τ)) - other terms
104 // For accurate theta, would need full BS theta calculation
105 double sqrtTau = System.Math.Sqrt(taus[i]);
106 double theta = -(sigmas[i] * vegas[i]) / (2 * sqrtTau * daysPerYear);
107
108 results[i] = new GreeksResult
109 {
110 Price = prices[i],
111 Delta = deltas[i],
112 Gamma = gammas[i],
113 Vega = vegas[i],
114 Theta = theta
115 };
116 }
