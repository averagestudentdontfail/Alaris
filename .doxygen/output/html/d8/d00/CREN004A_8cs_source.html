218 private double PriceSingleBoundary(
219 double spot, double strike, double tau, double r, double q, double sigma, bool isCall)
220 {
221 // Step 1: Compute initial boundary guess B∞ using QD+ approximation
222 double bInfinity = ComputeQdPlusInitialGuess(strike, tau, r, q, sigma, isCall);
223
224 // Step 2: Generate Chebyshev nodes in time domain [0, τ]
225 double[] timeNodes = CRCH001A.ChebyshevNodes(_chebyshevNodes, 0.0, tau);
226
227 // Step 3: Initialize boundary values
228 double[] boundaryValues = new double[_chebyshevNodes];
229 for (int i = 0; i < _chebyshevNodes; i++)
230 {
231 boundaryValues[i] = bInfinity;
232 }
233
234 // Step 4: Fixed-point iteration to refine boundary
235 boundaryValues = FixedPointIteration(
236 boundaryValues, timeNodes, strike, tau, r, q, sigma, isCall);
237
238 // Step 5: Integrate along refined boundary to get option value
239 double europeanValue = BlackScholesEuropean(spot, strike, tau, r, q, sigma, isCall);
240 double earlyExercisePremium = IntegrateEarlyExercisePremium(
241 spot, strike, tau, r, q, sigma, isCall, timeNodes, boundaryValues);
242
243 double americanValue = europeanValue + earlyExercisePremium;
244
245 // Ensure at least intrinsic value
246 double intrinsic = CalculateIntrinsicValue(spot, strike, isCall ? OptionType.Call : OptionType.Put);
247 return System.Math.Max(americanValue, intrinsic);
248 }

250 // ========== Double Boundary Pricing (Negative Rate Regimes) ==========
251
252 private double PriceDoubleBoundary(
253 double spot, double strike, double tau, double r, double q, double sigma, bool isCall)
254 {
255 // For double boundary cases, we may not get valid QD+ boundaries
256 // Fall back to heuristic initial guesses if needed
257
258 double upperInit, lowerInit;
259
260 try
261 {
262 CRAP001A qdPlus = new CRAP001A(spot, strike, tau, r, q, sigma, isCall);
263 (upperInit, lowerInit) = qdPlus.CalculateBoundaries();
264 }
265 catch (InvalidOperationException)
266 {
267 upperInit = double.NaN;
268 lowerInit = double.NaN;
269 }
270 catch (ArgumentException)
271 {
272 upperInit = double.NaN;
273 lowerInit = double.NaN;
274 }
275
276 // Set sensible defaults if QD+ fails
277 if (isCall)
278 {
279 // Call double boundary (0 < r < q): upper and lower both above strike
280 if (!double.IsFinite(upperInit) || upperInit <= strike)
281 {
282 upperInit = strike * 1.5; // Upper boundary well above strike
283 }
284 if (!double.IsFinite(lowerInit) || lowerInit <= strike)
285 {
286 lowerInit = strike * 1.1; // Lower boundary just above strike
287 }
288 // Ensure ordering
289 if (lowerInit >= upperInit)
290 {
291 upperInit = lowerInit * 1.2;
292 }
293 }
294 else
295 {
296 // Put double boundary (q < r < 0): upper and lower both below strike
297 if (!double.IsFinite(upperInit) || upperInit >= strike)
298 {
299 upperInit = strike * 0.9; // Upper boundary just below strike
300 }
301 if (!double.IsFinite(lowerInit) || lowerInit >= upperInit)
302 {
303 lowerInit = strike * 0.5; // Lower boundary well below strike
304 }
305 // Ensure ordering
306 if (lowerInit >= upperInit)
307 {
308 lowerInit = upperInit * 0.8;
309 }
310 }
311
312 // Generate Chebyshev nodes
313 double[] timeNodes = CRCH001A.ChebyshevNodes(_chebyshevNodes, 0.0, tau);
314
315 // Initialize both boundaries
316 double[] upperBoundary = new double[_chebyshevNodes];
317 double[] lowerBoundary = new double[_chebyshevNodes];
318 for (int i = 0; i < _chebyshevNodes; i++)
319 {
320 upperBoundary[i] = upperInit;
321 lowerBoundary[i] = lowerInit;
322 }
323
324 // Apply FP-B' stabilized iteration (Healy approach)
325 // Skip iteration for call double-boundary if not well-understood
326 if (!isCall)
327 {
328 (upperBoundary, lowerBoundary) = DoubleBoundaryFixedPointIteration(
329 upperBoundary, lowerBoundary, timeNodes, strike, tau, r, q, sigma, isCall);
330
331 // Optionally refine with Kim solver for higher precision
332 if (_fixedPointIterations > 2)
333 {
334 try
335 {
336 CRSL002A kimSolver = new CRSL002A(spot, strike, tau, r, q, sigma, isCall, _chebyshevNodes);
337 (double[] refinedUpper, double[] refinedLower, double _) =
338 kimSolver.SolveBoundaries(upperBoundary[^1], lowerBoundary[^1]);
339 upperBoundary = refinedUpper;
340 lowerBoundary = refinedLower;
341 }
342 catch (InvalidOperationException)
343 {
344 // Keep the FP-B' boundaries if Kim fails
345 }
346 }
347 }
348
349 // Compute option value from refined boundaries
350 double europeanValue = BlackScholesEuropean(spot, strike, tau, r, q, sigma, isCall);
351 double eep = IntegrateDoubleBoundaryPremium(
352 spot, strike, tau, r, q, sigma, isCall, timeNodes, upperBoundary, lowerBoundary);
353
354 // Ensure no NaN propagation
355 if (!double.IsFinite(eep))
356 {
357 eep = 0.0;
358 }
359
360 double americanValue = europeanValue + System.Math.Abs(eep);
361 double intrinsic = CalculateIntrinsicValue(spot, strike, isCall ? OptionType.Call : OptionType.Put);
362 return System.Math.Max(americanValue, intrinsic);
363 }

367 private double[] FixedPointIteration(
368 double[] boundary,
369 double[] timeNodes,
370 double strike,
371 double tau,
372 double r,
373 double q,
374 double sigma,
375 bool isCall)
376 {
377 double[] current = (double[])boundary.Clone();
378
379 for (int iter = 0; iter < _fixedPointIterations; iter++)
380 {
381 double[] next = new double[_chebyshevNodes];
382
383 for (int i = 0; i < _chebyshevNodes; i++)
384 {
385 double t = timeNodes[i];
386 double tauRemaining = tau - t;
387
388 if (tauRemaining < NumericalEpsilon)
389 {
390 next[i] = strike;
391 continue;
392 }
393
394 // Compute the fixed-point update
395 next[i] = FixedPointUpdate(current, timeNodes, i, strike, tau, r, q, sigma, isCall);
396 }
397
398 // Check convergence
399 double maxChange = 0.0;
400 for (int i = 0; i < _chebyshevNodes; i++)
401 {
402 maxChange = System.Math.Max(maxChange, System.Math.Abs(next[i] - current[i]));
403 }
404
405 current = next;
406
407 if (maxChange < Tolerance * strike)
408 {
409 break;
410 }
411 }
412
413 return current;
414 }
415
416 private double FixedPointUpdate(
417 double[] boundary,
418 double[] timeNodes,
419 int index,
420 double strike,
421 double tau,
422 double r,
423 double q,
424 double sigma,
425 bool isCall)
426 {
427 double t = timeNodes[index];
428 double tauRemaining = tau - t;
429 double B = boundary[index];
430
431 // Compute D1, D2
432 double d1 = (System.Math.Log(B / strike) + ((r - q + (0.5 * sigma * sigma)) * tauRemaining))
433 / (sigma * System.Math.Sqrt(tauRemaining));
434 double d2 = d1 - (sigma * System.Math.Sqrt(tauRemaining));
435
436 double eta = isCall ? 1.0 : -1.0;
437
438 // Compute the integral term using quadrature
439 double integralN = ComputeIntegralN(boundary, timeNodes, index, strike, r, q, sigma, isCall);
440 double integralD = ComputeIntegralD(boundary, timeNodes, index, strike, r, q, sigma, isCall);
441
442 // FP-A equation: B = K * N(d) / D(d)
443 double Nd2 = CRMF001A.NormalCDF(eta * d2);
444 double Nd1 = CRMF001A.NormalCDF(eta * d1);
445
446 double numerator = 1.0 - (System.Math.Exp(-r * tauRemaining) * Nd2) - integralN;
447 double denominator = 1.0 - (System.Math.Exp(-q * tauRemaining) * Nd1) - integralD;
448
449 if (System.Math.Abs(denominator) < NumericalEpsilon)
450 {
451 return B;
452 }
453
454 double newB = strike * numerator / denominator;
455
456 // Apply constraints
457 if (isCall)
458 {
459 newB = System.Math.Max(newB, strike * 1.001);
460 }
461 else
462 {
463 newB = System.Math.Clamp(newB, strike * 0.01, strike * 0.999);
464 }
465
466 return newB;
467 }

469 private (double[] Upper, double[] Lower) DoubleBoundaryFixedPointIteration(
470 double[] upper,
471 double[] lower,
472 double[] timeNodes,
473 double strike,
474 double tau,
475 double r,
476 double q,
477 double sigma,
478 bool isCall)
479 {
480 double[] currentUpper = (double[])upper.Clone();
481 double[] currentLower = (double[])lower.Clone();
482
483 for (int iter = 0; iter < _fixedPointIterations; iter++)
484 {
485 double[] nextUpper = new double[_chebyshevNodes];
486 double[] nextLower = new double[_chebyshevNodes];
487
488 for (int i = 0; i < _chebyshevNodes; i++)
489 {
490 // FP-B' stabilization: update upper first, then use updated upper for lower
491 nextUpper[i] = FixedPointUpdate(currentUpper, timeNodes, i, strike, tau, r, q, sigma, isCall);
492
493 // Use just-computed upper for lower boundary update
494 double[] tempUpper = (double[])currentUpper.Clone();
495 tempUpper[i] = nextUpper[i];
496 nextLower[i] = FixedPointUpdateLower(currentLower, tempUpper, timeNodes, i, strike, tau, r, q, sigma, isCall);
497
498 // Enforce ordering constraint
499 if (!isCall && nextLower[i] >= nextUpper[i])
500 {
501 double mid = (nextUpper[i] + nextLower[i]) / 2.0;
502 nextUpper[i] = mid + NumericalEpsilon;
503 nextLower[i] = mid - NumericalEpsilon;
504 }
505 }
506
507 currentUpper = nextUpper;
508 currentLower = nextLower;
509 }
510
511 return (currentUpper, currentLower);
512 }

514 private double FixedPointUpdateLower(
515 double[] lower,
516 double[] upper,
517 double[] timeNodes,
518 int index,
519 double strike,
520 double tau,
521 double r,
522 double q,
523 double sigma,
524 bool isCall)
525 {
526 // Simplified lower boundary update using FP-B' modification
527 double t = timeNodes[index];
528 double tauRemaining = tau - t;
529
530 if (tauRemaining < NumericalEpsilon)
531 {
532 return strike * 0.5;
533 }
534
535 double BL = lower[index];
536 double d1 = (System.Math.Log(BL / strike) + ((r - q + (0.5 * sigma * sigma)) * tauRemaining))
537 / (sigma * System.Math.Sqrt(tauRemaining));
538 double d2 = d1 - (sigma * System.Math.Sqrt(tauRemaining));
539
540 double Nd2 = CRMF001A.NormalCDF(-d2);
541 double Nd1 = CRMF001A.NormalCDF(-d1);
542
543 double numerator = 1.0 - (System.Math.Exp(-r * tauRemaining) * Nd2);
544 double denominator = 1.0 - (System.Math.Exp(-q * tauRemaining) * Nd1);
545
546 if (System.Math.Abs(denominator) < NumericalEpsilon)
547 {
548 return BL;
549 }
550
551 double newBL = strike * numerator / denominator;
552 double upperBound = upper[index] - NumericalEpsilon;
553 double lowerBound = strike * 0.01;
554
555 // Ensure min <= max for clamp
556 if (lowerBound > upperBound)
557 {
558 lowerBound = upperBound - NumericalEpsilon;
559 }
560
561 return System.Math.Clamp(newBL, lowerBound, upperBound);
562 }

566 private double ComputeIntegralN(
567 double[] boundary, double[] timeNodes, int index,
568 double strike, double r, double q, double sigma, bool isCall)
569 {
570 double t = timeNodes[index];
571 if (index >= _chebyshevNodes - 1)
572 {
573 return 0.0;
574 }
575
576 double tMax = timeNodes[^1];
577 double B = boundary[index];
578
579 Func<double, double> integrand = s =>
580 {
581 double tau = s - t;
582 if (tau < NumericalEpsilon)
583 {
584 return 0.0;
585 }
586
587 double Bs = InterpolateBoundary(boundary, timeNodes, s);
588 double d2 = (System.Math.Log(B / Bs) + ((r - q - (0.5 * sigma * sigma)) * tau))
589 / (sigma * System.Math.Sqrt(tau));
590
591 double eta = isCall ? 1.0 : -1.0;
592 return r * System.Math.Exp(-r * tau) * CRMF001A.NormalCDF(eta * d2);
593 };
594
595 if (_useTanhSinh)
596 {
597 return CRGQ001A.TanhSinhIntegrate(integrand, t, tMax, Tolerance);
598 }
599
600 return CRGQ001A.IntegrateWithScheme(integrand, _quadratureScheme, t, tMax);
601 }

640 private double IntegrateEarlyExercisePremium(
641 double spot, double strike, double tau, double r, double q, double sigma, bool isCall,
642 double[] timeNodes, double[] boundary)
643 {
644 // The early exercise premium integral (Kim 1990, Andersen-Lake-Offengenden 2016):
645 // For a put: EEP = ∫₀^τ [r*K*e^(-r*s)*N(-d₂(S,B(τ-s),s)) - q*S*e^(-q*s)*N(-d₁(S,B(τ-s),s))] ds
646 // For a call: EEP = ∫₀^τ [q*S*e^(-q*s)*N(d₁(S,B(τ-s),s)) - r*K*e^(-r*s)*N(d₂(S,B(τ-s),s))] ds
647
648 Func<double, double> integrand = s =>
649 {
650 if (s < NumericalEpsilon)
651 {
652 return 0.0;
653 }
654
655 // Time remaining at evaluation point
656 double tRemaining = tau - s;
657 if (tRemaining < NumericalEpsilon)
658 {
659 tRemaining = NumericalEpsilon;
660 }
661
662 // Interpolate boundary at time (tau - s) from start
663 double Bt = InterpolateBoundary(boundary, timeNodes, tRemaining);
664 if (Bt <= 0)
665 {
666 return 0.0;
667 }
668
669 double sqrtS = System.Math.Sqrt(s);
670 double d1 = (System.Math.Log(spot / Bt) + ((r - q + (0.5 * sigma * sigma)) * s))
671 / (sigma * sqrtS);
672 double d2 = d1 - (sigma * sqrtS);
673
674 double discountR = System.Math.Exp(-r * s);
675 double discountQ = System.Math.Exp(-q * s);
676
677 if (isCall)
678 {
679 // Call: dividend income from early exercise
680 double term1 = q * spot * discountQ * CRMF001A.NormalCDF(d1);
681 double term2 = r * strike * discountR * CRMF001A.NormalCDF(d2);
682 return term1 - term2;
683 }
684 else
685 {
686 // Put: interest income from early exercise
687 double term1 = r * strike * discountR * CRMF001A.NormalCDF(-d2);
688 double term2 = q * spot * discountQ * CRMF001A.NormalCDF(-d1);
689 return term1 - term2;
690 }
691 };
692
693 if (_useTanhSinh)
694 {
695 return CRGQ001A.TanhSinhIntegrate(integrand, 0.0, tau, Tolerance);
696 }
697
698 return CRGQ001A.IntegrateWithScheme(integrand, _quadratureScheme, 0.0, tau);
699 }

701 private double IntegrateDoubleBoundaryPremium(
702 double spot, double strike, double tau, double r, double q, double sigma, bool isCall,
703 double[] timeNodes, double[] upper, double[] lower)
704 {
705 // For double boundary, integrate the difference at both boundaries
706 Func<double, double> integrand = t =>
707 {
708 double tauRemaining = tau - t;
709 if (tauRemaining < NumericalEpsilon)
710 {
711 return 0.0;
712 }
713
714 double Bu = InterpolateBoundary(upper, timeNodes, t);
715 double Bl = InterpolateBoundary(lower, timeNodes, t);
716 double sqrtTau = System.Math.Sqrt(tauRemaining);
717
718 // Upper boundary contribution
719 double d2u = (System.Math.Log(spot / Bu) + ((r - q - (0.5 * sigma * sigma)) * tauRemaining))
720 / (sigma * sqrtTau);
721 // Lower boundary contribution
722 double d2l = (System.Math.Log(spot / Bl) + ((r - q - (0.5 * sigma * sigma)) * tauRemaining))
723 / (sigma * sqrtTau);
724
725 double termU = r * strike * System.Math.Exp(-r * tauRemaining) * CRMF001A.NormalCDF(-d2u);
726 double termL = r * strike * System.Math.Exp(-r * tauRemaining) * CRMF001A.NormalCDF(-d2l);
727
728 return termL - termU;
729 };
730
731 if (_useTanhSinh)
732 {
733 return CRGQ001A.TanhSinhIntegrate(integrand, 0.0, tau, Tolerance);
734 }
735
736 return CRGQ001A.IntegrateWithScheme(integrand, _quadratureScheme, 0.0, tau);
737 }

746 private static double ComputeQdPlusInitialGuess(
747 double strike, double tau, double r, double q, double sigma, bool isCall)
748 {
749 // Barone-Adesi Whaley-style initial guess
750 double h = 1.0 - System.Math.Exp(-r * tau);
751 double sigma2 = sigma * sigma;
752
753 double a = (2.0 * r) / sigma2;
754 double b = (2.0 * (r - q)) / sigma2 - 1.0;
755
756 double discriminant = (b * b) + (4.0 * a / h);
757 if (discriminant < 0)
758 {
759 return strike;
760 }
761
762 double sqrtDisc = System.Math.Sqrt(discriminant);
763 double lambda = isCall ? (-b + sqrtDisc) / 2.0 : (-b - sqrtDisc) / 2.0;
764
765 // Early exercise boundary approximation
766 if (isCall)
767 {
768 return strike * lambda / (lambda - 1.0);
769 }
770 else
771 {
772 return strike * lambda / (lambda + 1.0);
773 }
774 }

776 private static double BlackScholesEuropean(
777 double spot, double strike, double tau, double r, double q, double sigma, bool isCall)
778 {
779 double sqrtT = System.Math.Sqrt(tau);
780 double d1 = (System.Math.Log(spot / strike) + ((r - q + (0.5 * sigma * sigma)) * tau)) / (sigma * sqrtT);
781 double d2 = d1 - (sigma * sqrtT);
782
783 double discountS = System.Math.Exp(-q * tau);
784 double discountK = System.Math.Exp(-r * tau);
785
786 if (isCall)
787 {
788 return (spot * discountS * CRMF001A.NormalCDF(d1)) - (strike * discountK * CRMF001A.NormalCDF(d2));
