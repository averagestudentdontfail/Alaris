102 public static STTM004A Create(
103 DateTime valuationDate,
104 DateTime expirationDate,
105 DateTime? earningsDate,
106 TimeConstraints constraints)
107 {
108 ArgumentNullException.ThrowIfNull(constraints);
109
110 // Basic validation
111 if (expirationDate <= valuationDate)
112 {
113 throw new ArgumentException(
114 $"Expiration date ({expirationDate:yyyy-MM-dd}) must be after valuation date ({valuationDate:yyyy-MM-dd}).",
115 nameof(expirationDate));
116 }
117
118 // Calculate time values
119 int dte = CalculateTradingDays(valuationDate, expirationDate);
120 double timeToExpiry = dte / TradingDaysPerYear;
121
122 // Validate time constraints
123 if (timeToExpiry < constraints.MinTimeToExpiry)
124 {
125 throw new ArgumentException(
126 $"Time to expiry ({timeToExpiry:F6} years) is below minimum ({constraints.MinTimeToExpiry:F6} years).",
127 nameof(expirationDate));
128 }
129
130 if (timeToExpiry > constraints.MaxTimeToExpiry)
131 {
132 throw new ArgumentException(
133 $"Time to expiry ({timeToExpiry:F2} years) exceeds maximum ({constraints.MaxTimeToExpiry:F2} years).",
134 nameof(expirationDate));
135 }
136
137 // Process earnings date
138 int? daysToEarnings = null;
139 double? timeToEarnings = null;
140
141 if (earningsDate.HasValue)
142 {
143 if (earningsDate.Value <= valuationDate)
144 {
145 // Earnings already passed - post-earnings regime
146 daysToEarnings = -CalculateTradingDays(earningsDate.Value, valuationDate);
147 timeToEarnings = daysToEarnings.Value / TradingDaysPerYear;
148 }
149 else if (earningsDate.Value < expirationDate)
150 {
151 // Earnings before expiry - pre-earnings regime
152 daysToEarnings = CalculateTradingDays(valuationDate, earningsDate.Value);
153 timeToEarnings = daysToEarnings.Value / TradingDaysPerYear;
154 }
155 // else: earnings after expiry, treat as no earnings event
156 }
157
158 return new STTM004A
159 {
160 ValuationDate = valuationDate,
161 ExpirationDate = expirationDate,
162 EarningsDate = earningsDate,
163 TimeToExpiry = timeToExpiry,
164 TimeToEarnings = timeToEarnings,
165 DaysToExpiry = dte,
166 DaysToEarnings = daysToEarnings
167 };
168 }

254 public ValidationResult ValidatePreEarnings(STTM004A timeParams)
255 {
256 ArgumentNullException.ThrowIfNull(timeParams);
257
258 var errors = new List<string>();
259
260 if (!timeParams.IsPreEarnings)
261 {
262 errors.Add("Not in pre-earnings regime.");
263 }
264 else if (timeParams.DaysToEarnings.HasValue)
265 {
266 if (timeParams.DaysToEarnings.Value < MinDaysBeforeEarnings)
267 {
268 errors.Add($"Days to earnings ({timeParams.DaysToEarnings.Value}) below minimum ({MinDaysBeforeEarnings}).");
269 }
270
271 if (timeParams.DaysToEarnings.Value > MaxDaysBeforeEarnings)
272 {
273 errors.Add($"Days to earnings ({timeParams.DaysToEarnings.Value}) exceeds maximum ({MaxDaysBeforeEarnings}).");
274 }
275 }
276
277 if (timeParams.TimeToExpiry < MinTimeToExpiry)
278 {
279 errors.Add($"Time to expiry ({timeParams.TimeToExpiry:F4}) below minimum ({MinTimeToExpiry:F4}).");
280 }
281
282 if (timeParams.TimeToExpiry > MaxTimeToExpiry)
283 {
284 errors.Add($"Time to expiry ({timeParams.TimeToExpiry:F2}) exceeds maximum ({MaxTimeToExpiry:F2}).");
285 }
286
287 return new ValidationResult(errors.Count == 0, errors);
288 }
