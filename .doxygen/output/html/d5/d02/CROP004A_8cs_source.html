82 public DividendSchedule(IEnumerable<CashDividend> dividends)
83 {
84 ArgumentNullException.ThrowIfNull(dividends);
85 List<CashDividend> filtered = new List<CashDividend>();
86
87 foreach (CashDividend dividend in dividends)
88 {
89 if (dividend.IsValid)
90 {
91 filtered.Add(dividend);
92 }
93 }
94
95 if (filtered.Count == 0)
96 {
97 _dividends = ImmutableArray<CashDividend>.Empty;
98 return;
99 }
100
101 filtered.Sort(static (left, right) => left.ExDividendDate.CompareTo(right.ExDividendDate));
102 _dividends = filtered.ToImmutableArray();
103 }

154 public double ComputePresentValue(DateTime valuationDate, DateTime expiryDate, double riskFreeRate)
155 {
156 double totalPV = 0.0;
157 if (_dividends.IsDefaultOrEmpty)
158 {
159 return totalPV;
160 }
161
162 for (int i = 0; i < _dividends.Length; i++)
163 {
164 CashDividend dividend = _dividends[i];
165 if (dividend.ExDividendDate <= valuationDate || dividend.ExDividendDate > expiryDate)
166 {
167 continue;
168 }
169
170 double timeToDiv = (dividend.ExDividendDate - valuationDate).TotalDays / 365.0;
171 totalPV += dividend.Amount * System.Math.Exp(-riskFreeRate * timeToDiv);
172 }
173
174 return totalPV;
175 }

185 public double AdjustSpotForDividends(double spot, DateTime valuationDate, DateTime expiryDate, double riskFreeRate)

253 public static DividendSchedule FromContinuousYield(
254 double spot,
255 double continuousYield,
256 DateTime startDate,
257 DateTime endDate,
258 int frequency = 4)
259 {
260 if (continuousYield <= 0)
261 {
262 return Empty;
263 }
264
265 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(spot, nameof(spot));
266 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(frequency, nameof(frequency));
267
268 if (endDate < startDate)
269 {
270 throw new ArgumentOutOfRangeException(nameof(endDate), "End date must not be earlier than start date.");
271 }
272
273 List<CashDividend> dividends = new List<CashDividend>();
274 TimeSpan period = TimeSpan.FromDays(365.0 / frequency);
275 double dividendAmount = spot * (System.Math.Exp(continuousYield / frequency) - 1);
276
277 DateTime currentDate = startDate.Add(period);
278 while (currentDate <= endDate)
279 {
280 dividends.Add(new CashDividend(currentDate, dividendAmount));
281 currentDate = currentDate.Add(period);
282 }
283
284 return new DividendSchedule(dividends);
285 }
