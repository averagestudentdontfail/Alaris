81 public MaturityEntryResult EvaluateEntry(string symbol, double timeToExpiry)

128 public MaturityExitResult EvaluateExit(string symbol, double timeToExpiry)
129 {
130 if (timeToExpiry <= _forceExitMaturity)
131 {
132 SafeLog(LogLevel.Information,
133 "Force exit triggered for {Symbol}: T={T:F4} at/below threshold {Threshold:F4}",
134 symbol, timeToExpiry, _forceExitMaturity);
135
136 return new MaturityExitResult
137 {
138 RequiresExit = true,
139 TimeToExpiry = timeToExpiry,
140 Reason = $"Maturity {timeToExpiry:F4}y at/below exit threshold",
141 UrgencyLevel = timeToExpiry <= NearExpiryThreshold
142 ? ExitUrgency.Immediate
143 : ExitUrgency.Normal
144 };
145 }
146
147 if (timeToExpiry <= _minEntryMaturity)
148 {
149 // In warning zone - no new entries but don't force exit yet
150 SafeLog(LogLevel.Debug,
151 "Position {Symbol} in warning zone: T={T:F4}",
152 symbol, timeToExpiry);
153
154 return new MaturityExitResult
155 {
156 RequiresExit = false,
157 TimeToExpiry = timeToExpiry,
158 Reason = "In warning zone - monitor closely",
159 UrgencyLevel = ExitUrgency.None
160 };
161 }
162
163 return new MaturityExitResult
164 {
165 RequiresExit = false,
166 TimeToExpiry = timeToExpiry,
167 Reason = "Maturity within normal range",
168 UrgencyLevel = ExitUrgency.None
169 };
170 }

179 public double CalculateTimeToExpiry(DateTime expiryDate, DateTime currentDate)
180 {
181 // Use trading calendar if available for accurate business day count
182 if (_calendar is not null)
183 {
184 return _calendar.GetTimeToExpiryInYears(currentDate, expiryDate);
185 }
186
187 // Fallback: approximate using 5/7 calendar-to-trading day ratio
188 TimeSpan span = expiryDate - currentDate;
189 double calendarDays = span.TotalDays;
190 double tradingDays = calendarDays * (5.0 / 7.0);
191 return Math.Max(0, tradingDays / 252.0);
192 }
