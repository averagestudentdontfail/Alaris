53 public ValidationResult Validate()
54 {
55 List<string> errors = new();
56
57 if (V0 <= 0)
58 {
59 errors.Add("V0 (initial variance) must be positive.");
60 }
61
62 if (Theta <= 0)
63 {
64 errors.Add("Theta (long-run variance) must be positive.");
65 }
66
67 if (Kappa <= 0)
68 {
69 errors.Add("Kappa (mean reversion speed) must be positive.");
70 }
71
72 if (SigmaV <= 0)
73 {
74 errors.Add("SigmaV (vol-of-vol) must be positive.");
75 }
76
77 if (Rho < -1 || Rho > 1)
78 {
79 errors.Add("Rho (correlation) must be in [-1, 1].");
80 }
81
82 // Check Feller condition for variance positivity
83 if (!SatisfiesFellerCondition())
84 {
85 errors.Add($"Feller condition violated: 2*kappa*theta ({(2 * Kappa * Theta):F4}) must be > sigma_v^2 ({(SigmaV * SigmaV):F4}).");
86 }
87
88 return new ValidationResult(errors.Count == 0, errors);
89 }

94 public bool SatisfiesFellerCondition() =>
95 (2 * Kappa * Theta) > (SigmaV * SigmaV);

101 public double ExpectedVariance(double t) =>
102 Theta + ((V0 - Theta) * Math.Exp(-Kappa * t));

107 public double VarianceOfVariance(double t)
108 {
109 double expKt = Math.Exp(-Kappa * t);
110 return (V0 * SigmaV * SigmaV / Kappa * expKt * (1 - expKt)) +
111 (Theta * SigmaV * SigmaV / (2 * Kappa) * (1 - expKt) * (1 - expKt));
112 }

201 public double ComputeTheoreticalIVApproximate(double spot, double strike, double timeToExpiry)
202 {
203 ValidateInputs(spot, strike, timeToExpiry);
204
205 // Use moment-matching approximation for efficiency
206 double logMoneyness = Math.Log(strike / spot);
207
208 // Expected total variance
209 double expectedVariance = ComputeExpectedIntegratedVariance(timeToExpiry);
210 double baseIV = Math.Sqrt(expectedVariance / timeToExpiry);
211
212 // Skew adjustment from correlation
213 double skewAdjustment = ComputeSkewAdjustment(logMoneyness, timeToExpiry);
214
215 // Smile curvature from vol-of-vol
216 double smileAdjustment = ComputeSmileAdjustment(logMoneyness, timeToExpiry);
217
218 double iv = baseIV + skewAdjustment + smileAdjustment;
219 return Math.Max(iv, 0.001);
220 }
221
225 private double ComputeExpectedIntegratedVariance(double t)
226 {
227 // E[int_0^T V_s ds] = theta*T + (V0 - theta)/kappa * (1 - exp(-kappa*T))
228 double expKt = Math.Exp(-_params.Kappa * t);
229 return (_params.Theta * t) +
230 ((_params.V0 - _params.Theta) / _params.Kappa * (1 - expKt));
231 }
232
236 private double ComputeSkewAdjustment(double logMoneyness, double timeToExpiry)
237 {
238 // First-order approximation of skew from rho
239 double sqrtV = Math.Sqrt(_params.V0);
240 return _params.Rho * _params.SigmaV * logMoneyness / (2 * sqrtV * timeToExpiry);
241 }
242
246 private double ComputeSmileAdjustment(double logMoneyness, double timeToExpiry)
247 {
248 // Second-order approximation (smile curvature)
249 double k2 = logMoneyness * logMoneyness;
250 double sigmaV2 = _params.SigmaV * _params.SigmaV;
251 return sigmaV2 * k2 / (24 * _params.V0 * timeToExpiry);
252 }

261 public Complex CharacteristicFunction(Complex u, double t)
262 {
263 Complex i = Complex.ImaginaryOne;
264
265 double kappa = _params.Kappa;
266 double theta = _params.Theta;
267 double sigmaV = _params.SigmaV;
268 double rho = _params.Rho;
269 double v0 = _params.V0;
270 double r = _params.RiskFreeRate;
271 double d = _params.DividendYield;
272
273 // Heston (1993) parameters
274 Complex xi = kappa - (rho * sigmaV * i * u);
275 Complex d_h = Complex.Sqrt(
276 (xi * xi) + (sigmaV * sigmaV * ((i * u) + (u * u))));
277
278 Complex g = (xi - d_h) / (xi + d_h);
279 Complex exp_dt = Complex.Exp(-d_h * t);
280
281 Complex C = ((r - d) * i * u * t) +
282 (kappa * theta / (sigmaV * sigmaV) *
283 (((xi - d_h) * t) - (2 * Complex.Log((1 - (g * exp_dt)) / (1 - g)))));
284
285 Complex D = (xi - d_h) / (sigmaV * sigmaV) *
286 ((1 - exp_dt) / (1 - (g * exp_dt)));
287
288 return Complex.Exp(C + (D * v0));
289 }

345 public static HestonParameters Calibrate(
346 double spot,
347 IReadOnlyList<(double Strike, int DTE, double MarketIV)> marketData,
348 double riskFreeRate,
349 double dividendYield)
350 {
351 ArgumentNullException.ThrowIfNull(marketData);
352
353 if (marketData.Count < 5)
354 {
355 throw new ArgumentException("Need at least 5 market observations for calibration.", nameof(marketData));
356 }
357
358 // Use Levenberg-Marquardt for production-grade calibration
359 var optimizer = new Numerical.STPR004A
360 {
361 MaxIterations = 200,
362 ParameterTolerance = 1e-8,
363 ObjectiveTolerance = 1e-8
364 };
365
366 // Parameter vector: [v0, theta, kappa, sigmaV, rho]
367 // Initial guess from ATM market data or defaults
368 double atmIV = marketData
369 .Where(x => Math.Abs(x.Strike - spot) / spot < 0.05)
370 .Select(x => x.MarketIV)
371 .DefaultIfEmpty(0.25)
372 .Average();
373
374 double[] initialGuess =
375 {
376 atmIV * atmIV, // v0
377 atmIV * atmIV, // theta
378 2.0, // kappa
379 0.3, // sigmaV
380 -0.7 // rho
381 };
382
383 // Parameter bounds
384 double[] lowerBounds = { 0.001, 0.001, 0.01, 0.01, -0.99 };
385 double[] upperBounds = { 2.0, 2.0, 10.0, 2.0, 0.99 };
386
387 // Residual function
388 double[] Residuals(double[] x)
389 {
390 HestonParameters candidateParams = new()
391 {
392 V0 = x[0],
393 Theta = x[1],
394 Kappa = x[2],
395 SigmaV = x[3],
396 Rho = x[4],
397 RiskFreeRate = riskFreeRate,
398 DividendYield = dividendYield
399 };
400
401 // Validate Feller condition and basic constraints
402 if (!candidateParams.Validate().IsValid)
403 {
404 // Return large residuals for invalid parameters
405 return Enumerable.Repeat(100.0, marketData.Count).ToArray();
406 }
407
408 STIV001A model = new(candidateParams);
409 double[] residuals = new double[marketData.Count];
410
411 for (int i = 0; i < marketData.Count; i++)
412 {
413 (double strike, int dte, double marketIV) = marketData[i];
414 double timeToExpiry = TradingCalendarDefaults.DteToYears(dte);
415
416 try
417 {
418 double modelIV = model.ComputeTheoreticalIV(spot, strike, timeToExpiry);
419 residuals[i] = modelIV - marketIV;
420 }
421#pragma warning disable CA1031 // Do not catch general exception types
422 catch (Exception)
423#pragma warning restore CA1031
424 {
425 // Pricing can fail for many reasons during calibration (e.g., invalid parameters,
426 // numerical instability, convergence issues). Penalize rather than failing the entire calibration.
427 residuals[i] = 10.0;
428 }
429 }
430
431 return residuals;
432 }
433
434 // Run optimization
435 OptimizationResult result = optimizer.Minimize(Residuals, initialGuess, lowerBounds, upperBounds);
436
437 if (result.Converged)
438 {
439 return new HestonParameters
440 {
441 V0 = result.OptimalParameters[0],
442 Theta = result.OptimalParameters[1],
443 Kappa = result.OptimalParameters[2],
444 SigmaV = result.OptimalParameters[3],
445 Rho = result.OptimalParameters[4],
446 RiskFreeRate = riskFreeRate,
447 DividendYield = dividendYield
448 };
449 }
450
451 // Fallback: Try grid search with coarse grid to find better initial guess
452 return CalibrateGridSearch(spot, marketData, riskFreeRate, dividendYield);
453 }

460 private static HestonParameters CalibrateGridSearch(
461 double spot,
462 IReadOnlyList<(double Strike, int DTE, double MarketIV)> marketData,
463 double riskFreeRate,
464 double dividendYield)
465 {
466 double bestError = double.MaxValue;
467 HestonParameters? bestParams = null;
468 object lockObj = new();
469
470 // Parameter grid (coarse grid for demo)
471 double[] v0s = { 0.02, 0.04, 0.06, 0.09 };
472 double[] thetas = { 0.02, 0.04, 0.06 };
473 double[] kappas = { 1.0, 2.0, 3.0, 5.0 };
474 double[] sigmaVs = { 0.2, 0.3, 0.4, 0.5 };
475 double[] rhos = { -0.9, -0.7, -0.5, -0.3 };
476
477 // Flatten parameter combinations for parallel processing
478 var combinations = from v0 in v0s
479 from theta in thetas
480 from kappa in kappas
481 from sigmaV in sigmaVs
482 from rho in rhos
483 select (v0, theta, kappa, sigmaV, rho);
484
485 // Parallel grid search over all parameter combinations
486 Parallel.ForEach(combinations, combination =>
487 {
488 var (v0, theta, kappa, sigmaV, rho) = combination;
489 HestonParameters candidateParams = new()
490 {
491 V0 = v0,
492 Theta = theta,
493 Kappa = kappa,
494 SigmaV = sigmaV,
495 Rho = rho,
496 RiskFreeRate = riskFreeRate,
497 DividendYield = dividendYield
498 };
499
500 if (!candidateParams.Validate().IsValid)
501 {
502 return;
503 }
504
505 STIV001A model = new(candidateParams);
506 double error = ComputeCalibrationError(model, spot, marketData);
507
508 lock (lockObj)
509 {
510 if (error < bestError)
511 {
512 bestError = error;
513 bestParams = candidateParams;
514 }
515 }
516 });
517
518 return bestParams ?? HestonParameters.DefaultEquity;
519 }

521 private static double ComputeCalibrationError(
522 STIV001A model,
523 double spot,
524 IReadOnlyList<(double Strike, int DTE, double MarketIV)> marketData)
525 {
526 double totalError = 0;
527
528 foreach ((double strike, int dte, double marketIV) in marketData)
529 {
530 double timeToExpiry = TradingCalendarDefaults.DteToYears(dte);
531 double modelIV = model.ComputeTheoreticalIV(spot, strike, timeToExpiry);
532 double error = modelIV - marketIV;
533 totalError += error * error;
534 }
535
536 return totalError / marketData.Count;
