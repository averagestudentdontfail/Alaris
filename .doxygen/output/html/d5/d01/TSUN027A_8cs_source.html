1// TSUN027A.cs - Risk Component Unit Tests
2// Component ID: TSUN027A
3//
4// Tests for Risk Management Components:
5// - STRK001A: Kelly Criterion position sizing
6// - STRK002A: Position size validation
7// - STMG001A: Maturity guard for entry/exit
8//
9// Mathematical Invariants Tested:
10// 1. Kelly Formula: f* = (p*b - q) / b where p + q = 1
11// 2. Net-of-Cost Kelly: f*_net ≤ f* (costs reduce optimal bet)
12// 3. Fractional Kelly: Applied fraction ∈ (0, 1]
13// 4. Position Bounds: 0 ≤ allocation ≤ MaxAllocation
14// 5. Maturity Ordering: MinEntry > ForceExit thresholds
15//
16// References:
17// - Kelly, J.L. (1956) "A New Interpretation of Information Rate"
18// - Thorp, E.O. (2008) "The Kelly Criterion in Blackjack, Sports Betting, and the Stock Market"

36 private static List<Trade> CreateWinningTradeHistory(int count, double avgWin)
37 {
38 var trades = new List<Trade>();
39 for (int i = 0; i < count; i++)
40 {
41 trades.Add(new Trade
42 {
43 EntryDate = DateTime.Today.AddDays(-i - 1),
44 ExitDate = DateTime.Today.AddDays(-i),
45 ProfitLoss = avgWin * (0.8 + 0.4 * (i % 3) / 2.0), // Vary around avgWin
46 Symbol = "TEST",
47 Strategy = "CalendarSpread"
48 });
49 }
50 return trades;
51 }

53 private static List<Trade> CreateMixedTradeHistory(int wins, int losses, double avgWin, double avgLoss)
54 {
55 var trades = new List<Trade>();
56
57 for (int i = 0; i < wins; i++)
58 {
59 trades.Add(new Trade
60 {
61 EntryDate = DateTime.Today.AddDays(-i - losses - 1),
62 ExitDate = DateTime.Today.AddDays(-i - losses),
63 ProfitLoss = avgWin,
64 Symbol = "TEST",
65 Strategy = "CalendarSpread"
66 });
67 }
68
69 for (int i = 0; i < losses; i++)
70 {
71 trades.Add(new Trade
72 {
73 EntryDate = DateTime.Today.AddDays(-i - 1),
74 ExitDate = DateTime.Today.AddDays(-i),
75 ProfitLoss = -avgLoss,
76 Symbol = "TEST",
77 Strategy = "CalendarSpread"
78 });
79 }
80
81 return trades;
82 }

156 [Fact]
157 public void NetOfCostKelly_ZeroCosts_MatchesStandardFormula()
158 {
159 // Arrange
160 double p = 0.6; // Win probability
161 double q = 0.4; // Loss probability
162 double avgWin = 200;
163 double avgLoss = 100;
164 double b = avgWin / avgLoss; // Win/loss ratio = 2
165
166 // Expected: f* = (0.6 * 2 - 0.4) / 2 = (1.2 - 0.4) / 2 = 0.4
167 double expectedKelly = (p * b - q) / b;
168
169 // Act
170 double actualKelly = STRK001A.CalculateNetOfCostKelly(p, avgWin, avgLoss, 0);
171
172 // Assert
173 actualKelly.Should().BeApproximately(expectedKelly, 1e-10,
174 "Zero-cost Kelly should match standard formula");
175 }
