60 public async Task<TradingOpportunity> EvaluateOpportunity(
61 string symbol,
62 DateTime earningsDate,
63 DateTime evaluationDate,
64 double portfolioValue,
65 IReadOnlyList<Trade> historicalTrades)
66 {
67 ArgumentException.ThrowIfNullOrWhiteSpace(symbol);
68
69 if (_logger != null)
70 {
71 LogEvaluatingOpportunity(_logger, symbol, earningsDate, null);
72 }
73
74 TradingOpportunity opportunity = new TradingOpportunity
75 {
76 Symbol = symbol,
77 EarningsDate = earningsDate,
78 EvaluationDate = evaluationDate
79 };
80
81 try
82 {
83 // Generate signal
84 STCR004A signal = _signalGenerator.Generate(symbol, earningsDate, evaluationDate);
85 opportunity.STCR004A = signal;
86
87 if (signal.Strength == STCR004AStrength.Avoid)
88 {
89 SafeLog(() => LogSTCR004AAvoid(_logger!, symbol, null));
90 return opportunity;
91 }
92
93 // Price the calendar spread
94 STPR001AParameters spreadParams = CreateSpreadParameters(signal, evaluationDate);
95 STPR001APricing spreadPricing = await _pricingEngine.PriceSTPR001A(spreadParams).ConfigureAwait(false);
96 opportunity.SpreadPricing = spreadPricing;
97
98 // Calculate position size
99 STRK002A positionSize = _positionSizer.CalculateFromHistory(
100 portfolioValue,
101 historicalTrades,
102 spreadPricing.SpreadCost,
103 signal);
104 opportunity.STRK002A = positionSize;
105
106 SafeLog(() => LogOpportunityEvaluated(_logger!, symbol, signal.Strength, positionSize.Contracts, spreadPricing.SpreadCost, null));
107 }
108 catch (ArgumentException ex)
109 {
110 SafeLog(() => LogErrorEvaluatingOpportunity(_logger!, symbol, ex));
111 throw;
112 }
113 catch (InvalidOperationException ex)
114 {
115 SafeLog(() => LogErrorEvaluatingOpportunity(_logger!, symbol, ex));
116 throw;
117 }
118
119 return opportunity;
120 }

125 private STPR001AParameters CreateSpreadParameters(STCR004A signal, DateTime evaluationDate)
126 {
127 // Determine appropriate expiration dates
128 DateTime frontExpiry = FindFrontMonthExpiry(signal.EarningsDate, evaluationDate);
129 DateTime backExpiry = FindBackMonthExpiry(frontExpiry);
130
131 return new STPR001AParameters
132 {
133 UnderlyingPrice = signal.ExpectedMove, // This should be actual price from market data
134 Strike = signal.ExpectedMove, // This should be ATM strike
135 FrontExpiry = Alaris.Core.Time.CRTM005A.FromDateTime(frontExpiry),
136 BackExpiry = Alaris.Core.Time.CRTM005A.FromDateTime(backExpiry),
137 ImpliedVolatility = signal.ImpliedVolatility30,
138 RiskFreeRate = 0.05, // Should come from market data
139 DividendYield = 0.0, // Should come from market data
140 OptionType = Alaris.Core.Options.OptionType.Call,
141 ValuationDate = Alaris.Core.Time.CRTM005A.FromDateTime(evaluationDate)
142 };

145 private DateTime FindFrontMonthExpiry(DateTime earningsDate, DateTime evaluationDate)
146 {
147 // Find the Friday after earnings (standard monthly expiration)
148 int daysAfterEarnings = 0;
149 while (true)
150 {
151 DateTime candidate = earningsDate.AddDays(daysAfterEarnings);
152 if (candidate.DayOfWeek == DayOfWeek.Friday && candidate > earningsDate)
153 {
154 return candidate;
155 }
156
157 daysAfterEarnings++;
158 if (daysAfterEarnings > 14) // Safety limit
159 {
160 return earningsDate.AddDays(7);
161 }
162 }
163 }

165 private DateTime FindBackMonthExpiry(DateTime frontExpiry)
166 {
167 // Typically 28-35 days after front month
168 return frontExpiry.AddDays(30);
