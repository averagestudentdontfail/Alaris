616 private static int ExtractSequence(string sessionId)
617 {
618 // Format: BT001A-YYYYMMDD-YYYYMMDD
619 if (sessionId.Length >= 5 && sessionId.StartsWith("BT", StringComparison.Ordinal))
620 {
621 var sequenceStr = sessionId.Substring(2, 3);
622 if (int.TryParse(sequenceStr, out var sequence))
623 {
624 return sequence;
625 }
626 }
627 return 0;
628 }
