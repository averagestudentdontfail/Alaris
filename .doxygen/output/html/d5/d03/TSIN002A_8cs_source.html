363 private static List<PriceBar> GenerateSamplePriceBars(int count)
364 {
365 var random = new Random(42);
366 var bars = new List<PriceBar>();
367 var basePrice = 150.0;
368
369 for (int i = 0; i < count; i++)
370 {
371 var open = basePrice + random.NextDouble() * 5 - 2.5;
372 var close = open + random.NextDouble() * 3 - 1.5;
373 var high = System.Math.Max(open, close) + random.NextDouble() * 2;
374 var low = System.Math.Min(open, close) - random.NextDouble() * 2;
375
376 bars.Add(new PriceBar
377 {
378 Date = DateTime.Today.AddDays(-count + i),
379 Open = open,
380 High = high,
381 Low = low,
382 Close = close,
383 Volume = random.Next(1_000_000, 5_000_000)
384 });
385
386 basePrice = close;
387 }
388
389 return bars;
390 }

392 private static List<Trade> GenerateSampleTrades(int count)
393 {
394 var random = new Random(42);
395 var trades = new List<Trade>();
396
397 for (int i = 0; i < count; i++)
398 {
399 var profitLoss = random.NextDouble() > 0.55 ? // 55% win rate
400 random.NextDouble() * 500 + 100 : // Win
401 -(random.NextDouble() * 300 + 50); // Loss
402
403 trades.Add(new Trade
404 {
405 EntryDate = DateTime.Today.AddDays(-60 + i * 2),
406 ExitDate = DateTime.Today.AddDays(-60 + i * 2 + 7),
407 ProfitLoss = profitLoss,
408 Symbol = "TEST"
409 });
410 }
411
412 return trades;
413 }
