173 private (RehedgeAction Action, string Rationale) DetermineAction(
174 double delta,
175 double gamma,
176 double moneynessDeviation,
177 int daysToEarnings)
178 {
179 double absDelta = Math.Abs(delta);
180
181 // Priority 1: Delta exceeds threshold - must re-centre
182 if (absDelta > _deltaRehedgeThreshold)
183 {
184 return (
185 RehedgeAction.RecenterRequired,
186 $"Delta ({delta:F4}) exceeds threshold ({_deltaRehedgeThreshold:F2}). " +
187 "Close position and re-establish at current ATM strike."
188 );
189 }
190
191 // Priority 2: Close to earnings with elevated gamma and moneyness deviation
192 if (daysToEarnings <= 2 && gamma < _gammaWarningThreshold && moneynessDeviation > _moneynessAlertThreshold)
193 {
194 return (
195 RehedgeAction.ExitPosition,
196 $"Elevated gap risk with {daysToEarnings} days to earnings. " +
197 $"Gamma ({gamma:F4}) and moneyness deviation ({moneynessDeviation:P1}) indicate exit."
198 );
199 }
200
201 // Priority 3: Elevated gamma with significant moneyness deviation
202 if (gamma < _gammaWarningThreshold && moneynessDeviation > _moneynessAlertThreshold)
203 {
204 return (
205 RehedgeAction.MonitorClosely,
206 $"Elevated gamma risk ({gamma:F4}) with underlying {moneynessDeviation * 100:F1}% " +
207 "from strike. Consider reducing position size."
208 );
209 }
210
211 // Priority 4: Approaching delta threshold
212 if (absDelta > _deltaRehedgeThreshold * 0.7)
213 {
214 return (
215 RehedgeAction.MonitorClosely,
216 $"Delta ({delta:F4}) approaching threshold ({_deltaRehedgeThreshold:F2}). " +
217 "Monitor for further drift."
218 );
219 }
220
221 // All clear
222 return (
223 RehedgeAction.Hold,
224 "Position within acceptable parameters."
225 );
226 }
