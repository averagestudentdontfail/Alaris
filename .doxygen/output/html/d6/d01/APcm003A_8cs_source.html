72 private static int DownloadData(DataSettings settings)
73 {
74 var tickers = settings.Tickers ?? settings.Ticker;
75 if (tickers is null)

78 return 1;

81 // Validate dates

83 var toDate = settings.ToDate ?? DateTime.Now.ToString("yyyyMMdd");

103 table.AddColumn("[white]Value[/]");

105 table.AddRow("Type", settings.DataType);
106 table.AddRow("Resolution", settings.Resolution);

109 table.AddRow("To", endDate.ToString("yyyy-MM-dd"));

138 totalBars += bars;

140 // Rate limiting: wait 12 seconds between requests for free Polygon tier (5 calls/min)

153 private static int DownloadTickerData(string ticker, DateTime startDate, DateTime endDate, string resolution, string apiKey, string dataPath)
154 {
155 return AnsiConsole.Status()
156 .Spinner(Spinner.Known.Dots)
157 .Start($"[yellow]Downloading {Markup.Escape(ticker)}...[/]", ctx =>
158 {
159 try
160 {
161 using var httpClient = new HttpClient { Timeout = TimeSpan.FromSeconds(60) };
162
163 // Map resolution to Polygon timespan
164 var (multiplier, timespan) = resolution.ToLowerInvariant() switch
165 {
166 "minute" => (1, "minute"),
167 "hour" => (1, "hour"),
168 "daily" => (1, "day"),
169 _ => (1, "day")
170 };
171
172 // Polygon aggregates endpoint
173 var url = $"https://api.polygon.io/v2/aggs/ticker/{ticker}/range/{multiplier}/{timespan}/{startDate:yyyy-MM-dd}/{endDate:yyyy-MM-dd}?adjusted=true&sort=asc&limit=50000&apiKey={apiKey}";
174
175 var response = httpClient.GetFromJsonAsync<PolygonAggregatesResponse>(url).GetAwaiter().GetResult();
176
177 if (response?.Results == null || response.Results.Length == 0)
178 {
179 AnsiConsole.MarkupLine($" [yellow]{Markup.Escape(ticker)}: No data returned[/]");
180 return 0;
181 }
182
183 // Create output directory
184 var securityType = "equity";
185 var market = "usa";
186 var leanResolution = resolution.ToLowerInvariant() switch
187 {
188 "minute" => "minute",
189 "hour" => "hour",
190 "daily" => "daily",
191 _ => "daily"
192 };
193
194 var outputDir = System.IO.Path.Combine(dataPath, securityType, market, leanResolution, ticker.ToLowerInvariant());
195 System.IO.Directory.CreateDirectory(outputDir);
196
197 // Write data in LEAN format
198 var barCount = 0;
199 if (leanResolution == "daily")
200 {
201 // Daily format: single CSV file
202 var filePath = System.IO.Path.Combine(outputDir, $"{ticker.ToLowerInvariant()}.csv");
203 using var writer = new System.IO.StreamWriter(filePath);
204
205 foreach (var bar in response.Results)
206 {
207 var date = DateTimeOffset.FromUnixTimeMilliseconds(bar.Timestamp).UtcDateTime;
208 // LEAN daily format: Date,Open,High,Low,Close,Volume (all prices * 10000)
209 var line = $"{date:yyyyMMdd 00:00},{bar.Open * 10000m:F0},{bar.High * 10000m:F0},{bar.Low * 10000m:F0},{bar.Close * 10000m:F0},{bar.Volume}";
210 writer.WriteLine(line);
211 barCount++;
212 }
213 }
214 else
215 {
216 // Minute/Hour format: files by date
217 var barsByDate = response.Results.GroupBy(b =>
218 DateTimeOffset.FromUnixTimeMilliseconds(b.Timestamp).UtcDateTime.Date);
219
220 foreach (var dateGroup in barsByDate)
221 {
222 var fileName = $"{dateGroup.Key:yyyyMMdd}_{ticker.ToLowerInvariant()}_trade.csv";
223 var filePath = System.IO.Path.Combine(outputDir, fileName);
224
225 using var writer = new System.IO.StreamWriter(filePath);
226
227 foreach (var bar in dateGroup.OrderBy(b => b.Timestamp))
228 {
229 var time = DateTimeOffset.FromUnixTimeMilliseconds(bar.Timestamp).UtcDateTime;
230 // LEAN minute format: Milliseconds,Open,High,Low,Close,Volume
231 var ms = (long)(time - dateGroup.Key).TotalMilliseconds;
232 var line = $"{ms},{bar.Open * 10000m:F0},{bar.High * 10000m:F0},{bar.Low * 10000m:F0},{bar.Close * 10000m:F0},{bar.Volume}";
233 writer.WriteLine(line);
234 barCount++;
235 }
236 }
237 }
238
239 AnsiConsole.MarkupLine($" [green]{Markup.Escape(ticker)}:[/] {barCount:N0} bars saved");
240 return barCount;
241 }
242 catch (Exception ex)
243 {
244 AnsiConsole.MarkupLine($" [red]{Markup.Escape(ticker)}: Error - {Markup.Escape(ex.Message)}[/]");
245 return 0;
246 }
247 });
248 }

357 private static string? GetPolygonApiKey()
358 {
359 // Try to find and read appsettings.local.jsonc
360 var paths = new[]
361 {
362 "appsettings.local.jsonc",
363 "../appsettings.local.jsonc",
364 "../../appsettings.local.jsonc"
365 };
366
367 foreach (var path in paths)
368 {
369 if (System.IO.File.Exists(path))
370 {
371 try
372 {
373 var json = System.IO.File.ReadAllText(path);
374 // Simple extraction - look for "ApiKey" in Polygon section
375 var lines = json.Split('\n');
376 var inPolygon = false;
377 foreach (var line in lines)
378 {
379 var trimmed = line.Trim();
380 if (trimmed.StartsWith("//")) continue;
381
382 if (trimmed.Contains("\"Polygon\""))
383 {
384 inPolygon = true;
385 }
386 else if (inPolygon && trimmed.Contains("\"ApiKey\""))
387 {
388 var colonIdx = trimmed.IndexOf(':');
389 if (colonIdx > 0)
390 {
391 var value = trimmed[(colonIdx + 1)..].Trim().Trim(',', '"', ' ');
392 if (!string.IsNullOrEmpty(value))
393 return value;
394 }
395 }
396 else if (inPolygon && trimmed.StartsWith("}"))
397 {
398 inPolygon = false;
399 }
400 }
401 }
402 catch
403 {
404 // Ignore parsing errors
405 }
406 }
407 }
408
409 return null;
410 }
