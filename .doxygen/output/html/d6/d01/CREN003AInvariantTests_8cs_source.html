181 private static double BlackScholesEuropean(double spot, double strike, double tau, double r, double q, double sigma, bool isCall)
182 {
183 if (tau <= 0)
184 {
185 return isCall ? System.Math.Max(0, spot - strike) : System.Math.Max(0, strike - spot);
186 }
187
188 double sqrtT = System.Math.Sqrt(tau);
189 double d1 = (System.Math.Log(spot / strike) + ((r - q + (0.5 * sigma * sigma)) * tau)) / (sigma * sqrtT);
190 double d2 = d1 - (sigma * sqrtT);
191
192 double discountS = System.Math.Exp(-q * tau);
193 double discountK = System.Math.Exp(-r * tau);
194
195 if (isCall)
196 {
197 return (spot * discountS * NormalCDF(d1)) - (strike * discountK * NormalCDF(d2));
198 }
199
200 return (strike * discountK * NormalCDF(-d2)) - (spot * discountS * NormalCDF(-d1));
201 }
202
203 private static double NormalCDF(double x)
204 {
205 return 0.5 * (1.0 + Erf(x / System.Math.Sqrt(2.0)));
206 }
207
208 private static double Erf(double x)
209 {
210 double a1 = 0.254829592;
211 double a2 = -0.284496736;
212 double a3 = 1.421413741;
213 double a4 = -1.453152027;
214 double a5 = 1.061405429;
215 double p = 0.3275911;
216
217 int sign = x < 0 ? -1 : 1;
218 x = System.Math.Abs(x);
219 double t = 1.0 / (1.0 + (p * x));
220 double y = 1.0 - ((((((a5 * t) + a4) * t) + a3) * t + a2) * t + a1) * t * System.Math.Exp(-x * x);
221 return sign * y;
222 }
