110 {

114
115 double calendarResult = _calendar.GetTimeToExpiryInYears(start, end);
116
117 // Approximate: 7 calendar days * 5/7 = 5 trading days
118 // Actual: Dec 23 (Mon), Dec 24 (Tue), Dec 25 (holiday), Dec 26 (Thu), Dec 27 (Fri) = 4 days
119 // Plus Dec 28-29 weekend excluded, Dec 30 is end (excluded) = 4 business days
120
121 calendarResult.Should().BeLessThan(5.0 / 252.0,
122 "calendar should account for Christmas holiday");
123 calendarResult.Should().BeGreaterThan(0);
