30 public static double ComputePrice(
31 double spot,
32 double strike,
33 double timeToExpiry,
34 HestonParameters @params,
35 bool isCall = true)
36 {
37 ArgumentNullException.ThrowIfNull(@params);
38
39 if (spot <= 0 || strike <= 0 || timeToExpiry <= 0)
40 {
41 throw new ArgumentException("Spot, strike, and time to expiry must be positive.");
42 }
43
44 // Compute using Heston semi-analytical formula
45 // Call price = S*exp(-qT)*P1 - K*exp(-rT)*P2
46 // where P1 and P2 are probabilities computed via characteristic function
47
48 double discountFactor = Math.Exp(-@params.RiskFreeRate * timeToExpiry);
49 double forwardFactor = Math.Exp(-@params.DividendYield * timeToExpiry);
50
51 double p1 = ComputeProbability(spot, strike, timeToExpiry, @params, 1);
52 double p2 = ComputeProbability(spot, strike, timeToExpiry, @params, 2);
53
54 double callPrice = (spot * forwardFactor * p1) - (strike * discountFactor * p2);
55
56 if (isCall)
57 {
58 return Math.Max(callPrice, 0);
59 }
60 else
61 {
62 // Put-call parity: P = C - S*exp(-dT) + K*exp(-rT)
63 double putPrice = callPrice - (spot * forwardFactor) + (strike * discountFactor);
64 return Math.Max(putPrice, 0);
65 }
66 }

105 private static double ComputeProbability(
106 double spot,
107 double strike,
108 double timeToExpiry,
109 HestonParameters @params,
110 int j)
111 {
112 double logMoneyness = Math.Log(spot / strike);
113
114 // Integrate characteristic function
115 // P_j = 0.5 + (1/pi) * integral from 0 to inf of Re[(exp(i*phi*log(S/K)) * f_j(phi)) / (i*phi)]
116 // where f_j is the characteristic function of the log-return
117
118 double Integrand(double phi)
119 {
120 // UPDATED: Removed the "if (phi < 1e-10) return 0;" check to ensure smoothness
121 // for high-order quadrature. The integration lower bound handles the singularity.
122
123 Complex iPhi = new Complex(0, phi);
124 Complex charFunc = CharacteristicFunction(phi, timeToExpiry, @params, j);
125 Complex exponent = Complex.Exp(iPhi * logMoneyness);
126
127 return (exponent * charFunc / iPhi).Real;
128 }
129
130 // Numerical integration from slightly above 0 to infinity using adaptive quadrature
131 // Uses STPR002C unified facade for automatic AVX2 dispatch
132 (double integralValue, double _) = STPR002C.IntegrateToInfinity(
133 Integrand,
134 1e-8,
135 absoluteTolerance: 1e-8,
136 relativeTolerance: 1e-6);
137
138 double probability = 0.5 + (integralValue / Math.PI);
139
140 return Math.Clamp(probability, 0, 1);
141 }

148 private static Complex CharacteristicFunction(
149 double phi,
150 double timeToExpiry,
151 HestonParameters @params,
152 int j)
153 {
154 Complex i = Complex.ImaginaryOne;
155
156 double kappa = @params.Kappa;
157 double theta = @params.Theta;
158 double sigmaV = @params.SigmaV;
159 double rho = @params.Rho;
160 double v0 = @params.V0;
161 double r = @params.RiskFreeRate;
162 double q = @params.DividendYield;
163 double sigmaSq = sigmaV * sigmaV;
164
165 // Parameters u_j and b_j depend on which probability we're computing
166 // For P1 (stock measure): u = 0.5, b = kappa - rho*sigmaV
167 // For P2 (risk-neutral): u = -0.5, b = kappa
168 double u = j == 1 ? 0.5 : -0.5;
169 double b = j == 1 ? kappa - (rho * sigmaV) : kappa;
170
171 // a = b - rho*sigma*i*phi
172 Complex a = b - (rho * sigmaV * i * phi);
173
174 // d² = a² + sigma²*(phi² - 2*u*i*phi)
175 Complex dSquared = (a * a) + (sigmaSq * ((phi * phi) - (2 * u * i * phi)));
176 Complex d = Complex.Sqrt(dSquared);
177
178 // Branch cut handling: ensure Re(d) > 0 for numerical stability
179 if (d.Real < 0)
180 {
181 d = -d;
182 }
183
184 // g = (a - d) / (a + d)
185 Complex g = (a - d) / (a + d);
186
187 Complex expMinusDT = Complex.Exp(-d * timeToExpiry);
188
189 // C term: drift contribution + variance mean-reversion contribution
190 // C = (r-q)*i*phi*T + (kappa*theta/sigma²) * [(a-d)*T - 2*ln((1-g*exp(-dT))/(1-g))]
191 Complex logTerm = Complex.Log((1 - (g * expMinusDT)) / (1 - g));
192 Complex C = ((r - q) * i * phi * timeToExpiry) +
193 (kappa * theta / sigmaSq *
194 (((a - d) * timeToExpiry) - (2 * logTerm)));
195
196 // D term: initial variance contribution
197 // D = (a-d)/sigma² * (1 - exp(-dT)) / (1 - g*exp(-dT))
198 Complex D = (a - d) / sigmaSq *
199 ((1 - expMinusDT) / (1 - (g * expMinusDT)));
200
201 // Characteristic function of log-return ln(S_T/S_0)
202 return Complex.Exp(C + (D * v0));
203 }

208 private static double BlackScholesPrice(
209 double spot,
210 double strike,
211 double timeToExpiry,
212 double riskFreeRate,
213 double dividendYield,
214 double volatility,
215 bool isCall)
216 {
217 double sqrtT = Math.Sqrt(timeToExpiry);
218 double d1 = (Math.Log(spot / strike) +
219 ((riskFreeRate - dividendYield + (0.5 * volatility * volatility)) * timeToExpiry)) /
220 (volatility * sqrtT);
221 double d2 = d1 - (volatility * sqrtT);
222
223 double forward = spot * Math.Exp(-dividendYield * timeToExpiry);
224 double discountFactor = Math.Exp(-riskFreeRate * timeToExpiry);
225
226 if (isCall)
227 {
228 return (forward * NormalCDF(d1)) - (strike * discountFactor * NormalCDF(d2));
229 }
230 else
231 {
232 return (strike * discountFactor * NormalCDF(-d2)) - (forward * NormalCDF(-d1));
233 }
234 }
