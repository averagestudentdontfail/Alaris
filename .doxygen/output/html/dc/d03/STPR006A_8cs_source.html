31 public static double ComputePrice(
32 double spot,
33 double strike,
34 double timeToExpiry,
35 KouParameters @params,
36 bool isCall = true)
37 {
38 ArgumentNullException.ThrowIfNull(@params);
39
40 if (spot <= 0 || strike <= 0 || timeToExpiry <= 0)
41 {
42 throw new ArgumentException("Spot, strike, and time to expiry must be positive.");
43 }
44
45 // Compute using characteristic function approach
46 double discountFactor = Math.Exp(-@params.RiskFreeRate * timeToExpiry);
47 double forwardFactor = Math.Exp(-@params.DividendYield * timeToExpiry);
48
49 // Use Carr-Madan formula: C(K) = exp(-alpha*k) / pi * integral...
50 // where k = log(K), alpha is damping parameter
51 double logStrike = Math.Log(strike);
52 double alpha = 1.5; // Damping parameter for call options
53
54 double Integrand(double v)
55 {
56 if (v < 1e-10)
57 {
58 return 0;
59 }
60
61 Complex iV = new Complex(0, v);
62 Complex charFunc = CharacteristicFunction(v - ((alpha + 1) * Complex.ImaginaryOne), spot, timeToExpiry, @params);
63
64 // Carr-Madan denominator: α² + α - v² + i*v*(2α + 1)
65 Complex denominator = (alpha * alpha) + alpha - (v * v) + (iV * ((2 * alpha) + 1));
66
67 return (Complex.Exp(-iV * logStrike) * charFunc / denominator).Real;
68 }
69
70 // Numerical integration using STPR002C unified facade for automatic AVX2 dispatch
71 (double integralValue, double _) = STPR002C.IntegrateToInfinity(
72 Integrand,
73 0,
74 absoluteTolerance: 1e-6,
75 relativeTolerance: 1e-4);
76
77 double callPrice = discountFactor * Math.Exp(-alpha * logStrike) * integralValue / Math.PI;
78
79 if (isCall)
80 {
81 return Math.Max(callPrice, 0);
82 }
83 else
84 {
85 // Put-call parity
86 double putPrice = callPrice - (spot * forwardFactor) + (strike * discountFactor);
87 return Math.Max(putPrice, 0);
88 }
89 }

124 private static Complex CharacteristicFunction(
125 Complex u,
126 double spot,
127 double timeToExpiry,
128 KouParameters @params)
129 {
130 Complex i = Complex.ImaginaryOne;
131
132 double sigma = @params.Sigma;
133 double lambda = @params.Lambda;
134 double p = @params.P;
135 double eta1 = @params.Eta1;
136 double eta2 = @params.Eta2;
137 double r = @params.RiskFreeRate;
138 double d = @params.DividendYield;
139
140 // Martingale correction: kappa = E[e^Y - 1] = E[e^Y] - 1
141 // For double exponential: E[e^Y] = p*eta1/(eta1-1) + (1-p)*eta2/(eta2+1)
142 double kappa = (p * eta1 / (eta1 - 1)) + ((1 - p) * eta2 / (eta2 + 1)) - 1.0;
143
144 // Characteristic exponent with martingale drift correction
145 // psi(u) = (r - d - 0.5*sigma^2 - lambda*kappa)*i*u - 0.5*sigma^2*u^2 + lambda*jump_transform
146 Complex diffusionTerm = ((r - d - (lambda * kappa)) * i * u) - (0.5 * sigma * sigma * u * (u + i));
147
148 // Jump transform: E[exp(i*u*Y)] where Y = log(V)
149 // For double exponential: integral of exp(i*u*y) * f_Y(y) dy
150 Complex jumpTransform = Complex.Zero;
151 if (lambda > 0)
152 {
153 // Upward jump contribution
154 Complex upwardTerm = p * eta1 / (eta1 - (i * u));
155
156 // Downward jump contribution
157 Complex downwardTerm = (1 - p) * eta2 / (eta2 + (i * u));
158
159 jumpTransform = lambda * (upwardTerm + downwardTerm - 1);
160 }
161
162 Complex exponent = (diffusionTerm + jumpTransform) * timeToExpiry;
163
164 return Complex.Exp(exponent + (i * u * Math.Log(spot)));
165 }

170 private static double BlackScholesPrice(
171 double spot,
172 double strike,
173 double timeToExpiry,
174 double riskFreeRate,
175 double dividendYield,
176 double volatility,
177 bool isCall)
178 {
179 double d1 = (Math.Log(spot / strike) +
180 ((riskFreeRate - dividendYield + (0.5 * volatility * volatility)) * timeToExpiry)) /
181 (volatility * Math.Sqrt(timeToExpiry));
182 double d2 = d1 - (volatility * Math.Sqrt(timeToExpiry));
183
184 double forward = spot * Math.Exp(-dividendYield * timeToExpiry);
185 double discountFactor = Math.Exp(-riskFreeRate * timeToExpiry);
186
187 if (isCall)
188 {
189 return (forward * NormalCDF(d1)) - (strike * discountFactor * NormalCDF(d2));
190 }
191 else
192 {
193 return (strike * discountFactor * NormalCDF(-d2)) - (forward * NormalCDF(-d1));
194 }
195 }
