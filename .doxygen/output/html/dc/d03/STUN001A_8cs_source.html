121 public override IEnumerable<Symbol> Select(QCAlgorithm algorithm, IEnumerable<Fundamental> fundamental)
122 {
123 var currentDate = algorithm.Time.Date;
124
125 // Step 1: Apply fundamental filters (volume, price)
126 var filtered = fundamental
127 .Where(f => f.HasFundamentalData)
128 .Select(f => new
129 {
130 Fundamental = f,
131 DollarVolume = (decimal)f.DollarVolume,
132 Price = (decimal)f.Price
133 })
134 .Where(f => f.DollarVolume >= _minimumDollarVolume)
135 .Where(f => f.Price >= _minimumPrice)
136 .OrderByDescending(f => f.DollarVolume)
137 .Take(_maxCoarseSymbols)
138 .Select(f => f.Fundamental)
139 .ToList();
140
141 // Step 2: Query earnings calendar (with caching)
142 var earningsSymbols = GetEarningsSymbolsForWindow(
143 algorithm,
144 currentDate.AddDays(_daysBeforeEarningsMin),
145 currentDate.AddDays(_daysBeforeEarningsMax));
146
147 // Step 3: Filter to symbols with upcoming earnings
148 var selected = filtered
149 .Where(f => earningsSymbols.Contains(f.Symbol.Value))
150 .Take(_maxSymbols)
151 .Select(f => f.Symbol)
152 .ToList();
153
154 if (_logger != null)
155 {
156 LogUniverseSelection(_logger, filtered.Count, earningsSymbols.Count, selected.Count, null);
157 }
158
159 return selected;
160 }

224 private HashSet<string> GetEarningsSymbolsForWindow(
225 QCAlgorithm algorithm,
226 DateTime startDate,
227 DateTime endDate)
228 {
229 // Cache earnings query for same day
230 if (_lastEarningsQueryDate == algorithm.Time.Date)
231 {
232 return _cachedEarningsSymbols;
233 }
234
235 try
236 {
237 // Query earnings provider (synchronous for LEAN compatibility)
238 var symbols = _earningsProvider.GetSymbolsWithEarningsAsync(
239 startDate,
240 endDate,
241 CancellationToken.None)
242 .GetAwaiter().GetResult();
243
244 _cachedEarningsSymbols = new HashSet<string>(
245 symbols,
246 StringComparer.OrdinalIgnoreCase);
247 _lastEarningsQueryDate = algorithm.Time.Date;
248
249 if (_logger != null)
250 {
251 LogEarningsQuery(_logger, startDate, endDate, _cachedEarningsSymbols.Count, null);
252 }
253
254 return _cachedEarningsSymbols;
255 }
256 catch (Exception ex)
257 {
258 algorithm.Error($"Failed to query earnings calendar: {ex.Message}");
259 return _cachedEarningsSymbols; // Return cached if query fails
260 }
