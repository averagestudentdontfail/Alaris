320 private static OptionChain GenerateOptionChainWithEarningsIV(
321 DateTime evaluationDate,
322 double spotPrice,
323 DateTime earningsDate)
324 {
325 var expiries = new List<OptionExpiry>();
326 var random = new Random(42);
327
328 int daysToEarnings = (earningsDate - evaluationDate).Days;
329
330 // Generate 3 monthly expiries with inverted term structure
331 for (int monthOffset = 1; monthOffset <= 3; monthOffset++)
332 {
333 DateTime expiryDate = evaluationDate.AddDays(monthOffset * 30);
334 int dte = monthOffset * 30;
335 var calls = new List<OptionContract>();
336 var puts = new List<OptionContract>();
337
338 // Base IV decreases with time for inverted structure
339 // Front month (30 DTE): ~28% IV (elevated due to earnings)
340 // Back month (90 DTE): ~22% IV (normal baseline)
341 double baseIV;
342 if (monthOffset == 1 && dte >= daysToEarnings)
343 {
344 // Front month containing earnings: add premium
345 baseIV = 0.20 + EarningsIVPremium;
346 }
347 else
348 {
349 // Normal IV that increases slightly with time (base term structure)
350 baseIV = 0.20 + (monthOffset * 0.01);
351 }
352
353 // Ensure inverted structure: front month must be higher
354 if (monthOffset == 1)
355 {
356 baseIV = 0.28; // Front month: 28%
357 }
358 else if (monthOffset == 2)
359 {
360 baseIV = 0.24; // Mid month: 24%
361 }
362 else
363 {
364 baseIV = 0.22; // Back month: 22%
365 }
366
367 // Generate strikes around ATM
368 for (int strikeOffset = -2; strikeOffset <= 2; strikeOffset++)
369 {
370 double strike = Math.Round(spotPrice + (strikeOffset * 5), 2);
371
372 // IV skew: OTM puts have higher IV
373 double skew = strikeOffset * -0.005;
374 double iv = baseIV + skew + ((random.NextDouble() * 0.01) - 0.005);
375
376 // Generate call
377 double callTheo = SimulateOptionPrice(
378 spotPrice, strike, dte, 0.05, 0.0, iv, isCall: true).Price;
379 OptionContract call = new OptionContract
380 {
381 Strike = strike,
382 ImpliedVolatility = iv,
383 Bid = callTheo - 0.10,
384 Ask = callTheo + 0.10,
385 LastPrice = callTheo,
386 Volume = random.Next(1000, 8000),
387 OpenInterest = random.Next(2000, 15000)
388 };
389 calls.Add(call);
390
391 // Generate put
392 double putTheo = SimulateOptionPrice(
393 spotPrice, strike, dte, 0.05, 0.0, iv, isCall: false).Price;
394 OptionContract put = new OptionContract
395 {
396 Strike = strike,
397 ImpliedVolatility = iv,
398 Bid = putTheo - 0.10,
399 Ask = putTheo + 0.10,
400 LastPrice = putTheo,
401 Volume = random.Next(1000, 8000),
402 OpenInterest = random.Next(2000, 15000)
403 };
404 puts.Add(put);
405 }
406
407 OptionExpiry expiry = new OptionExpiry
408 {
409 ExpiryDate = expiryDate
410 };
411
412 foreach (OptionContract c in calls)
413 {
414 expiry.Calls.Add(c);
415 }
416
417 foreach (OptionContract p in puts)
418 {
419 expiry.Puts.Add(p);
420 }
421
422 expiries.Add(expiry);
423 }
424
425 OptionChain optionChain = new OptionChain
426 {
427 UnderlyingPrice = spotPrice
428 };
429
430 foreach (OptionExpiry exp in expiries)
431 {
432 optionChain.Expiries.Add(exp);
433 }
434
435 return optionChain;
436 }

451 private static LeungSantoliMetrics CalculateLeungSantoliMetrics(
452 SimulatedMarketData marketData,
453 DateTime earningsDate,
454 DateTime evaluationDate,
455 double realisedVolatility,
456 TermStructureResult termResult)
457 {
458 // Count historical earnings samples with valid price gaps
459 int historicalSamples = 0;
460 double sumSquaredReturns = 0.0;
461
462 var earningsDateSet = new HashSet<DateTime>(
463 marketData.HistoricalEarningsDates.Select(d => d.Date));
464
465 for (int i = 1; i < marketData.PriceHistory.Count; i++)
466 {
467 PriceBar current = marketData.PriceHistory[i];
468 PriceBar previous = marketData.PriceHistory[i - 1];
469
470 // Check if this is a post-earnings bar
471 if (earningsDateSet.Contains(previous.Date.Date))
472 {
473 // Calculate overnight gap return
474 double gapReturn = Math.Log(current.Open / previous.Close);
475 sumSquaredReturns += gapReturn * gapReturn;
476 historicalSamples++;
477 }
478 }
479
480 // Calibrate jump volatility from historical earnings moves
481 double jumpVolatility;
482 bool isCalibrated = historicalSamples >= 4;
483
484 if (isCalibrated)
485 {
486 // Annualised jump volatility from quarterly earnings
487 jumpVolatility = Math.Sqrt(sumSquaredReturns / historicalSamples);
488 }
489 else
490 {
491 // Default: use empirical average (15-25% for typical equities)
492 jumpVolatility = 0.18;
493 }
494
495 // Time to earnings in years
496 int daysToEarnings = (earningsDate - evaluationDate).Days;
497 double timeToEarnings = TradingCalendarDefaults.DteToYears(daysToEarnings);
498
499 // Leung-Santoli theoretical IV: I(t) = sqrt(sigma^2 + sigmaE^2 / (T-t))
500 double theoreticalIV = Math.Sqrt(
501 (realisedVolatility * realisedVolatility) +
502 (jumpVolatility * jumpVolatility / timeToEarnings));
503
504 // Mispricing signal: market IV - theoretical IV
505 double mispricingSignal = termResult.IV30 - theoreticalIV;
506
507 // Expected IV crush after earnings
508 double postEarningsIV = realisedVolatility; // Returns to base volatility
509 double expectedIVCrush = termResult.IV30 - postEarningsIV;
510 double ivCrushRatio = expectedIVCrush / termResult.IV30;
511
512 return new LeungSantoliMetrics
513 {
514 HistoricalSamples = historicalSamples,
515 IsCalibrated = isCalibrated,
516 JumpVolatility = jumpVolatility,
517 BaseVolatility = realisedVolatility,
518 TheoreticalIV = theoreticalIV,
519 MarketIV = termResult.IV30,
520 MispricingSignal = mispricingSignal,
521 ExpectedIVCrush = expectedIVCrush,
522 IVCrushRatio = ivCrushRatio
523 };
524 }

594 BackPrice = backPrice,
595 SpreadCost = spreadCost,
596 IsCredit = isCredit,
597 SpreadDelta = spreadDelta,
598 SpreadGamma = spreadGamma,
599 SpreadTheta = spreadTheta,
600 SpreadVega = spreadVega,

603 BreakEven = atmStrike,
604 RiskFreeRate = riskFreeRate,
605 DividendYield = dividendYield
606 };

625 private static (double Price, double Delta, double Gamma, double Theta, double Vega) PriceWithDoubleValidated(
626 double spot,
627 double strike,
628 int dte,
629 double rate,
630 double div,
631 double vol,
632 bool isCall)
633 {
634 double timeToExpiry = TradingCalendarDefaults.DteToYears(dte);
635
636 // For calls in negative rate regimes (q < r < 0), use put-call parity
637 // because Healy (2021) double boundary is designed for puts
638 double price;
639 if (isCall && div < rate && rate < 0)
640 {
641 // Price the corresponding put
642 double putPrice = PricingEngine.Price(spot, strike, timeToExpiry, rate, div, vol, Alaris.Core.Options.OptionType.Put);
643
644 // Apply put-call parity: C = P + S·exp(-qT) - K·exp(-rT)
645 double forwardFactor = Math.Exp(-div * timeToExpiry);
646 double discountFactor = Math.Exp(-rate * timeToExpiry);
647 price = putPrice + (spot * forwardFactor) - (strike * discountFactor);
648 }
649 else
650 {
651 // Use CRAP001A directly for puts or other regimes
652 price = PricingEngine.Price(spot, strike, timeToExpiry, rate, div, vol, isCall ? Alaris.Core.Options.OptionType.Call : Alaris.Core.Options.OptionType.Put);
653 }
654
655 // Adaptive step sizes proportional to spot and volatility
656 double ds = spot * 0.005; // 0.5% of spot price
657 double dv = 0.001; // 0.1% volatility bump
658 double dt = 1.0 / TradingCalendarDefaults.TradingDaysPerYear;
659
660 // Helper function to price with put-call parity if needed
661 double PriceOption(double s, double k, double t, double r, double q, double v, bool call)
662 {
663 return PricingEngine.Price(s, k, t, r, q, v, call ? Alaris.Core.Options.OptionType.Call : Alaris.Core.Options.OptionType.Put);
664 }
665
666 // Delta: central difference with adaptive step
667 double priceUp = PriceOption(spot + ds, strike, timeToExpiry, rate, div, vol, isCall);
668 double priceDown = PriceOption(spot - ds, strike, timeToExpiry, rate, div, vol, isCall);
669 double delta = (priceUp - priceDown) / (2.0 * ds);
670
671 // Gamma: central difference second derivative
672 double gamma = (priceUp - (2.0 * price) + priceDown) / (ds * ds);
673
674 // Vega: forward difference with vol bump
675 double priceVegaUp = PriceOption(spot, strike, timeToExpiry, rate, div, vol + dv, isCall);
676 double vega = (priceVegaUp - price) / dv * 0.01;
677
678 // Theta: forward difference for time decay
679 double theta = 0.0;
680 if (timeToExpiry > dt)
681 {
682 double priceTheta = PriceOption(spot, strike, timeToExpiry - dt, rate, div, vol, isCall);
683 theta = (priceTheta - price) / dt / TradingCalendarDefaults.TradingDaysPerYear;
684 }
685
686 // Validate Greeks against physical constraints
687 bool greeksValid = ValidateGreeks(delta, gamma, vega, isCall);
688
689 if (!greeksValid)
690 {
691 // Fall back to Black-Scholes analytical Greeks
692 (delta, gamma, theta, vega) = CalculateBlackScholesGreeks(
693 spot, strike, timeToExpiry, rate, div, vol, isCall);
694 }
695
696 return (price, delta, gamma, theta, vega);
697 }

710 private static bool ValidateGreeks(double delta, double gamma, double vega, bool isCall)
711 {
712 // Check delta bounds
713 if (Math.Abs(delta) > MaxAbsDelta)
714 {
715 return false;
716 }
717
718 // Check gamma bounds
719 if (Math.Abs(gamma) > MaxAbsGamma)
720 {
721 return false;
722 }
723
724 // Check vega bounds
725 if (Math.Abs(vega) > MaxAbsVega)
726 {
727 return false;
728 }
729
730 // Check for NaN or infinity
731 if (double.IsNaN(delta) || double.IsInfinity(delta) ||
732 double.IsNaN(gamma) || double.IsInfinity(gamma) ||
733 double.IsNaN(vega) || double.IsInfinity(vega))
734 {
735 return false;
736 }
737
738 return true;
739 }

744 private static (double Delta, double Gamma, double Theta, double Vega) CalculateBlackScholesGreeks(
745 double spot,
746 double strike,
747 double timeToExpiry,
748 double rate,
749 double div,
750 double vol,
751 bool isCall)
752 {
753 double sqrtT = Math.Sqrt(timeToExpiry);
754 double d1 = (Math.Log(spot / strike) + ((rate - div + (0.5 * vol * vol)) * timeToExpiry)) / (vol * sqrtT);
755 double d2 = d1 - (vol * sqrtT);
756
757 double nd1 = NormalCdf(d1);
758 double nd2 = NormalCdf(d2);
759 double npd1 = NormalPdf(d1);
760
761 double expQT = Math.Exp(-div * timeToExpiry);
762 double expRT = Math.Exp(-rate * timeToExpiry);
763
764 double delta;
765 double theta;
766
767 if (isCall)
768 {
769 delta = expQT * nd1;
770 theta = ((-spot * npd1 * vol * expQT / (2.0 * sqrtT)) -
771 (rate * strike * expRT * nd2) +
772 (div * spot * expQT * nd1)) / TradingCalendarDefaults.TradingDaysPerYear;
773 }
774 else
775 {
776 delta = expQT * (nd1 - 1.0);
777 theta = ((-spot * npd1 * vol * expQT / (2.0 * sqrtT)) +
778 (rate * strike * expRT * (1.0 - nd2)) -
779 (div * spot * expQT * (1.0 - nd1))) / TradingCalendarDefaults.TradingDaysPerYear;
780 }
781
782 double gamma = expQT * npd1 / (spot * vol * sqrtT);
783 double vega = spot * sqrtT * npd1 * expQT * 0.01; // Per 1% vol change
784
785 return (delta, gamma, theta, vega);
786 }

865 private static (double Price, double Delta, double Gamma, double Theta, double Vega) SimulateOptionPrice(
866 double spot,
867 double strike,
868 int dte,
869 double rate,
870 double div,
871 double vol,
872 bool isCall)
873 {
874 double t = TradingCalendarDefaults.DteToYears(dte);
875 double sqrtT = Math.Sqrt(t);
876
877 double d1 = (Math.Log(spot / strike) + ((rate - div + (0.5 * vol * vol)) * t)) / (vol * sqrtT);
878 double d2 = d1 - (vol * sqrtT);
879
880 double nd1 = NormalCdf(d1);
881 double nd2 = NormalCdf(d2);
882 double npd1 = NormalPdf(d1);
883
884 double price;
885 double delta;
886 double theta;
887
888 if (isCall)
889 {
890 price = (spot * Math.Exp(-div * t) * nd1) - (strike * Math.Exp(-rate * t) * nd2);
891 delta = Math.Exp(-div * t) * nd1;
892 theta = ((-spot * npd1 * vol * Math.Exp(-div * t) / (2.0 * sqrtT)) -
893 (rate * strike * Math.Exp(-rate * t) * nd2) +
894 (div * spot * Math.Exp(-div * t) * nd1)) / TradingCalendarDefaults.TradingDaysPerYear;
895 }
896 else
897 {
898 price = (strike * Math.Exp(-rate * t) * (1.0 - nd2)) - (spot * Math.Exp(-div * t) * (1.0 - nd1));
899 delta = Math.Exp(-div * t) * (nd1 - 1.0);
900 theta = ((-spot * npd1 * vol * Math.Exp(-div * t) / (2.0 * sqrtT)) +
901 (rate * strike * Math.Exp(-rate * t) * (1.0 - nd2)) -
902 (div * spot * Math.Exp(-div * t) * (1.0 - nd1))) / TradingCalendarDefaults.TradingDaysPerYear;
903 }
904
905 double gamma = Math.Exp(-div * t) * npd1 / (spot * vol * sqrtT);
906 double vega = spot * sqrtT * npd1 * Math.Exp(-div * t) * 0.01;
907
908 return (price, delta, gamma, theta, vega);
909 }

1598 private static double CalculateRealisedVolatility(List<PriceBar> priceHistory)
1599 {
1600 if (priceHistory.Count < 2)
1601 {
1602 return 0.20; // Default fallback
1603 }
1604
1605 // Yang-Zhang OHLC-based volatility estimation
1606 int n = priceHistory.Count;
1607 double sumOC = 0.0;
1608 double sumCC = 0.0;
1609 double sumRS = 0.0;
1610
1611 for (int i = 1; i < n; i++)
1612 {
1613 PriceBar current = priceHistory[i];
1614 PriceBar previous = priceHistory[i - 1];
1615
1616 // Overnight return
1617 double overnight = Math.Log(current.Open / previous.Close);
1618 sumOC += overnight * overnight;
1619
1620 // Close-to-close return
1621 double closeToClose = Math.Log(current.Close / previous.Close);
1622 sumCC += closeToClose * closeToClose;
1623
1624 // Rogers-Satchell component
1625 double h = Math.Log(current.High / current.Open);
1626 double l = Math.Log(current.Low / current.Open);
1627 double c = Math.Log(current.Close / current.Open);
1628 sumRS += (h * (h - c)) + (l * (l - c));
1629 }
1630
1631 double varOC = sumOC / (n - 1);
1632 double varCC = sumCC / (n - 1);
1633 double varRS = sumRS / (n - 1);
1634
1635 // Yang-Zhang: weighted combination
1636 double k = 0.34 / (1.34 + ((n + 1.0) / (n - 1.0)));
1637 double variance = varOC + (k * varCC) + ((1.0 - k) * varRS);
1638
1639 // Annualise
1640 return Math.Sqrt(variance * TradingCalendarDefaults.TradingDaysPerYear);
1641 }

1646 private static TermStructureResult AnalyseTermStructure(OptionChain chain, DateTime evalDate)
1647 {
1648 var points = new List<(int dte, double iv)>();
1649
1650 foreach (OptionExpiry expiry in chain.Expiries)
1651 {
1652 int dte = (int)(expiry.ExpiryDate - evalDate).TotalDays;
1653
1654 // Get ATM IV (closest to spot)
1655 OptionContract atmCall = expiry.Calls
1656 .OrderBy(c => Math.Abs(c.Strike - chain.UnderlyingPrice))
1657 .First();
1658
1659 points.Add((dte, atmCall.ImpliedVolatility));
1660 }
1661
1662 // Calculate slope using linear regression
1663 double n = points.Count;
1664 double sumX = points.Sum(p => (double)p.dte);
1665 double sumY = points.Sum(p => p.iv);
1666 double sumXY = points.Sum(p => p.dte * p.iv);
1667 double sumX2 = points.Sum(p => (double)p.dte * p.dte);
1668
1669 double slope = ((n * sumXY) - (sumX * sumY)) / ((n * sumX2) - (sumX * sumX));
1670 bool isInverted = slope < 0;
1671 bool meetsCriterion = slope <= -0.00406;
1672
1673 // 30-day IV
1674 double iv30 = points.First(p => p.dte >= 25 && p.dte <= 35).iv;
1675
1676 return new TermStructureResult
1677 {
1678 Slope = slope,
1679 IsInverted = isInverted,
1680 MeetsTradingCriterion = meetsCriterion,
1681 IV30 = iv30,
1682 Points = points
1683 };
