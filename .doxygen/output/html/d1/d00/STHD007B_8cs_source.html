4// DESIGN RATIONALE:
5// The PD controller formulation was rejected because:
6// 1. No feedback loop - our exit doesn't affect IV; the plant evolves independently
7// 2. Non-linear decision boundary - rate-level relationship is scenario-dependent
8// 3. Discrete decision - binary exit/hold, not continuous control signal

10// This component implements explicit rule-based logic with interpretable conditions.

90 public ExitSignal Evaluate(double crushCaptured, double daysElapsed, double daysRemaining)
91 {
92 // Rate Estimation
93
94 double rawRate = 0;
95 if (_isInitialised)
96 {
97 // Rate of change in crush captured per day
98 rawRate = crushCaptured - _previousCrushCaptured;
99 }
100
101 // Exponential smoothing to reduce noise
102 _smoothedRate = (_params.RateSmoothingAlpha * rawRate)
103 + ((1 - _params.RateSmoothingAlpha) * _smoothedRate);
104
105 _previousCrushCaptured = crushCaptured;
106 _isInitialised = true;
107 _updateCount++;
108
109 // Rule-Based Exit Evaluation
110
111 // Rule 1: Target capture - we got what we came for
112 if (crushCaptured >= _params.TargetCrushRatio)
113 {
114 return CreateExitSignal(
115 ExitReason.TargetCaptured,
116 $"Captured {crushCaptured:P0} of expected crush (target: {_params.TargetCrushRatio:P0})",
117 crushCaptured,
118 _smoothedRate,
119 daysRemaining);
120 }
121
122 // Rule 2: Stall detection - significant capture but rate has stalled
123 if (crushCaptured >= _params.StallCrushThreshold &&
124 Math.Abs(_smoothedRate) < _params.StallRateThreshold &&
125 _updateCount >= _params.MinUpdatesForStall)
126 {
127 return CreateExitSignal(
128 ExitReason.CrushStalled,
129 $"Crush stalled at {crushCaptured:P0} (rate: {_smoothedRate:F4}/day)",
130 crushCaptured,
131 _smoothedRate,
132 daysRemaining);
133 }
134
135 // Rule 3: Protective exit - trade not working
136 if (crushCaptured < _params.MinExpectedCrush &&
137 _smoothedRate > _params.AdverseRateThreshold &&
138 daysElapsed > _params.MaxWaitDays)
139 {
140 return CreateExitSignal(
141 ExitReason.TradeNotWorking,
142 $"Insufficient crush ({crushCaptured:P0}) after {daysElapsed:F1} days, rate unfavourable",
143 crushCaptured,
144 _smoothedRate,
145 daysRemaining);
146 }
147
148 // Rule 4: Time decay - approaching expiry with partial profit
149 if (crushCaptured >= _params.PartialCaptureThreshold &&
150 daysRemaining < _params.MinDaysRemaining)
151 {
152 return CreateExitSignal(
153 ExitReason.TimeDecay,
154 $"Locking in {crushCaptured:P0} with only {daysRemaining:F1} days remaining",
155 crushCaptured,
156 _smoothedRate,
157 daysRemaining);
158 }
159
160 // Rule 5: Rate reversal warning - IV moving against us
161 if (crushCaptured >= _params.PartialCaptureThreshold &&
162 _smoothedRate < _params.ReversalRateThreshold &&
163 _updateCount >= _params.MinUpdatesForStall)
164 {
165 return CreateExitSignal(
166 ExitReason.RateReversal,
167 $"IV reversing with {crushCaptured:P0} captured (rate: {_smoothedRate:F4}/day)",
168 crushCaptured,
169 _smoothedRate,
170 daysRemaining);
171 }
172
173 // Default: Hold
174 SafeLog(() => LogHoldSignal(_logger!, crushCaptured, _smoothedRate, null));
175
176 return new ExitSignal(
177 Action: ExitAction.Hold,
178 Reason: ExitReason.None,
179 Message: "Continue holding",
180 CrushCaptured: crushCaptured,
181 SmoothedRate: _smoothedRate,
182 DaysRemaining: daysRemaining,
183 Confidence: ComputeHoldConfidence(crushCaptured, _smoothedRate, daysRemaining));
184 }

193 public static double ComputeCrushCaptured(double ivObserved, double ivExpected, double expectedCrush)
194 {
195 if (expectedCrush <= 0)
196 {
197 return 0;
198 }
199
200 // How much IV has dropped from expected theoretical level
201 double actualCrush = ivExpected - ivObserved;
202 return actualCrush / expectedCrush;
203 }

212 public static double ComputeExpectedCrush(
213 double baseVolatility,
214 double earningsJumpVolatility,
215 double timeToExpiryAtEntry)
216 {
217 if (timeToExpiryAtEntry <= 0)
218 {
219 return 0;
220 }
221
222 // Pre-EA IV from L&S: I(t) = √(σ² + σ_e²/(T-t))
223 double preEaIV = Math.Sqrt(
224 (baseVolatility * baseVolatility) +
225 (earningsJumpVolatility * earningsJumpVolatility / timeToExpiryAtEntry));
226
227 // Expected crush = pre-EA IV - base vol
228 return preEaIV - baseVolatility;
229 }

250 private static double ComputeExitConfidence(ExitReason reason, double crushCaptured, double rate)
251 {
252 return reason switch
253 {
254 ExitReason.TargetCaptured => 0.95, // Very confident
255 ExitReason.CrushStalled => 0.75 + (0.15 * Math.Min(1, crushCaptured)),
256 ExitReason.TimeDecay => 0.80,
257 ExitReason.TradeNotWorking => 0.70,
258 ExitReason.RateReversal => 0.65 + (0.20 * Math.Min(1, crushCaptured)),
259 _ => 0.50
260 };
261 }

263 private double ComputeHoldConfidence(double crushCaptured, double rate, double daysRemaining)
264 {
265 // Higher confidence in hold when:
266 // - Crush is progressing (rate negative = IV declining)
267 // - Plenty of time remaining
268 // - Crush is not yet at target
269
270 double rateScore = rate < 0 ? 0.3 : 0.0; // Reward declining IV
271 double timeScore = Math.Min(0.3, daysRemaining / 20.0); // More time = more confidence
272 double progressScore = 0.4 * (1 - (crushCaptured / _params.TargetCrushRatio)); // Room to grow
273
274 return Math.Min(0.95, 0.5 + rateScore + timeScore + progressScore);
275 }
