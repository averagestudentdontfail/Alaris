81 public DoubleBoundaryResult Solve()
82 {
83 // Near-expiry guard: Handle T near zero where QD+ has numerical issues
84 CREX001A nearExpiryHandler = new CREX001A();
85 NearExpiryValidation validation = nearExpiryHandler.Validate(_maturity);
86
87 if (validation.Recommendation == NearExpiryRecommendation.UseIntrinsic ||
88 validation.Recommendation == NearExpiryRecommendation.UseBlended)
89 {
90 // Very near expiry or in blending zone: return intrinsic-based boundaries
91 return CreateNearExpiryResult(validation);
92 }
93
94 // Stage 1: QD+ approximation for initial boundaries
95 (double upperInitial, double lowerInitial) = CalculateInitialBoundaries();
96
97 // Check regime and refinement settings
98 bool isDoubleBoundary = DetectDoubleBoundaryRegime();
99
100 if (!isDoubleBoundary)
101 {
102 return CreateSingleBoundaryResult(upperInitial, lowerInitial);
103 }
104
105 if (!_useRefinement)
106 {
107 return CreateQdOnlyResult(upperInitial, lowerInitial);
108 }
109
110 // Stage 2: Kim refinement with FP-B' stabilization
111 return ApplyKimRefinement(upperInitial, lowerInitial);
112 }

117 private DoubleBoundaryResult CreateNearExpiryResult(NearExpiryValidation validation)
118 {
119 double theoreticalSpread = System.Math.Max(0.01, _volatility * System.Math.Sqrt(_maturity));
120
121 double nearExpiryUpper = _isCall
122 ? _strike * (1.0 + (0.3 * theoreticalSpread))
123 : _strike * (1.0 - (0.3 * theoreticalSpread));
124
125 double nearExpiryLower = _isCall
126 ? _strike * (1.0 + (0.1 * theoreticalSpread))
127 : _strike * (1.0 - theoreticalSpread);
128
129 if (!_isCall && nearExpiryLower >= nearExpiryUpper)
130 {
131 nearExpiryLower = nearExpiryUpper * 0.99;
132 }
133
134 return new DoubleBoundaryResult
135 {
136 UpperBoundary = nearExpiryUpper,
137 LowerBoundary = nearExpiryLower,
138 QdUpperBoundary = nearExpiryUpper,
139 QdLowerBoundary = nearExpiryLower,
140 CrossingTime = 0.0,
141 IsRefined = false,
142 Method = $"Near-Expiry Handler ({validation.Reason})",
143 IsValid = true,
144 Iterations = 0
145 };
146 }

201 private DoubleBoundaryResult ApplyKimRefinement(double upperInitial, double lowerInitial)
202 {
203 CRSL002A kimSolver = new CRSL002A(
204 _spot, _strike, _maturity, _rate,
205 _dividendYield, _volatility, _isCall,
206 _collocationPoints);
207
208 (double[] upperRefined, double[] lowerRefined, double crossingTime) = kimSolver.SolveBoundaries(
209 upperInitial, lowerInitial);
210
211 // Extract boundary values at maturity
212 int lastIndex = upperRefined.Length - 1;
213 double upperFinal = upperRefined[lastIndex];
214 double lowerFinal = lowerRefined[lastIndex];
215
216 return new DoubleBoundaryResult
217 {
218 UpperBoundary = upperFinal,
219 LowerBoundary = lowerFinal,
220 QdUpperBoundary = upperInitial,
221 QdLowerBoundary = lowerInitial,
222 CrossingTime = crossingTime,
223 IsRefined = true,
224 Method = "QD+ + FP-B' Kim Refinement",
225 IsValid = ValidateBoundaries(upperFinal, lowerFinal),
226 Iterations = _collocationPoints,
227 UpperBoundaryPath = upperRefined,
228 LowerBoundaryPath = lowerRefined
229 };

235 private bool DetectDoubleBoundaryRegime()
236 {
237 const double HysteresisEpsilon = 0.0005;
238
239 if (!_isCall)
240 {
241 bool rateClearlyNegative = _rate < -HysteresisEpsilon;
242 bool dividendClearlyBelowRate = _dividendYield < _rate - HysteresisEpsilon;
243 return dividendClearlyBelowRate && rateClearlyNegative;
244 }
245 else
246 {
247 bool rateClearlyPositive = _rate > HysteresisEpsilon;
248 bool rateClearlyBelowDividend = _rate < _dividendYield - HysteresisEpsilon;
249 return rateClearlyPositive && rateClearlyBelowDividend;
250 }
251 }

256 private bool ValidateBoundaries(double upper, double lower)
257 {
258 if (double.IsNaN(upper) || double.IsNaN(lower))
259 {
260 return false;
261 }
262
263 if (double.IsInfinity(upper) && double.IsInfinity(lower))
264 {
265 return false;
266 }
267
268 if (!_isCall)
269 {
270 if (!double.IsPositiveInfinity(upper) && upper > _strike)
271 {
272 return false;
273 }
274
275 if (!double.IsNegativeInfinity(lower) && lower < 0)
276 {
277 return false;
278 }
279
280 if (!double.IsInfinity(upper) && !double.IsInfinity(lower) && lower >= upper)
281 {
282 return false;
283 }
284 }
285 else
286 {
287 if (!double.IsPositiveInfinity(upper) && upper < _strike)
288 {
289 return false;
290 }
291
292 if (!double.IsNegativeInfinity(lower) && lower < 0)
293 {
294 return false;
295 }
296
297 if (!double.IsInfinity(upper) && !double.IsInfinity(lower) && lower >= upper)
298 {
299 return false;
300 }
301 }
302
303 return true;
304 }
