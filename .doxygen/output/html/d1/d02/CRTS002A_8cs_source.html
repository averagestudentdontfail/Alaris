81 public CRTS002AFlatVol(CRTM005A referenceDate, double volatility, IDayCounter? dayCounter = null)
82 {
83 if (volatility < 0)
84 {
85 throw new ArgumentException("Volatility cannot be negative", nameof(volatility));
86 }
87
88 ReferenceDate = referenceDate;
89 _volatility = volatility;
90 DayCounter = dayCounter ?? DayCounters.Actual365Fixed;

94 public CRTM005A ReferenceDate { get; }

97 public IDayCounter DayCounter { get; }

100 public double BlackVol(double time, double strike)
101 {
102 // Strike-independent for flat vol
103 _ = strike;
104 return _volatility;
105 }
106
108 public double BlackVariance(double time, double strike)
109 {
110 double vol = BlackVol(time, strike);
111 return vol * vol * System.Math.Max(time, 0.0);
112 }
113
115 public double BlackVol(CRTM005A date, double strike)
116 {
117 double time = DayCounter.YearFraction(ReferenceDate, date);
118 return BlackVol(time, strike);
119 }
120
