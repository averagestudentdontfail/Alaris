57 public double CalculateFreshness(DateTime signalTime, DateTime currentTime)
58 {
59 if (currentTime < signalTime)
60 {
61 return 1.0; // Signal is from the future (clock issue) - treat as fresh
62 }
63
64 double ageMinutes = (currentTime - signalTime).TotalMinutes;
65 return Math.Exp(-_decayRate * ageMinutes);
66 }

99 public TimeSpan TimeUntilStale(
100 DateTime signalTime,
101 DateTime currentTime,
102 double freshnessThreshold = DefaultFreshnessThreshold)
103 {
104 // Solve: e^(-λ×t) = threshold → t = -ln(threshold)/λ
105 double staleAgeMinutes = -Math.Log(freshnessThreshold) / _decayRate;
106 double currentAgeMinutes = (currentTime - signalTime).TotalMinutes;
107 double remainingMinutes = staleAgeMinutes - currentAgeMinutes;
108
109 return TimeSpan.FromMinutes(remainingMinutes);
