36 public static double ComputeTheoreticalIV(
37 double baseVolatility,
38 double earningsJumpVolatility,
39 double timeToExpiry)
40 {
41 if (baseVolatility < 0)
42 {
43 throw new ArgumentException("Base volatility must be non-negative.", nameof(baseVolatility));
44 }
45
46 if (earningsJumpVolatility < 0)
47 {
48 throw new ArgumentException("Earnings jump volatility must be non-negative.", nameof(earningsJumpVolatility));
49 }
50
51 if (timeToExpiry <= 0)
52 {
53 throw new ArgumentException("Time to expiry must be positive.", nameof(timeToExpiry));
54 }
55
56 // Ensure minimum time to avoid numerical issues
57 double effectiveTimeToExpiry = Math.Max(timeToExpiry, MinTimeToExpiry);
58
59 // Leung & Santoli (2014) formula: I(t) = sqrt(sigma^2 + sigma_e^2 / (T-t))
60 double varianceComponent = earningsJumpVolatility * earningsJumpVolatility / effectiveTimeToExpiry;
61 double totalVariance = (baseVolatility * baseVolatility) + varianceComponent;
62
63 return Math.Sqrt(totalVariance);
64 }

134 public static double ExtractEarningsJumpVolatility(
135 double marketIV,
136 double baseVolatility,
137 double timeToExpiry)
138 {
139 if (marketIV < 0)
140 {
141 throw new ArgumentException("Market IV must be non-negative.", nameof(marketIV));
142 }
143
144 if (baseVolatility < 0)
145 {
146 throw new ArgumentException("Base volatility must be non-negative.", nameof(baseVolatility));
147 }
148
149 if (timeToExpiry <= 0)
150 {
151 throw new ArgumentException("Time to expiry must be positive.", nameof(timeToExpiry));
152 }
153
154 // Rearranging I(t)^2 = sigma^2 + sigma_e^2/(T-t) gives:
155 // sigma_e = sqrt((I(t)^2 - sigma^2) * (T-t))
156 double ivSquared = marketIV * marketIV;
157 double baseSquared = baseVolatility * baseVolatility;
158 double variance = ivSquared - baseSquared;
159
160 if (variance <= 0)
161 {
162 // Market IV is below base volatility - no earnings premium implied
163 return 0;
164 }
165
166 return Math.Sqrt(variance * timeToExpiry);
167 }
