49 public async Task<List<string>> ScreenAsync(
50 DateTime date,
51 decimal minPrice = DefaultMinPrice,
52 decimal minDollarVolume = DefaultMinDollarVolume,
53 int maxSymbols = DefaultMaxSymbols,
54 CancellationToken cancellationToken = default)
55 {
56 if (string.IsNullOrEmpty(_apiKey))
57 {
58 _logger.LogError("Polygon API key not configured. Cannot perform stock screening.");
59 throw new InvalidOperationException(
60 "Polygon API key is required for stock screening. Configure 'Polygon:ApiKey' in appsettings.");
61 }
62
63 try
64 {
65 // Use the most recent trading day considering holidays
66 // Retry up to 5 days back if market is closed (Polygon returns 0 results)
67 var currentParamsDate = GetMostRecentTradingDay(date);
68 PolygonGroupedResponse? response = null;
69 var attempts = 0;
70 const int MaxAttempts = 5;
71
72 while (attempts < MaxAttempts)
73 {
74 var dateStr = currentParamsDate.ToString("yyyy-MM-dd");
75
76 _logger.LogInformation("Screening stocks for {Date} (Attempt {Attempt}/{Max})", dateStr, attempts + 1, MaxAttempts);
77
78 var url = $"https://api.polygon.io/v2/aggs/grouped/locale/us/market/stocks/{dateStr}?adjusted=true&apiKey={_apiKey}";
79 var options = new JsonSerializerOptions { PropertyNameCaseInsensitive = false };
80
81 try
82 {
83 response = await _httpClient.GetFromJsonAsync<PolygonGroupedResponse>(url, options, cancellationToken);
84 }
85 catch (HttpRequestException)
86 {
87 // Ignore transient HTTP errors in loop? Or count as attempt?
88 // For now, assume empty response if 404, etc.
89 }
90
91 if (response?.Results != null && response.Results.Length > 0)
92 {
93 break; // Found data!
94 }
95
96 _logger.LogWarning("No results from Polygon for {Date} (Holiday/Weekend?), trying previous day...", dateStr);
97
98 // Move back one day
99 currentParamsDate = currentParamsDate.AddDays(-1);
100 // Skip weekends again just in case
101 while (currentParamsDate.DayOfWeek == DayOfWeek.Saturday || currentParamsDate.DayOfWeek == DayOfWeek.Sunday)
102 {
103 currentParamsDate = currentParamsDate.AddDays(-1);
104 }
105 attempts++;
106 }
107
108 if (response?.Results == null || response.Results.Length == 0)
109 {
110 _logger.LogError("Failed to find any screening data after {Max} attempts. Aborting screening.", MaxAttempts);
111 return new List<string>(); // Return empty, caller decides fallback or fail.
112 }
113
114 // Filter and sort by dollar volume
115 var filteredSymbols = response.Results
116 .Where(r => !string.IsNullOrEmpty(r.Ticker))
117 .Where(r => !r.Ticker!.Contains('.')) // Exclude preferred/class shares
118 .Where(r => r.Ticker!.Length <= 5) // Exclude warrants/units (usually > 5 chars)
119 .Where(r => r.Close >= minPrice)
120 .Where(r => r.Close * (decimal)r.Volume >= minDollarVolume)
121 .OrderByDescending(r => r.Close * (decimal)r.Volume)
122 .Take(maxSymbols)
123 .Select(r => r.Ticker!.ToUpperInvariant())
124 .ToList();
125
126 _logger.LogInformation("Screened {Count} symbols from {Total} total",
127 filteredSymbols.Count, response.Results.Length);
128
129 if (filteredSymbols.Count == 0)
130 {
131 _logger.LogWarning("No symbols passed screening criteria.");
132 return new List<string>();
133 }
134
135 return filteredSymbols;
136 }
137 catch (HttpRequestException ex)
138 {
139 _logger.LogError(ex, "HTTP error during screening. Aborting - no fallback data substitution.");
140 throw new InvalidOperationException("Stock screening failed due to HTTP error. Check network connectivity.", ex);
141 }
142 catch (Exception ex)
143 {
144 _logger.LogError(ex, "Error during screening. Aborting - no fallback data substitution.");
145 throw new InvalidOperationException("Stock screening failed unexpectedly.", ex);
146 }
147 }

150 // If screening cannot be performed, the caller must handle the exception

156 private static DateTime GetMostRecentTradingDay(DateTime date)
157 {
158 var current = date.Date;
159
160 // If date is in future, use yesterday
161 if (current >= DateTime.Today)
162 {
163 current = DateTime.Today.AddDays(-1);
164 }
165
166 // Skip weekends
167 while (current.DayOfWeek == DayOfWeek.Saturday || current.DayOfWeek == DayOfWeek.Sunday)
168 {
169 current = current.AddDays(-1);
170 }
171
172 return current;
173 }
