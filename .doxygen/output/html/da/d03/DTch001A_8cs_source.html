71 TimeSpan? slidingExpiration = null)

75 _logger.LogDebug("Cache hit for key: {Key}", key);
76 return cachedValue!;
77 }
78
79 _logger.LogDebug("Cache miss for key: {Key}, fetching...", key);

97 options.SetAbsoluteExpiration(TimeSpan.FromMinutes(30));
98 }
99
100 _cache.Set(key, value, options);
101 _logger.LogDebug("Cached value for key: {Key}", key);
