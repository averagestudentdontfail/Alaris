89 apiKey: _apiKey,
90 cancellationToken);
91

94 _logger.LogWarning("No data returned from Polygon for {Symbol}", symbol);
95 return Array.Empty<PriceBar>();

97
98 List<PriceBar> bars = response.Results.Select(r => new PriceBar
99 {
100 Symbol = symbol,
101 Timestamp = DateTimeOffset.FromUnixTimeMilliseconds(r.Timestamp).DateTime,
102 Open = r.Open,
103 High = r.High,
104 Low = r.Low,
105 Close = r.Close,
106 Volume = (long)r.Volume

134 public async Task<OptionChainSnapshot> GetHistoricalOptionChainAsync(
135 string symbol,
136 DateTime asOfDate,
137 CancellationToken cancellationToken = default)
138 {
139 if (string.IsNullOrWhiteSpace(symbol))
140 throw new ArgumentException("Symbol cannot be null or whitespace", nameof(symbol));
141
142 DateTime referenceDate = asOfDate > DateTime.UtcNow.Date ? DateTime.UtcNow.Date : asOfDate;
143 DateTime twoYearsAgo = referenceDate.AddYears(-2).AddDays(1);
144 DateTime effectiveDate = asOfDate < twoYearsAgo ? twoYearsAgo : asOfDate;
145
146 if (asOfDate < twoYearsAgo)
147 {
148 _logger.LogWarning("asOfDate {Date:yyyy-MM-dd} is outside 2-year subscription limit, using {EffectiveDate:yyyy-MM-dd}",
149 asOfDate, effectiveDate);
150 }
151
152 _logger.LogInformation("Fetching historical option chain for {Symbol} as of {Date:yyyy-MM-dd}", symbol, effectiveDate);
153
154 // 1. Get historical spot price
155 decimal spotPrice = 0m;
156 try
157 {
158 DateTime spotStart = effectiveDate.AddDays(-5);
159 if (spotStart < twoYearsAgo) spotStart = twoYearsAgo;
160 IReadOnlyList<PriceBar> bars = await GetHistoricalBarsAsync(symbol, spotStart, effectiveDate, cancellationToken);
161 spotPrice = bars.Count > 0 ? bars[bars.Count - 1].Close : 0m;
162 }
163 catch (Exception ex)
164 {
165 _logger.LogWarning(ex, "Failed to get historical spot price for {Symbol}", symbol);
166 }
167
168 if (spotPrice == 0)
169 {
170 _logger.LogWarning("Spot price is 0 for {Symbol}, option chain will be incomplete", symbol);
171 return new OptionChainSnapshot { Symbol = symbol, SpotPrice = 0, Timestamp = effectiveDate, Contracts = new List<OptionContract>() };
172 }
173
174 // 2. Fetch option contracts using Reference API
175 string dateStr = effectiveDate.ToString("yyyy-MM-dd");
176 string expirationMin = effectiveDate.ToString("yyyy-MM-dd");
177 string expirationMax = effectiveDate.AddDays(60).ToString("yyyy-MM-dd");
178
179 _logger.LogDebug("Fetching reference contracts for {Symbol}", symbol);
180
181 try
182 {
183 PolygonOptionsContractsResponse refResponse = await _api.GetOptionsContractsAsync(
184 underlyingTicker: symbol,
185 asOfDate: dateStr,
186 expirationMin: expirationMin,
187 expirationMax: expirationMax,
188 limit: 250,
189 apiKey: _apiKey,
190 cancellationToken);
191
192 if (refResponse?.Results == null || refResponse.Results.Length == 0)
193 {
194 _logger.LogWarning("No reference options found for {Symbol} as of {Date}", symbol, dateStr);
195 return new OptionChainSnapshot { Symbol = symbol, SpotPrice = spotPrice, Timestamp = effectiveDate, Contracts = new List<OptionContract>() };
196 }
197
198 _logger.LogInformation("Found {Count} reference contracts for {Symbol}, fetching daily bars (max 50)...", refResponse.Results.Length, symbol);
199
200 // 3. Fetch daily bars for each contract - WITH PARALLELISM and LIMITS
201 PolygonOptionContract[] contractsToFetch = refResponse.Results.Take(50).ToArray();
202 ConcurrentBag<OptionContract> contracts = new();
203 bool subscriptionLimitHit = false;
204
205 // Use semaphore to limit concurrent requests (Polygon rate limits apply)
206 using SemaphoreSlim semaphore = new(5);
207 IEnumerable<Task> tasks = contractsToFetch.Select(async refContract =>
208 {
209 if (subscriptionLimitHit)
210 return;
211
212 await semaphore.WaitAsync(cancellationToken);
213 try
214 {
215 PolygonAggregatesResponse aggResponse = await _api.GetTickerAggregatesAsync(
216 refContract.Ticker,
217 dateStr,
218 dateStr,
219 adjusted: true,
220 apiKey: _apiKey,
221 cancellationToken);
222
223 if (aggResponse?.Results == null || aggResponse.Results.Length == 0)
224 return;
225
226 PolygonBar bar = aggResponse.Results[0];
227
228 // Parse contract details from OCC ticker format
229 (string underlying, decimal strike, DateTime expiration, OptionRight right) = ParseOptionTicker(refContract.Ticker);
230
231 // Calculate IV using CRMF001A Black-Scholes solver (industry standard for backtesting)
232 double riskFreeRate = 0.05;
233 double timeToExpiry = (expiration - effectiveDate).TotalDays / 365.25;
234 double impliedVol = CRMF001A.BSImpliedVolatility(
235 (double)spotPrice, (double)strike, timeToExpiry, riskFreeRate, 0.0,
236 (double)bar.Close, right == OptionRight.Call);
237
238 contracts.Add(new OptionContract
239 {
240 OptionSymbol = refContract.Ticker,
241 UnderlyingSymbol = underlying,
242 Strike = strike,
243 Expiration = expiration,
244 Right = right,
245 Bid = Math.Max(0, bar.Close - 0.10m),
246 Ask = bar.Close + 0.10m,
247 Last = bar.Close,
248 Volume = (long)bar.Volume,
249 OpenInterest = (long)bar.Volume,
250 ImpliedVolatility = !double.IsNaN(impliedVol) && impliedVol > 0 ? (decimal)impliedVol : null,
251 Timestamp = DateTimeOffset.FromUnixTimeMilliseconds(bar.Timestamp).DateTime
252 });
253 }
254 catch (OperationCanceledException)
255 {
256 // Timeout - skip this contract
257 }
258 catch (HttpRequestException ex) when (ex.StatusCode == System.Net.HttpStatusCode.Forbidden)
259 {
260 _logger.LogWarning(
261 "Options data for {Date} is outside the 2-year historical data window. Skipping remaining contracts.",
262 dateStr);
263 subscriptionLimitHit = true;
264 }
265 catch (Exception ex)
266 {
267 _logger.LogDebug(ex, "Failed to fetch pricing for contract {Ticker}", refContract.Ticker);
268 }
269 finally
270 {
271 semaphore.Release();
272 }
273 });
274
275 await Task.WhenAll(tasks);
276
277 _logger.LogInformation("Retrieved {Count} contracts with pricing for {Symbol}", contracts.Count, symbol);
278
279 return new OptionChainSnapshot
280 {
281 Symbol = symbol,
282 SpotPrice = spotPrice,
283 Timestamp = effectiveDate,
284 Contracts = contracts.ToList()
285 };
286 }
287 catch (Exception ex)
288 {
289 _logger.LogError(ex, "Error fetching historical option chain for {Symbol}", symbol);
290 throw;
291 }

345 #region Option Ticker Parsing
346
347 private static (string underlying, decimal strike, DateTime expiration, OptionRight right)
348 ParseOptionTicker(string ticker)
349 {
350 // Remove "O:" prefix if present
351 if (ticker.StartsWith("O:", StringComparison.OrdinalIgnoreCase))
352 ticker = ticker[2..];
353
354 // OCC format: UNDERLYING + YYMMDD + C/P + STRIKE(8 digits)
355 // But some tickers have digits in the underlying (e.g., AMD1, BRK.A -> BRKA)
356 // Strategy: Find C or P that's followed by exactly 8 digits (the strike)
357
358 int cpIndex = -1;
359 for (int i = ticker.Length - 9; i >= 6; i--) // Need at least 6 chars before (date)
360 {
361 char c = ticker[i];
362 if ((c == 'C' || c == 'P') &&
363 i + 9 == ticker.Length && // C/P + 8 strike digits = end of string
364 ticker[(i + 1)..].All(char.IsDigit))
365 {
366 cpIndex = i;
367 break;
368 }
369 }
370
371 if (cpIndex < 6)
372 {
373 throw new FormatException($"Unable to parse option ticker: {ticker}");
374 }
375
376 char rightChar = ticker[cpIndex];
377 string strikeStr = ticker[(cpIndex + 1)..];
378 string dateStr = ticker[(cpIndex - 6)..cpIndex];
379 string underlying = ticker[..(cpIndex - 6)];
380
381 // Parse with 2-digit year prefix
382 DateTime expiration = DateTime.ParseExact($"20{dateStr}", "yyyyMMdd", null);
383 OptionRight right = rightChar == 'C' ? OptionRight.Call : OptionRight.Put;
384 decimal strike = decimal.Parse(strikeStr) / 1000m;
385
386 return (underlying, strike, expiration, right);
387 }
