143 return 0;

146 private static (bool success, int stockCount, string? error) GenerateUniverseForDate(
147 DateTime date,
148 string apiKey,
149 string outputDir,
150 decimal minDollarVolume,
151 decimal minPrice)
152 {
153 try
154 {
155 using var httpClient = new HttpClient { Timeout = TimeSpan.FromSeconds(60) };
156
157 var dateStr = date.ToString("yyyy-MM-dd");
158 var url = $"https://api.polygon.io/v2/aggs/grouped/locale/us/market/stocks/{dateStr}?adjusted=true&apiKey={apiKey}";
159
160 // First get the raw response to check for errors
161 var httpResponse = httpClient.GetAsync(url).GetAwaiter().GetResult();
162 var content = httpResponse.Content.ReadAsStringAsync().GetAwaiter().GetResult();
163
164 if (!httpResponse.IsSuccessStatusCode)
165 {
166 return (false, 0, $"HTTP {(int)httpResponse.StatusCode}: {content.Substring(0, Math.Min(100, content.Length))}");
167 }
168
169 var response = System.Text.Json.JsonSerializer.Deserialize<PolygonGroupedResponse>(content);
170
171 if (response == null)
172 {
173 return (false, 0, "Failed to parse response");
174 }
175
176 if (response.Status != null && response.Status != "OK")
177 {
178 return (false, 0, $"API status: {response.Status}");
179 }
180
181 if (response.Results == null || response.Results.Length == 0)
182 {
183 return (false, 0, "No results in response");
184 }
185
186 // Filter stocks
187 var filteredStocks = response.Results
188 .Where(r => !string.IsNullOrEmpty(r.Ticker))
189 .Where(r => r.Close >= minPrice)
190 .Where(r => r.Close * (decimal)r.Volume >= minDollarVolume)
191 .OrderByDescending(r => r.Close * (decimal)r.Volume)
192 .Take(3000) // Top 3000 by dollar volume
193 .ToList();
194
195 if (filteredStocks.Count == 0)
196 {
197 return (false, 0, $"0 stocks after filtering (raw: {response.Results.Length})");
198 }
199
200 // Write coarse universe file in LEAN format
201 // Format: sid,symbol,close,volume,dollar_volume
202 var filePath = System.IO.Path.Combine(outputDir, $"{date:yyyyMMdd}.csv");
203 using var writer = new System.IO.StreamWriter(filePath);
204
205 foreach (var stock in filteredStocks)
206 {
207 var dollarVolume = stock.Close * (decimal)stock.Volume;
208 // LEAN coarse format: Symbol ID (hash), Symbol, Close Price, Volume, Dollar Volume
209 var symbolId = GetSymbolId(stock.Ticker!);
210 var line = $"{symbolId},{stock.Ticker!.ToLowerInvariant()},{stock.Close:F4},{stock.Volume:F0},{dollarVolume:F0}";
211 writer.WriteLine(line);
212 }
213
214 return (true, filteredStocks.Count, null);
215 }
216 catch (Exception ex)
217 {
218 return (false, 0, ex.Message);
219 }
220 }

222 private static string GetSymbolId(string ticker)
223 {
224 // Generate a consistent hash-based ID for the symbol
225 // This is a simplified version - LEAN uses more complex SID generation
226 var hash = ticker.GetHashCode();
227 return $"{Math.Abs(hash):X8}";
228 }

230 private static List<DateTime> GetTradingDays(DateTime start, DateTime end)
231 {
232 var days = new List<DateTime>();
233 var current = start;
234
235 while (current <= end)
236 {
237 // Skip weekends
238 if (current.DayOfWeek != DayOfWeek.Saturday && current.DayOfWeek != DayOfWeek.Sunday)
239 {
240 days.Add(current);
241 }
242 current = current.AddDays(1);
243 }
244
245 return days;
246 }

369 private static string? GetPolygonApiKey()
370 {
371 var paths = new[] { "appsettings.local.jsonc", "../appsettings.local.jsonc", "../../appsettings.local.jsonc" };
372 foreach (var path in paths)
373 {
374 if (System.IO.File.Exists(path))
375 {
376 try
377 {
378 var json = System.IO.File.ReadAllText(path);
379 var lines = json.Split('\n');
380 var inPolygon = false;
381 foreach (var line in lines)
382 {
383 var trimmed = line.Trim();
384 if (trimmed.StartsWith("//")) continue;
385 if (trimmed.Contains("\"Polygon\"")) inPolygon = true;
386 else if (inPolygon && trimmed.Contains("\"ApiKey\""))
387 {
388 var colonIdx = trimmed.IndexOf(':');
389 if (colonIdx > 0)
390 {
391 var value = trimmed[(colonIdx + 1)..].Trim().Trim(',', '"', ' ');
392 if (!string.IsNullOrEmpty(value)) return value;
393 }
394 }
395 else if (inPolygon && trimmed.StartsWith("}")) inPolygon = false;
396 }
397 }
398 catch { /* Ignore parsing errors */ }
399 }
400 }
401 return null;
402 }
