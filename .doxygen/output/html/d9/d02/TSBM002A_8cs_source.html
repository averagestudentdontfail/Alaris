36 [Theory]
37 [InlineData(100.0, 100.0, 1.0, 0.05, 0.02, 0.20, OptionType.Put)]
38 [InlineData(100.0, 100.0, 0.5, 0.05, 0.02, 0.30, OptionType.Put)]
39 [InlineData(100.0, 100.0, 2.0, 0.03, 0.01, 0.25, OptionType.Put)]

41 [InlineData(120.0, 100.0, 1.0, 0.05, 0.02, 0.20, OptionType.Put)] // OTM
42 public void Accuracy_SpectralVsFD_StandardRegime(
43 double spot, double strike, double tau, double r, double q, double sigma, OptionType optionType)
44 {
45 double spectralPrice = _spectralAccurate.Price(spot, strike, tau, r, q, sigma, optionType);
46 double fdPrice = _fdEngine.Price(spot, strike, tau, r, q, sigma, optionType);
47
48 double absoluteDiff = System.Math.Abs(spectralPrice - fdPrice);
49 double relativeDiff = absoluteDiff / fdPrice;
50
51 _output.WriteLine($"Spot={spot}, K={strike}, τ={tau}, r={r}, q={q}, σ={sigma}");
52 _output.WriteLine($" Spectral: {spectralPrice:F6}");
53 _output.WriteLine($" FD: {fdPrice:F6}");
54 _output.WriteLine($" Absolute Diff: {absoluteDiff:F6}");
55 _output.WriteLine($" Relative Diff: {relativeDiff:P4}");
56
57 // Spectral should be within 5% of FD for standard regimes
58 Assert.True(relativeDiff < 0.10,
59 $"Spectral {spectralPrice:F4} differs from FD {fdPrice:F4} by {relativeDiff:P2}");
60 }

82 // ========== Performance Comparison ==========
83
84 [Fact]
85 public void Performance_SpectralFastVsFD_100Options()
86 {
87 const int iterations = 100;
88 var random = new Random(42);
89
90 // Warm up
91 _ = _spectralFast.Price(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
92 _ = _fdEngine.Price(100, 100, 1.0, 0.05, 0.02, 0.20, OptionType.Put);
93
94 // Generate random option parameters
95 var testCases = new (double Spot, double Strike, double Tau, double R, double Q, double Sigma)[iterations];
96 for (int i = 0; i < iterations; i++)
97 {
98 testCases[i] = (
99 Spot: 80.0 + (random.NextDouble() * 40.0), // 80-120
100 Strike: 100.0,
101 Tau: 0.25 + (random.NextDouble() * 1.75), // 0.25-2 years
102 R: 0.01 + (random.NextDouble() * 0.08), // 1%-9%
103 Q: random.NextDouble() * 0.05, // 0%-5%
104 Sigma: 0.10 + (random.NextDouble() * 0.40) // 10%-50%
105 );
106 }
107
108 // Spectral Fast timing
109 var swSpectral = Stopwatch.StartNew();
110 for (int i = 0; i < iterations; i++)
111 {
112 var (spot, strike, tau, r, q, sigma) = testCases[i];
113 _ = _spectralFast.Price(spot, strike, tau, r, q, sigma, OptionType.Put);
114 }
115 swSpectral.Stop();
116 double spectralAvgMs = swSpectral.Elapsed.TotalMilliseconds / iterations;
117
118 // FD timing
119 var swFD = Stopwatch.StartNew();
120 for (int i = 0; i < iterations; i++)
121 {
122 var (spot, strike, tau, r, q, sigma) = testCases[i];
123 _ = _fdEngine.Price(spot, strike, tau, r, q, sigma, OptionType.Put);
124 }
125 swFD.Stop();
126 double fdAvgMs = swFD.Elapsed.TotalMilliseconds / iterations;
127
128 double speedup = fdAvgMs / spectralAvgMs;
129
130 _output.WriteLine($"Performance Benchmark (N={iterations}):");
131 _output.WriteLine($" Spectral Fast: {spectralAvgMs:F4} ms/option");
132 _output.WriteLine($" FD Engine: {fdAvgMs:F4} ms/option");
133 _output.WriteLine($" Speedup: {speedup:F1}x");
134
135 // Spectral should be at least as fast as FD (may not be 10x due to JIT etc)
136 Assert.True(spectralAvgMs < fdAvgMs * 2.0,
137 $"Spectral ({spectralAvgMs:F4}ms) should not be much slower than FD ({fdAvgMs:F4}ms)");
138 }

201 // ========== RMSE Calculation ==========
202
203 [Fact]
204 public void RMSE_SpectralVsFD_Portfolio()
205 {
206 // Test portfolio of 20 options across different strikes/maturities
207 var testCases = new (double Spot, double Strike, double Tau)[]
208 {
209 (100, 80, 0.25), (100, 90, 0.25), (100, 100, 0.25), (100, 110, 0.25), (100, 120, 0.25),
210 (100, 80, 0.50), (100, 90, 0.50), (100, 100, 0.50), (100, 110, 0.50), (100, 120, 0.50),
211 (100, 80, 1.00), (100, 90, 1.00), (100, 100, 1.00), (100, 110, 1.00), (100, 120, 1.00),
212 (100, 80, 2.00), (100, 90, 2.00), (100, 100, 2.00), (100, 110, 2.00), (100, 120, 2.00),
213 };
214
215 double r = 0.05, q = 0.02, sigma = 0.20;
216 double sumSquaredErrors = 0.0;
217 int count = 0;
218
219 foreach (var (spot, strike, tau) in testCases)
220 {
221 double spectral = _spectralAccurate.Price(spot, strike, tau, r, q, sigma, OptionType.Put);
222 double fd = _fdEngine.Price(spot, strike, tau, r, q, sigma, OptionType.Put);
223
224 double error = spectral - fd;
225 sumSquaredErrors += error * error;
226 count++;
227
228 _output.WriteLine($"K={strike:F0}, τ={tau:F2}: Spectral={spectral:F4}, FD={fd:F4}, Diff={error:F4}");
229 }
230
231 double rmse = System.Math.Sqrt(sumSquaredErrors / count);
232 _output.WriteLine($"\nRMSE: {rmse:F6}");
233
234 // RMSE should be < 0.5 (within $0.50 of FD reference)
235 Assert.True(rmse < 0.50,
236 $"RMSE {rmse:F4} exceeds threshold 0.50");
237 }
