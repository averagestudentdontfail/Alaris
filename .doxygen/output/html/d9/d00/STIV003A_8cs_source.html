39 public ModelSelectionResult SelectBestModel(ModelSelectionContext context)
40 {
41 ArgumentNullException.ThrowIfNull(context);
42
43 // Detect regime
44 STTM002A regime = context.Regime ?? STTM002A.Detect(context.TimeParams);
45
46 // Get candidate models based on regime
47 IReadOnlyList<RecommendedModel> candidates = GetCandidateModels(regime);
48
49 // Evaluate each candidate
50 List<ModelEvaluation> evaluations = new();

52 foreach (RecommendedModel candidate in candidates)

54 ModelEvaluation evaluation = EvaluateModel(candidate, context, regime);
55 evaluations.Add(evaluation);

69
70 return new ModelSelectionResult
71 {
72 SelectedModel = best.ModelType,
73 Regime = regime,
74 Evaluations = evaluations.AsReadOnly(),
75 BestEvaluation = best,
76 SelectionReason = GenerateSelectionReason(best, regime)

111 private ModelEvaluation EvaluateModel(
112 RecommendedModel modelType,
113 ModelSelectionContext context,
114 STTM002A regime)
115 {
116 // Get model complexity (number of parameters)
117 int complexity = GetModelComplexity(modelType);
118
119 // Compute fit metrics
120 FitMetrics fitMetrics = ComputeFitMetrics(modelType, context);
121
122 // Validate martingale condition
123 bool martingaleValid = _martingaleValidator.Validate(modelType, context);
124
125 // Compute information criteria
126 int n = context.MarketIVs?.Count ?? 1;
127 double aic = ComputeAIC(fitMetrics.MSE, complexity, n);
128 double bic = ComputeBIC(fitMetrics.MSE, complexity, n);
129
130 // Composite score (lower is better)
131 double compositeScore = ComputeCompositeScore(fitMetrics, aic, martingaleValid);
132
133 return new ModelEvaluation
134 {
135 ModelType = modelType,
136 FitMetrics = fitMetrics,
137 Complexity = complexity,
138 AIC = aic,
139 BIC = bic,
140 MartingaleValid = martingaleValid,
141 CompositeScore = compositeScore
142 };

148 private static FitMetrics ComputeFitMetrics(
149 RecommendedModel modelType,
150 ModelSelectionContext context)
151 {
152 if (context.MarketIVs == null || context.MarketIVs.Count == 0)
153 {
154 return FitMetrics.Default;
155 }
156
157 IReadOnlyList<double> modelIVs = ComputeModelIVs(modelType, context);
158
159 if (modelIVs.Count != context.MarketIVs.Count)
160 {
161 return FitMetrics.Default;
162 }
163
164 // Compute error metrics
165 double sumSquaredError = 0;
166 double sumAbsError = 0;
167 double maxError = 0;
168
169 for (int i = 0; i < context.MarketIVs.Count; i++)
170 {
171 double error = modelIVs[i] - context.MarketIVs[i].IV;
172 sumSquaredError += error * error;
173 sumAbsError += Math.Abs(error);
174 maxError = Math.Max(maxError, Math.Abs(error));
175 }
176
177 int n = context.MarketIVs.Count;
178 double mse = sumSquaredError / n;
179 double rmse = Math.Sqrt(mse);
180 double mae = sumAbsError / n;
181
182 // Compute R-squared
183 double meanIV = context.MarketIVs.Average(x => x.IV);
184 double totalSS = context.MarketIVs.Sum(x => (x.IV - meanIV) * (x.IV - meanIV));
185 double rSquared = totalSS > 0 ? 1 - (sumSquaredError / totalSS) : 0;
186
187 return new FitMetrics
188 {
189 MSE = mse,
190 RMSE = rmse,
191 MAE = mae,
192 MaxError = maxError,
193 RSquared = Math.Max(0, rSquared)
194 };
195 }

200 private static IReadOnlyList<double> ComputeModelIVs(
201 RecommendedModel modelType,
202 ModelSelectionContext context)
203 {
204 List<double> result = new();
205
206 if (context.MarketIVs == null)
207 {
208 return result;
209 }
210
211 foreach ((double strike, int dte, double _) in context.MarketIVs)
212 {
213 double timeToExpiry = TradingCalendarDefaults.DteToYears(dte);
214 double modelIV = modelType switch
215 {
216 RecommendedModel.BlackScholes =>
217 context.BaseVolatility,
218
219 RecommendedModel.LeungSantoli when context.EarningsJumpVolatility.HasValue =>
220 STIV004A.ComputeTheoreticalIV(
221 context.BaseVolatility,
222 context.EarningsJumpVolatility.Value,
223 timeToExpiry),
224
225 RecommendedModel.Heston when context.HestonParams != null =>
226 new STIV001A(context.HestonParams)
227 .ComputeTheoreticalIV(context.Spot, strike, timeToExpiry),
228
229 RecommendedModel.Kou when context.KouParams != null =>
230 new STIV002A(context.KouParams)
231 .ComputeTheoreticalIV(context.Spot, strike, timeToExpiry),
232
233 _ => context.BaseVolatility
234 };
235
236 result.Add(modelIV);
237 }
238
239 return result;
240 }

287 private static double ComputeCompositeScore(FitMetrics fit, double aic, bool martingaleValid)
288 {
289 // Weighted combination
290 double score = (0.4 * fit.RMSE * 100) + // RMSE in percentage points
291 (0.3 * aic / 100) + // Normalized AIC
292 (0.2 * (1 - fit.RSquared)) + // R-squared penalty
293 (0.1 * fit.MaxError * 100); // Maximum error
294
295 // Penalize martingale violation
296 if (!martingaleValid)
297 {
298 score += 10;
299 }
300
301 return score;
302 }

316public sealed class MartingaleValidator
317{
321 private const double DriftTolerance = 0.001;
322
329 public bool Validate(RecommendedModel modelType, ModelSelectionContext context)
330 {
331 ArgumentNullException.ThrowIfNull(context);
332
333 return modelType switch
334 {
335 RecommendedModel.BlackScholes => true, // Always satisfies by construction
336 RecommendedModel.LeungSantoli => ValidateLeungSantoli(context),
337 RecommendedModel.Heston => ValidateHeston(context),
338 RecommendedModel.Kou => ValidateKou(context),
339 _ => true
340 };
341 }

347 private static bool ValidateLeungSantoli(ModelSelectionContext context)
348 {
349 // L&S model doesn't affect drift - martingale preserved
350 // The jump only affects variance, not expected return
351 return true;
352 }
353
358 private static bool ValidateHeston(ModelSelectionContext context)
359 {
360 if (context.HestonParams == null)
361 {
362 return true;
363 }
364
365 // Heston uses drift = r - d by construction
366 // Martingale requires: E[S_T / B_T] = S_0 / B_0
367 // This is satisfied when stock drift = r - d
368 double impliedDrift = context.HestonParams.RiskFreeRate -
369 context.HestonParams.DividendYield;
370 double expectedDrift = context.RiskFreeRate - context.DividendYield;
371
372 return Math.Abs(impliedDrift - expectedDrift) < DriftTolerance;
373 }
374
379 private static bool ValidateKou(ModelSelectionContext context)
380 {
381 if (context.KouParams == null)
382 {
383 return true;
384 }
385
386 // Check that drift is properly compensated
387 double kappa = context.KouParams.ComputeKappa();
388 double adjustedDrift = context.KouParams.ComputeMartingaleDrift();
389 double expectedDrift = context.RiskFreeRate - context.DividendYield -
390 (context.KouParams.Lambda * kappa);
391
392 return Math.Abs(adjustedDrift - expectedDrift) < DriftTolerance;
393 }
394
398 public static double ComputeJumpCompensation(KouParameters kouParams)
399 {
400 ArgumentNullException.ThrowIfNull(kouParams);
401 return kouParams.Lambda * kouParams.ComputeKappa();
402 }
