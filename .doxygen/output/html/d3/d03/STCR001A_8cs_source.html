108 public STCR004A Generate(
109 string symbol,
110 DateTime earningsDate,
111 DateTime evaluationDate,
112 IReadOnlyList<DateTime>? historicalEarningsDates)
113 {
114 ArgumentException.ThrowIfNullOrWhiteSpace(symbol);
115
116 SafeLog(() => LogGeneratingSTCR004A(_logger!, symbol, earningsDate, null));
117
118 STCR004A signal = new STCR004A
119 {
120 Symbol = symbol,
121 EarningsDate = earningsDate,
122 STCR004ADate = evaluationDate
123 };
124
125 try
126 {
127 // Fetch and validate market data
128 (List<PriceBar> priceHistory, STDT002A optionChain) = FetchMarketData(symbol, evaluationDate);
129 if (priceHistory.Count < 30)
130 {
131 SafeLog(() => LogInsufficientPriceHistory(_logger!, symbol, null));
132 signal.Strength = STCR004AStrength.Avoid;
133 return signal;
134 }

137 // Options data must be bootstrapped via CLI before running backtest

139 {

141 throw new InvalidOperationException(
142 $"No valid options data with IV available for {symbol} on {evaluationDate:yyyy-MM-dd}. " +
143 "Run 'alaris backtest run --auto-bootstrap' to download historical options data from Polygon API.");
144 }
145
146 // Calculate signal metrics (including L&S model if historical data provided)
147 CalculateSTCR004AMetrics(signal, priceHistory, optionChain, earningsDate, evaluationDate, historicalEarningsDates);
148
149 // Log final signal
150 SafeLog(() => LogSTCR004AGenerated(_logger!, symbol, signal.Strength, signal.IVRVRatio, signal.STTM001ASlope, signal.AverageVolume, null));
151 }
152 catch (ArgumentException ex)
153 {
154 SafeLog(() => LogErrorGeneratingSTCR004A(_logger!, symbol, ex));

156 }
157 catch (InvalidOperationException ex)

160 throw;
161 }

185 private void CalculateSTCR004AMetrics(
186 STCR004A signal,
187 List<PriceBar> priceHistory,
188 STDT002A optionChain,
189 DateTime earningsDate,
190 DateTime evaluationDate,
191 IReadOnlyList<DateTime>? historicalEarningsDates)
192 {
193 // Calculate 30-day Yang-Zhang realized volatility
194 signal.RealizedVolatility30 = _yangZhang.Calculate(priceHistory, 30);
195
196 // Extract and analyze term structure
197 List<STTM001APoint> termPoints = ExtractSTTM001APoints(optionChain, evaluationDate);
198 if (termPoints.Count < 2)
199 {
200 SafeLog(() => LogInsufficientSTTM001A(_logger!, signal.Symbol, null));
201 signal.Strength = STCR004AStrength.Avoid;
202 return;
203 }
204
205 STTM001AAnalysis termAnalysis = _termAnalyzer.Analyze(termPoints);
206 signal.STTM001ASlope = termAnalysis.Slope;
207 signal.ImpliedVolatility30 = termAnalysis.GetIVAt(30);
208
209 // Calculate IV/RV ratio
210 signal.IVRVRatio = (signal.RealizedVolatility30 > 0)
211 ? (signal.ImpliedVolatility30 / signal.RealizedVolatility30)
212 : 0;
213
214 // Calculate average volume (Index-based loop for performance)
215 long totalVolume = 0;
216 int volumeStart = Math.Max(0, priceHistory.Count - 30);
217 int volumeCount = priceHistory.Count - volumeStart;
218 for (int i = volumeStart; i < priceHistory.Count; i++)
219 {
220 totalVolume += priceHistory[i].Volume;
221 }
222 signal.AverageVolume = volumeCount > 0 ? totalVolume / volumeCount : 0;
223
224 // Calculate expected move and volatility spread
225 signal.ExpectedMove = CalculateExpectedMove(optionChain, earningsDate, evaluationDate);
226 signal.VolatilitySpread = CalculateVolatilitySpread(optionChain, evaluationDate);
227
228 // Leung & Santoli (2014) Model Calculations
229 CalculateLeungSantoliMetrics(signal, priceHistory, optionChain, earningsDate, evaluationDate, historicalEarningsDates);
230
231 // Evaluate criteria
232 signal.Criteria["Volume"] = signal.AverageVolume >= MinAverageVolume;
233 signal.Criteria["IV/RV"] = signal.IVRVRatio >= MinIvRvRatio;
234 signal.Criteria["TermSlope"] = signal.STTM001ASlope <= MaxTermSlope;
235
236 signal.EvaluateStrength();
237 }

288 private bool TryCalibrateFromTermStructure(STCR004A signal, STDT002A optionChain, DateTime evaluationDate)
289 {
290 List<STTM001APoint> termPoints = ExtractSTTM001APoints(optionChain, evaluationDate);
291 if (termPoints.Count < 2)
292 {
293 return false;
294 }
295
296 // Find min and second min dte - ZERO ALLOC
297 IReadOnlyList<STTM001APoint> points = termPoints;
298 STTM001APoint? nearTerm = CRFN001A.FindMinBy(points, p => p.DaysToExpiry);
299 STTM001APoint? farTerm = CRFN001A.FindMinBy(points, p => p.DaysToExpiry, p => p != nearTerm);
300
301 if (nearTerm == null || farTerm == null)
302 {
303 return false;
304 }
305
306 double? sigmaE = STIV005A.STTM001AEstimator(nearTerm.ImpliedVolatility, nearTerm.DaysToExpiry, farTerm.ImpliedVolatility, farTerm.DaysToExpiry);
307 double? baseVol = STIV005A.BaseVolatilityEstimator(nearTerm.ImpliedVolatility, nearTerm.DaysToExpiry, farTerm.ImpliedVolatility, farTerm.DaysToExpiry);
308
309 if (sigmaE.HasValue && sigmaE.Value > 0)
310 {
311 signal.EarningsJumpVolatility = sigmaE.Value;
312 signal.BaseVolatility = baseVol ?? signal.RealizedVolatility30;
313 signal.IsLeungSantoliCalibrated = true;
314 signal.HistoricalEarningsCount = 0;
315 return true;
316 }

324 private static void ComputeTheoreticalMetrics(
325 STCR004A signal,
326 STDT002A optionChain,
327 DateTime earningsDate,
328 DateTime evaluationDate)
329 {
330 // Find the front-month expiry (closest to but after earnings) - ZERO ALLOC
331 OptionExpiry? targetExpiry = CRFN001A.FindMinBy<OptionExpiry>(
332 optionChain.Expiries,
333 e => (e.ExpiryDate - earningsDate).TotalDays,
334 e => e.ExpiryDate >= earningsDate);
335
336 if (targetExpiry == null)
337 {
338 return;
339 }
340
341 int dte = targetExpiry.GetDaysToExpiry(evaluationDate);
342 if (dte <= 0)
343 {
344 return;
345 }
346
347 double timeToExpiry = TradingCalendarDefaults.DteToYears(dte);
348
349 // Compute theoretical IV using L&S formula: I(t) = sqrt(sigma^2 + sigma_e^2/(T-t))
350 signal.TheoreticalIV = STIV004A.ComputeTheoreticalIV(
351 signal.BaseVolatility,
352 signal.EarningsJumpVolatility,
353 timeToExpiry);
354
355 // Compute mispricing signal (market IV - theoretical IV)
356 signal.IVMispricingSTCR004A = signal.ImpliedVolatility30 - signal.TheoreticalIV;
357
358 // Compute expected IV crush
359 signal.ExpectedIVCrush = STIV004A.ComputeExpectedIVCrush(
360 signal.BaseVolatility,
361 signal.EarningsJumpVolatility,
362 timeToExpiry);
363
364 signal.IVCrushRatio = STIV004A.ComputeIVCrushRatio(
365 signal.BaseVolatility,
366 signal.EarningsJumpVolatility,
367 timeToExpiry);
368 }

373 private List<STTM001APoint> ExtractSTTM001APoints(STDT002A chain, DateTime evaluationDate)
374 {
375 List<STTM001APoint> points = new List<STTM001APoint>();
376 double underlyingPrice = chain.UnderlyingPrice;
377
378 foreach (OptionExpiry expiry in chain.Expiries.OrderBy(e => e.ExpiryDate))
379 {
380 int dte = expiry.GetDaysToExpiry(evaluationDate);
381 if ((dte < 1) || (dte > 60))
382 {
383 continue;
384 }
385
386 // Find ATM options - ZERO ALLOC
387 OptionContract? atmCall = CRFN001A.FindMinBy<OptionContract>(
388 expiry.Calls,
389 c => Math.Abs(c.Strike - underlyingPrice),
390 c => (c.OpenInterest > 0) && (c.ImpliedVolatility > 0));
391
392 OptionContract? atmPut = CRFN001A.FindMinBy<OptionContract>(
393 expiry.Puts,
394 p => Math.Abs(p.Strike - underlyingPrice),
395 p => (p.OpenInterest > 0) && (p.ImpliedVolatility > 0));
396
397 if ((atmCall is not null) && (atmPut is not null))
398 {
399 // Average the call and put IV for ATM
400 double avgIV = (atmCall.ImpliedVolatility + atmPut.ImpliedVolatility) / 2.0;
401
402 points.Add(new STTM001APoint
403 {
404 DaysToExpiry = dte,
405 ImpliedVolatility = avgIV,
406 Strike = atmCall.Strike
407 });
408 }
409 }
410
411 return points;
412 }

417 private double CalculateExpectedMove(STDT002A chain, DateTime earningsDate, DateTime evaluationDate)
418 {
419 // Find the expiry closest to (but after) earnings date - ZERO ALLOC
420 OptionExpiry? targetExpiry = CRFN001A.FindMinBy<OptionExpiry>(
421 chain.Expiries,
422 e => (e.ExpiryDate - earningsDate).TotalDays,
423 e => e.ExpiryDate >= earningsDate);
424
425 if (targetExpiry is null)
426 {
427 return 0;
428 }
429
430 double underlyingPrice = chain.UnderlyingPrice;
431
432 // Find ATM straddle - ZERO ALLOC
433 OptionContract? atmCall = CRFN001A.FindMinBy<OptionContract>(
434 targetExpiry.Calls,
435 c => Math.Abs(c.Strike - underlyingPrice),
436 c => (c.Bid > 0) && (c.Ask > 0));
437
438 OptionContract? atmPut = CRFN001A.FirstWhere<OptionContract>(
439 targetExpiry.Puts,
440 p => (p.Bid > 0) && (p.Ask > 0) && (Math.Abs(p.Strike - (atmCall?.Strike ?? 0)) < 0.01));
441
442 if ((atmCall is null) || (atmPut is null))
443 {
444 return 0;
445 }
446
447 double straddlePrice = atmCall.MidPrice + atmPut.MidPrice;
448 return straddlePrice / underlyingPrice;
449 }

454 private double CalculateVolatilitySpread(STDT002A chain, DateTime evaluationDate)
455 {
456 List<(double spread, double weight)> spreads = new List<(double spread, double weight)>();
457 double underlyingPrice = chain.UnderlyingPrice;
458
459 foreach (OptionExpiry expiry in chain.Expiries)
460 {
461 int dte = expiry.GetDaysToExpiry(evaluationDate);
462 if ((dte < 10) || (dte > 60))
463 {
464 continue;
465 }
466
467 // Match put-call pairs
468 IEnumerable<(double Strike, double Spread, double Weight)> pairs = from call in expiry.Calls
469 join put in expiry.Puts on call.Strike equals put.Strike
470 where (call.OpenInterest > 0) && (put.OpenInterest > 0)
471 && (call.ImpliedVolatility > 0) && (put.ImpliedVolatility > 0)
472 select (call.Strike, put.ImpliedVolatility - call.ImpliedVolatility, (call.OpenInterest + put.OpenInterest) / 2.0);
473
474 foreach ((double Strike, double Spread, double Weight) pair in pairs)
475 {
476 spreads.Add((pair.Spread, pair.Weight));
477 }
478 }
479
480 if (spreads.Count == 0)
481 {
482 return 0;
483 }
484
485 double totalWeight = spreads.Sum(s => s.weight);
486 return spreads.Sum(s => s.spread * s.weight) / totalWeight;
487 }
