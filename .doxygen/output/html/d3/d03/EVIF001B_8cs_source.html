240 private static byte[] SerializeEnvelope(EVCR003A envelope)
241 {
242 // Get UTF-8 bytes for variable-length fields
243 byte[] eventTypeBytes = Encoding.UTF8.GetBytes(envelope.EventType);
244 byte[] eventDataBytes = Encoding.UTF8.GetBytes(envelope.EventData);
245 byte[] aggregateIdBytes = Encoding.UTF8.GetBytes(envelope.AggregateId ?? "");
246 byte[] aggregateTypeBytes = Encoding.UTF8.GetBytes(envelope.AggregateType ?? "");
247 byte[] correlationIdBytes = Encoding.UTF8.GetBytes(envelope.CorrelationId ?? "");
248 byte[] initiatedByBytes = Encoding.UTF8.GetBytes(envelope.InitiatedBy ?? "");
249
250 int variableLength = eventTypeBytes.Length + eventDataBytes.Length +
251 aggregateIdBytes.Length + aggregateTypeBytes.Length +
252 correlationIdBytes.Length + initiatedByBytes.Length;
253 int totalLength = HeaderSize + variableLength;
254
255 byte[] buffer = new byte[totalLength];
256 Span<byte> span = buffer;
257
258 // Write header
259 int offset = 0;
260 BinaryPrimitives.WriteInt64LittleEndian(span[offset..], envelope.SequenceNumber);
261 offset += 8;
262
263 envelope.EventId.TryWriteBytes(span[offset..]);
264 offset += 16;
265
266 BinaryPrimitives.WriteInt64LittleEndian(span[offset..], envelope.StoredAtUtc.Ticks);
267 offset += 8;
268
269 BinaryPrimitives.WriteInt32LittleEndian(span[offset..], eventTypeBytes.Length);
270 offset += 4;
271 BinaryPrimitives.WriteInt32LittleEndian(span[offset..], eventDataBytes.Length);
272 offset += 4;
273 BinaryPrimitives.WriteInt32LittleEndian(span[offset..], aggregateIdBytes.Length);
274 offset += 4;
275 BinaryPrimitives.WriteInt32LittleEndian(span[offset..], aggregateTypeBytes.Length);
276 offset += 4;
277 BinaryPrimitives.WriteInt32LittleEndian(span[offset..], correlationIdBytes.Length);
278 offset += 4;
279 BinaryPrimitives.WriteInt32LittleEndian(span[offset..], initiatedByBytes.Length);
280 offset += 4;
281 BinaryPrimitives.WriteInt32LittleEndian(span[offset..], totalLength);
282 offset += 4;
283 BinaryPrimitives.WriteInt32LittleEndian(span[offset..], 0); // Reserved
284 offset += 4;
285
286 // Write variable-length data
287 eventTypeBytes.CopyTo(span[offset..]);
288 offset += eventTypeBytes.Length;
289 eventDataBytes.CopyTo(span[offset..]);
290 offset += eventDataBytes.Length;
291 aggregateIdBytes.CopyTo(span[offset..]);
292 offset += aggregateIdBytes.Length;
293 aggregateTypeBytes.CopyTo(span[offset..]);
294 offset += aggregateTypeBytes.Length;
295 correlationIdBytes.CopyTo(span[offset..]);
296 offset += correlationIdBytes.Length;
297 initiatedByBytes.CopyTo(span[offset..]);
298
299 return buffer;
300 }

302 private static EVCR003A DeserializeEnvelope(ReadOnlySpan<byte> buffer)
303 {
304 int offset = 0;
305
306 long sequenceNumber = BinaryPrimitives.ReadInt64LittleEndian(buffer[offset..]);
307 offset += 8;
308
309 Guid eventId = new(buffer.Slice(offset, 16));
310 offset += 16;
311
312 long storedTicks = BinaryPrimitives.ReadInt64LittleEndian(buffer[offset..]);
313 DateTime storedAtUtc = new(storedTicks, DateTimeKind.Utc);
314 offset += 8;
315
316 int eventTypeLen = BinaryPrimitives.ReadInt32LittleEndian(buffer[offset..]);
317 offset += 4;
318 int eventDataLen = BinaryPrimitives.ReadInt32LittleEndian(buffer[offset..]);
319 offset += 4;
320 int aggregateIdLen = BinaryPrimitives.ReadInt32LittleEndian(buffer[offset..]);
321 offset += 4;
322 int aggregateTypeLen = BinaryPrimitives.ReadInt32LittleEndian(buffer[offset..]);
323 offset += 4;
324 int correlationIdLen = BinaryPrimitives.ReadInt32LittleEndian(buffer[offset..]);
325 offset += 4;
326 int initiatedByLen = BinaryPrimitives.ReadInt32LittleEndian(buffer[offset..]);
327 offset += 4;
328 // Skip totalLength and reserved
329 offset += 8;
330
331 string eventType = Encoding.UTF8.GetString(buffer.Slice(offset, eventTypeLen));
332 offset += eventTypeLen;
333 string eventData = Encoding.UTF8.GetString(buffer.Slice(offset, eventDataLen));
334 offset += eventDataLen;
335 string? aggregateId = aggregateIdLen > 0 ?
336 Encoding.UTF8.GetString(buffer.Slice(offset, aggregateIdLen)) : null;
337 offset += aggregateIdLen;
338 string? aggregateType = aggregateTypeLen > 0 ?
339 Encoding.UTF8.GetString(buffer.Slice(offset, aggregateTypeLen)) : null;
340 offset += aggregateTypeLen;
341 string? correlationId = correlationIdLen > 0 ?
342 Encoding.UTF8.GetString(buffer.Slice(offset, correlationIdLen)) : null;
343 offset += correlationIdLen;
344 string? initiatedBy = initiatedByLen > 0 ?
345 Encoding.UTF8.GetString(buffer.Slice(offset, initiatedByLen)) : null;
346
347 return new EVCR003A
348 {
349 EventId = eventId,
350 SequenceNumber = sequenceNumber,
351 StoredAtUtc = storedAtUtc,
352 EventType = eventType,
353 EventData = eventData,

362 private async IAsyncEnumerable<EVCR003A> ReadEventsAsync(
363 [System.Runtime.CompilerServices.EnumeratorCancellation] CancellationToken cancellationToken = default)
364 {
365 if (!File.Exists(_eventsPath))
366 {
367 yield break;
368 }
369
370 await using FileStream fs = new(
371 _eventsPath,
372 FileMode.Open,
373 FileAccess.Read,
374 FileShare.ReadWrite);
375
376 byte[] headerBuffer = new byte[HeaderSize];
377 byte[]? recordBuffer = null;
378
379 while (fs.Position < fs.Length && !cancellationToken.IsCancellationRequested)
380 {
381 // Read header to get record length
382 int headerRead = await fs.ReadAsync(headerBuffer, cancellationToken).ConfigureAwait(false);
383 if (headerRead < HeaderSize)
384 {
385 break; // Incomplete header, file may be corrupted/truncated
386 }
387
388 int totalLength = BinaryPrimitives.ReadInt32LittleEndian(headerBuffer.AsSpan(56));
389 if (totalLength <= 0 || totalLength > MaxRecordSize)
390 {
391 break; // Invalid record length
392 }
393
394 // Read full record
395 if (recordBuffer == null || recordBuffer.Length < totalLength)
396 {
397 recordBuffer = new byte[totalLength];
398 }
399
400 headerBuffer.CopyTo(recordBuffer, 0);
401 int remainingBytes = totalLength - HeaderSize;
402 if (remainingBytes > 0)
403 {
404 int dataRead = await fs.ReadAsync(
405 recordBuffer.AsMemory(HeaderSize, remainingBytes),
406 cancellationToken).ConfigureAwait(false);
407 if (dataRead < remainingBytes)
408 {
409 break; // Incomplete record
410 }
411 }
412
413 EVCR003A envelope;
414 try
415 {
416 envelope = DeserializeEnvelope(recordBuffer.AsSpan(0, totalLength));
417 }
418 catch
419 {
420 // Skip corrupted records
421 continue;
422 }
423
424 yield return envelope;
425 }

428 private long RecoverSequence()
429 {
430 // Try to read sequence file first
431 if (File.Exists(_sequencePath))
432 {
433 string content = File.ReadAllText(_sequencePath);
434 if (long.TryParse(content.Trim(), out long seq))
435 {
436 return seq;
437 }
438 }
439
440 // Fall back to scanning binary events file
441 if (!File.Exists(_eventsPath))
442 {
443 return 0;
444 }
445
446 long maxSequence = 0;
447 using FileStream fs = new(_eventsPath, FileMode.Open, FileAccess.Read, FileShare.ReadWrite);
448 byte[] headerBuffer = new byte[HeaderSize];
449
450 while (fs.Position < fs.Length)
451 {
452 int bytesRead = fs.Read(headerBuffer, 0, HeaderSize);
453 if (bytesRead < HeaderSize)
454 {
455 break;
456 }
457
458 long seq = BinaryPrimitives.ReadInt64LittleEndian(headerBuffer);
459 maxSequence = Math.Max(maxSequence, seq);
460
461 int totalLength = BinaryPrimitives.ReadInt32LittleEndian(headerBuffer.AsSpan(56));
462 if (totalLength <= HeaderSize || totalLength > MaxRecordSize)
463 {
464 break;
465 }
466
467 // Skip to next record
468 int skipBytes = totalLength - HeaderSize;
469 if (skipBytes > 0)
470 {
471 fs.Seek(skipBytes, SeekOrigin.Current);
472 }
473 }
474
475 return maxSequence;
476 }
