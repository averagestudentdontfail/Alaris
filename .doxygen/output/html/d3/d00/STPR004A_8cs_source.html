42 public OptimizationResult Minimize(
43 Func<double[], double[]> residuals,
44 double[] initialGuess,
45 double[]? lowerBounds = null,
46 double[]? upperBounds = null)
47 {
48 ArgumentNullException.ThrowIfNull(residuals);
49 ArgumentNullException.ThrowIfNull(initialGuess);
50
51 // Use the strictest tolerance provided
52 double effectiveTolerance = Math.Min(Tolerance, Math.Min(ParameterTolerance, ObjectiveTolerance));
53
54 IObjectiveModel obj = new LeastSquaresObjective(residuals);
55
56 LevenbergMarquardtMinimizer solver = new LevenbergMarquardtMinimizer(
57 gradientTolerance: effectiveTolerance,
58 stepTolerance: effectiveTolerance,
59 functionTolerance: effectiveTolerance,
60 maximumIterations: MaxIterations);
61
62 NonlinearMinimizationResult result = solver.FindMinimum(obj, Vector<double>.Build.DenseOfArray(initialGuess));
63
64 return CreateResult(result, residuals);
65 }
66
67 private OptimizationResult CreateResult(
68 NonlinearMinimizationResult result,
69 Func<double[], double[]> residuals)
70 {
71 double[] finalParams = result.MinimizingPoint.ToArray();
72 double[] finalResiduals = residuals(finalParams);
73 double sumSq = finalResiduals.Sum(r => r * r);
74
75 return new OptimizationResult
76 {
77 OptimalParameters = finalParams,
78 OptimalValue = 0.5 * sumSq,
79 RMSE = Math.Sqrt(sumSq / finalResiduals.Length),
80 Iterations = result.Iterations,
81 Status = MapStatus(result.ReasonForExit)
82 };
83 }

109 public void EvaluateAt(Vector<double> point)
110 {
111 _point = point;
112 double[] r = _residuals(point.ToArray());
113 _residualsVector = Vector<double>.Build.DenseOfArray(r);
114
115 // Custom Numerical Jacobian (Central Difference)
116 _jacobian = ComputeJacobian(point);
117
118 _evaluated = true;
119 FunctionEvaluations++;
120 JacobianEvaluations++;
121 }
122
123 private Matrix<double> ComputeJacobian(Vector<double> point)
124 {
125 int n = point.Count;
126 int m = _residualsVector.Count;
127 Matrix<double> J = Matrix<double>.Build.Dense(m, n);
128 double h = 1e-8;
129
130 // We need to mutate point, so copy it or restore it.
131 // Vector<double> is mutable.
132
133 for (int j = 0; j < n; j++)
134 {
135 double original = point[j];
136
137 point[j] = original + h;
138 Vector<double> rPlus = Vector<double>.Build.DenseOfArray(_residuals(point.ToArray()));
139
140 point[j] = original - h;
141 Vector<double> rMinus = Vector<double>.Build.DenseOfArray(_residuals(point.ToArray()));
142
143 point[j] = original; // Restore
144
145 Vector<double> col = (rPlus - rMinus) / (2 * h);
146 J.SetColumn(j, col);
147 }
148 return J;
149 }
