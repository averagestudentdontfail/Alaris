483 private static string? SelectSession(string action)
484 {

486 var sessions = service.ListAsync().Result;
487
488 if (sessions.Count == 0)
489 {

753 private static void ViewLogs()
754 {
755 AnsiConsole.MarkupLine("[yellow]Recent System Activity[/]");
756 AnsiConsole.WriteLine();
757
758 var logPath = System.IO.Path.Combine("Alaris.Simulation", "Output", "Logs");
759 if (!Directory.Exists(logPath))
760 {
761 AnsiConsole.MarkupLine("[grey]No logs found.[/]");
762 return;
763 }
764
765 var logs = Directory.GetFiles(logPath, "*.log")
766 .OrderByDescending(f => File.GetLastWriteTime(f))
767 .Take(5)
768 .ToList();
769
770 if (logs.Count == 0)
771 {
772 AnsiConsole.MarkupLine("[grey]No log files found.[/]");
773 return;
774 }
775
776 foreach (var log in logs)
777 {
778 var lastWrite = File.GetLastWriteTime(log);
779 AnsiConsole.MarkupLine($"[blue]{System.IO.Path.GetFileName(log)}[/] - {lastWrite:yyyy-MM-dd HH:mm}");
780 }
