78 public OptimizationResult Minimize(
79 Func<double[], double> objective,
80 double[] lowerBounds,
81 double[] upperBounds)
82 {
83 ArgumentNullException.ThrowIfNull(objective);
84 ArgumentNullException.ThrowIfNull(lowerBounds);
85 ArgumentNullException.ThrowIfNull(upperBounds);
86
87 int dimension = lowerBounds.Length;
88
89 if (upperBounds.Length != dimension)
90 {
91 throw new ArgumentException("Bounds must have same dimension.");
92 }
93
94 // Auto-select population size if not specified
95 int popSize = PopulationSize > 0 ? PopulationSize : 10 * dimension;
96
97 // CA5394: Random is acceptable for optimization algorithms (no security context)
98#pragma warning disable CA5394
99 Random rng = RandomSeed.HasValue ? new Random(RandomSeed.Value) : new Random();
100#pragma warning restore CA5394
101
102 // Initialize population randomly
103 double[][] population = new double[popSize][];
104 double[] fitness = new double[popSize];
105
106 for (int i = 0; i < popSize; i++)
107 {
108 population[i] = new double[dimension];
109 for (int j = 0; j < dimension; j++)
110 {
111#pragma warning disable CA5394 // Random is acceptable for optimization algorithms (no security context)
112 population[i][j] = lowerBounds[j] +
113 (rng.NextDouble() * (upperBounds[j] - lowerBounds[j]));
114#pragma warning restore CA5394
115 }
116
117 fitness[i] = EvaluateIndividual(population, i, objective);
118 }
119
120 // Track best solution
121 int bestIndex = FindMinIndex(fitness);
122 double[] bestSolution = GetIndividual(population, bestIndex);
123 double bestFitness = fitness[bestIndex];
124
125 int generation = 0;
126 int stagnationCount = 0;
127 const int maxStagnation = 50;
128
129 while (generation < MaxGenerations)
130 {
131 double[][] newPopulation = new double[popSize][];
132 double[] newFitness = new double[popSize];
133
134 for (int i = 0; i < popSize; i++)
135 {
136 newPopulation[i] = new double[dimension];
137
138 // Select three random distinct individuals (different from i)
139 int a = SelectRandomIndex(rng, popSize, i);
140 int b = SelectRandomIndex(rng, popSize, i, a);
141 int c = SelectRandomIndex(rng, popSize, i, a, b);
142
143 // Create mutant vector: v = x_a + F * (x_b - x_c)
144 double[] mutant = new double[dimension];
145 for (int j = 0; j < dimension; j++)
146 {
147 mutant[j] = population[a][j] +
148 (DifferentialWeight * (population[b][j] - population[c][j]));
149
150 // Ensure bounds
151 mutant[j] = Math.Clamp(mutant[j], lowerBounds[j], upperBounds[j]);
152 }
153
154 // Crossover: create trial vector
155 double[] trial = new double[dimension];
156#pragma warning disable CA5394 // Random is acceptable for optimization algorithms (no security context)
157 int forcedIndex = rng.Next(dimension); // Ensure at least one parameter from mutant
158#pragma warning restore CA5394
159
160 for (int j = 0; j < dimension; j++)
161 {
162#pragma warning disable CA5394 // Random is acceptable for optimization algorithms (no security context)
163 if (rng.NextDouble() < CrossoverProbability || j == forcedIndex)
164#pragma warning restore CA5394
165 {
166 trial[j] = mutant[j];
167 }
168 else
169 {
170 trial[j] = population[i][j];
171 }
172 }
173
174 // Selection: keep better of trial and current
175 double trialFitness = objective(trial);
176
177 if (trialFitness < fitness[i])
178 {
179 // Trial is better - accept it
180 newPopulation[i] = trial;
181 newFitness[i] = trialFitness;
182
183 // Update global best
184 if (trialFitness < bestFitness)
185 {
186 bestSolution = (double[])trial.Clone();
187 bestFitness = trialFitness;
188 stagnationCount = 0;
189 }
190 }
191 else
192 {
193 // Keep current individual
194 newPopulation[i] = (double[])population[i].Clone();
195 newFitness[i] = fitness[i];
196 }
197 }
198
199 population = newPopulation;
200 fitness = newFitness;
201 generation++;
202 stagnationCount++;
203
204 // Check convergence: population diversity
205 double fitnessStd = ComputeStandardDeviation(fitness);
206 if (fitnessStd < Tolerance)
207 {
208 return CreateResult(bestSolution, bestFitness, generation,
209 OptimizationStatus.ObjectiveConvergence);
210 }
211
212 // Early stopping if stagnant
213 if (stagnationCount > maxStagnation)
214 {
215 return CreateResult(bestSolution, bestFitness, generation,
216 OptimizationStatus.ParameterConvergence);
217 }
218 }
219
220 return CreateResult(bestSolution, bestFitness, generation,
221 OptimizationStatus.MaxIterationsReached);
222 }
