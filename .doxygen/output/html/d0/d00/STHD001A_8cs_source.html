111 public STHD002A Analyse(
112 string symbol,
113 IReadOnlyList<double> frontIVChanges,
114 IReadOnlyList<double> backIVChanges)
115 {
116 ArgumentException.ThrowIfNullOrWhiteSpace(symbol);
117 ArgumentNullException.ThrowIfNull(frontIVChanges);
118 ArgumentNullException.ThrowIfNull(backIVChanges);
119
120 if (frontIVChanges.Count != backIVChanges.Count)
121 {
122 throw new ArgumentException(
123 $"IV change series must have equal length. Front: {frontIVChanges.Count}, Back: {backIVChanges.Count}",
124 nameof(backIVChanges));
125 }
126
127 int observations = frontIVChanges.Count;
128
129 // Check minimum observations
130 if (observations < _minimumObservations)
131 {
132 SafeLog(() => LogInsufficientData(
133 _logger!,
134 symbol,
135 observations,
136 _minimumObservations,
137 null));
138
139 return new STHD002A
140 {
141 Symbol = symbol,
142 Correlation = double.NaN,
143 Threshold = _maxAcceptableCorrelation,
144 Observations = observations,
145 MinimumObservations = _minimumObservations,
146 PassesFilter = false,
147 HasSufficientData = false,
148 Interpretation = "Insufficient data for correlation analysis."
149 };
150 }
151
152 // Compute Pearson correlation using MathNet.Numerics
153 double correlation = Correlation.Pearson(frontIVChanges, backIVChanges);
154
155 // Handle edge case where correlation is NaN (e.g., zero variance)
156 if (double.IsNaN(correlation))
157 {
158 return new STHD002A
159 {
160 Symbol = symbol,
161 Correlation = double.NaN,
162 Threshold = _maxAcceptableCorrelation,
163 Observations = observations,
164 MinimumObservations = _minimumObservations,
165 PassesFilter = false,
166 HasSufficientData = true,
167 Interpretation = "Correlation undefined (zero variance in one or both series)."
168 };
169 }
170
171 // Determine interpretation
172 string interpretation = DetermineInterpretation(correlation);
173
174 // Evaluate against threshold
175 bool passesFilter = correlation < _maxAcceptableCorrelation;
176
177 var result = new STHD002A
178 {
179 Symbol = symbol,
180 Correlation = correlation,
181 Threshold = _maxAcceptableCorrelation,
182 Observations = observations,
183 MinimumObservations = _minimumObservations,
184 PassesFilter = passesFilter,
185 HasSufficientData = true,
186 Interpretation = interpretation
187 };
188
189 SafeLog(() => LogCorrelationResult(_logger!, symbol, correlation, interpretation, null));
190
191 return result;
192 }

223 private static double[] ComputeChanges(IReadOnlyList<double> levels)
224 {
225 if (levels.Count < 2)
226 {
227 return [];
228 }
229
230 double[] changes = new double[levels.Count - 1];
231 for (int i = 1; i < levels.Count; i++)
232 {
233 changes[i - 1] = levels[i] - levels[i - 1];
234 }
235
236 return changes;
237 }

244 private static string DetermineInterpretation(double correlation)
245 {
246 double absCorrelation = Math.Abs(correlation);
247
248 return absCorrelation switch
249 {
250 < 0.30 => "Weak correlation - Strong vega independence between tenors.",
251 < 0.50 => "Moderate-weak correlation - Good independence for calendar spread.",
252 < 0.70 => "Moderate correlation - Acceptable for calendar spread with monitoring.",
253 < 0.85 => "Strong correlation - Elevated sympathetic collapse risk.",
254 _ => "Very strong correlation - Calendar spread thesis at risk."
255 };
