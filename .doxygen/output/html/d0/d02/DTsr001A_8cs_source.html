81 public static int EncodePriceBars(IReadOnlyList<PriceBar> bars, Span<byte> buffer)
82 {
83 int offset = 0;
84
85 // Version byte
86 buffer[offset++] = FormatVersion;
87
88 // Write count header
89 BinaryPrimitives.WriteInt32LittleEndian(buffer[offset..], bars.Count);
90 offset += 4;
91
92 // Write symbol once (first bar's symbol, 16 bytes fixed)
93 string symbol = bars.Count > 0 ? bars[0].Symbol : string.Empty;
94 WriteFixedString(buffer[offset..], symbol, 16);
95 offset += 16;
96
97 // Write each bar
98 foreach (var bar in bars)
99 {
100 int written = EncodePriceBar(bar, buffer[offset..]);
101 offset += written;
102 }
103
104 return offset;
105 }

110 public static ReadOnlyCollection<PriceBar> DecodePriceBars(ReadOnlySpan<byte> buffer)
111 {
112 int offset = 0;
113
114 // Version check
115 byte version = buffer[offset++];
116 if (version != FormatVersion)
117 {
118 throw new InvalidOperationException($"Unsupported binary format version: {version}");
119 }
120
121 int count = BinaryPrimitives.ReadInt32LittleEndian(buffer[offset..]);
122 offset += 4;
123
124 string symbol = ReadFixedString(buffer[offset..], 16);
125 offset += 16;
126
127 var result = new List<PriceBar>(count);
128
129 for (int i = 0; i < count; i++)
130 {
131 result.Add(DecodePriceBar(buffer[offset..], symbol));
132 offset += 64; // Fixed size per bar (header + fields)
133 }
134
135 return result.AsReadOnly();
136 }

143 public static int EncodeOptionContract(OptionContract contract, Span<byte> buffer)
144 {
145 int offset = 0;
146
147 // Fixed fields first (for direct memory access)
148 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], PLSR001A.ToMantissa(contract.Strike));
149 offset += 8;
150 BinaryPrimitives.WriteInt32LittleEndian(buffer[offset..], (int)(contract.Expiration - DateTime.UnixEpoch).TotalDays);
151 offset += 4;
152 buffer[offset++] = contract.Right == OptionRight.Call ? (byte)0 : (byte)1;
153 // Padding
154 buffer[offset++] = 0;
155 buffer[offset++] = 0;
156 buffer[offset++] = 0;
157 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], PLSR001A.ToMantissa(contract.Bid));
158 offset += 8;
159 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], PLSR001A.ToMantissa(contract.Ask));
160 offset += 8;
161 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], PLSR001A.ToMantissa(contract.Last ?? 0m));
162 offset += 8;
163 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], PLSR001A.ToMantissa(contract.ImpliedVolatility ?? 0m));
164 offset += 8;
165 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], PLSR001A.ToMantissa(contract.Delta ?? 0m));
166 offset += 8;
167 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], PLSR001A.ToMantissa(contract.Gamma ?? 0m));
168 offset += 8;
169 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], PLSR001A.ToMantissa(contract.Theta ?? 0m));
170 offset += 8;
171 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], PLSR001A.ToMantissa(contract.Vega ?? 0m));
172 offset += 8;
173 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.OpenInterest);
174 offset += 8;
175 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.Volume);
176 offset += 8;
177 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], new DateTimeOffset(contract.Timestamp, TimeSpan.Zero).ToUnixTimeMilliseconds());
178 offset += 8;
179
180 // Variable-length strings at end
181 WriteFixedString(buffer[offset..], contract.UnderlyingSymbol, 16);
182 offset += 16;
183 WriteFixedString(buffer[offset..], contract.OptionSymbol, 32);
184 offset += 32;
185
186 return offset;

264 public static int EncodeOptionChainSnapshot(OptionChainSnapshot snapshot, Span<byte> buffer)
265 {
266 int offset = 0;
267
268 // Header: version byte for format compatibility
269 buffer[offset++] = FormatVersion;
270
271 // Timestamp (8 bytes)
272 BinaryPrimitives.WriteInt64LittleEndian(
273 buffer[offset..],
274 new DateTimeOffset(snapshot.Timestamp, TimeSpan.Zero).ToUnixTimeMilliseconds());
275 offset += 8;
276
277 // Spot price (8 bytes)
278 BinaryPrimitives.WriteInt64LittleEndian(
279 buffer[offset..],
280 PLSR001A.ToMantissa(snapshot.SpotPrice));
281 offset += 8;
282
283 // Symbol (16 bytes fixed)
284 WriteFixedString(buffer[offset..], snapshot.Symbol, 16);
285 offset += 16;
286
287 // Contract count (4 bytes)
288 BinaryPrimitives.WriteInt32LittleEndian(
289 buffer[offset..],
290 snapshot.Contracts.Count);
291 offset += 4;
292
293 // Contracts
294 foreach (var contract in snapshot.Contracts)
295 {
296 int written = EncodeOptionContract(contract, buffer[offset..]);
297 offset += written;
298 }
299
300 return offset;
301 }

306 public static OptionChainSnapshot DecodeOptionChainSnapshot(ReadOnlySpan<byte> buffer)
307 {
308 int offset = 0;
309
310 // Version check
311 byte version = buffer[offset++];
312 if (version != FormatVersion)
313 {
314 throw new InvalidOperationException($"Unsupported binary cache version: {version}");
315 }
316
317 // Timestamp
318 long timestampMs = BinaryPrimitives.ReadInt64LittleEndian(buffer[offset..]);
319 offset += 8;
320
321 // Spot price
322 long spotMantissa = BinaryPrimitives.ReadInt64LittleEndian(buffer[offset..]);
323 offset += 8;
324
325 // Symbol
326 string symbol = ReadFixedString(buffer[offset..], 16);
327 offset += 16;
328
329 // Contract count
330 int contractCount = BinaryPrimitives.ReadInt32LittleEndian(buffer[offset..]);
331 offset += 4;
332
333 // Contracts
334 var contracts = new List<OptionContract>(contractCount);
335 for (int i = 0; i < contractCount; i++)
336 {
337 contracts.Add(DecodeOptionContract(buffer[offset..]));
338 offset += OptionContractEncodedSize;
339 }
340
341 return new OptionChainSnapshot
342 {
343 Symbol = symbol,
344 Timestamp = DateTimeOffset.FromUnixTimeMilliseconds(timestampMs).UtcDateTime,
345 SpotPrice = PLSR001A.FromMantissa(spotMantissa),
346 Contracts = contracts
347 };
348 }

352 private static void WriteFixedString(Span<byte> buffer, string? value, int length)
353 {
354 buffer[..length].Clear();
355 if (!string.IsNullOrEmpty(value))
356 {
357 int bytesToWrite = Math.Min(value.Length, length);
358 System.Text.Encoding.ASCII.GetBytes(value.AsSpan(0, bytesToWrite), buffer);
359 }
360 }
361
362 private static string ReadFixedString(ReadOnlySpan<byte> buffer, int length)
363 {
364 int actualLength = buffer[..length].IndexOf((byte)0);
365 if (actualLength < 0)
366 {
367 actualLength = length;
368 }
369
370 return System.Text.Encoding.ASCII.GetString(buffer[..actualLength]);
371 }
