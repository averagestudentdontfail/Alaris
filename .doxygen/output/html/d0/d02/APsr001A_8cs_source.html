28 public static int EncodeSessionMetadata(APmd001A session, Span<byte> buffer)
29 {
30 ArgumentNullException.ThrowIfNull(session);
31
32 int offset = 0;
33
34 // Version byte
35 buffer[offset++] = FormatVersion;
36
37 // SessionId (fixed 64 chars)
38 WriteFixedString(buffer[offset..], session.SessionId, 64);
39 offset += 64;
40
41 // StartDate as epoch days
42 BinaryPrimitives.WriteInt32LittleEndian(buffer[offset..], (int)(session.StartDate - DateTime.UnixEpoch).TotalDays);
43 offset += 4;
44
45 // EndDate as epoch days
46 BinaryPrimitives.WriteInt32LittleEndian(buffer[offset..], (int)(session.EndDate - DateTime.UnixEpoch).TotalDays);
47 offset += 4;
48
49 // CreatedAt as epoch ms
50 BinaryPrimitives.WriteInt64LittleEndian(
51 buffer[offset..],
52 new DateTimeOffset(session.CreatedAt, TimeSpan.Zero).ToUnixTimeMilliseconds());
53 offset += 8;
54
55 // UpdatedAt as epoch ms
56 BinaryPrimitives.WriteInt64LittleEndian(
57 buffer[offset..],
58 new DateTimeOffset(session.UpdatedAt, TimeSpan.Zero).ToUnixTimeMilliseconds());
59 offset += 8;
60
61 // Status (1 byte enum)
62 buffer[offset++] = (byte)session.Status;
63
64 // SessionPath (fixed 256 chars)
65 WriteFixedString(buffer[offset..], session.SessionPath, 256);
66 offset += 256;
67
68 // Symbol count + symbols
69 BinaryPrimitives.WriteInt32LittleEndian(buffer[offset..], session.Symbols.Count);
70 offset += 4;
71 foreach (string symbol in session.Symbols)
72 {
73 WriteFixedString(buffer[offset..], symbol, 16);
74 offset += 16;
75 }
76
77 return offset;
78 }

94 public static APmd001A DecodeSessionMetadata(ReadOnlySpan<byte> buffer)
95 {
96 int offset = 0;
97
98 // Version check
99 byte version = buffer[offset++];
100 if (version != FormatVersion)
101 {
102 throw new InvalidOperationException($"Unsupported session format version: {version}");
103 }
104
105 // SessionId
106 string sessionId = ReadFixedString(buffer[offset..], 64);
107 offset += 64;
108
109 // Dates
110 int startDays = BinaryPrimitives.ReadInt32LittleEndian(buffer[offset..]);
111 offset += 4;
112 int endDays = BinaryPrimitives.ReadInt32LittleEndian(buffer[offset..]);
113 offset += 4;
114
115 // Timestamps
116 long createdMs = BinaryPrimitives.ReadInt64LittleEndian(buffer[offset..]);
117 offset += 8;
118 long updatedMs = BinaryPrimitives.ReadInt64LittleEndian(buffer[offset..]);
119 offset += 8;
120
121 // Status
122 SessionStatus status = (SessionStatus)buffer[offset++];
123
124 // SessionPath
125 string sessionPath = ReadFixedString(buffer[offset..], 256);
126 offset += 256;
127
128 // Symbols
129 int symbolCount = BinaryPrimitives.ReadInt32LittleEndian(buffer[offset..]);
130 offset += 4;
131 List<string> symbols = new(symbolCount);
132 for (int i = 0; i < symbolCount; i++)
133 {
134 symbols.Add(ReadFixedString(buffer[offset..], 16));
135 offset += 16;
136 }
137
138 return new APmd001A
139 {
140 SessionId = sessionId,
141 StartDate = DateTime.UnixEpoch.AddDays(startDays),
142 EndDate = DateTime.UnixEpoch.AddDays(endDays),
143 CreatedAt = DateTimeOffset.FromUnixTimeMilliseconds(createdMs).UtcDateTime,
144 UpdatedAt = DateTimeOffset.FromUnixTimeMilliseconds(updatedMs).UtcDateTime,
145 Status = status,
146 SessionPath = sessionPath,
147 Symbols = symbols
148 };
149 }

152 {
