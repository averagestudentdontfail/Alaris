118 [Fact]
119 public void CalculateStickyDeltaIVChange_SpotUp_ReturnsPositiveChange()
120 {
121 // Arrange: Under sticky-delta with negative skew, when spot moves up,
122 // the original strike becomes OTM put-like (lower strike relative to spot)
123 // which has HIGHER IV due to negative skew.
124 // Formula: -(-0.10) * (+0.05) / (0.30 * sqrt(T)) > 0
125 STIV007A parameters = CreateSampleParameters(atmIV: 0.30, skew: -0.10, spot: 100.0);
126
127 // Act: Spot moved from 100 to 105 (5% up)
128 double ivChange = _interpolator.CalculateStickyDeltaIVChange(
129 parameters, originalStrike: 100.0, originalSpot: 100.0, currentSpot: 105.0);
130
131 // Assert: With negative skew, spot up → original strike is now OTM put → higher IV
132 ivChange.Should().BeGreaterThan(0);
133 }

198 private static List<STIV008A> CreateSampleChain(double spotPrice, double atmIV)
199 {
200 List<STIV008A> chain = new List<STIV008A>();
201
202 // Create strikes from 85% to 115% of spot with typical equity skew
203 double[] strikePcts = { 0.85, 0.90, 0.95, 1.00, 1.05, 1.10, 1.15 };
204
205 foreach (double pct in strikePcts)
206 {
207 double strike = spotPrice * pct;
208 // Equity skew: lower strikes have higher IV
209 double skewAdjustment = -0.10 * (1.0 - pct);
210 double iv = atmIV * (1.0 + skewAdjustment);
211
212 chain.Add(new STIV008A
213 {
214 Strike = strike,
215 ImpliedVolatility = iv,
216 IsCall = pct >= 1.0
217 });
218 }
219
220 return chain;
221 }
