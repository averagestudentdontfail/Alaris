63 public async Task RunAsync(CancellationToken cancellationToken = default)

127 private Panel BuildEquityPanel()
128 {
129 lock (_lock)
130 {
131 if (_equityHistory.Count == 0)

139 // Build simple text-based chart using bar characters
140 decimal maxEquity = _equityHistory.Max(e => e.Equity);
141 decimal minEquity = _equityHistory.Min(e => e.Equity);
142 decimal range = maxEquity - minEquity;
143
144 if (range == 0) range = 1;
145
146 const int chartHeight = 8;
147 var chartLines = new string[chartHeight];
148
149 // Sample points for display
150 var points = _equityHistory.TakeLast(50).ToList();
151
152 for (int row = 0; row < chartHeight; row++)
153 {
154 decimal threshold = maxEquity - (range * row / (chartHeight - 1));
155 var line = new System.Text.StringBuilder();
156
157 foreach (var point in points)
158 {
159 char c = point.Equity >= threshold ? '█' : ' ';
160 line.Append(c);
161 }
162
163 chartLines[row] = line.ToString();
164 }
165
166 decimal currentEquity = _equityHistory.Last().Equity;
167 decimal startEquity = _equityHistory.First().Equity;
168 decimal pnl = currentEquity - startEquity;
169 string pnlColor = pnl >= 0 ? "green" : "red";
170
171 var content = new Rows(
172 new Text(string.Join("\n", chartLines)),
173 new Markup($"\n[bold]Current:[/] ${currentEquity:N2} ([{pnlColor}]{pnl:+#,##0.00;-#,##0.00}[/])"));
174
175 return new Panel(content)
176 .Header("[bold]Equity Curve[/]")
177 .Border(BoxBorder.Rounded)
178 .BorderColor(Color.Blue);
179 }

182 private Panel BuildMetricsPanel()
183 {
184 lock (_lock)
185 {
186 if (_equityHistory.Count < 2)
187 {
188 return new Panel(new Markup("[grey]Calculating...[/]"))
189 .Header("[bold]Performance[/]")
190 .Border(BoxBorder.Rounded)
191 .BorderColor(Color.Grey);
192 }
193
194 decimal startEquity = _equityHistory.First().Equity;
195 decimal currentEquity = _equityHistory.Last().Equity;
196 decimal returnPct = startEquity > 0 ? (currentEquity - startEquity) / startEquity * 100 : 0;
197
198 // Calculate drawdown
199 decimal peak = startEquity;
200 decimal maxDrawdown = 0;
201 foreach (var e in _equityHistory)
202 {
203 if (e.Equity > peak) peak = e.Equity;
204 decimal dd = (peak - e.Equity) / peak * 100;
205 if (dd > maxDrawdown) maxDrawdown = dd;
206 }
207
208 var table = new Table()
209 .Border(TableBorder.None)
210 .HideHeaders()
211 .AddColumn("Metric")
212 .AddColumn("Value");
213
214 string returnColor = returnPct >= 0 ? "green" : "red";
215 table.AddRow("Return", $"[{returnColor}]{returnPct:+0.00;-0.00}%[/]");
216 table.AddRow("Max DD", $"[red]{maxDrawdown:F2}%[/]");
217 table.AddRow("Trades", $"{_trades.Count}");
218
219 int winners = _trades.Count(t => t.PnL > 0);
220 int losers = _trades.Count(t => t.PnL < 0);
221 double winRate = _trades.Count > 0 ? (double)winners / _trades.Count * 100 : 0;
222 table.AddRow("Win Rate", $"{winRate:F0}%");
223
224 return new Panel(table)
225 .Header("[bold]Performance[/]")
226 .Border(BoxBorder.Rounded)
227 .BorderColor(Color.Green);
228 }
229 }
