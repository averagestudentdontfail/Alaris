104 private IEnumerable<Symbol> SelectSymbols(QCAlgorithm algorithm, DateTime dateTime)
105 {
106 var date = dateTime.Date;
107
108 // Load universe from file if needed
109 if (_lastUniverseDate != date)
110 {
111 LoadUniverseForDate(algorithm, date);
112 LoadEarningsSymbols(algorithm, date);
113 _lastUniverseDate = date;
114 }
115
116 if (_cachedUniverse.Count == 0)
117 {
118 algorithm.Debug($"STUN001B: No universe data for {date:yyyy-MM-dd}");
119 return Enumerable.Empty<Symbol>();
120 }
121
122 // Filter by earnings
123 var withEarnings = _cachedUniverse
124 .Where(u => _cachedEarningsSymbols.Contains(u.Ticker))
125 .Take(_maxSymbols)
126 .ToList();
127
128 _logger?.LogInformation(
129 "Universe selection for {Date}: {TotalCount} stocks → {EarningsCount} with earnings",
130 date, _cachedUniverse.Count, withEarnings.Count);
131
132 algorithm.Debug($"STUN001B: {date:yyyy-MM-dd} - {_cachedUniverse.Count} stocks → {withEarnings.Count} with earnings");
133
134 // Convert to LEAN symbols
135 return withEarnings
136 .Select(u => algorithm.AddEquity(u.Ticker, Resolution.Daily).Symbol);
137 }

147 private void LoadUniverseForDate(QCAlgorithm algorithm, DateTime date)
148 {
149 _cachedUniverse.Clear();
150
151 // Check for static session symbols (Backtest Mode override)
152 var sessionSymbols = Environment.GetEnvironmentVariable("ALARIS_SESSION_SYMBOLS");
153 if (!string.IsNullOrEmpty(sessionSymbols))
154 {
155 algorithm.Debug("STUN001B: Using static session universe from environment");
156 var tickers = sessionSymbols.Split(Separators, StringSplitOptions.RemoveEmptyEntries | StringSplitOptions.TrimEntries);
157
158 foreach (var ticker in tickers)
159 {
160 _cachedUniverse.Add(new UniverseEntry
161 {
162 SymbolId = ticker,
163 Ticker = ticker.ToUpperInvariant(),
164 Close = 1000m, // Dummy high value to pass filter
165 Volume = 1_000_000,
166 DollarVolume = 100_000_000m // Dummy high value to pass filter
167 });
168 }
169
170 // Session symbols are explicit; filters are not applied in this path.
171 return;
172 }
173
174 var filePath = System.IO.Path.Combine(_dataPath, "equity", "usa", "fundamental", "coarse", $"{date:yyyyMMdd}.csv");
175
176 if (!System.IO.File.Exists(filePath))
177 {
178 // Try to find the most recent available file
179 var universeDir = System.IO.Path.Combine(_dataPath, "equity", "usa", "fundamental", "coarse");
180 if (!System.IO.Directory.Exists(universeDir))
181 {
182 algorithm.Error($"STUN001B: Universe directory not found: {universeDir}");
183 algorithm.Error("STUN001B: Run 'alaris universe generate' to create universe files");
184 return;
185 }
186
187 var files = System.IO.Directory.GetFiles(universeDir, "*.csv")
188 .OrderByDescending(f => f)
189 .ToList();
190
191 if (files.Count == 0)
192 {
193 algorithm.Error("STUN001B: No universe files found. Run 'alaris universe generate'");
194 return;
195 }
196
197 // Find the most recent file before or on the date
198 var targetDate = date.ToString("yyyyMMdd");
199 filePath = files.FirstOrDefault(f => string.Compare(System.IO.Path.GetFileNameWithoutExtension(f), targetDate, StringComparison.Ordinal) <= 0)
200 ?? files.Last();
201
202 algorithm.Debug($"STUN001B: Using universe file: {System.IO.Path.GetFileName(filePath)}");
203 }
204
205 try
206 {
207 var lines = System.IO.File.ReadAllLines(filePath);
208 foreach (var line in lines)
209 {
210 var parts = line.Split(',');
211 if (parts.Length >= 5)
212 {
213 _cachedUniverse.Add(new UniverseEntry
214 {
215 SymbolId = parts[0],
216 Ticker = parts[1].ToUpperInvariant(),
217 Close = decimal.TryParse(parts[2], NumberStyles.Number, CultureInfo.InvariantCulture, out var c) ? c : 0,
218 Volume = double.TryParse(parts[3], NumberStyles.Float, CultureInfo.InvariantCulture, out var v) ? v : 0,
219 DollarVolume = decimal.TryParse(parts[4], NumberStyles.Number, CultureInfo.InvariantCulture, out var dv) ? dv : 0
220 });
221 }
222 }
223
224 // Apply filters
225 _cachedUniverse = _cachedUniverse
226 .Where(u => u.Close >= _minimumPrice)
227 .Where(u => u.DollarVolume >= _minimumDollarVolume)
228 .OrderByDescending(u => u.DollarVolume)
229 .ToList();
230
231 algorithm.Debug($"STUN001B: Loaded {_cachedUniverse.Count} stocks from {System.IO.Path.GetFileName(filePath)}");
232 }
233 catch (Exception ex)
234 {
235 algorithm.Error($"STUN001B: Error reading universe file: {ex.Message}");
236 }
237 }

244 private void LoadEarningsSymbols(QCAlgorithm algorithm, DateTime date)
245 {
246 _cachedEarningsSymbols.Clear();
247
248 try
249 {
250 // Target earnings window (5-7 days ahead of simulation date)
251 var startDate = date.AddDays(_daysBeforeEarningsMin);
252 var endDate = date.AddDays(_daysBeforeEarningsMax);
253
254 algorithm.Debug($"STUN001B: Checking for earnings between {startDate:MM-dd} and {endDate:MM-dd}");
255
256 // Use batch query instead of per-symbol loop (prevents 100+ API calls)
257 var symbolsWithEarnings = _earningsProvider.GetSymbolsWithEarningsAsync(
258 startDate, endDate, CancellationToken.None)
259 .GetAwaiter().GetResult();
260
261 // Filter to symbols in our universe
262 var universeSet = new HashSet<string>(
263 _cachedUniverse.Select(u => u.Ticker),
264 StringComparer.OrdinalIgnoreCase);
265
266 foreach (var symbol in symbolsWithEarnings)
267 {
268 if (universeSet.Contains(symbol))
269 {
270 _cachedEarningsSymbols.Add(symbol.ToUpperInvariant());
271 }
272 }
273
274 algorithm.Debug($"STUN001B: Found {_cachedEarningsSymbols.Count} symbols with earnings in {startDate:MM-dd} to {endDate:MM-dd}");
275 }
276 catch (Exception ex)
277 {
278 algorithm.Error($"STUN001B: Error querying earnings calendar: {ex.Message}");
279 }
