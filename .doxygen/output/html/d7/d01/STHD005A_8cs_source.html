82 public STHD006A Validate(
83 Signal signal,
84 STCS002A frontLegParams,
85 STCS002A backLegParams,
86 IReadOnlyList<double> frontIVHistory,
87 IReadOnlyList<double> backIVHistory,
88 int backMonthVolume,
89 int backMonthOpenInterest,
90 double spotPrice,
91 double strikePrice,
92 SpreadGreeks spreadGreeks,
93 int daysToEarnings)
94 {
95 ArgumentNullException.ThrowIfNull(signal);
96 ArgumentNullException.ThrowIfNull(frontLegParams);
97 ArgumentNullException.ThrowIfNull(backLegParams);
98 ArgumentNullException.ThrowIfNull(frontIVHistory);
99 ArgumentNullException.ThrowIfNull(backIVHistory);
100 ArgumentNullException.ThrowIfNull(spreadGreeks);
101
102 var checks = new List<ValidationCheck>();
103
104 try
105 {
106 // 1. Execution cost validation
107 STCS007A costResult = _costValidator.Validate(signal, frontLegParams, backLegParams);
108 checks.Add(new ValidationCheck
109 {
110 Name = "Execution Cost Survival",
111 Passed = costResult.OverallPass,
112 Detail = costResult.Summary
113 });
114
115 // 2. Vega correlation analysis
116 STHD002A vegaResult = _vegaAnalyser.AnalyseFromLevels(signal.Symbol, frontIVHistory, backIVHistory);
117 checks.Add(new ValidationCheck
118 {
119 Name = "Vega Independence",
120 Passed = vegaResult.PassesFilter,
121 Detail = vegaResult.Summary
122 });
123
124 // 3. Liquidity validation
125 STCS009A liquidityResult = _liquidityValidator.Validate(
126 signal.Symbol,
127 frontLegParams.Contracts,
128 backMonthVolume,
129 backMonthOpenInterest);
130 checks.Add(new ValidationCheck
131 {
132 Name = "Liquidity Assurance",
133 Passed = liquidityResult.DefinedRiskAssured,
134 Detail = liquidityResult.Summary
135 });
136
137 // 4. Gamma risk assessment
138 STHD004A gammaResult = _gammaManager.Evaluate(
139 signal.Symbol,
140 spreadGreeks.Delta,
141 spreadGreeks.Gamma,
142 spreadGreeks.Vega,
143 spreadGreeks.Theta,
144 spotPrice,
145 strikePrice,
146 daysToEarnings);
147 checks.Add(new ValidationCheck
148 {
149 Name = "Gamma Risk",
150 Passed = gammaResult.RecommendedAction == RehedgeAction.Hold,
151 Detail = gammaResult.Summary
152 });
153
154 // Determine overall result
155 int passedCount = checks.Count(c => c.Passed);
156 bool overallPass = checks.All(c => c.Passed);
157
158 // Determine recommended contracts (may be adjusted by liquidity)
159 int recommendedContracts = liquidityResult.RecommendedContracts;
160
161 // Compute adjusted debit
162 double adjustedDebit = costResult.SpreadCost.ExecutionDebit;
163
164 var result = new STHD006A
165 {
166 BaseSignal = signal,
167 Checks = checks,
168 OverallPass = overallPass,
169 AdjustedDebit = adjustedDebit,
170 RecommendedContracts = recommendedContracts,
171 ProductionReady = overallPass && signal.Strength == SignalStrength.Recommended,
172 CostValidation = costResult,
173 VegaCorrelation = vegaResult,
174 LiquidityValidation = liquidityResult,
175 GammaAssessment = gammaResult
176 };
177
178 SafeLog(() => LogValidationComplete(
179 _logger!,
180 signal.Symbol,
181 overallPass,
182 passedCount,
183 checks.Count,
184 null));
185
186 return result;
187 }
188 catch (InvalidOperationException ex)
189 {
190 SafeLog(() => LogValidationError(_logger!, signal.Symbol, ex.Message, ex));
191
192 // Return failed result with error information
193 checks.Add(new ValidationCheck
194 {
195 Name = "Validation Error",
196 Passed = false,
197 Detail = ex.Message
198 });
199
200 return new STHD006A
201 {
202 BaseSignal = signal,
203 Checks = checks,
204 OverallPass = false,
205 AdjustedDebit = 0,
206 RecommendedContracts = 0,
207 ProductionReady = false,
208 CostValidation = null,
209 VegaCorrelation = null,
210 LiquidityValidation = null,
211 GammaAssessment = null
212 };
213 }
214 }
