60 public double CalculateEarlyExerciseRisk(
61 double dividendAmount,
62 double strike,
63 double riskFreeRate,
64 double timeFromExToExpiry)
65 {
66 ArgumentOutOfRangeException.ThrowIfNegative(dividendAmount);
67 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(strike);
68 ArgumentOutOfRangeException.ThrowIfNegative(timeFromExToExpiry);
69
70 if (dividendAmount <= 0)
71 {
72 return 0.0;
73 }
74
75 // Merton threshold: K × (1 - e^(-r×τ))
76 double threshold = strike * (1.0 - Math.Exp(-riskFreeRate * timeFromExToExpiry));
77
78 // Early exercise is optimal if D > threshold
79 if (dividendAmount > threshold)
80 {
81 // Risk proportional to how much dividend exceeds threshold
82 double excess = (dividendAmount - threshold) / dividendAmount;
83 return Math.Min(1.0, 0.5 + (0.5 * excess));
84 }
85
86 // Partial risk if close to threshold
87 double ratio = dividendAmount / Math.Max(threshold, 0.001);
88 return Math.Min(0.5, ratio * 0.5);
89 }

101 public STDD003A EvaluateDividendRisk(
102 IReadOnlyList<STDD002A> dividendSchedule,
103 double spotPrice,
104 double strike,
105 DateTime shortLegExpiry,
106 DateTime earningsDate,
107 double riskFreeRate)
108 {
109 ArgumentNullException.ThrowIfNull(dividendSchedule);
110 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(spotPrice);
111 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(strike);
112
113 // Define window: 7 days before earnings to short leg expiry
114 DateTime windowStart = earningsDate.AddDays(-7);
115 DateTime windowEnd = shortLegExpiry;
116
117 IReadOnlyList<STDD002A> exDatesInWindow = FindExDatesInWindow(
118 dividendSchedule, windowStart, windowEnd);
119
120 if (exDatesInWindow.Count == 0)
121 {
122 return new STDD003A
123 {
124 HasExDateInWindow = false,
125 ExDatesFound = exDatesInWindow,
126 MaxEarlyExerciseRisk = 0.0,
127 RiskLevel = STDD004A.None,
128 SignalAdjustment = 1.0,
129 Rationale = "No ex-dates in earnings window"
130 };
131 }
132
133 // Calculate early exercise risk for each ex-date
134 double maxRisk = 0.0;
135 foreach (STDD002A dividend in exDatesInWindow)
136 {
137 double timeToExpiry = (shortLegExpiry - dividend.ExDate).TotalDays / 365.0;
138 double risk = CalculateEarlyExerciseRisk(
139 dividend.Amount, strike, riskFreeRate, timeToExpiry);
140
141 SafeLog(() => LogExDateDetected(_logger!, dividend.ExDate, dividend.Amount, risk, null));
142
143 maxRisk = Math.Max(maxRisk, risk);
144 }
145
146 // Determine risk level and signal adjustment
147 STDD004A riskLevel;
148 double signalAdjustment;
149 if (maxRisk > 0.70)
150 {
151 riskLevel = STDD004A.High;
152 signalAdjustment = 0.0; // Avoid trade
153 }
154 else if (maxRisk > 0.40)
155 {
156 riskLevel = STDD004A.Elevated;
157 signalAdjustment = 0.50; // Reduce exposure
158 }
159 else if (maxRisk > 0.20)
160 {
161 riskLevel = STDD004A.Moderate;
162 signalAdjustment = 0.75;
163 }
164 else
165 {
166 riskLevel = STDD004A.Low;
167 signalAdjustment = 1.0;
168 }
169
170 return new STDD003A
171 {
172 HasExDateInWindow = true,
173 ExDatesFound = exDatesInWindow,
174 MaxEarlyExerciseRisk = maxRisk,
175 RiskLevel = riskLevel,
176 SignalAdjustment = signalAdjustment,
177 Rationale = riskLevel switch
178 {
179 STDD004A.High => $"High early exercise risk ({maxRisk:P0}) - avoid trade",
180 STDD004A.Elevated => $"Elevated early exercise risk ({maxRisk:P0}) - reduce position",
181 STDD004A.Moderate => $"Moderate early exercise risk ({maxRisk:P0})",
182 _ => "Low early exercise risk from dividends"
183 }
184 };
185 }
