448 [Fact]
449 public async Task EVIF001A_ConcurrentAppends_MaintainsSequenceIntegrity()
450 {
451 // Arrange
452 var store = new EVIF001A();
453 int eventCount = 100;
454 var tasks = new List<Task<EVCR003A>>();
455
456 // Act - Concurrent appends
457 for (int i = 0; i < eventCount; i++)
458 {
459 int index = i; // Capture for closure
460 tasks.Add(Task.Run(async () =>
461 {
462 var domainEvent = new TestDomainEvent($"Event-{index}");
463 return await store.AppendAsync(domainEvent);
464 }));
465 }
466
467 var envelopes = await Task.WhenAll(tasks);
468
469 // Assert - All sequence numbers should be unique
470 var sequenceNumbers = envelopes.Select(e => e.SequenceNumber).ToList();
471 sequenceNumbers.Distinct().Count().Should().Be(eventCount,
472 "All sequence numbers must be unique even with concurrent appends");
473
474 // Assert - Sequence numbers should span 1 to eventCount
475 sequenceNumbers.Min().Should().Be(1);
476 sequenceNumbers.Max().Should().Be(eventCount);
477 }
