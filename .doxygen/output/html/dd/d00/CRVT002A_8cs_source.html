95 [MethodImpl(MethodImplOptions.AggressiveInlining)]
96 public static Vector256<double> VectorExp256(Vector256<double> x)
97 {
98 if (!Avx2.IsSupported)
99 {
100 return VectorExp256Fallback(x);
101 }
102
103 // Clamp to avoid overflow/underflow
104 x = Avx.Max(x, ExpMin);
105 x = Avx.Min(x, ExpMax);
106
107 // Range reduction: exp(x) = 2^k * exp(r) where r = x - k*ln(2)
108 // k = round(x / ln(2))
109 Vector256<double> k = Avx.RoundToNearestInteger(Avx.Multiply(x, ExpInvLn2));
110
111 // r = x - k*ln2_hi - k*ln2_lo (for precision)
112 Vector256<double> r = Avx.Subtract(x, Avx.Multiply(k, ExpLn2Hi));
113 r = Avx.Subtract(r, Avx.Multiply(k, ExpLn2Lo));
114
115 // Polynomial approximation of exp(r) - 1 for r in [-ln2/2, ln2/2]
116 // Using Horner's method: p = c1 + r*(c2 + r*(c3 + r*(c4 + r*(c5 + r*c6))))
117 Vector256<double> r2 = Avx.Multiply(r, r);
118
119 Vector256<double> p = ExpC6;
120 p = Fma.IsSupported
121 ? Fma.MultiplyAdd(p, r, ExpC5)
122 : Avx.Add(Avx.Multiply(p, r), ExpC5);
123 p = Fma.IsSupported
124 ? Fma.MultiplyAdd(p, r, ExpC4)
125 : Avx.Add(Avx.Multiply(p, r), ExpC4);
126 p = Fma.IsSupported
127 ? Fma.MultiplyAdd(p, r, ExpC3)
128 : Avx.Add(Avx.Multiply(p, r), ExpC3);
129 p = Fma.IsSupported
130 ? Fma.MultiplyAdd(p, r, ExpC2)
131 : Avx.Add(Avx.Multiply(p, r), ExpC2);
132 p = Fma.IsSupported
133 ? Fma.MultiplyAdd(p, r, ExpC1)
134 : Avx.Add(Avx.Multiply(p, r), ExpC1);
135
136 // exp(r) ≈ 1 + r*p
137 p = Fma.IsSupported
138 ? Fma.MultiplyAdd(r, p, One)
139 : Avx.Add(Avx.Multiply(r, p), One);
140
141 // Scale by 2^k using bit manipulation
142 // 2^k = reinterpret((1023 + k) << 52) as double
143 Vector256<long> kLong = Avx2.ConvertToVector256Int64(Avx.ConvertToVector128Int32WithTruncation(k));
144 Vector256<long> bias = Vector256.Create(1023L);
145 Vector256<long> exp = Avx2.Add(kLong, bias);
146 exp = Avx2.ShiftLeftLogical(exp, 52);
147 Vector256<double> scale = exp.AsDouble();
148
149 return Avx.Multiply(p, scale);
150 }

161 [MethodImpl(MethodImplOptions.AggressiveInlining)]
162 public static Vector256<double> VectorLog256(Vector256<double> x)
163 {
164 if (!Avx2.IsSupported)
165 {
166 return VectorLog256Fallback(x);
167 }
168
169 // Extract exponent and mantissa: x = 2^e * m where m in [1, 2)
170 Vector256<long> xBits = x.AsInt64();
171 Vector256<long> expMask = Vector256.Create(0x7FF0000000000000L);
172 Vector256<long> mantMask = Vector256.Create(0x000FFFFFFFFFFFFFL);
173 Vector256<long> expBias = Vector256.Create(1023L);
174
175 // Get exponent
176 Vector256<long> expBits = Avx2.And(xBits, expMask);
177 expBits = Avx2.ShiftRightLogical(expBits, 52);
178 Vector256<double> e = Avx.Subtract(
179 Avx.ConvertToVector256Double(expBits.AsInt32().GetLower()),
180 Vector256.Create(1023.0));
181
182 // Get mantissa in [1, 2)
183 Vector256<long> mantBits = Avx2.Or(
184 Avx2.And(xBits, mantMask),
185 Vector256.Create(0x3FF0000000000000L));
186 Vector256<double> m = mantBits.AsDouble();
187
188 // Reduce to [sqrt(2)/2, sqrt(2)] for better polynomial convergence
189 Vector256<double> needsAdjust = Avx.Compare(m, LogSqrt2, FloatComparisonMode.OrderedGreaterThanSignaling);
190 m = Avx.BlendVariable(Avx.Multiply(m, Half), m, needsAdjust);
191 e = Avx.BlendVariable(Avx.Add(e, One), e, needsAdjust);
192
193 // Polynomial approximation of log(1 + f) where f = m - 1
194 Vector256<double> f = Avx.Subtract(m, One);
195 Vector256<double> f2 = Avx.Multiply(f, f);
196
197 // Simple polynomial: log(1+f) ≈ f - f²/2 + f³/3 - f⁴/4 + ...
198 Vector256<double> logM = f;
199 logM = Avx.Subtract(logM, Avx.Multiply(f2, Half));
200 logM = Avx.Add(logM, Avx.Multiply(Avx.Multiply(f2, f), Vector256.Create(1.0 / 3.0)));
201 logM = Avx.Subtract(logM, Avx.Multiply(Avx.Multiply(f2, f2), Vector256.Create(0.25)));
202
203 // log(x) = e * ln(2) + log(m)
204 return Fma.IsSupported
205 ? Fma.MultiplyAdd(e, LogLn2, logM)
206 : Avx.Add(Avx.Multiply(e, LogLn2), logM);
207 }

210 // VectorErf256 - AVX2 Vectorized Error Function
211 // =========================================================================

222 [MethodImpl(MethodImplOptions.AggressiveInlining)]
223 public static Vector256<double> VectorErf256(Vector256<double> x)
224 {
225 if (!Avx2.IsSupported)
226 {
227 return VectorErf256Fallback(x);
228 }
229
230 // Save sign and work with |x|
231 Vector256<double> sign = Avx.And(x, Vector256.Create(-0.0).AsDouble());
232 Vector256<double> absX = Avx.AndNot(Vector256.Create(-0.0).AsDouble(), x);
233
234 // t = 1 / (1 + p*|x|)
235 Vector256<double> t = Avx.Add(One, Avx.Multiply(ErfP, absX));
236 t = Avx.Divide(One, t);
237
238 // Polynomial: a1*t + a2*t² + a3*t³ + a4*t⁴ + a5*t⁵
239 // Using Horner: t * (a1 + t*(a2 + t*(a3 + t*(a4 + t*a5))))
240 Vector256<double> poly = ErfA5;
241 poly = Fma.IsSupported
242 ? Fma.MultiplyAdd(poly, t, ErfA4)
243 : Avx.Add(Avx.Multiply(poly, t), ErfA4);
244 poly = Fma.IsSupported
245 ? Fma.MultiplyAdd(poly, t, ErfA3)
246 : Avx.Add(Avx.Multiply(poly, t), ErfA3);
247 poly = Fma.IsSupported
248 ? Fma.MultiplyAdd(poly, t, ErfA2)
249 : Avx.Add(Avx.Multiply(poly, t), ErfA2);
250 poly = Fma.IsSupported
251 ? Fma.MultiplyAdd(poly, t, ErfA1)
252 : Avx.Add(Avx.Multiply(poly, t), ErfA1);
253 poly = Avx.Multiply(poly, t);
254
255 // erf(|x|) = 1 - poly * exp(-x²)
256 Vector256<double> negX2 = Avx.Multiply(Avx.Multiply(absX, absX), Vector256.Create(-1.0));
257 Vector256<double> expNegX2 = VectorExp256(negX2);
258 Vector256<double> result = Avx.Subtract(One, Avx.Multiply(poly, expNegX2));
259
260 // Apply sign
261 return Avx.Xor(result, sign);
262 }
