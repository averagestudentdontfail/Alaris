67 public static double Erf(double x)
68 {
69 // Save sign
70 int sign = x >= 0 ? 1 : -1;
71 x = System.Math.Abs(x);
72
73 // Abramowitz & Stegun constants
74 const double a1 = 0.254829592;
75 const double a2 = -0.284496736;
76 const double a3 = 1.421413741;
77 const double a4 = -1.453152027;
78 const double a5 = 1.061405429;
79 const double p = 0.3275911;
80
81 double t = 1.0 / (1.0 + (p * x));
82 double t2 = t * t;
83 double t3 = t2 * t;
84 double t4 = t3 * t;
85 double t5 = t4 * t;
86
87 // Horner's form for the polynomial
88 double poly = (a1 * t) + (a2 * t2) + (a3 * t3) + (a4 * t4) + (a5 * t5);
89 double y = 1.0 - (poly * System.Math.Exp(-x * x));
90
91 return sign * y;
92 }

343 public static double BSImpliedVolatility(
344 double S, double K, double tau, double r, double q,
345 double marketPrice, bool isCall,
346 double tolerance = 1e-7,
347 int maxIterations = 10)
348 {
349 // Validate inputs
350 if (marketPrice <= 0 || S <= 0 || K <= 0 || tau <= 0)
351 {
352 return double.NaN;
353 }
354
355 // Discount factors
356 double discountDiv = System.Math.Exp(-q * tau);
357 double discountRate = System.Math.Exp(-r * tau);
358 double fwd = S * discountDiv;
359 double strike = K * discountRate;
360 double maxPrice = isCall ? fwd : strike;
361
362 // Check intrinsic value bounds
363 double intrinsic = isCall
364 ? System.Math.Max(fwd - strike, 0)
365 : System.Math.Max(strike - fwd, 0);
366
367 if (marketPrice < intrinsic - tolerance)
368 {
369 return double.NaN; // Price below intrinsic value - no valid IV
370 }
371
372 if (marketPrice > maxPrice + tolerance)
373 {
374 return double.NaN; // Price above theoretical bound - no valid IV
375 }
376
377 // Initial guess: Corrado-Miller (1996) approximation
378 double sigma = CorradoMillerGuess(S, K, tau, r, q, marketPrice, isCall);
379 sigma = ClampVolatility(sigma);
380
381 // 3rd-order Householder refinement (cubic convergence)
382 // σ_{n+1} = σ_n - f/f' × (1 + f×f''/(2×f'²))
383 // Typically converges in 2 iterations vs 3-5 for Newton-Raphson
384 for (int i = 0; i < maxIterations; i++)
385 {
386 double price = BSPrice(S, K, tau, sigma, r, q, isCall);
387 double vega = BSVega(S, K, tau, sigma, r, q);
388
389 double f = price - marketPrice;
390
391 if (System.Math.Abs(f) < tolerance)
392 {
393 return sigma;
394 }
395
396 if (vega < MachineEpsilon)
397 {
398 // Vega too small - use Brent fallback
399 break;
400 }
401
402 // Compute Householder(3) correction
403 double volga = BSVolga(S, K, tau, sigma, r, q);
404 double vegaSq = vega * vega;
405
406 // Householder(3): σ - f/f' × (1 + f×f''/(2×f'²))
407 double householderCorrection = 1.0 + (f * volga / (2.0 * vegaSq));
408 double step = f / vega * householderCorrection;
409 double newSigma = sigma - step;
410
411 // Check for divergence
412 if (newSigma <= 0 || newSigma > MaxVolatility || double.IsNaN(newSigma))
413 {
414 break; // Fall through to Brent
415 }
416
417 sigma = newSigma;
418 }
419
420 // Brent's method fallback for guaranteed convergence
421 return BSImpliedVolatilityBrent(S, K, tau, r, q, marketPrice, isCall, tolerance);
422 }

433 private static double CorradoMillerGuess(
434 double S, double K, double tau, double r, double q,
435 double marketPrice, bool isCall)
436 {
437 double discountDiv = System.Math.Exp(-q * tau);
438 double discountRate = System.Math.Exp(-r * tau);
439 double F = S * discountDiv; // Forward price
440 double Kd = K * discountRate; // Discounted strike
441
442 // Convert put to call price via put-call parity for uniform treatment
443 double C = isCall ? marketPrice : marketPrice + F - Kd;
444
445 // Corrado-Miller formula
446 double diff = F - Kd;
447 double avg = (F + Kd) / 2;
448 double sqrtTau = System.Math.Sqrt(tau);
449
450 // Handle at-the-money specially
451 if (System.Math.Abs(diff) < MachineEpsilon * avg)
452 {
453 // ATM: σ ≈ C × √(2π) / (S × √τ)
454 return C * SqrtTwoPi / (S * sqrtTau);
455 }
456
457 // General case: Corrado-Miller with Hallerbach's geometric mean adjustment
458 double term1 = C - (diff / 2);
459 double term2Sq = (term1 * term1) - (diff * diff / System.Math.PI);
460
461 if (term2Sq < 0)
462 {
463 // Fallback to simple Brenner-Subrahmanyam for edge cases
464 return System.Math.Sqrt(2 * System.Math.PI / tau) * marketPrice / S;
465 }
466
467 double term2 = System.Math.Sqrt(term2Sq);
468 double sigma = SqrtTwoPi / sqrtTau * ((term1 / avg) + (term2 / avg));
469
470 return System.Math.Max(sigma, MinVolatility);
471 }

480 private static double BSImpliedVolatilityBrent(
481 double S, double K, double tau, double r, double q,
482 double marketPrice, bool isCall,
483 double tolerance = 1e-7,
484 int maxIterations = 100)
485 {
486 double a = MinVolatility;
487 double b = MaxVolatility;
488
489 double fa = BSPrice(S, K, tau, a, r, q, isCall) - marketPrice;
490 double fb = BSPrice(S, K, tau, b, r, q, isCall) - marketPrice;
491
492 // Check if root is bracketed
493 if (fa * fb > 0)
494 {
495 // Not bracketed - return NaN or best guess
496 return System.Math.Abs(fa) < System.Math.Abs(fb) ? a : b;
497 }
498
499 // Ensure |f(a)| >= |f(b)|
500 if (System.Math.Abs(fa) < System.Math.Abs(fb))
501 {
502 (a, b) = (b, a);
503 (fa, fb) = (fb, fa);
504 }
505
506 double c = a;
507 double fc = fa;
508 bool mflag = true;
509 double d = 0;
510
511 for (int i = 0; i < maxIterations; i++)
512 {
513 // Converged?
514 if (System.Math.Abs(fb) < tolerance || System.Math.Abs(b - a) < tolerance)
515 {
516 return b;
517 }
518
519 double s;
520 if (System.Math.Abs(fa - fc) > MachineEpsilon &&
521 System.Math.Abs(fb - fc) > MachineEpsilon)
522 {
523 // Inverse quadratic interpolation
524 s = (a * fb * fc / ((fa - fb) * (fa - fc))) +
525 (b * fa * fc / ((fb - fa) * (fb - fc))) +
526 (c * fa * fb / ((fc - fa) * (fc - fb)));
527 }
528 else
529 {
530 // Secant method
531 s = b - (fb * (b - a) / (fb - fa));
532 }
533
534 // Conditions for using bisection instead
535 double temp1 = ((3 * a) + b) / 4;
536 bool cond1 = !((s > temp1 && s < b) || (s > b && s < temp1));
537 bool cond2 = mflag && System.Math.Abs(s - b) >= System.Math.Abs(b - c) / 2;
538 bool cond3 = !mflag && System.Math.Abs(s - b) >= System.Math.Abs(c - d) / 2;
539 bool cond4 = mflag && System.Math.Abs(b - c) < tolerance;
540 bool cond5 = !mflag && System.Math.Abs(c - d) < tolerance;
541
542 if (cond1 || cond2 || cond3 || cond4 || cond5)
543 {
544 // Bisection
545 s = (a + b) / 2;
546 mflag = true;
547 }
548 else
549 {
550 mflag = false;
551 }
552
553 double fs = BSPrice(S, K, tau, s, r, q, isCall) - marketPrice;
554 d = c;
555 c = b;
556 fc = fb;
557
558 if (fa * fs < 0)
559 {
560 b = s;
561 fb = fs;
562 }
563 else
564 {
565 a = s;
566 fa = fs;
567 }
568
569 // Ensure |f(a)| >= |f(b)|
570 if (System.Math.Abs(fa) < System.Math.Abs(fb))
571 {
572 (a, b) = (b, a);
573 (fa, fb) = (fb, fa);
574 }
575 }
576
577 return b; // Return best estimate
578 }
