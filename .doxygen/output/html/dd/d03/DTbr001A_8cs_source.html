84 public async Task<MarketDataSnapshot> GetMarketDataSnapshotAsync(
85 string symbol,
86 DateTime? evaluationDate = null,
87 CancellationToken cancellationToken = default)
88 {
89 // Use provided date or fall back to UTC now for live trading
90 var effectiveDate = evaluationDate ?? DateTime.UtcNow;
91 _logger.LogInformation("Building market data snapshot for {Symbol} as of {Date:yyyy-MM-dd}", symbol, effectiveDate);
92
93 try
94 {
95 // Step 1: Fetch primary data concurrently
96 // Note: Spot price is derived from historical bars, NOT from /prev endpoint
97 // The /prev endpoint only returns yesterday's close and doesn't work for historical dates in backtests
98 var historicalBarsTask = GetHistoricalBarsForRvCalculationAsync(symbol, effectiveDate, cancellationToken);
99 var optionChainTask = GetOptionChainWithCacheFallbackAsync(symbol, effectiveDate, cancellationToken);
100 var earningsTask = GetEarningsDataAsync(symbol, effectiveDate, cancellationToken);
101 var avgVolumeTask = _marketDataProvider.GetAverageVolume30DayAsync(symbol, effectiveDate, cancellationToken);
102
103 await Task.WhenAll(
104 historicalBarsTask,
105 optionChainTask,
106 earningsTask,
107 avgVolumeTask);
108
109 var historicalBars = await historicalBarsTask;
110 var optionChain = await optionChainTask;
111 var (nextEarnings, historicalEarnings) = await earningsTask;
112 var avgVolume = await avgVolumeTask;
113
114 // Derive spot price from the most recent bar's close price
115 // This works for both live (recent bars) and backtesting (historical bars)
116 decimal spotPrice;
117 if (historicalBars.Count > 0)
118 {
119 // Get the bar closest to evaluation date (but not after it)
120 var relevantBar = historicalBars
121 .Where(b => b.Timestamp.Date <= effectiveDate.Date)
122 .OrderByDescending(b => b.Timestamp)
123 .FirstOrDefault();
124
125 spotPrice = relevantBar?.Close ?? historicalBars[^1].Close;
126 _logger.LogDebug("Derived spot price {Price} from historical bars for {Symbol}", spotPrice, symbol);
127 }
128 else
129 {
130 // Fallback to live spot price only if no historical data
131 spotPrice = await _marketDataProvider.GetSpotPriceAsync(symbol, cancellationToken);
132 }
133
134 // For backtesting, use a reasonable historical risk-free rate
135 // The Treasury API returns 0% for future dates, so we use a sensible default
136 decimal riskFreeRate;
137 if (evaluationDate.HasValue && evaluationDate.Value > DateTime.UtcNow.Date)
138 {
139 // Future date in backtest - use typical historical rate
140 riskFreeRate = 0.05m; // 5% typical for 2023-2024
141 _logger.LogDebug("Using default risk-free rate {Rate}% for backtest date {Date}",
142 riskFreeRate * 100, effectiveDate);
143 }
144 else
145 {
146 riskFreeRate = await _riskFreeRateProvider.GetCurrentRateAsync(cancellationToken);
147 }
148
149 // Step 2: Construct snapshot
150 var snapshot = new MarketDataSnapshot
151 {
152 Symbol = symbol,
153 Timestamp = effectiveDate,
154 SpotPrice = spotPrice,
155 HistoricalBars = historicalBars,
156 OptionChain = optionChain,
157 NextEarnings = nextEarnings,
158 HistoricalEarnings = historicalEarnings,
159 RiskFreeRate = riskFreeRate,
160 DividendYield = 0.005m, // TODO: Fetch actual dividend yield
161 AverageVolume30Day = avgVolume
162 };
163
164 // Step 3: Run data quality validation
165 var validationResults = RunDataQualityValidation(snapshot);
166
167 // Step 4: Check for critical failures
168 var criticalFailures = validationResults
169 .Where(r => r.Status == ValidationStatus.Failed)
170 .ToList();
171
172 if (criticalFailures.Count > 0)
173 {
174 var errors = string.Join("; ", criticalFailures.Select(r => r.Message));
175 _logger.LogError(
176 "Data quality validation failed for {Symbol}: {Errors}",
177 symbol, errors);
178
179 throw new InvalidOperationException(
180 $"Data quality validation failed: {errors}");
181 }
182
183 // Step 5: Log warnings
184 var warnings = validationResults
185 .Where(r => r.Status == ValidationStatus.PassedWithWarnings)
186 .SelectMany(r => r.Warnings)
187 .ToList();
188
189 if (warnings.Count > 0)
190 {
191 _logger.LogWarning(
192 "Data quality warnings for {Symbol}: {Warnings}",
193 symbol, string.Join("; ", warnings));
194 }
195
196 _logger.LogInformation(
197 "Successfully built market data snapshot for {Symbol} " +
198 "(Bars: {BarCount}, Options: {OptionCount}, Earnings: {EarningsCount})",
199 symbol,
200 historicalBars.Count,
201 optionChain.Contracts.Count,
202 historicalEarnings.Count);
203
204 return snapshot;
205 }
206 catch (Exception ex) when (ex is not InvalidOperationException)
207 {
208 _logger.LogError(ex, "Failed to build market data snapshot for {Symbol}", symbol);
209 throw new InvalidOperationException($"Failed to build market data snapshot for {symbol}", ex);
210 }
211 }

349 private async Task<OptionChainSnapshot?> FindNearestCacheAsync(
350 string optionsDir,
351 string symbolLower,
352 DateTime evaluationDate,
353 CancellationToken cancellationToken)
354 {
355 try
356 {
357 // Find all cache files for this symbol with date suffixes
358 var pattern = $"{symbolLower}_*.sbe";
359 var binaryFiles = Directory.GetFiles(optionsDir, pattern);
360 var jsonPattern = $"{symbolLower}_*.json";
361 var jsonFiles = Directory.GetFiles(optionsDir, jsonPattern);
362
363 // Parse dates from filenames and find nearest earlier date
364 var availableDates = new List<(DateTime date, string path, bool isBinary)>();
365
366 foreach (var file in binaryFiles)
367 {
368 var filename = Path.GetFileNameWithoutExtension(file);
369 var datePart = filename.Substring(symbolLower.Length + 1);
370 if (DateTime.TryParseExact(datePart, "yyyyMMdd", null, System.Globalization.DateTimeStyles.None, out var date))
371 {
372 availableDates.Add((date, file, true));
373 }
374 }
375
376 foreach (var file in jsonFiles)
377 {
378 var filename = Path.GetFileNameWithoutExtension(file);
379 var datePart = filename.Substring(symbolLower.Length + 1);
380 if (DateTime.TryParseExact(datePart, "yyyyMMdd", null, System.Globalization.DateTimeStyles.None, out var date))
381 {
382 // Only add if not already in list (binary takes precedence)
383 if (!availableDates.Exists(x => x.date == date))
384 {
385 availableDates.Add((date, file, false));
386 }
387 }
388 }
389
390 if (availableDates.Count == 0)
391 {
392 return null;
393 }
394
395 // Find nearest date <= evaluationDate, or if none, nearest date > evaluationDate
396 var nearestEarlier = availableDates
397 .Where(x => x.date <= evaluationDate)
398 .OrderByDescending(x => x.date)
399 .FirstOrDefault();
400
401 var selected = nearestEarlier != default
402 ? nearestEarlier
403 : availableDates.OrderBy(x => x.date).First();
404
405 _logger.LogDebug("Using cached options from {CacheDate} for {EvalDate} (delta: {Days} days)",
406 selected.date, evaluationDate, Math.Abs((evaluationDate - selected.date).Days));
407
408 if (selected.isBinary)
409 {
410 return await LoadBinaryCacheAsync(selected.path, cancellationToken);
411 }
412 else
413 {
414 var json = await File.ReadAllTextAsync(selected.path, cancellationToken);
415 return JsonSerializer.Deserialize<OptionChainSnapshot>(json, JsonOptions);
416 }
417 }
418 catch (Exception ex)
419 {
420 _logger.LogWarning(ex, "Error finding nearest cache for {Symbol}", symbolLower);
421 return null;
422 }
423 }

534 .Where(e => e.Date > evaluationDate.Date)
535 .OrderBy(e => e.Date)
536 .FirstOrDefault();
537

540 _logger.LogDebug(

542 symbol, nextEarnings.Date.ToString("yyyy-MM-dd"));
543 }
544 }
545
546 // Filter historical to only include dates BEFORE evaluation date
547 var historicalBeforeEvaluation = historical
548 .Where(e => e.Date <= evaluationDate.Date)
549 .ToList();

551 _logger.LogDebug(
552 "Earnings data for {Symbol}: Next={NextDate}, Historical={HistoricalCount}",
