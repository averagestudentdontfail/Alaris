30 public static int EncodeEventEnvelope(EVCR003A envelope, Span<byte> buffer)
31 {
32 ArgumentNullException.ThrowIfNull(envelope);
33
34 int offset = 0;
35
36 // Version byte
37 buffer[offset++] = FormatVersion;
38
39 // EventId (16 bytes GUID)
40 envelope.EventId.TryWriteBytes(buffer[offset..]);
41 offset += 16;
42
43 // SequenceNumber (8 bytes)
44 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], envelope.SequenceNumber);
45 offset += 8;
46
47 // StoredAtUtc as Unix epoch ms (8 bytes)
48 BinaryPrimitives.WriteInt64LittleEndian(
49 buffer[offset..],
50 new DateTimeOffset(envelope.StoredAtUtc, TimeSpan.Zero).ToUnixTimeMilliseconds());
51 offset += 8;
52
53 // EventType (fixed 64 chars)
54 WriteFixedString(buffer[offset..], envelope.EventType, 64);
55 offset += 64;
56
57 // AggregateId (fixed 64 chars, nullable)
58 WriteFixedString(buffer[offset..], envelope.AggregateId, 64);
59 offset += 64;
60
61 // CorrelationId (fixed 36 chars, nullable)
62 WriteFixedString(buffer[offset..], envelope.CorrelationId, 36);
63 offset += 36;
64
65 // CausationId (fixed 36 chars, nullable)
66 WriteFixedString(buffer[offset..], envelope.CausationId, 36);
67 offset += 36;
68
69 // InitiatedBy (fixed 64 chars, nullable)
70 WriteFixedString(buffer[offset..], envelope.InitiatedBy, 64);
71 offset += 64;
72
73 // EventData length + variable data
74 byte[] eventDataBytes = System.Text.Encoding.UTF8.GetBytes(envelope.EventData);
75 BinaryPrimitives.WriteInt32LittleEndian(buffer[offset..], eventDataBytes.Length);
76 offset += 4;
77 eventDataBytes.CopyTo(buffer[offset..]);
78 offset += eventDataBytes.Length;
79
80 return offset;
81 }

86 public static EVCR003A DecodeEventEnvelope(ReadOnlySpan<byte> buffer)
87 {
88 int offset = 0;
89
90 // Version check
91 byte version = buffer[offset++];
92 if (version != FormatVersion)
93 {
94 throw new InvalidOperationException($"Unsupported event format version: {version}");
95 }
96
97 // EventId
98 var eventId = new Guid(buffer.Slice(offset, 16));
99 offset += 16;
100
101 // SequenceNumber
102 long sequenceNumber = BinaryPrimitives.ReadInt64LittleEndian(buffer[offset..]);
103 offset += 8;
104
105 // StoredAtUtc
106 long storedAtMs = BinaryPrimitives.ReadInt64LittleEndian(buffer[offset..]);
107 offset += 8;
108
109 // EventType
110 string eventType = ReadFixedString(buffer[offset..], 64);
111 offset += 64;
112
113 // AggregateId
114 string? aggregateId = ReadNullableFixedString(buffer[offset..], 64);
115 offset += 64;
116
117 // CorrelationId
118 string? correlationId = ReadNullableFixedString(buffer[offset..], 36);
119 offset += 36;
120
121 // CausationId
122 string? causationId = ReadNullableFixedString(buffer[offset..], 36);
123 offset += 36;
124
125 // InitiatedBy
126 string? initiatedBy = ReadNullableFixedString(buffer[offset..], 64);
127 offset += 64;
128
129 // EventData
130 int eventDataLength = BinaryPrimitives.ReadInt32LittleEndian(buffer[offset..]);
131 offset += 4;
132 string eventData = System.Text.Encoding.UTF8.GetString(buffer.Slice(offset, eventDataLength));
133
134 return new EVCR003A
135 {
136 EventId = eventId,
137 SequenceNumber = sequenceNumber,
138 StoredAtUtc = DateTimeOffset.FromUnixTimeMilliseconds(storedAtMs).UtcDateTime,
139 EventType = eventType,
140 EventData = eventData,
141 AggregateId = aggregateId,
142 CorrelationId = correlationId,
143 CausationId = causationId,
144 InitiatedBy = initiatedBy
145 };
146 }
