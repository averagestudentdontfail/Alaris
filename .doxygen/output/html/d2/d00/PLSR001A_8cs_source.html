45 [MethodImpl(MethodImplOptions.AggressiveInlining)]
46 public static int EncodePriceBar(in PriceBarData bar, Span<byte> buffer)
47 {
48 const int blockLength = 56; // Fixed field block size
49 int offset = 0;
50
51 // Write header
52 BinaryPrimitives.WriteUInt16LittleEndian(buffer[offset..], blockLength);
53 offset += 2;
54 BinaryPrimitives.WriteUInt16LittleEndian(buffer[offset..], PriceBarTemplateId);
55 offset += 2;
56 BinaryPrimitives.WriteUInt16LittleEndian(buffer[offset..], SchemaId);
57 offset += 2;
58 BinaryPrimitives.WriteUInt16LittleEndian(buffer[offset..], SchemaVersion);
59 offset += 2;
60
61 // Write fixed fields
62 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], bar.TimestampEpochMs);
63 offset += 8;
64 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], bar.OpenMantissa);
65 offset += 8;
66 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], bar.HighMantissa);
67 offset += 8;
68 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], bar.LowMantissa);
69 offset += 8;
70 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], bar.CloseMantissa);
71 offset += 8;
72 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], bar.Volume);
73 offset += 8;
74
75 return offset;
76 }

102 [MethodImpl(MethodImplOptions.AggressiveInlining)]
103 public static int EncodeOptionContract(in OptionContractData contract, Span<byte> buffer)
104 {
105 const int blockLength = 128; // Fixed field block size
106 int offset = 0;
107
108 // Write header
109 BinaryPrimitives.WriteUInt16LittleEndian(buffer[offset..], blockLength);
110 offset += 2;
111 BinaryPrimitives.WriteUInt16LittleEndian(buffer[offset..], OptionContractTemplateId);
112 offset += 2;
113 BinaryPrimitives.WriteUInt16LittleEndian(buffer[offset..], SchemaId);
114 offset += 2;
115 BinaryPrimitives.WriteUInt16LittleEndian(buffer[offset..], SchemaVersion);
116 offset += 2;
117
118 // Write fixed fields
119 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.StrikeMantissa);
120 offset += 8;
121 BinaryPrimitives.WriteInt32LittleEndian(buffer[offset..], contract.ExpirationDays);
122 offset += 4;
123 buffer[offset] = (byte)contract.Right;
124 offset += 1;
125 // Padding for alignment
126 buffer[offset] = 0;
127 buffer[offset + 1] = 0;
128 buffer[offset + 2] = 0;
129 offset += 3;
130 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.BidMantissa);
131 offset += 8;
132 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.AskMantissa);
133 offset += 8;
134 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.LastMantissa);
135 offset += 8;
136 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.ImpliedVolatilityMantissa);
137 offset += 8;
138 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.DeltaMantissa);
139 offset += 8;
140 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.GammaMantissa);
141 offset += 8;
142 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.ThetaMantissa);
143 offset += 8;
144 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.VegaMantissa);
145 offset += 8;
146 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.OpenInterest);
147 offset += 8;
148 BinaryPrimitives.WriteInt64LittleEndian(buffer[offset..], contract.Volume);
149 offset += 8;
150
151 // Write symbol (fixed 16 chars)
152 WriteFixedString(buffer[offset..], contract.Symbol, 16);
153 offset += 16;
154
155 return offset;
156 }

161 [MethodImpl(MethodImplOptions.AggressiveInlining)]
162 public static OptionContractData DecodeOptionContract(ReadOnlySpan<byte> buffer)
163 {
164 int offset = HeaderSize;
165
166 var contract = new OptionContractData
167 {
168 StrikeMantissa = BinaryPrimitives.ReadInt64LittleEndian(buffer[offset..]),
169 ExpirationDays = BinaryPrimitives.ReadInt32LittleEndian(buffer[(offset + 8)..]),
170 Right = (OptionRightEnum)buffer[offset + 12]
171 };
172 offset += 16; // Including padding
173
174 contract.BidMantissa = BinaryPrimitives.ReadInt64LittleEndian(buffer[offset..]);
175 contract.AskMantissa = BinaryPrimitives.ReadInt64LittleEndian(buffer[(offset + 8)..]);
176 contract.LastMantissa = BinaryPrimitives.ReadInt64LittleEndian(buffer[(offset + 16)..]);
177 contract.ImpliedVolatilityMantissa = BinaryPrimitives.ReadInt64LittleEndian(buffer[(offset + 24)..]);
178 contract.DeltaMantissa = BinaryPrimitives.ReadInt64LittleEndian(buffer[(offset + 32)..]);
179 contract.GammaMantissa = BinaryPrimitives.ReadInt64LittleEndian(buffer[(offset + 40)..]);
180 contract.ThetaMantissa = BinaryPrimitives.ReadInt64LittleEndian(buffer[(offset + 48)..]);
181 contract.VegaMantissa = BinaryPrimitives.ReadInt64LittleEndian(buffer[(offset + 56)..]);
182 contract.OpenInterest = BinaryPrimitives.ReadInt64LittleEndian(buffer[(offset + 64)..]);
183 contract.Volume = BinaryPrimitives.ReadInt64LittleEndian(buffer[(offset + 72)..]);
184 offset += 80;
185
186 contract.Symbol = ReadFixedString(buffer[offset..], 16);
187
188 return contract;
189 }

208 [MethodImpl(MethodImplOptions.AggressiveInlining)]
209 private static void WriteFixedString(Span<byte> buffer, string? value, int length)
210 {
211 buffer[..length].Clear();
212 if (!string.IsNullOrEmpty(value))
213 {
214 int bytesToWrite = Math.Min(value.Length, length);
215 Encoding.ASCII.GetBytes(value.AsSpan(0, bytesToWrite), buffer);
216 }
217 }
218
222 private static string ReadFixedString(ReadOnlySpan<byte> buffer, int length)
223 {
224 int actualLength = buffer[..length].IndexOf((byte)0);
225 if (actualLength < 0)
226 {
227 actualLength = length;
228 }
229
230 return Encoding.ASCII.GetString(buffer[..actualLength]);
231 }
