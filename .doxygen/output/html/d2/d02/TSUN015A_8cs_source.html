175 private static List<Trade> GenerateHistoricalTrades(int count, int avgHoldingDays)
176 {
177 var trades = new List<Trade>();
178 var baseDate = DateTime.Today.AddDays(-count * 2.5); // Space out entries
179
180 for (int i = 0; i < count; i++)
181 {
182 var entryDate = baseDate.AddDays(i * 2.5);
183 var holdingPeriod = avgHoldingDays + (i % 3 - 1); // Vary ±1 day
184
185 trades.Add(new Trade
186 {
187 Symbol = i % 2 == 0 ? "AAPL" : "MSFT",
188 EntryDate = entryDate,
189 ExitDate = entryDate.AddDays(holdingPeriod),
190 ProfitLoss = (i % 3 == 0) ? -100 : 150 // 2/3 winners
191 });
192 }
193
194 return trades;
195 }
