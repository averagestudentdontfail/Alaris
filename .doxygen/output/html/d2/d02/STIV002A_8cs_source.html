89 public double ComputeKappa()
90 {
91 if (Lambda == 0)
92 {
93 return 0;
94 }
95
96 // kappa = p * eta1/(eta1-1) + (1-p) * eta2/(eta2+1) - 1
97 double upwardContribution = P * Eta1 / (Eta1 - 1);
98 double downwardContribution = (1 - P) * Eta2 / (Eta2 + 1);
99 return upwardContribution + downwardContribution - 1;
100 }

105 public double ComputeMartingaleDrift()

179 public double ComputeTheoreticalIVApproximate(double spot, double strike, double timeToExpiry)
180 {
181 ValidateInputs(spot, strike, timeToExpiry);
182
183 // Log-moneyness
184 double k = Math.Log(strike / spot);
185
186 // Total variance from diffusion
187 double diffusionVariance = _params.Sigma * _params.Sigma * timeToExpiry;
188
189 // Jump contribution to variance
190 double jumpVariance = ComputeJumpVariance(timeToExpiry);
191
192 // Total variance
193 double totalVariance = diffusionVariance + jumpVariance;
194
195 // Skew adjustment based on moneyness
196 double skewAdjustment = ComputeSkewAdjustment(k, timeToExpiry);
197
198 // Approximate IV
199 double approximateIV = Math.Sqrt(totalVariance / timeToExpiry) + skewAdjustment;
200
201 return Math.Max(approximateIV, 0.001);
202 }

207 private double ComputeJumpVariance(double timeToExpiry)
208 {
209 if (_params.Lambda == 0)
210 {
211 return 0;
212 }
213
214 // Second moment of log-jump: E[Y^2] where Y = log(V)
215 // For double-exponential: E[Y^2] = 2*p/eta1^2 + 2*(1-p)/eta2^2
216 double secondMoment = (2 * _params.P / (_params.Eta1 * _params.Eta1)) +
217 (2 * (1 - _params.P) / (_params.Eta2 * _params.Eta2));
218
219 // First moment: E[Y] = p/eta1 - (1-p)/eta2
220 double firstMoment = (_params.P / _params.Eta1) - ((1 - _params.P) / _params.Eta2);
221
222 // Variance of single jump
223 double jumpVariance = secondMoment - (firstMoment * firstMoment);
224
225 // Total jump variance = lambda * t * Var(Y)
226 return _params.Lambda * timeToExpiry * jumpVariance;
227 }
228
233 private double ComputeSkewAdjustment(double logMoneyness, double timeToExpiry)
234 {
235 if (_params.Lambda == 0)
236 {
237 return 0;
238 }
239
240 // Skewness coefficient from jump distribution
241 // Negative skew when downward jumps are larger/more frequent
242 double skewness = ComputeJumpSkewness();
243
244 // Adjust IV based on moneyness and skewness
245 // OTM puts (k < 0) have higher IV when skewness is negative
246 double sqrtT = Math.Sqrt(timeToExpiry);
247 return -skewness * logMoneyness / (6 * sqrtT) * (_params.Lambda / 10);
248 }
249
253 private double ComputeJumpSkewness()
254 {
255 // Third moment contribution
256 double thirdMoment = (6 * _params.P / Math.Pow(_params.Eta1, 3)) -
257 (6 * (1 - _params.P) / Math.Pow(_params.Eta2, 3));
258
259 // Normalize by variance^(3/2)
260 double variance = (2 * _params.P / (_params.Eta1 * _params.Eta1)) +
261 (2 * (1 - _params.P) / (_params.Eta2 * _params.Eta2));
262
263 if (variance <= 0)
264 {
265 return 0;
266 }
267
268 return thirdMoment / Math.Pow(variance, 1.5);
269 }

338 public static KouParameters Calibrate(
339 double spot,
340 IReadOnlyList<(double Strike, int DTE, double MarketIV)> marketData,
341 double riskFreeRate,
342 double dividendYield)
343 {
344 ArgumentNullException.ThrowIfNull(marketData);
345
346 if (marketData.Count < 5)
347 {
348 throw new ArgumentException("Need at least 5 market observations for calibration.", nameof(marketData));
349 }
350
351 // Use Differential Evolution for production-grade global optimization
352 // Jump-diffusion models often have multi-modal error surfaces
353 var optimizer = new Numerical.STPR005A
354 {
355 MaxGenerations = 500,
356 PopulationSize = 50, // 10 * dimension (5 parameters)
357 DifferentialWeight = 0.8,
358 CrossoverProbability = 0.9,
359 Tolerance = 1e-6
360 };
361
362 // Parameter vector: [sigma, lambda, p, eta1, eta2]
363 // Parameter bounds
364 double[] lowerBounds = { 0.05, 0.0, 0.0, 1.01, 0.1 };
365 double[] upperBounds = { 1.0, 20.0, 1.0, 50.0, 50.0 };
366
367 // Objective function (mean squared error)
368 double Objective(double[] x)
369 {
370 KouParameters candidateParams = new()
371 {
372 Sigma = x[0],
373 Lambda = x[1],
374 P = x[2],
375 Eta1 = x[3],
376 Eta2 = x[4],
377 RiskFreeRate = riskFreeRate,
378 DividendYield = dividendYield
379 };
380
381 // Validate constraints
382 if (!candidateParams.Validate().IsValid)
383 {
384 return 1e10; // Large penalty for invalid parameters
385 }
386
387 STIV002A model = new(candidateParams);
388
389 double sumSquaredError = 0;
390 int validCount = 0;
391
392 foreach ((double strike, int dte, double marketIV) in marketData)
393 {
394 double timeToExpiry = TradingCalendarDefaults.DteToYears(dte);
395
396 try
397 {
398 double modelIV = model.ComputeTheoreticalIV(spot, strike, timeToExpiry);
399 double error = modelIV - marketIV;
400 sumSquaredError += error * error;
401 validCount++;
402 }
403#pragma warning disable CA1031 // Do not catch general exception types
404 catch (Exception)
405#pragma warning restore CA1031
406 {
407 // Pricing can fail for many reasons during calibration (e.g., invalid parameters,
408 // numerical instability, convergence issues). Penalize rather than failing the entire calibration.
409 sumSquaredError += 100.0;
410 }
411 }
412
413 return validCount > 0 ? sumSquaredError / validCount : 1e10;
414 }
415
416 // Run optimization
417 OptimizationResult result = optimizer.Minimize(Objective, lowerBounds, upperBounds);
418
419 if (result.Converged && result.OptimalValue < 1e9)
420 {
421 return new KouParameters
422 {
423 Sigma = result.OptimalParameters[0],
424 Lambda = result.OptimalParameters[1],
425 P = result.OptimalParameters[2],
426 Eta1 = result.OptimalParameters[3],
427 Eta2 = result.OptimalParameters[4],
428 RiskFreeRate = riskFreeRate,
429 DividendYield = dividendYield
430 };
431 }
432
433 // Fallback: Try grid search if DE fails
434 return CalibrateGridSearch(spot, marketData, riskFreeRate, dividendYield);
435 }

442 private static KouParameters CalibrateGridSearch(
443 double spot,
444 IReadOnlyList<(double Strike, int DTE, double MarketIV)> marketData,
445 double riskFreeRate,
446 double dividendYield)
447 {
448 double bestError = double.MaxValue;
449 KouParameters? bestParams = null;
450
451 // Parameter grid
452 double[] sigmas = { 0.15, 0.20, 0.25, 0.30 };
453 double[] lambdas = { 1.0, 3.0, 5.0, 10.0 };
454 double[] ps = { 0.3, 0.4, 0.5 };
455 double[] eta1s = { 5.0, 10.0, 20.0 };
456 double[] eta2s = { 3.0, 5.0, 10.0 };
457
458 foreach (double sigma in sigmas)
459 {
460 foreach (double lambda in lambdas)
461 {
462 foreach (double p in ps)
463 {
464 foreach (double eta1 in eta1s)
465 {
466 foreach (double eta2 in eta2s)
467 {
468 KouParameters candidateParams = new()
469 {
470 Sigma = sigma,
471 Lambda = lambda,
472 P = p,
473 Eta1 = eta1,
474 Eta2 = eta2,
475 RiskFreeRate = riskFreeRate,
476 DividendYield = dividendYield
477 };
478
479 if (!candidateParams.Validate().IsValid)
480 {
481 continue;
482 }
483
484 STIV002A model = new(candidateParams);
485 double error = ComputeCalibrationError(model, spot, marketData);
486
487 if (error < bestError)
488 {
489 bestError = error;
490 bestParams = candidateParams;
491 }
492 }
493 }
494 }
495 }
496 }
497
498 return bestParams ?? KouParameters.DefaultEquity;
499 }
