59 public STRK002A CalculateFromHistory(
60 double portfolioValue,
61 IReadOnlyList<Trade> historicalTrades,
62 double spreadCost,
63 STCR004A signal)
64 {
65 ArgumentNullException.ThrowIfNull(historicalTrades);
66 ArgumentNullException.ThrowIfNull(signal);
67
68 if (portfolioValue <= 0)
69 {
70 throw new ArgumentException("Portfolio value must be positive", nameof(portfolioValue));
71 }
72
73 if (spreadCost <= 0)
74 {
75 throw new ArgumentException("Spread cost must be positive", nameof(spreadCost));
76 }
77
78 STRK002A positionSize = new STRK002A
79 {
80 MaxLossPerContract = spreadCost * ContractMultiplier
81 };
82
83 // Need sufficient trade history for meaningful statistics
84 if (historicalTrades.Count < 20)
85 {
86 SafeLog(() => LogInsufficientTradeHistory(_logger!, historicalTrades.Count, null));
87 return GetMinimumPosition(portfolioValue, spreadCost);
88 }
89
90 try
91 {
92 // Calculate win rate
93 List<Trade> winningTrades = historicalTrades.Where(t => t.ProfitLoss > 0).ToList();
94 List<Trade> losingTrades = historicalTrades.Where(t => t.ProfitLoss <= 0).ToList();
95
96 double winRate = (double)winningTrades.Count / historicalTrades.Count;
97 double lossRate = 1 - winRate;
98
99 if (winRate <= 0 || winRate >= 1)
100 {
101 SafeLog(() => LogInvalidWinRate(_logger!, winRate, null));
102 return GetMinimumPosition(portfolioValue, spreadCost);
103 }
104
105 // Calculate average win and loss amounts
106 double avgWin = (winningTrades.Count > 0) ? winningTrades.Average(t => t.ProfitLoss) : 0;
107 double avgLoss = (losingTrades.Count > 0) ? Math.Abs(losingTrades.Average(t => t.ProfitLoss)) : (spreadCost * ContractMultiplier);
108
109 if (avgWin <= 0 || avgLoss <= 0)
110 {
111 SafeLog(() => LogInvalidAverageWinLoss(_logger!, null));
112 return GetMinimumPosition(portfolioValue, spreadCost);
113 }
114
115 double winLossRatio = avgWin / avgLoss;
116
117 // Kelly formula: f* = (p*b - q) / b
118 double fullKellyPercent = ((winRate * winLossRatio) - lossRate) / winLossRatio;
119
120 // Apply fractional Kelly for safety
121 double kellyPercent = fullKellyPercent * FractionalKelly;
122
123 // Cap at maximum allocation
124 double allocationPercent = Math.Max(0, Math.Min(kellyPercent, MaxAllocation));
125
126 // Adjust based on signal strength
127 allocationPercent = AdjustForSTCR004AStrength(allocationPercent, signal);
128
129 // Calculate position size
130 double dollarAllocation = portfolioValue * allocationPercent;
131 int contracts = (int)Math.Floor(dollarAllocation / (spreadCost * ContractMultiplier));
132
133 positionSize.Contracts = Math.Max(contracts, 0);
134 positionSize.AllocationPercent = allocationPercent;
135 positionSize.DollarAllocation = dollarAllocation;
136 positionSize.TotalRisk = contracts * spreadCost * ContractMultiplier;
137 positionSize.ExpectedProfitPerContract = avgWin;
138 positionSize.KellyFraction = fullKellyPercent;
139
140 SafeLog(() => LogPositionCalculated(_logger!, signal.Symbol, positionSize.Contracts, allocationPercent, fullKellyPercent, null));
141
142 return positionSize;
143 }
144 catch (DivideByZeroException ex)
145 {
146 SafeLog(() => LogErrorCalculatingPosition(_logger!, signal.Symbol, ex));
147 throw;
148 }
149 catch (OverflowException ex)
150 {
151 SafeLog(() => LogErrorCalculatingPosition(_logger!, signal.Symbol, ex));
152 throw;
153 }
154 catch (InvalidOperationException ex)
155 {
156 SafeLog(() => LogErrorCalculatingPosition(_logger!, signal.Symbol, ex));
157 throw;
158 }
159 }

170 public static double CalculateNetOfCostKelly(
171 double winProbability,
172 double avgWin,
173 double avgLoss,
174 double roundTripCost)
175 {
176 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(avgWin);
177 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(avgLoss);
178 ArgumentOutOfRangeException.ThrowIfNegative(roundTripCost);
179
180 if (winProbability <= 0 || winProbability >= 1)
181 {
182 return 0.0;
183 }
184
185 double lossProbability = 1.0 - winProbability;
186
187 // Adjust win amount for transaction costs
188 double effectiveWin = avgWin - roundTripCost;
189 if (effectiveWin <= 0)
190 {
191 return 0.0; // Costs exceed edge - no positive expectancy
192 }
193
194 // Adjusted win/loss ratio
195 double effectiveWinLossRatio = effectiveWin / avgLoss;
196
197 // Modified Kelly formula with costs
198 double kellyFraction = ((winProbability * effectiveWinLossRatio) - lossProbability) / effectiveWinLossRatio;
199
200 // Never bet if negative expectancy
201 return Math.Max(0.0, kellyFraction);
202 }
