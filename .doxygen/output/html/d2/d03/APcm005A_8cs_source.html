171 private static (DateTime start, DateTime end) ResolveDateRange(BacktestCreateSettings settings)
172 {
173 // Option 1: --years specified (e.g., "2024" or "2023,2024")
174 if (!string.IsNullOrEmpty(settings.Years))
175 {
176 var yearStrings = settings.Years.Split(',', StringSplitOptions.RemoveEmptyEntries | StringSplitOptions.TrimEntries);
177 var years = new List<int>();
178
179 foreach (var yearStr in yearStrings)
180 {
181 if (int.TryParse(yearStr, out var year) && year >= 2000 && year <= 2100)
182 {
183 years.Add(year);
184 }
185 else
186 {
187 AnsiConsole.MarkupLine($"[red]Invalid year: {yearStr}. Use format: 2024 or 2023,2024[/]");
188 return (DateTime.MinValue, DateTime.MinValue);
189 }
190 }
191
192 if (years.Count == 0)
193 {
194 AnsiConsole.MarkupLine("[red]No valid years specified[/]");
195 return (DateTime.MinValue, DateTime.MinValue);
196 }
197
198 var minYear = years.Min();
199 var maxYear = years.Max();
200 var startDate = new DateTime(minYear, 1, 1);
201 var endDate = maxYear == DateTime.Now.Year
202 ? DateTime.Now.Date.AddDays(-1) // Current year: end yesterday
203 : new DateTime(maxYear, 12, 31); // Past years: end Dec 31
204
205 AnsiConsole.MarkupLine($"[cyan]Using year range: {minYear}-{maxYear}[/]");
206 return (startDate, endDate);
207 }
208
209 // Option 2: Explicit --start/--end
210 if (!string.IsNullOrEmpty(settings.StartDate) || !string.IsNullOrEmpty(settings.EndDate))
211 {
212 if (string.IsNullOrEmpty(settings.StartDate) || !DateTime.TryParse(settings.StartDate, out var start))
213 {
214 AnsiConsole.MarkupLine("[red]Invalid or missing start date. Use YYYY-MM-DD format[/]");
215 return (DateTime.MinValue, DateTime.MinValue);
216 }
217
218 if (string.IsNullOrEmpty(settings.EndDate) || !DateTime.TryParse(settings.EndDate, out var end))
219 {
220 AnsiConsole.MarkupLine("[red]Invalid or missing end date. Use YYYY-MM-DD format[/]");
221 return (DateTime.MinValue, DateTime.MinValue);
222 }
223
224 return (start, end);
225 }
226
227 // Option 3: Default - 2 years ending yesterday (with 1-month buffer for options data)
228 // Polygon's Options Starter plan has a 2-year limit, but options aggregate data
229 // at the boundary often returns 403. Using 23 months ensures reliable data access.
230 var yesterday = DateTime.Now.Date.AddDays(-1);
231 var twoYearsAgo = yesterday.AddYears(-2).AddMonths(1); // 23 months instead of 24
232 AnsiConsole.MarkupLine($"[cyan]Using default 2-year range: {twoYearsAgo:yyyy-MM-dd} to {yesterday:yyyy-MM-dd}[/]");
233 return (twoYearsAgo, yesterday);
234 }

312 private static IReadOnlyList<DateTime> GetOptionsRequiredDatesFromSession(string sessionDataPath, IEnumerable<string> symbols)
313 {
314 var dates = new HashSet<DateTime>();
315 var nasdaqPath = System.IO.Path.Combine(sessionDataPath, "earnings", "nasdaq");
316
317 if (!Directory.Exists(nasdaqPath))
318 {
319 return Array.Empty<DateTime>();
320 }
321
322 var symbolSet = new HashSet<string>(symbols, StringComparer.OrdinalIgnoreCase);
323
324 foreach (var file in Directory.GetFiles(nasdaqPath, "*.json"))
325 {
326 try
327 {
328 var json = File.ReadAllText(file);
329 // Simple parsing - look for symbol matches
330 foreach (var symbol in symbolSet)
331 {
332 if (json.Contains($"\"{symbol}\"", StringComparison.OrdinalIgnoreCase))
333 {
334 // Extract date from filename
335 var dateStr = System.IO.Path.GetFileNameWithoutExtension(file);
336 if (DateTime.TryParse(dateStr, out var earningsDate))
337 {
338 // Add evaluation dates (5-7 days before)
339 for (int d = 5; d <= 7; d++)
340 {
341 var evalDate = earningsDate.AddDays(-d);
342 if (evalDate.DayOfWeek != DayOfWeek.Saturday &&
343 evalDate.DayOfWeek != DayOfWeek.Sunday)
344 {
345 dates.Add(evalDate);
346 }
347 }
348 }
349 }
350 }
351 }
352 catch
353 {
354 // Ignore parsing errors
355 }
356 }
357
358 return dates.OrderBy(d => d).ToList();
359 }

476public sealed class BacktestRunCommand : AsyncCommand<BacktestRunSettings>
477{
478 public override async Task<int> ExecuteAsync(CommandContext context, BacktestRunSettings settings)
479 {
480 // Initialize FSM for deterministic state tracking
481 var fsm = PLWF002A.Create();
482 fsm.OnTransition += record =>
483 {
484 if (record.Succeeded)
485 AnsiConsole.MarkupLine($"[grey]FSM: {record.FromState} → {record.ToState}[/]");
486 };
487
488 var service = new APsv001A();
489
490 string? sessionId = settings.SessionId;
491
492 if (settings.Latest || string.IsNullOrEmpty(sessionId))
493 {
494 var sessions = await service.ListAsync();
495 var latest = sessions.Count > 0 ? sessions[0] : null;
496
497 if (latest == null)
498 {
499 AnsiConsole.MarkupLine("[red]No sessions found. Create one with:[/] [cyan]alaris backtest create[/]");
500 return 1;
501 }
502
503 sessionId = latest.SessionId;
504 AnsiConsole.MarkupLine($"[grey]Using latest session: {sessionId}[/]");
505 }
506
507 var session = await service.GetAsync(sessionId);
508 if (session == null)
509 {
510 AnsiConsole.MarkupLine($"[red]Session not found: {sessionId}[/]");
511 return 1;
512 }
513
514 // FSM: Idle → SessionSelected
515 fsm.Fire(BacktestEvent.SelectSession);
516
517 AnsiConsole.MarkupLine($"[blue]Running session:[/] {session.SessionId}");
518 AnsiConsole.MarkupLine($"[blue]Date Range:[/] {session.StartDate:yyyy-MM-dd} to {session.EndDate:yyyy-MM-dd}");
519 AnsiConsole.MarkupLine($"[blue]Symbols:[/] {string.Join(", ", session.Symbols.Take(10))}{(session.Symbols.Count > 10 ? "..." : "")}");
520 AnsiConsole.WriteLine();
521
522 // FSM: SessionSelected → DataChecking
523 fsm.Fire(BacktestEvent.CheckData);
524
525 var dataPath = service.GetDataPath(session.SessionId);
526 var (pricesMissing, earningsMissing, optionsMissing) = CheckDataAvailability(dataPath, session);
527
528 if (pricesMissing || earningsMissing || optionsMissing)
529 {
530 // FSM: DataChecking → DataBootstrapping
531 fsm.Fire(BacktestEvent.DataMissing);
532
533 AnsiConsole.MarkupLine("[yellow]⚠ Missing data detected:[/]");
534 if (pricesMissing) AnsiConsole.MarkupLine(" [grey]• Price data not found[/]");
535 if (earningsMissing) AnsiConsole.MarkupLine(" [grey]• Earnings cache empty[/]");
536 if (optionsMissing) AnsiConsole.MarkupLine(" [grey]• Options data empty (mandatory)[/]");
537 AnsiConsole.WriteLine();
538
539 if (settings.AutoBootstrap)
540 {
541 try
542 {
543 await BootstrapDataAsync(session, service, dataPath);
544 // FSM: DataBootstrapping → DataChecking (re-check after bootstrap)
545 fsm.Fire(BacktestEvent.BootstrapComplete);
546 // FSM: DataChecking → ExecutingLean (data should be ready now)
547 fsm.Fire(BacktestEvent.DataReady);
548 }
549 catch (Exception ex)
550 {
551 AnsiConsole.MarkupLine($"[red]Bootstrap failed: {ex.Message}[/]");
552 fsm.Fire(BacktestEvent.BootstrapFailed);
553 return 1;
554 }
555 }
556 else
557 {
558 var download = AnsiConsole.Confirm("Download missing data now?", defaultValue: true);
559 if (download)
560 {
561 try
562 {
563 await BootstrapDataAsync(session, service, dataPath);
564 fsm.Fire(BacktestEvent.BootstrapComplete);
565 fsm.Fire(BacktestEvent.DataReady);
566 }
567 catch (Exception ex)
568 {
569 AnsiConsole.MarkupLine($"[red]Bootstrap failed: {ex.Message}[/]");
570 fsm.Fire(BacktestEvent.BootstrapFailed);
571 return 1;
572 }
573 }
574 else
575 {
576 AnsiConsole.MarkupLine("[red]Cannot proceed without mandatory market data.[/]");
577 AnsiConsole.MarkupLine("[grey]Strategy requires historical options data with IV for backtesting.[/]");
578 fsm.Fire(BacktestEvent.BootstrapFailed);
579 return 1;
580 }
581 }
582 AnsiConsole.WriteLine();
583 }
584 else
585 {
586 // FSM: DataChecking → ExecutingLean
587 fsm.Fire(BacktestEvent.DataReady);
588 AnsiConsole.MarkupLine("[green]✓[/] Data available for session");
589 AnsiConsole.WriteLine();
590 }
591
592 // Update status to Running
593 await service.UpdateAsync(session with { Status = SessionStatus.Running });
594
595 // Execute LEAN with session-specific paths
596 int exitCode;
597 if (!settings.NoMonitor)
598 {
599 exitCode = await ExecuteLeanWithMonitoringAsync(session, service);
600 }
601 else
602 {
603 exitCode = await ExecuteLeanForSession(session, service);
604 }
605
606 // FSM: ExecutingLean → Completed/Failed
607 if (exitCode == 0)
608 {
609 fsm.Fire(BacktestEvent.LeanCompleted);
610 }
611 else
612 {
613 fsm.Fire(BacktestEvent.LeanFailed);
614 }
615
616 // Update status based on result
617 var finalStatus = exitCode == 0 ? SessionStatus.Completed : SessionStatus.Failed;
618 await service.UpdateAsync(session with
619 {
620 Status = finalStatus,
621 ExitCode = exitCode
622 });
623
624 if (exitCode == 0)
625 {
626 AnsiConsole.MarkupLine($"[green]✓[/] Session completed successfully");
627 }
628 else
629 {
630 AnsiConsole.MarkupLine($"[red]✗[/] Session failed with exit code {exitCode}");
631 }
632
633 // Log FSM audit trail
634 AnsiConsole.WriteLine();
635 AnsiConsole.MarkupLine($"[grey]FSM transitions: {fsm.History.Count}, Final state: {fsm.CurrentState}[/]");
636
637 return exitCode;
638 }

768 private static List<DateTime> GetEarningsDatesFromCache(string dataPath, DateTime startDate, DateTime endDate)
769 {
770 var earningsDates = new List<DateTime>();
771 var earningsPath = System.IO.Path.Combine(dataPath, "earnings", "nasdaq");
772
773 if (!Directory.Exists(earningsPath))
774 {
775 return earningsDates;
776 }
777
778 foreach (var file in Directory.GetFiles(earningsPath, "*.json"))
779 {
780 var filename = System.IO.Path.GetFileNameWithoutExtension(file);
781 if (DateTime.TryParseExact(filename, "yyyy-MM-dd", null, System.Globalization.DateTimeStyles.None, out var date))
782 {
783 if (date >= startDate && date <= endDate)
784 {
785 // Check if file has any earnings events
786 try
787 {
788 var content = File.ReadAllText(file);
789 if (content.Length > 10 && !content.Contains("[]")) // Not empty
790 {
791 earningsDates.Add(date);
792 }
793 }
794 catch
795 {
796 // Skip files that can't be read
797 }
798 }
799 }
800 }
801
802 return earningsDates;

817 private static async Task<int> ExecuteLeanForSession(APmd001A session, APsv001A service)
818 {
819 // Find config and launcher
820 var configPath = FindConfigPath();
821 var launcherPath = FindLeanLauncher();
822
823 if (configPath == null || launcherPath == null)
824 {
825 AnsiConsole.MarkupLine("[red]Could not find config.json or LEAN launcher[/]");
826 return 1;
827 }
828
829 // CRITICAL: Inject session data path into LEAN config before engine starts
830 // Environment variables (QC_DATA_FOLDER) do NOT work - LEAN reads from config.json only
831 // Config.Set() also doesn't work because it only affects THIS process, not the subprocess
832 // Solution: Pass --data-folder as a command-line argument to the LEAN subprocess
833 var sessionDataPath = service.GetDataPath(session.SessionId);
834
835 var psi = new ProcessStartInfo
836 {
837 FileName = "dotnet",
838 Arguments = $"run --project \"{launcherPath}\" -- --data-folder \"{sessionDataPath}\" --close-automatically true",
839 WorkingDirectory = System.IO.Path.GetDirectoryName(configPath),
840 UseShellExecute = false,
841 RedirectStandardOutput = true,
842 RedirectStandardError = true
843 };
844
845 // Set session-specific environment variables
846 psi.Environment["QC_ENVIRONMENT"] = "backtesting";
847 // Inject data folder override for LEAN engine to pick up session data automatically
848 psi.Environment["QC_DATA_FOLDER"] = service.GetDataPath(session.SessionId);
849 // Inject static universe symbols for STUN001B to bypass universe file requirement
850 psi.Environment["ALARIS_SESSION_SYMBOLS"] = string.Join(",", session.Symbols);
851
852 psi.Environment["ALARIS_SESSION_ID"] = session.SessionId;
853 psi.Environment["ALARIS_SESSION_PATH"] = session.SessionPath;
854 psi.Environment["ALARIS_SESSION_DATA"] = service.GetDataPath(session.SessionId);
855 psi.Environment["ALARIS_SESSION_RESULTS"] = service.GetResultsPath(session.SessionId);
856 psi.Environment["ALARIS_BACKTEST_STARTDATE"] = session.StartDate.ToString("yyyy-MM-dd");
857 psi.Environment["ALARIS_BACKTEST_ENDDATE"] = session.EndDate.ToString("yyyy-MM-dd");
858
859 using var process = Process.Start(psi);
860 if (process == null)
861 {
862 return 1;
863 }
864
865 process.OutputDataReceived += (_, e) =>
866 {
867 if (e.Data != null) Console.WriteLine(e.Data);
868 };
869
870 process.ErrorDataReceived += (_, e) =>
871 {
872 if (e.Data != null)
873 {
874 var escaped = Markup.Escape(e.Data);
875 AnsiConsole.MarkupLine($"[red]{escaped}[/]");
876 }
877 };
878
879 process.BeginOutputReadLine();
880 process.BeginErrorReadLine();
881
882 await process.WaitForExitAsync();
883 return process.ExitCode;
884 }
