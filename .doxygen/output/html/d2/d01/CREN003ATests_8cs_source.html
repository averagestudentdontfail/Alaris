56 #region Intrinsic Value Floor
57
58 [Theory]
59 [InlineData(150, 100, 0.25, 0.30, 0.05, 0.02)] // Deep ITM call
60 [InlineData(50, 100, 0.25, 0.30, 0.05, 0.02)] // Deep ITM put
61 [InlineData(105, 100, 0.1, 0.20, 0.03, 0.01)] // Slightly ITM call
62 [InlineData(95, 100, 0.1, 0.20, 0.03, 0.01)] // Slightly ITM put
63 public void Price_AtLeastIntrinsicValue(
64 double spot, double strike, double tau, double sigma, double r, double q)
65 {
66 // Tolerance accounts for FD grid resolution numerical precision
67 // For deep ITM options (S/K = 0.5), FD can have ~0.001% error
68 const double IntrinsicTolerance = 0.01; // Allow 1 cent deviation
69
70 // Call intrinsic = max(0, S - K)
71 double callIntrinsic = System.Math.Max(0, spot - strike);
72 double americanCall = _engine.Price(spot, strike, tau, r, q, sigma, OptionType.Call);
73 Assert.True(americanCall >= callIntrinsic - IntrinsicTolerance,
74 $"American call {americanCall:F6} should be >= intrinsic {callIntrinsic:F6}");
75
76 // Put intrinsic = max(0, K - S)
77 double putIntrinsic = System.Math.Max(0, strike - spot);
78 double americanPut = _engine.Price(spot, strike, tau, r, q, sigma, OptionType.Put);
79 Assert.True(americanPut >= putIntrinsic - IntrinsicTolerance,
80 $"American put {americanPut:F6} should be >= intrinsic {putIntrinsic:F6}");
81 }

199 [Fact]
200 public void Price_NearExpiryOption_StillValid()
201 {
202 double tau = 1.0 / 252.0; // 1 trading day
203 double americanCall = _engine.Price(100, 100, tau, 0.05, 0.02, 0.30, OptionType.Call);
204 double americanPut = _engine.Price(100, 100, tau, 0.05, 0.02, 0.30, OptionType.Put);
205
206 // ATM options (S=K) have 0 intrinsic value, so near-expiry can return 0 or small positive
207 // The invariant is: value >= 0 (non-negative) and for longer expiry, value > 0
208 Assert.True(americanCall >= 0, $"Near-expiry call should be non-negative, got {americanCall}");
209 Assert.True(americanPut >= 0, $"Near-expiry put should be non-negative, got {americanPut}");
210
211 // For slightly longer expiry (1 week), ATM should have positive time value
212 double weekTau = 5.0 / 252.0;
213 double weekCall = _engine.Price(100, 100, weekTau, 0.05, 0.02, 0.30, OptionType.Call);
214 double weekPut = _engine.Price(100, 100, weekTau, 0.05, 0.02, 0.30, OptionType.Put);
215 Assert.True(weekCall > 0, $"1-week ATM call should have positive value, got {weekCall}");
216 Assert.True(weekPut > 0, $"1-week ATM put should have positive value, got {weekPut}");
217 }
