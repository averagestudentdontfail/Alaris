52 public STIV007A CalibrateFromChain(
53 IReadOnlyList<STIV008A> chain,
54 double spotPrice,
55 double timeToExpiry,
56 double riskFreeRate)
57 {
58 ArgumentNullException.ThrowIfNull(chain);
59
60 if (chain.Count < 3)
61 {
62 throw new ArgumentException("At least 3 strikes required for calibration", nameof(chain));
63 }
64
65 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(spotPrice);
66 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(timeToExpiry);
67
68 // Find ATM strike (closest to spot)
69 STIV008A atmQuote = chain
70 .OrderBy(q => Math.Abs(q.Strike - spotPrice))
71 .First();
72
73 double atmIV = atmQuote.ImpliedVolatility;
74
75 // Calculate skew from 25-delta options if available
76 double skewCoefficient = CalculateSkewFromChain(chain, spotPrice, atmIV, timeToExpiry);

99 public double InterpolateIV(STIV007A parameters, double strike)
100 {
101 ArgumentNullException.ThrowIfNull(parameters);
102 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(strike);
103
104 // Calculate moneyness (log-moneyness normalised by vol×sqrt(T))
105 double logMoneyness = Math.Log(strike / parameters.SpotPrice);
106 double normalisedMoneyness = logMoneyness /
107 (parameters.AtmImpliedVolatility * Math.Sqrt(Math.Max(parameters.TimeToExpiry, MinTimeToExpiry)));
108
109 // Apply skew adjustment
110 // σ(K) = σ_ATM × (1 + ρ × normalised_moneyness)
111 double skewAdjustment = parameters.SkewCoefficient * normalisedMoneyness;
112
113 // Clamp adjustment to prevent extreme values
114 skewAdjustment = Math.Clamp(skewAdjustment, -0.30, 0.30);
115
116 double interpolatedIV = parameters.AtmImpliedVolatility * (1.0 + skewAdjustment);
117
118 // Ensure IV remains positive and reasonable
119 return Math.Clamp(interpolatedIV, 0.05, 3.0);
120 }

131 public double CalculateStickyDeltaIVChange(
132 STIV007A parameters,
133 double originalStrike,
134 double originalSpot,
135 double currentSpot)
136 {
137 ArgumentNullException.ThrowIfNull(parameters);
138 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(originalStrike);
139 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(originalSpot);
140 ArgumentOutOfRangeException.ThrowIfNegativeOrZero(currentSpot);
141
142 double spotReturn = (currentSpot - originalSpot) / originalSpot;
143 double sqrtT = Math.Sqrt(Math.Max(parameters.TimeToExpiry, MinTimeToExpiry));
144
145 // Under sticky-delta: IV at original strike changes as spot moves
146 // dσ/dS = -ρ / (S × σ × √T) approximately
147 double ivChange = -parameters.SkewCoefficient * spotReturn /
148 (parameters.AtmImpliedVolatility * sqrtT);
149
150 return ivChange;
151 }

160 public STIV009A EvaluateSkewExposure(
161 STIV007A parameters,
162 double currentMoneyness,
163 double entryMoneyness)
164 {
165 ArgumentNullException.ThrowIfNull(parameters);
166
167 double moneynessShift = currentMoneyness - entryMoneyness;
168 double logMoneynessShift = Math.Log(currentMoneyness / entryMoneyness);
169
170 // Calculate skew P&L impact
171 // Approximate: ΔPnL_skew ≈ Vega × Δσ(from skew)
172 double skewIVImpact = parameters.SkewCoefficient * logMoneynessShift /
173 (parameters.AtmImpliedVolatility * Math.Sqrt(Math.Max(parameters.TimeToExpiry, MinTimeToExpiry)));
174
175 STIV010A riskLevel = Math.Abs(moneynessShift) switch
176 {
177 < 0.02 => STIV010A.Low,
178 < 0.05 => STIV010A.Moderate,
179 < 0.10 => STIV010A.Elevated,
180 _ => STIV010A.High
181 };
182
183 return new STIV009A
184 {
185 MoneynessShift = moneynessShift,
186 EstimatedIVImpact = skewIVImpact,
187 RiskLevel = riskLevel,
188 ShouldRecenter = riskLevel >= STIV010A.Elevated,
189 Rationale = riskLevel >= STIV010A.Elevated
190 ? $"Position has drifted {Math.Abs(moneynessShift):P1} from ATM - consider recentering"
191 : "Skew exposure within acceptable bounds"
192 };
193 }

198 private static double CalculateSkewFromChain(
199 IReadOnlyList<STIV008A> chain,
200 double spotPrice,
201 double atmIV,
202 double timeToExpiry)
203 {
204 // Find puts around 25-delta (≈ 0.85-0.90 moneyness) and calls around 25-delta (≈ 1.05-1.10 moneyness)
205 List<STIV008A> lowStrikes = chain
206 .Where(q => q.Strike < spotPrice * 0.95)
207 .OrderByDescending(q => q.Strike)
208 .Take(3)
209 .ToList();
210
211 List<STIV008A> highStrikes = chain
212 .Where(q => q.Strike > spotPrice * 1.05)
213 .OrderBy(q => q.Strike)
214 .Take(3)
215 .ToList();
216
217 if (lowStrikes.Count == 0 || highStrikes.Count == 0)
218 {
219 return DefaultSkewCoefficient;
220 }
221
222 double avgLowIV = lowStrikes.Average(q => q.ImpliedVolatility);
223 double avgHighIV = highStrikes.Average(q => q.ImpliedVolatility);
224
225 // Risk reversal = Vol(OTM puts) - Vol(OTM calls)
226 double riskReversal = avgLowIV - avgHighIV;
227
228 // Convert to skew coefficient (normalise by ATM IV)
229 double skew = riskReversal / (2.0 * atmIV);
230
231 // Clamp to reasonable range
232 return Math.Clamp(skew, -0.50, 0.10);
233 }
