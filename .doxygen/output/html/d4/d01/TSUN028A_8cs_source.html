306 [Fact]
307 public void TermStructureSlope_IsNegativeForInvertedStructure()
308 {
309 // Arrange
310 double frontMonthIV = 0.28;
311 double backMonthIV = 0.22;
312 int frontDTE = 30;
313 int backDTE = 90;
314
315 // Act - Calculate slope: (backIV - frontIV) / (backDTE - frontDTE)
316 double slope = (backMonthIV - frontMonthIV) / (backDTE - frontDTE);
317
318 // Assert - Inverted term structure has negative slope
319 slope.Should().BeLessThan(0,
320 "Inverted term structure should have negative slope");
321
322 // The slope is -0.001 per DTE, which is -0.06/60 = -0.001
323 slope.Should().BeApproximately(-0.001, 0.0001);
324 }
