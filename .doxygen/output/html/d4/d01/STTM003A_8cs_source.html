264 private static DateTime GetGoodFriday(int year)
265 {
266 // Meeus/Jones/Butcher algorithm for Gregorian calendar
267 int a = year % 19;
268 int b = year / 100;
269 int c = year % 100;
270 int d = b / 4;
271 int e = b % 4;
272 int f = (b + 8) / 25;
273 int g = (b - f + 1) / 3;
274 int h = ((19 * a) + b - d - g + 15) % 30;
275 int i = c / 4;
276 int k = c % 4;
277 int l = (32 + (2 * e) + (2 * i) - h - k) % 7;
278 int m = (a + (11 * h) + (22 * l)) / 451;
279 int month = (h + l - (7 * m) + 114) / 31;
280 int day = ((h + l - (7 * m) + 114) % 31) + 1;
281
282 DateTime easter = new DateTime(year, month, day);
283 // Good Friday is 2 days before Easter
284 return easter.AddDays(-2);
285 }
