170 public async Task<OptionContract> GetSnapshotQuoteAsync(
171 string underlyingSymbol,
172 decimal strike,
173 DateTime expiration,
174 OptionRight right,
175 CancellationToken cancellationToken = default)
176 {
177 if (string.IsNullOrWhiteSpace(underlyingSymbol))
178 throw new ArgumentException("Underlying symbol cannot be null or whitespace", nameof(underlyingSymbol));
179
180 _logger.LogInformation(
181 "Requesting snapshot quote for {Symbol} {Strike} {Right} expiring {Expiration:yyyy-MM-dd}",
182 underlyingSymbol, strike, right, expiration);
183
184 var reqId = Interlocked.Increment(ref _nextRequestId);
185 var tcs = new TaskCompletionSource<OptionContract>();
186
187 if (!_pendingRequests.TryAdd(reqId, tcs))
188 throw new InvalidOperationException($"Request ID {reqId} already in use");
189
190 // Initialize quote state
191 var state = new QuoteState
192 {
193 UnderlyingSymbol = underlyingSymbol,
194 Strike = strike,
195 Expiration = expiration,
196 Right = right
197 };
198 _quoteStates[reqId] = state;
199
200 try
201 {
202 using var timeoutCts = CancellationTokenSource.CreateLinkedTokenSource(cancellationToken);
203 timeoutCts.CancelAfter(_quoteTimeout);
204
205 if (_isBacktestMode)
206 {
207 // Backtest mode: generate simulated quote
208 await GenerateBacktestQuoteAsync(reqId, state);
209 }
210 else if (!_isConnected)
211 {
212 // Not connected - fallback to simulated quote with warning
213 _logger.LogWarning("IB Gateway not connected - using simulated quote for {Symbol}", underlyingSymbol);
214 await GenerateBacktestQuoteAsync(reqId, state);
215 }
216 else
217 {
218 // Live mode: request from IB Gateway
219 await RequestLiveQuoteAsync(reqId, state);
220 }
221
222 var quote = await tcs.Task.WaitAsync(timeoutCts.Token);
223
224 _logger.LogInformation(
225 "Received quote for {Symbol}: Bid={Bid:F2}, Ask={Ask:F2}, Mid={Mid:F2}",
226 underlyingSymbol, quote.Bid, quote.Ask, (quote.Bid + quote.Ask) / 2);
227
228 return quote;
229 }
230 catch (OperationCanceledException)
231 {
232 _logger.LogError("Snapshot quote request timed out for {Symbol} after {Timeout}s",
233 underlyingSymbol, _quoteTimeout.TotalSeconds);
234 throw new TimeoutException($"Snapshot quote request timed out after {_quoteTimeout.TotalSeconds} seconds");
235 }
236 finally
237 {
238 _pendingRequests.TryRemove(reqId, out _);
239 _quoteStates.TryRemove(reqId, out _);
240 }
241 }

247 private Task GenerateBacktestQuoteAsync(int reqId, QuoteState state)
248 {
249 return Task.Run(async () =>
250 {
251 // Small delay to simulate network latency
252 await Task.Delay(50);
253
254 if (_pendingRequests.TryGetValue(reqId, out var tcs))
255 {
256 // Generate realistic mid price based on strike and DTE
257 var daysToExpiry = (state.Expiration - DateTime.UtcNow.Date).Days;
258 var atm = state.Strike; // Assume ATM for simulation
259
260 // Base price: ~$2-5 for ATM options with 30 DTE
261 var dteFactor = daysToExpiry / 30.0m;
262 var basePrice = 2.0m + (dteFactor * 1.5m);
263
264 // Realistic bid-ask spread: wider for short-dated, narrower for liquid
265 var spreadWidth = Math.Max(0.05m, basePrice * 0.03m); // 3% of price, min $0.05
266
267 var mid = basePrice;
268 var bid = mid - (spreadWidth / 2);
269 var ask = mid + (spreadWidth / 2);
270
271 var quote = new OptionContract
272 {
273 UnderlyingSymbol = state.UnderlyingSymbol,
274 OptionSymbol = FormatOptionSymbol(state),
275 Strike = state.Strike,
276 Expiration = state.Expiration,
277 Right = state.Right,
278 Bid = Math.Max(0.01m, bid),
279 Ask = ask,
280 Timestamp = DateTime.UtcNow,
281 Volume = 100 + (daysToExpiry * 10), // More volume for longer-dated
282 OpenInterest = 500 + (daysToExpiry * 20)
283 };
284
285 tcs.TrySetResult(quote);
286 }
287 });
288 }

315 public async Task<DTmd002A> GetDTmd002AAsync(
316 string underlyingSymbol,
317 decimal strike,
318 DateTime frontExpiration,
319 DateTime backExpiration,
320 OptionRight right,
321 CancellationToken cancellationToken = default)
322 {
323 _logger.LogInformation(
324 "Requesting calendar spread quote for {Symbol} {Strike} {Right}",
325 underlyingSymbol, strike, right);
326
327 // Request both legs concurrently
328 var frontTask = GetSnapshotQuoteAsync(
329 underlyingSymbol, strike, frontExpiration, right, cancellationToken);

331 var backTask = GetSnapshotQuoteAsync(

333
334 await Task.WhenAll(frontTask, backTask);
335
336 var frontLeg = await frontTask;
337 var backLeg = await backTask;
338
339 var calendarSpread = new DTmd002A

341 UnderlyingSymbol = underlyingSymbol,
342 Strike = strike,
343 FrontLeg = frontLeg,
344 BackLeg = backLeg,
345 Timestamp = DateTime.UtcNow
346 };
347
348 _logger.LogInformation(
349 "Calendar spread quote: Bid={Bid:F2}, Ask={Ask:F2}, Mid={Mid:F2}, Width={Width:F2}",

351 calendarSpread.SpreadAsk,
352 calendarSpread.SpreadMid,
353 calendarSpread.SpreadWidth);

362 internal void OnTickPrice(int reqId, int field, decimal price)

364 if (!_quoteStates.TryGetValue(reqId, out var state))

367 // Field codes: 1=Bid, 2=Ask
