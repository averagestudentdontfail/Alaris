70 public int GetBusinessDaysBetween(DateTime startDate, DateTime endDate)
71 {
72 if (endDate <= startDate)
73 {
74 return 0;
75 }
76
77 // Include first day, exclude last day (standard convention for T-to-expiry)
78 int count = 0;
79 DateTime current = startDate.Date;
80 DateTime end = endDate.Date;
81
82 // Bounded loop - max reasonable range is ~10 years (Rule 3)
83 const int maxDays = 365 * 10;
84 int iterations = 0;
85
86 while (current < end && iterations < maxDays)
87 {
88 if (IsBusinessDay(current))
89 {
90 count++;
91 }
92 current = current.AddDays(1);
93 iterations++;
94 }
95
96 return count;
97 }

100 public bool IsBusinessDay(DateTime targetDate)
101 {

166 private static HashSet<DateTime> ComputeNYSEHolidays(int year)
167 {
168 HashSet<DateTime> holidays = [];
169
170 // 1. New Year's Day
171 DateTime newYears = new(year, 1, 1);
172 holidays.Add(AdjustWeekendHoliday(newYears));
173
174 // 2. Martin Luther King Jr. Day (3rd Monday of January)
175 DateTime mlkDay = GetNthWeekdayOfMonth(year, 1, DayOfWeek.Monday, 3);
176 holidays.Add(mlkDay);
177
178 // 3. Presidents' Day (3rd Monday of February)
179 DateTime presidentsDay = GetNthWeekdayOfMonth(year, 2, DayOfWeek.Monday, 3);
180 holidays.Add(presidentsDay);
181
182 // 4. Good Friday (Friday before Easter)
183 DateTime easter = ComputeEasterSunday(year);
184 DateTime goodFriday = easter.AddDays(-2);
185 holidays.Add(goodFriday);
186
187 // 5. Memorial Day (Last Monday of May)
188 DateTime memorialDay = GetLastWeekdayOfMonth(year, 5, DayOfWeek.Monday);
189 holidays.Add(memorialDay);
190
191 // 6. Juneteenth (June 19) - Official NYSE holiday starting 2021
192 if (year >= 2021)
193 {
194 DateTime juneteenth = new(year, 6, 19);
195 holidays.Add(AdjustWeekendHoliday(juneteenth));
196 }
197
198 // 7. Independence Day (July 4)
199 DateTime independenceDay = new(year, 7, 4);
200 holidays.Add(AdjustWeekendHoliday(independenceDay));
201
202 // 8. Labor Day (1st Monday of September)
203 DateTime laborDay = GetNthWeekdayOfMonth(year, 9, DayOfWeek.Monday, 1);
204 holidays.Add(laborDay);
205
206 // 9. Thanksgiving (4th Thursday of November)
207 DateTime thanksgiving = GetNthWeekdayOfMonth(year, 11, DayOfWeek.Thursday, 4);
208 holidays.Add(thanksgiving);
209
210 // 10. Christmas (December 25)
211 DateTime christmas = new(year, 12, 25);
212 holidays.Add(AdjustWeekendHoliday(christmas));
213
214 return holidays;
215 }

222 private static DateTime AdjustWeekendHoliday(DateTime holiday)
223 {
224 return holiday.DayOfWeek switch

226 DayOfWeek.Saturday => holiday.AddDays(-1), // Friday
227 DayOfWeek.Sunday => holiday.AddDays(1), // Monday
228 _ => holiday

239 private static DateTime GetNthWeekdayOfMonth(int year, int month, DayOfWeek dayOfWeek, int n)
240 {
241 DateTime first = new(year, month, 1);
242 int daysToAdd = ((int)dayOfWeek - (int)first.DayOfWeek + 7) % 7;
243 DateTime firstOccurrence = first.AddDays(daysToAdd);
244 return firstOccurrence.AddDays((n - 1) * 7);
245 }
246
250 private static DateTime GetLastWeekdayOfMonth(int year, int month, DayOfWeek dayOfWeek)
251 {
252 DateTime lastDay = new(year, month, DateTime.DaysInMonth(year, month));
253 int daysToSubtract = ((int)lastDay.DayOfWeek - (int)dayOfWeek + 7) % 7;
254 return lastDay.AddDays(-daysToSubtract);
255 }

263 private static DateTime ComputeEasterSunday(int year)
264 {
265 int a = year % 19;
266 int b = year / 100;
267 int c = year % 100;
268 int d = b / 4;
269 int e = b % 4;
270 int f = (b + 8) / 25;
271 int g = (b - f + 1) / 3;
272 int h = ((19 * a) + b - d - g + 15) % 30;
273 int i = c / 4;
274 int k = c % 4;
275 int l = (32 + (2 * e) + (2 * i) - h - k) % 7;
276 int m = (a + (11 * h) + (22 * l)) / 451;
277 int month = (h + l - (7 * m) + 114) / 31;
278 int day = ((h + l - (7 * m) + 114) % 31) + 1;
279
280 return new DateTime(year, month, day);
281 }
