11 #region Actual/365 Fixed Tests
12
13 [Fact]
14 public void Actual365Fixed_OneYear_ReturnsOne()
15 {
16 var start = new CRTM005A(1, CRTM005AMonth.January, 2024);
17 var end = new CRTM005A(1, CRTM005AMonth.January, 2025);
18
19 double yearFraction = DayCounters.Actual365Fixed.YearFraction(start, end);
20
21 // 2024 is leap year: 366 days
22 Assert.Equal(366.0 / 365.0, yearFraction, precision: 6);
23 }
24
25 [Fact]
26 public void Actual365Fixed_HalfYear_ReturnsHalf()
27 {
28 var start = new CRTM005A(1, CRTM005AMonth.January, 2025);
29 var end = new CRTM005A(2, CRTM005AMonth.July, 2025);
30
31 double yearFraction = DayCounters.Actual365Fixed.YearFraction(start, end);
32
33 // Jan-Jun = 31+28+31+30+31+30+1 = 182 days
34 Assert.True(yearFraction > 0.49 && yearFraction < 0.51, $"Year fraction {yearFraction} should be ~0.5");
35 }
36
37 [Fact]
38 public void Actual365Fixed_SameDate_ReturnsZero()
39 {
40 var date = new CRTM005A(15, CRTM005AMonth.June, 2024);
41
42 double yearFraction = DayCounters.Actual365Fixed.YearFraction(date, date);
43
44 Assert.Equal(0.0, yearFraction);
45 }
46
47 [Fact]
48 public void Actual365Fixed_DayCount_ReturnsActualDays()
49 {
50 var start = new CRTM005A(1, CRTM005AMonth.January, 2024);
51 var end = new CRTM005A(1, CRTM005AMonth.February, 2024);
52
53 int days = DayCounters.Actual365Fixed.DayCount(start, end);
54
55 Assert.Equal(31, days);
56 }
