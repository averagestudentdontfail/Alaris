134 public CRTM005A AddMonths(int months)
135 {
136 DateTime current = ToDateTime();
137 int targetMonth = current.Month + months;
138 int targetYear = current.Year + ((targetMonth - 1) / 12);
139
140 // Handle negative month values
141 if (targetMonth <= 0)
142 {
143 targetYear += (targetMonth / 12) - 1;
144 targetMonth = 12 + (targetMonth % 12);
145 if (targetMonth == 0)
146 {
147 targetMonth = 12;
148 }
149 }
150 else
151 {
152 targetMonth = ((targetMonth - 1) % 12) + 1;
153 }
154
155 // Cap day at end of month
156 int daysInTargetMonth = DateTime.DaysInMonth(targetYear, targetMonth);
157 int targetDay = System.Math.Min(current.Day, daysInTargetMonth);
158
159 return new CRTM005A(targetDay, targetMonth, targetYear);
160 }

167 public CRTM005A AddYears(int years)

172 // Handle Feb 29 on non-leap years
