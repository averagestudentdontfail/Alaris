267 public TransitionResult<TState, TEvent> Fire(TEvent @event)
268 {
269 // Freeze on first use (defensive immutability)
270 _frozen = true;
271
272 var lockTaken = false;
273 try
274 {
275 _stateLock.Enter(ref lockTaken);
276
277 var key = (_currentState, @event);
278
279 if (!_transitions.TryGetValue(key, out var transition))
280 {
281 var result = TransitionResult<TState, TEvent>.InvalidTransition(_currentState, @event);
282 RecordTransition(result);
283 return result;
284 }
285
286 // Check guard condition
287 if (transition.Guard != null && !transition.Guard())
288 {
289 var result = TransitionResult<TState, TEvent>.GuardFailed(_currentState, @event, transition.Target);
290 RecordTransition(result);
291 return result;
292 }
293
294 var previousState = _currentState;
295
296 // Execute exit action (Moore model)
297 try
298 {
299 if (_stateExitActions.TryGetValue(previousState, out var exitAction))
300 exitAction?.Invoke();
301 }
302 catch (Exception ex)
303 {
304 var result = TransitionResult<TState, TEvent>.ActionFailed(previousState, @event, ex, "Exit action failed");
305 RecordTransition(result);
306 return result;
307 }
308
309 // Execute transition action (Mealy model)
310 try
311 {
312 transition.Action?.Invoke();
313 }
314 catch (Exception ex)
315 {
316 var result = TransitionResult<TState, TEvent>.ActionFailed(_currentState, @event, ex, "Transition action failed");
317 RecordTransition(result);
318 return result;
319 }
320
321 // Transition to new state
322 _currentState = transition.Target;
323
324 // Execute entry action (Moore model)
325 try
326 {
327 if (_stateEntryActions.TryGetValue(_currentState, out var entryAction))
328 entryAction?.Invoke();
329 }
330 catch (Exception ex)
331 {
332 // State already changed, log but don't revert
333 var result = TransitionResult<TState, TEvent>.ActionFailed(previousState, @event, ex, "Entry action failed");
334 RecordTransition(result);
335 return result;
336 }
337
338 var successResult = TransitionResult<TState, TEvent>.Success(previousState, @event, _currentState);
339 RecordTransition(successResult);
340 return successResult;
341 }
342 finally
343 {
344 if (lockTaken) _stateLock.Exit();
345 }
346 }

428 public IReadOnlyList<string> Validate()
429 {
430 var issues = new List<string>();
431
432 // Check that initial state has at least one outgoing transition
433 var hasInitialTransition = _transitions.Keys.Any(k => k.Item1.Equals(_initialState));
434 if (!hasInitialTransition)
435 {
436 issues.Add($"Initial state '{_initialState}' has no outgoing transitions");
437 }
438
439 // Build reachability graph
440 var allStates = _transitions.Values.Select(t => t.Target)
441 .Concat(_transitions.Keys.Select(k => k.Item1))
442 .Concat(new[] { _initialState })
443 .ToHashSet();
444
445 var reachableStates = new HashSet<TState> { _initialState };
446 var frontier = new Queue<TState>();
447 frontier.Enqueue(_initialState);
448
449 while (frontier.Count > 0)
450 {
451 var current = frontier.Dequeue();
452 var outgoing = _transitions
453 .Where(kvp => kvp.Key.Item1.Equals(current))
454 .Select(kvp => kvp.Value.Target);
455
456 foreach (var target in outgoing)
457 {
458 if (reachableStates.Add(target))
459 {
460 frontier.Enqueue(target);
461 }
462 }
463 }
464
465 // Check for unreachable states
466 var unreachable = allStates.Except(reachableStates).ToList();
467 foreach (var state in unreachable)
468 {
469 issues.Add($"State '{state}' is unreachable from initial state");
470 }
471
472 // Check for dead states (non-terminal states with no outgoing transitions)
473 foreach (var state in allStates)
474 {
475 if (_terminalStates.Contains(state))
476 continue;
477
478 var hasOutgoing = _transitions.Keys.Any(k => k.Item1.Equals(state));
479 if (!hasOutgoing)
480 {
481 issues.Add($"Non-terminal state '{state}' has no outgoing transitions (dead state)");
482 }
483 }
484
485 // Check that all terminal states are reachable
486 foreach (var terminal in _terminalStates)
487 {
488 if (!reachableStates.Contains(terminal))
489 {
490 issues.Add($"Terminal state '{terminal}' is unreachable");
491 }
492 }
493
494 return issues;
495 }

520 private bool DetectCycles()
521 {
522 var visited = new HashSet<TState>();
523 var inStack = new HashSet<TState>();
524
525 foreach (var state in _transitions.Keys.Select(k => k.Item1).Distinct())
526 {

528 return true;
529 }

531 return false;

534 private bool DetectCyclesDfs(TState state, HashSet<TState> visited, HashSet<TState> inStack)
535 {
536 if (inStack.Contains(state))
537 return true;
538
539 if (visited.Contains(state))
540 return false;
541
542 visited.Add(state);
543 inStack.Add(state);
544
545 var outgoing = _transitions
546 .Where(kvp => kvp.Key.Item1.Equals(state))
547 .Select(kvp => kvp.Value.Target);
548
549 foreach (var target in outgoing)
550 {
551 if (DetectCyclesDfs(target, visited, inStack))
552 return true;
553 }
554
555 inStack.Remove(state);
556 return false;
557 }
