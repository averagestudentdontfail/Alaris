\subsection{1. Rule Summary}\label{rule-summary}

\subsubsection{Option Pricing}\label{option-pricing}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  \textbf{American \(\geq\) European}: American option price shall
  exceed European price.
\item
  \textbf{American \(\geq\) Intrinsic}: American option price shall
  exceed intrinsic value.
\item
  \textbf{Regime Selection}: Use spectral collocation for
  \(\sigma \leq 100\%\); fallback to FD otherwise.
\end{enumerate}

\subsubsection{Signal Generation}\label{signal-generation}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\setcounter{enumi}{3}
\tightlist
\item
  \textbf{IV/RV Ratio}: Signal requires IV/RV \(\geq 1.25\).
\item
  \textbf{Term Structure}: Signal requires slope \(\leq -0.00406\).
\item
  \textbf{Liquidity}: Signal requires 30-day avg volume
  \(\geq 1,500,000\).
\end{enumerate}

\subsubsection{Position Sizing}\label{position-sizing}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\setcounter{enumi}{6}
\tightlist
\item
  \textbf{Kelly Fraction}: Use \(\kappa = 0.02\) (Recommended) or 0.01
  (Consider).
\item
  \textbf{Liquidity Limit}: Position \(\leq 5\%\) of daily volume,
  \(\leq 2\%\) of open interest.
\end{enumerate}

\subsubsection{Fault Detection}\label{fault-detection}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\setcounter{enumi}{8}
\tightlist
\item
  \textbf{Delta Drift}: Halt if \(|\Delta_{\text{net}}| > 0.10\).
\item
  \textbf{Vega Correlation}: Reject if \(|\rho_{\text{FB}}| > 0.70\).
\item
  \textbf{Cost Survival}: Reject if post-cost IV/RV \textless{} 1.20.
\end{enumerate}

\subsection{4. Option Pricing Framework}\label{option-pricing-framework}

\subsubsection{\texorpdfstring{Rule 1 (American \(\geq\)
European)}{Rule 1 (American \textbackslash geq European)}}\label{rule-1-american-geq-european}

The American option price shall satisfy:
\[V_{\text{American}}(S, K, \tau) \geq V_{\text{European}}(S, K, \tau)\]

\subsubsection{\texorpdfstring{Rule 2 (American \(\geq\)
Intrinsic)}{Rule 2 (American \textbackslash geq Intrinsic)}}\label{rule-2-american-geq-intrinsic}

The American option price shall satisfy:
\[V_{\text{Put}}(S, K, \tau) \geq (K - S)^+\]
\[V_{\text{Call}}(S, K, \tau) \geq (S - K)^+\]

\subsubsection{Rule 3 (Early Exercise
Premium)}\label{rule-3-early-exercise-premium}

The American value decomposes as:
\[V(\tau, S) = v(\tau, S) + \mathcal{P}(\tau, S)\]

where \(v\) is European value and \(\mathcal{P}\) is early exercise
premium.

\subsection{5. Rate Regime
Classification}\label{rate-regime-classification}

\subsubsection{Standard Regime}\label{standard-regime}

\textbf{Condition:} \(r \geq 0\) or \(r \geq q\)

\textbf{Boundary:} Single upper boundary \(B(\tau)\) satisfying:
\[\lim_{\tau \to \infty} B(\tau) = K \cdot \frac{\lambda_-}{\lambda_- - 1}\]

where \(\lambda_-\) is the negative root of:
\[\frac{1}{2}\sigma^2\lambda^2 + (r - q - \frac{\sigma^2}{2})\lambda - r = 0\]

\subsubsection{Double Boundary Regime}\label{double-boundary-regime}

\textbf{Condition:} \(q < r < 0\)

\textbf{Exercise Region:} \(\{S : Y(\tau) \leq S \leq B(\tau)\}\)

\textbf{Implementation:} FP-B' stabilised iteration Healy (2021).

\subsection{7. Implied Volatility
Computation}\label{implied-volatility-computation}

\subsubsection{Algorithm}\label{algorithm}

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\tightlist
\item
  \textbf{Initial Guess:} Corrado-Miller Corrado and Miller Jr (1996)
  approximation
\item
  \textbf{Refinement:} Householder(3) iteration (cubic convergence)
\item
  \textbf{Fallback:} Brent's method (guaranteed convergence)
\end{enumerate}

\subsubsection{Householder(3) Formula}\label{householder3-formula}

\[\sigma_{n+1} = \sigma_n - \frac{f(\sigma_n)}{f'(\sigma_n)} \cdot \left(1 + \frac{f\cdot f''}{2(f')^2}\right)\]

where \(f(\sigma) = V_{\text{BS}}(\sigma) - V_{\text{market}}\).

\subsubsection{Convergence}\label{convergence-1}

Typical: 2 iterations (vs 3-5 for Newton).

\subsubsection{Bounds}\label{bounds}

\begin{itemize}
\tightlist
\item
  Minimum: 0.001 (0.1\%)
\item
  Maximum: 5.0 (500\%)
\end{itemize}

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\subsection{8. Trading Signal
Generation}\label{trading-signal-generation}

\subsubsection{Rule 4 (IV/RV Ratio)}\label{rule-4-ivrv-ratio}

A signal requires: \[\frac{\sigma_I^{30}}{\sigma_R^{30}} \geq 1.25\]

where \(\sigma_I^{30}\) is 30-day implied volatility and
\(\sigma_R^{30}\) is Yang-Zhang realised volatility.

\subsubsection{Rule 5 (Term Structure)}\label{rule-5-term-structure}

A signal requires: \[\nabla_\tau \sigma_I \leq -0.00406\]

(Negative slope indicates front-month premium.)

\subsubsection{Rule 6 (Liquidity)}\label{rule-6-liquidity}

A signal requires: \[\bar{V}^{30} \geq 1,500,000\]

where \(\bar{V}^{30}\) is 30-day average daily volume.

\subsubsection{Signal Strength}\label{signal-strength}

\begin{longtable}[]{@{}ll@{}}
\toprule\noalign{}
Condition & Strength \\
\midrule\noalign{}
\endhead
\bottomrule\noalign{}
\endlastfoot
All three criteria & Recommended \\
Two criteria & Consider \\
Otherwise & Avoid \\
\end{longtable}

\subsubsection{Yang-Zhang Estimator}\label{yang-zhang-estimator}

\[\sigma_{YZ}^2 = \sigma_o^2 + k\sigma_c^2 + (1-k)\sigma_{RS}^2\]

where \(k = 0.34 / (1.34 + \frac{n+1}{n-1})\).

\subsection{9. Position Sizing}\label{position-sizing-1}

\subsubsection{Rule 7 (Kelly Fraction)}\label{rule-7-kelly-fraction}

Position size shall be: \[f_{\text{actual}} = \kappa \cdot f^*\]

where \(f^* = \frac{\mu}{\sigma^2}\) is full Kelly and:

\begin{longtable}[]{@{}ll@{}}
\toprule\noalign{}
Signal & \(\kappa\) \\
\midrule\noalign{}
\endhead
\bottomrule\noalign{}
\endlastfoot
Recommended & 0.02 \\
Consider & 0.01 \\
Avoid & 0 \\
\end{longtable}

\subsubsection{Rule 8 (Liquidity Limits)}\label{rule-8-liquidity-limits}

Position shall satisfy:
\[\frac{n}{V_{\text{daily}}} \leq 0.05 \quad \text{and} \quad \frac{n}{OI} \leq 0.02\]

\subsection{10. Fault Detection}\label{fault-detection-1}

\subsubsection{Rule 9 (Delta Drift)}\label{rule-9-delta-drift}

Halt trading if: \[|\Delta_{\text{net}}| > 0.10\]

\subsubsection{Rule 10 (Vega
Correlation)}\label{rule-10-vega-correlation}

Reject trade if front-back IV correlation exceeds threshold:
\[|\rho_{FB}| > 0.70\]

\subsubsection{Rule 11 (Cost Survival)}\label{rule-11-cost-survival}

Reject trade if post-cost ratio insufficient:
\[\frac{\sigma_I}{\sigma_R} \cdot \left(1 - \frac{C_{\text{exec}}}{V_{\text{position}}}\right) < 1.20\]
